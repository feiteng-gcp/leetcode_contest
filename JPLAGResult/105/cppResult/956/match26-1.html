typedef long long ll;

class Solution {
    const ll P=1000000007;
public:
    int numMusicPlaylists(int N, int L, int K) {
        vector<ll> ans(N+1);
        vector<vector<ll>> C(N+1);
        C[0]={1};
        for(int n=1;n<=N;n++){
            C[n]=vector<ll>(n+1);
            C[n][0]=C[n][n]=1;
            for(int i=1;i<n;i++)C[n][i]=(C[n-1][i-1]+C[n-1][i])%P;
        }

        for(int n=K+1;n<=N;n++){
            ans[n]=1;
            for(int i=0;i<K;i++)ans[n]=ans[n]*(n-i)%P;
            for(int i=K;i<L;i++)ans[n]=ans[n]*(n-K)%P;
            for(int m=n-1;m>K;m--)
                ans[n]=(ans[n]-ans[m]*C[n][n-m]%P+P)%P;
        }
        return ans[N];
    }
};
