class Solution {
public:
    int numMusicPlaylists(int N, int L, int K) {
        mo = 1e9 + 7;
        int res = a(N, K+1);
        vector<vector<int>> dp(L+1, vector<int>(N+1, 0));
        dp[0][N - K - 1] = 1;
        for (int i = 1; i <= L - K - 1; i++) {
            for (int j = 0; j <= N - K - 1; j++) {
                dp[i][j] = (long long)dp[i-1][j] * min(N-j, N-j-K) % mo;
                dp[i][j] += (long long)dp[i-1][j+1] * (j+1) % mo;
                dp[i][j] %= mo;
            }
        }
        return (int)((long long)res * dp[L-K-1][0] % mo);
    }
private:
    int mo;
    int a(int n, int k) {
        int res = 1;
        for (int i = 0; i < k; i++)
            res = (long long)res * (n-i) % mo;
        return res;
    }
};
