const int MXN = 110, MOD = 1e9 + 7;
typedef long long ll;
class Solution {
    long long c[MXN][MXN], g[MXN];
public:
    int numMusicPlaylists(int N, int L, int K) {
        for (int i = 0; i <= L; ++i) {
            c[i][0] = 1;
            for (int j = 1; j <= i; ++j)
            {
                c[i][j] = (c[i - 1][j] + c[i - 1][j - 1]) % MOD;
            }
        }
        ll ans = 0;
        for (int n = K; n <= N; ++n) {
            g[n] = 1;
            for (int i = 0; i < L; ++i)
                g[n] = g[n] * max(n - i, n - K) % MOD;
            if (n + N & 1)
                ans = (ans - g[n] * c[N][n]) % MOD;
            else
                ans = (ans + g[n] * c[N][n]) % MOD;
        }
        return (ans + MOD) % MOD;
    }
};
