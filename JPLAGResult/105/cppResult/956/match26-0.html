typedef long long int64;
const int MOD = 1e9 + 7;

const int N = 100 + 10;
int64 C[N][N];
int64 dp[N];

class Solution {
public:
    void add(int64& x, int y) {
        x = (x + y) % MOD;
    }
    int64 calc(int n, int L, int k) {
        int64 ret = 1;
        for (int i = n; i >= n - k + 1; --i) ret = ret * i % MOD;
        for (int i = 0; i < L - k; ++i) ret = ret * (n - k) % MOD;
        return ret;
    }
    int numMusicPlaylists(int n, int L, int k) {
        for (int i = 0; i <= n; ++i) {
            C[i][0] = C[i][i] = 1;
            for (int j = 1; j < i; ++j) {
                C[i][j] = (C[i - 1][j - 1] + C[i - 1][j]) % MOD;
            }
        }
        int ret = 0;
        for (int i = n; i >= k; --i) {
            int cur = calc(i, L, k) * C[n][i] % MOD;
            if ((n - i) % 2 == 0) {
                ret = (ret + cur) % MOD;
            } else {
                ret = (ret + MOD - cur) % MOD;
            }
        }
        return ret;
    }
};
