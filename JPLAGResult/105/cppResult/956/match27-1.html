class Solution {
public:
    int N_;
    int K_;
    int L_;
    long long f[105][105];
    const int MOD = 1000000007;
    long long dp(int now, int seen) {
        long long& ret = f[now][seen];
        if (now == L_) {
            return seen == N_;
        }
        if (ret == -1) {
            ret = 0;
            if (seen != N_) {
                ret += dp(now + 1, seen + 1) * (N_ - seen);
                ret %= MOD;
            }
            ret += dp(now + 1, seen) * max(0, seen - K_);
            ret %= MOD;

        }
        return ret;
    }
    int numMusicPlaylists(int N, int L, int K) {
        N_ = N;
        K_ = K;
        L_ = L;
        memset(f, -1, sizeof(f));
        return dp(0, 0);
    }
};
