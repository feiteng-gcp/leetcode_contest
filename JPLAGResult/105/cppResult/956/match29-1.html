class Solution {
public:
    int numMusicPlaylists(int N, int L, int K) {
        auto r = make_unique<int[]>((N+1) * (L+1));
        for(int i = 0; i <= L; i++){

            r[i * (N + 1)] = (i == 0);
            for(int j = 1; j <= N; j++){
                if(i == 0){
                    r[i * (N + 1) + j] = 0;
                }else{
                    long v = r[(i-1) * (N+1) + j - 1];
                    if(j > K){
                        v += (long)r[(i-1) * (N+1) + j] * (long)(j - K);
                    }
                    v %= 1000000007;
                    r[i * (N + 1) + j] = (int)v;
                }
            }
        }
        long v = r[L * (N+1) + N];
        for(int i = 2; i <= N; i++){
            v *= i;
            v %= 1000000007;
        }
        return (int)v;
    }
};
