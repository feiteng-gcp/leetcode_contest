class Solution {
public:
    int numMusicPlaylists(int N, int L, int K) {
        int mod = 1000000007;
        vector<vector<long>> dp(N + 1, vector<long>(L + 1, 0));
        dp.front().front() = 1;
        for (int i = 1; i <= N; ++i)
        {
            for (int j = 1; j <= L; ++j)
            {
                dp[i][j] = ((dp[i - 1][j - 1] * (N - i + 1)) % mod + (dp[i][j - 1] * max(0, i - K)) % mod) % mod;
            }
        }
        return dp.back().back();
    }
};
