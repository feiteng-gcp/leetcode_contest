/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
public:
    queue<TreeNode* > q;
    TreeNode* tree_root = NULL;
    TreeNode* node = NULL;
        
    CBTInserter(TreeNode* root) {
        tree_root = root;
        q.push(root);
        
        while (!q.empty()) {
            node = q.front();
            q.pop();
            
            if (NULL == node->left) 
            {
                break;
            }
            else
            {
                q.push(node->left);
            }
                
                
            if (NULL == node->right) 
            {
                break;
            }
            else
            {
                q.push(node->right);
            }
        }
        return;


    }
    
    int insert(int v) {
        int res = node->val;
        
        //cout << node->val << endl;
        if (NULL == node->left)
        {
            node->left = new TreeNode(v);
            q.push(node->left);
        }
        else if (NULL == node->right)
        {
            node->right = new TreeNode(v);
            q.push(node->right);
            node = q.front();
            q.pop();
        }
        return res;
    }
    
    TreeNode* get_root() {
        return tree_root;
    }
};

 * CBTInserter obj = new CBTInserter(root);
 * int param_1 = obj.insert(v);
 * TreeNode* param_2 = obj.get_root();
 */
