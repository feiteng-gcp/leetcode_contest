/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
private:
    queue<TreeNode*> q;
    
    TreeNode* root;
    
public:
    CBTInserter(TreeNode* root) {
        q.push(root);
        while (!q.empty()) {
            TreeNode* u = q.front(); 
            if (u->left) q.push(u->left);
            if (u->right) {
                q.push(u->right);
                q.pop();
            } else {
                break;
            }
        }
        this->root = root;
    }
    
    int insert(int v) {

        int ret = q.front()->val;
        if (!q.front()->left) {
            q.front()->left = new TreeNode(v);
            q.push(q.front()->left);
        } else {
            q.front()->right = new TreeNode(v);
            q.push(q.front()->right);
            q.pop();
        }
        return ret;
    }
    
    TreeNode* get_root() {
        return root;
    }
};
