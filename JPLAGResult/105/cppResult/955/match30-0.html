/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
public:
    int cnt;
    TreeNode* rt;
    int getNum(TreeNode *rt) {
        if (rt == nullptr) return 0;
        return 1 + getNum(rt->left) + getNum(rt->right);
    }
    CBTInserter(TreeNode* root) {
        rt = root;
        cnt = getNum(rt);
    }
    
    int insert(int v) {
        cnt++;
        
        int st = -1;
        for (int i = 30; i >= 0; i--) {
            if ((cnt >> i) & 1) {
                st = i;
                break;
            }
        }
        TreeNode* p = rt;
        for (int i = st - 1; i > 0; i--) {
            int dir = (cnt >> i) & 1;
            if (dir) p = p->right;
            else p = p->left;
        }
        int res = p->val;
        if (cnt & 1) {
            p->right = new TreeNode(v);
        } else {
            p->left = new TreeNode(v);
        }
        return res;
    }
    
    TreeNode* get_root() {
        return rt;
    }
};
