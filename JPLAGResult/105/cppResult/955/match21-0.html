/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
    vector<TreeNode*> vs;
    TreeNode* rt;
public:
    CBTInserter(TreeNode* root) {
        vs.push_back(NULL);
        queue<TreeNode*> q;
        if (root) q.push(root);
        while (!q.empty()) {
            int sz = q.size();
            for (int i = 0; i < sz; i++) {
                auto cur = q.front(); q.pop();
                vs.push_back(cur);
                if (cur->left) q.push(cur->left);
                if (cur->right) q.push(cur->right);
            }
        }
        rt = root;
    }
    
    int insert(int v) {
        int n = vs.size();
        n = n / 2;
        if (vs[n]->left) {
            vs[n]->right = new TreeNode(v);
            vs.push_back(vs[n]->right);
        } else {
            vs[n]->left = new TreeNode(v);
            vs.push_back(vs[n]->left);
        }
        return vs[n]->val;
        
    }
    
    TreeNode* get_root() {
        return rt;
    }
};
