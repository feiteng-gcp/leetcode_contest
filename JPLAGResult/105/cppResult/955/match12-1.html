/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
public:
    TreeNode *root;
    queue<TreeNode*> q;
    CBTInserter(TreeNode* root) {
        this->root = root;
        queue<TreeNode*> p;
        p.push(root);
        while(!p.empty())
        {
            TreeNode *t = p.front();
            p.pop();
            if(t->right == NULL)
                q.push(t);
            if(t->left)
                p.push(t->left);
            if(t->right)
                p.push(t->right);
        }
    }
    
    int insert(int v) {
        if(q.front()->right)
            q.pop();
        if(q.front()->left == NULL)

        {
            q.front()->left = new TreeNode(v);
            q.push(q.front()->left);
        }
        else
        {
            q.front()->right = new TreeNode(v);
            q.push(q.front()->right);
        }
        return q.front()->val;
    }
    
    TreeNode* get_root() {
        return root;
    }
};
