class CBTInserter {
public:
  int size;
  vector<TreeNode*> nodes;
  void dfs(TreeNode *root, int id) {
    if (root == NULL) return;
    size++;
    nodes[id] = root;
    dfs(root->left, 2*id+1);
    dfs(root->right, 2*id+2);
  }
  CBTInserter(TreeNode* root) {
    size = 0;
    nodes = vector<TreeNode*>(50000, NULL);
    dfs(root, 0);
  }

  int insert(int v) {
    TreeNode *node = new TreeNode(v);
    int id = size++;
    nodes[id] = node;
    int p = (id-1) / 2;
    if (id == 2*p + 1) nodes[p]->left = node;
    else nodes[p]->right = node;
    return nodes[p]->val;
  }

  TreeNode* get_root() {
    return nodes[0];
  }
};
