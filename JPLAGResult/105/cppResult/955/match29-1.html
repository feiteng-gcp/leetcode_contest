/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
    vector<TreeNode *> tree;
public:
    CBTInserter(TreeNode* root) {
        tree.push_back(root);
        int cur = 0;
        while (cur < tree.size()) {
            auto p = tree[cur];
            if (p->left) tree.push_back(p->left);
            if (p->right) tree.push_back(p->right);
            cur++;
        }
    }
    
    int insert(int v) {
        TreeNode *n = new TreeNode(v);
        tree.push_back(n);

        int pos = tree.size() - 1;
        int par = (pos - 1) / 2;
        if (pos & 1)
            tree[par]->left = n;
        else
            tree[par]->right = n;
        return tree[par]->val;
    }
    
    TreeNode* get_root() {
        return tree[0];
    }
};
