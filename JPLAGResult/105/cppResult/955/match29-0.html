/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
public:
    TreeNode *head_root;
    TreeNode *arr[2000];
    int tree_size;
    
    void buildTree(TreeNode *root)
    {
        deque<TreeNode*> q,q2;
        q.push_back(root);
        tree_size = 0;
        while(!q.empty())
        {
            while(!q.empty())
            {
                TreeNode *curr = q.front();
                q.pop_front();
                arr[tree_size] = curr;
                tree_size++;
                if(curr->left != NULL)
                    q2.push_back(curr->left);
                if(curr->right != NULL)
                    q2.push_back(curr->right);
                
            }
            q.swap(q2);
            q2.clear();
        }
    }
    
    CBTInserter(TreeNode* root) {
        head_root = root;
        buildTree(root);
    }
    
    int parent(int i)
    {
        return (i-1)/2;
    }
    
    int insert(int v) {
        TreeNode *node = new TreeNode(v);
        arr[tree_size] = node;
        tree_size++;
        int p_i = parent(tree_size-1);
        if(p_i*2+1 == tree_size-1)
            arr[p_i]->left = node;
        else
            arr[p_i]->right = node;
        return arr[p_i]->val;
    }
    
    TreeNode* get_root() {
        return head_root;
    }
};
