/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
    TreeNode* root;
    
    int minH(TreeNode* root) {
        if (!root) return 0;
        return 1 + min(minH(root->left), minH(root->right));
    }
    
    void helper(TreeNode*& curr) {
        if (!curr->right) return;
        if (minH(curr->left) > minH(curr->right)) curr = curr->right;
        else curr = curr->left;
        helper(curr);
    }
public:
    CBTInserter(TreeNode* root) {
        this->root = root;
    }
    
    int insert(int v) {
        TreeNode *temp = root;
        helper(temp);
        if (!temp->left) {
            temp->left = new TreeNode(v);
        } else {
            temp->right = new TreeNode(v);
        }
        return temp->val;
    }
    
    TreeNode* get_root() {
        return root;
    }
};
