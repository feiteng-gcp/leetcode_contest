const int MXN = 12000;
class CBTInserter {
    int n;
    TreeNode* nd[MXN];
    void dfs(int m, TreeNode* u) {
        if (!u)
            return;
        n = max(n, m);
        nd[m] = u;
        dfs(m + m, u->left);
        dfs(m * 2 + 1, u->right);
    }
public:
    CBTInserter(TreeNode* root) {
        n = 0;
        dfs(1, root);
    }
    
    int insert(int v) {
        //cout << n << endl;
        nd[++n] = new TreeNode(v);
        //db(nd[1]->val);
        if (n & 1)
        {
            nd[n >> 1]->right = nd[n];
        }
        else
        {
            nd[n >> 1]->left = nd[n];
        }
        return nd[n / 2]->val;
    }
    
    TreeNode* get_root() {
        return nd[1];
    }
};
