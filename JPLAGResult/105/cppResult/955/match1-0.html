/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
public:
    CBTInserter(TreeNode* root) {
        tree.push_back(root);
        int f = 0;
        while(tree.size() > f){
            auto n = tree[f];
            f++;
            if(n->left){
                tree.push_back(n->left);
            }
            if(n->right){
                tree.push_back(n->right);
            }
        }
    }
    
    int insert(int v) {
        auto newvalue = new TreeNode(v);
        tree.push_back(newvalue);
        int p = tree.size() / 2;
        if(p * 2 == tree.size()){
            tree[p-1]->left = newvalue;
        }else{
            tree[p-1]->right = newvalue;
        }
        return tree[p-1]->val;
    }
    
    TreeNode* get_root() {
        return tree[0];
    }
private:
    vector<TreeNode*> tree;
};

 * CBTInserter obj = new CBTInserter(root);
 * int param_1 = obj.insert(v);
 * TreeNode* param_2 = obj.get_root();
 */
