/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
public:
    CBTInserter(TreeNode* root) {
        m_root = root;
        enq(root);
    }
    
    int insert(int v) {
        while(fathers.size() > 0 && (fathers.front()->left && fathers.front() -> right)){
            fathers.pop();
        }
        if(fathers.size() == 0)
            return -1;
        
        TreeNode *f = fathers.front();
        //cout << "cur " << f->val<<endl;
        if(f->left == NULL){
            f->left = new TreeNode(v);
            fathers.push(f->left);
        }
            
        else{
            f->right = new TreeNode(v);
            fathers.push(f->right);
            fathers.pop();
        }
        return f->val;
    }
    
    TreeNode* get_root() {
        return m_root;
    }
    
private:
    queue<TreeNode *> fathers;
    TreeNode *m_root;
    
    void enq(TreeNode *root){
        if(root == NULL)
            return;
        
        queue<TreeNode *> q;
        q.push(root);
        while(q.size() > 0){
            if(q.front()->left)
                q.push(q.front()->left);
            if(q.front()->right)
                q.push(q.front()->right);
            //cout << "enqueue " << q.front()->val<<endl;
            fathers.push(q.front());
            q.pop();
        }
    }
};
