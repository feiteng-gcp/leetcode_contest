/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class CBTInserter {
public:
    CBTInserter(TreeNode* root) {
        r = root;
    }
    
    int insert(int v) {
        return insert_rec(r, new TreeNode(v));
    }
    
    TreeNode* get_root() {
        return r;
    }
private:
    TreeNode* r;
    
    int insert_rec(TreeNode* root, TreeNode* node) {
        if (!root->left) {
            root->left = node;
            return root->val;
        }
        else if (!root->right) {
            root->right = node;
            return root->val;
        }
        
        int lrh = 0;
        int rrh = 0;
        TreeNode* p = root->left;
        while(p) {
            lrh++;
            p = p->right;
        }
        p = root->right;
        while(p) {
            rrh++;
            p = p->right;
        }
        if (lrh != rrh) {
            return insert_rec(root->right, node);
        }
        else {
            return insert_rec(root->left, node);
        }
    }
};
