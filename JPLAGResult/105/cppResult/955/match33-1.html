/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
public:
  queue<TreeNode *> lastFull, last;
  TreeNode *root;
  CBTInserter(TreeNode* _root) {
    root = _root;
    vector<TreeNode *> nodes;
    queue<TreeNode *> q;
    q.push(root);
    while (!q.empty()) {
      TreeNode *node = q.front(); q.pop();
      nodes.push_back(node);
      if (node->left != NULL) q.push(node->left);
      if (node->right != NULL) q.push(node->right);
    }
    for (int i = 0, j = 1; i < nodes.size(); i += j, j *= 2) {
      if (i + j + 2 * j > nodes.size()) {
        for (int k = i; k < i + j; k++) lastFull.push(nodes[k]);
        for (int k = i + j; k < min((int)nodes.size(), i + 3 * j); k++) last.push(nodes[k]);
        while (!lastFull.empty()) {
          TreeNode *node = lastFull.front();
          if (node->left != NULL && node->right != NULL) lastFull.pop();
          else break;
        }
        break;
      }
    }
  }
    
  int insert(int v) {
    if (lastFull.empty()) {
      while (!last.empty()) {
        lastFull.push(last.front());
        last.pop();
      }
    }

    TreeNode *node = lastFull.front();
    if (node->left == NULL) {
      node->left = new TreeNode(v);
      last.push(node->left);
    } else {
      node->right = new TreeNode(v);
      last.push(node->right);
      lastFull.pop();
    }
    return node->val;
  }
    
  TreeNode* get_root() {
    return root;
  }
};
