/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
public:
    TreeNode *rt;
    queue<TreeNode*>q;
    CBTInserter(TreeNode* root) {
        rt=root;
        while(!q.empty())q.pop();
        queue<TreeNode*>qu;
        qu.push(rt);
        while(!qu.empty()){
            TreeNode *x=qu.front();qu.pop();
            if(x->left&&x->right){
                qu.push(x->left);
                qu.push(x->right);
            }
            else{
                q.push(x);
                if(x->left)qu.push(x->left);
            }
        }
    }
    
    int insert(int v) {
        TreeNode *y=new TreeNode(v);
        assert(!q.empty());
        TreeNode *x=q.front();
        if(!x->left){
            x->left=y;
        }
        else{
            x->right=y;
            q.pop();
        }
        q.push(y);
        return x->val;
    }
    
    TreeNode* get_root() {
        return rt;
    }
};
