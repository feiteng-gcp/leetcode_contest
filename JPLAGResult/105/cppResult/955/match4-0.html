/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
private:
    vector<TreeNode*> v;
    
public:
    CBTInserter(TreeNode* root) {
        v.clear();
        queue<TreeNode*> que;
        que.push(root);
        v.push_back(root);
        while (que.size()) {
            auto cur = que.front();
            que.pop();
            if (cur->left) {
                que.push(cur->left);
                v.push_back(cur->left);
            }
            if (cur->right) {
                que.push(cur->right);
                v.push_back(cur->right);
            }
        }
    }
    
    int insert(int val) {
        int idx = v.size();
        v.push_back(new TreeNode(val));
        if (idx % 2) {
            v[idx / 2]->left = v.back();
            return v[idx / 2]->val;
        } else {
            v[(idx - 2) / 2]->right = v.back();
            return v[(idx - 2) / 2]->val;
        }
    }
    
    TreeNode* get_root() {
        return v[0];
    }
};
