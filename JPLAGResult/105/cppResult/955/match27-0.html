/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
public:
    deque<TreeNode*> q;
    TreeNode* r;
    CBTInserter(TreeNode* root) {
        r = root;
        q.push_back(root);
        int b = 0;
        while (b < q.size()) {
            TreeNode* n = q[b];
            if (n->left && n->right) {
                q.pop_front();
                q.push_back(n->left);
                q.push_back(n->right);
                // cout << q.size();
            } else if (n->left) {
                q.push_back(n->left);
                // cout << q.size();
                ++b;
            } else {
                ++b;
            }
        }
    }
    
    int insert(int v) {
        int ret = q[0]->val;
        if (q[0]->left == NULL) {
            q[0]->left = new TreeNode(v);
            q.push_back(q[0]->left);
        } else if (q[0]->right == NULL) {
            q.front()->right = new TreeNode(v);
            q.push_back(q[0]->right);
            q.pop_front();
        }
        return ret;
    }
    
    TreeNode* get_root() {
        return r;
    }
};
