/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
    queue<TreeNode *> Q;
    TreeNode *r;
public:
    CBTInserter(TreeNode* root) {
        r = root;
        if(root){
            Q.push(root);
            while(true){
                auto p = Q.front();
                if(p->left && p->right){
                    Q.pop();
                    Q.push(p->left);
                    Q.push(p->right);
                }
                else if(p->left && !p->right){
                    Q.push(p->left);
                    break;
                }
                else{
                    break;
                }
            }
        }
    }
    int insert(int v) {

        auto p = Q.front();
        if(!p->left){
            p->left = new TreeNode(v);
            Q.push(p->left);
        }
        else{
            p->right = new TreeNode(v);
            Q.pop();
            Q.push(p->right);
        }
        return p->val;
    }
    
    TreeNode* get_root() {
        return r;
    }
};

 * int param_1 = obj.insert(v);
 * TreeNode* param_2 = obj.get_root();
