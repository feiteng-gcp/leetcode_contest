/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class CBTInserter {
public:
    vector<TreeNode*> node_;
    TreeNode* root_;
    CBTInserter(TreeNode* root) {
        node_.push_back(nullptr);
        root_ = root;
        queue<TreeNode*> Q;
        Q.push(root);
        while (!Q.empty()) {
            TreeNode* now = Q.front();
            Q.pop();
            if (now->left != nullptr) {
                Q.push(now->left);
            }
            if (now->right != nullptr) {
                Q.push(now->right);
            }
            node_.push_back(now);
        }
    }
    
    int insert(int v) {
        TreeNode* now = new TreeNode(v);
        node_.push_back(now);
        int index = (node_.size() - 1) / 2;

        TreeNode* father = node_[index];
        if (node_.size() & 1) {
            father->left = now;
        } else {
            father->right = now;
        }
        return father->val;
    }
    
    TreeNode* get_root() {
        return root_;
    }
};
