class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int res = A[0], len = A.size();
        vector<int> dp(len), psum(len);
        dp[0] = psum[0] = A[0];
        for (int i = 1; i < len; i++) {
            dp[i] = max(dp[i-1], 0) + A[i];
            psum[i] = psum[i-1] + A[i];
            res = max(res, dp[i]);
        }
        int suf_max = A[len-1], suf_sum = A[len-1];
        for (int i = len-2; i >= 0; i--) {
            res = max(res, psum[i] + suf_max);
            suf_sum += A[i];
            suf_max = max(suf_max, suf_sum);
        }
        return res;
    }
};
