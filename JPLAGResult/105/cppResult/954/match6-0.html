class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int startIndex = 0;
        int gMax = A[startIndex % A.size()];
        int carry = gMax > 0 ? gMax : 0;
        for(int i = 1; i < A.size(); i++){
            int cur = A[(startIndex + i) % A.size()];
            int curMax = (carry >= 0 ? carry + cur : cur);
            carry = max(0, curMax);
            gMax = max(curMax, gMax);
        }
        
        vector<int> sumMaxFromLeft(A.size()), sumMaxFromRight(A.size());        
        int sum = 0;
        sumMaxFromLeft[0] = 0;
        for(int i = 1; i < A.size() ;i++){
            sum += A[i-1];
            sumMaxFromLeft[i] = max(sum, sumMaxFromLeft[i-1]);
        }
        sum = A[A.size() - 1];
        sumMaxFromRight[A.size() - 1] = sum;
        for(int i = A.size() - 2; i >= 0; i--){
            sum += A[i];
            sumMaxFromRight[i] = max(sum, sumMaxFromRight[i+1]);
        }
        for(int i = 0; i < A.size(); i++)
            gMax = max(gMax, sumMaxFromLeft[i] + sumMaxFromRight[i]);
        
        return gMax;
    }
};
