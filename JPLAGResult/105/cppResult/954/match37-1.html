class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        vector<int> B = A;
        for (int i = 0; i < A.size(); i++) {
            B.push_back(A[i]);
        }
        
        vector<int> sum(B.size(), 0);
        sum[0] = B[0];
        for (int i = 1; i < B.size(); i++) {
            sum[i] = sum[i-1] + B[i];
            // cout << sum[i] << endl;
        }
        // cout << endl;
        
        int max_val = -100000000;
        vector< pair<int, int> > min_sum(B.size(), make_pair(0,0));
        int front = 0;
        int rear = 1;
        for (int i = 0; i < sum.size(); i++) {
            
            while (front < rear && (i - min_sum[front].second >= A.size())) {
                // cout << i << " " << min_sum[front].second << endl;
                front++;
            }
            
            // cout <<"f " << front << endl;
            
            int tmp = sum[i] - min_sum[front].first;
            
            // cout << tmp << " " << sum[i] << " " << min_sum[front].first << " " << min_sum[front].second << endl;
            max_val = max(max_val, tmp);
            
            while (front < rear && sum[i] <= min_sum[rear-1].first) {
                rear--;
            }
            
            min_sum[rear++] = make_pair(sum[i], i);
        }
                
        return max_val;
        
    }
};
