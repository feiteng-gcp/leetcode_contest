class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        vector<int> B = A;
        B.insert(B.end(), A.begin(), A.end());
        for (int i = 1; i < B.size(); i++) B[i] += B[i - 1];
        vector<pair<int, int>> maxa, mina;
        maxa.push_back({0,-1});
        mina.push_back({0,-1});
        int maxc = 0, minc = 0;
        int result = A[0];
        for (int i = 0; i < B.size(); i++) {
            while (!maxa.empty() && maxa.back().first <= B[i]) {
                maxa.pop_back();
            }
            maxa.push_back({B[i], i});
            if (maxc >= maxa.size()) maxc = maxa.size() - 1;
            while (maxa[maxc].second <= i - (int)A.size()) maxc++;
            if (i < A.size()) continue;
            int t = maxa[maxc].first - B[i - A.size()];
            if (t > result) result = t;
        }
        return result;
    }
};
