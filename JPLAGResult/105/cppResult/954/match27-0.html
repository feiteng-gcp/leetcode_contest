class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int n = A.size();
        
        vector<int> B(n,0);
        int maxValue = -300000;
        
        int sum = 0;
        for (int i=0; i < n; ++i) {
            sum += A[i];
            B[i] = -A[i];
            maxValue = max(A[i], maxValue);
        }
        
        if (maxValue < 0) return maxValue;
        
        return max(sum + maxSubarray(B), maxSubarray(A));
    }
    
    int maxSubarray(vector<int>& B) {
        int n = B.size();
        int runningSum = 0, maxSum = -1000000000;
        
        for (int i = 0; i < n; ++i) {
            runningSum += B[i];
            if (runningSum > maxSum)
                maxSum = runningSum;
            if (runningSum < 0) runningSum = 0;
        }
        return maxSum;
    }
};
