class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
      int ans = INT_MIN;
      int n = A.size();
      vector<int> s(2*n+1, 0);
      for (int i = 0; i < n; ++i ) s[i+1] = s[i] + A[i];
      for (int i = n; i < n*2; ++i ) s[i+1] = s[i] + A[i-n];
      deque<int> dq;
      dq.push_back(0);
      for (int i = 1; i < 2*n; ++i ) {
        while (dq.front() + n < i) dq.pop_front();
        ans = max(ans, s[i] - s[dq.front()]);
        while (!dq.empty() && s[dq.back()]>=s[i]) dq.pop_back();
        dq.push_back(i);
      }
      return ans;
    }
};
