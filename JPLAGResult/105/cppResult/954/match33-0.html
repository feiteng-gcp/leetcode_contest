class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int size = A.size();
        vector<int> f(size, 0), g(size, 0), m(size, 0), n(size, 0);
        int largest = A[0], smallest = A[0];
        int sum = A[0];
        for (int i = 1; i < size-1; ++i) {
            largest = max(largest, sum);
            smallest = min(smallest, sum);
            f[i] = largest;
            //m[i] = smallest;
            sum += A[i];
        }
        sum = A[size-1];
        largest = A[size-1];
        smallest = A[size-1];
        for (int i = size-2; i >0; --i) {
            largest = max(largest, sum);
            smallest = min(smallest, sum);
            g[i] = largest;
            //n[i] = smallest;
            sum += A[i];
        }
        sum = 0;
        smallest = 0;
        for (int i = 0; i < size; ++i) {
            m[i] = smallest;
            sum += A[i];
            smallest = min(smallest, sum);
        }
        sum = 0;
        smallest = 0;
        for (int i = size-1; i >= 0; --i) {
            n[i] = smallest;
            sum += A[i];
            smallest = min(smallest, sum);
        }
        
        int res = sum;
        for (int i = 1; i < size-1; ++i) {
            //res = max(res, sum - m[i] - n[i]);
            res = max(res, f[i] + g[i]);
        }
        for (int i = 0; i < size; ++i) {
            res = max(res, sum - m[i] - n[i]);
        }
        /*
        for (int i = 0; i < size; ++ i) {
            cout << f[i] <<"     "  << g[i] << "    " << m[i] << "    " << n[i] <<endl;
        }
        */
        res = max(res, sum);
        return res;
    }
};
