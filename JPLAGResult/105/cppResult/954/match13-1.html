class Solution {
public:
    int kadane(vector<int> A)  {

        int maxSoFar = 0, maxEndingHere = 0;
        for(int i=0;i<A.size();i++) {
            maxEndingHere += A[i];
            maxEndingHere = max(maxEndingHere, 0);
            maxSoFar = max(maxSoFar, maxEndingHere);
        }
        return maxSoFar;
    }
    
    int maxSubarraySumCircular(vector<int>& A) {
        bool foundPositive = false;
        int maxi = INT_MIN;
        for(int i=0;i<A.size();i++) {
            if(A[i]>=0)
            {
                foundPositive = true;
                break;
            }
            maxi = max(maxi, A[i]);
        }
        if(!foundPositive) return maxi;
        
        for(int i=0;i<A.size();i++)
            A[i] = -A[i];
        
        int revMax = kadane(A);
        
        for(int i=0;i<A.size();i++)
            A[i] = -A[i];
        
        
        return max(kadane(A), accumulate(A.begin(), A.end(),0) + revMax);
    }
};
