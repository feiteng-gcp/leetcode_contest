class Solution {
public:
    int kadanes(vector<int>& A){
        int m_s_f=A[0],m_e_h =A[0];
        for(int i=1;i<A.size();i++){
            m_e_h = max(A[i],m_e_h+A[i]);
            m_s_f = max(m_s_f,m_e_h);
        }
        return m_s_f;
    }
    int maxSubarraySumCircular(vector<int>& A) {
        int flag = 0;
        for(int i=0;i<A.size();i++){
            if(A[i]<0)
                continue;
            else{
                flag = 1;
                break;
            }
        }
        if(flag){
        int max_sum = kadanes(A);
        int max_wrap =0;
        for(int i=0;i<A.size();i++){
            max_wrap+=A[i];
            A[i]=-A[i];
        }
        max_wrap+=kadanes(A);
        if(max_wrap>max_sum)
            return max_wrap;
        else
            return max_sum;
        }
        else{
            return *std::max_element(A.begin(),A.end());
        }
    }
};
