class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int max1 = helper(A);

        int pSum = 0;
        for(int i = 0; i < A.size(); ++i){
            pSum += A[i];
            A[i] = -A[i];
        }
        int max2 = pSum + helper(A);
        //cout<< max1<<endl<<max2<<endl;
        if (max2 == 0 && max1<0) return max1;
        return max(max1, max2);
    }
private:
    int helper(vector<int>& A){
        int partialSum = A[0], Vmax = A[0];
        int partialmin = min(0, partialSum);
        for(int i = 1 ; i<A.size(); ++i){
            partialSum += A[i];
            Vmax = max(Vmax, partialSum - partialmin);
            partialmin = min(partialSum, partialmin);
        }
        return Vmax;
    }
};
