class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int len = A.size();
        for (int i = 0; i < len; i++)
            A.push_back(A[i]);
        int ans = INT_MIN;
        map<int, int> mp;
        mp[0] = -1;
        for (int i = 0, sum = 0; i < A.size(); i++) {
            sum += A[i];
            while (mp.size() && i - mp.begin()->second > len)
                mp.erase(mp.begin());
            ans = max(ans, sum - mp.begin()->first);
            mp[sum] = i;
        }
        return ans;
    }
};
