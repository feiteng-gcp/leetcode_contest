typedef long long ll;
class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int N=A.size();
        vector<ll> B;
        multiset<ll> S;

        ll s=0;
        B.push_back(s);
        for(int i=0;i<A.size();++i){
            s+=A[i];
            B.push_back(s);
        }
        for(int i=0;i<A.size();++i){
            s+=A[i];
            B.push_back(s);
        }
        ll best=*max_element(A.begin(),A.end());
        for(int i=0;i<2*N;++i){
            S.insert(B[i]);
            if(i>=N)S.erase(S.find(B[i-N]));
            //printf("%d %d %lld\n",i,(int)S.size(),*S.begin());
            best=max(best, B[i+1]-*S.begin());
        }
        return best;
    }
};
