class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        vector<int> left;
        vector<int> right;
        int n = A.size();
        int l = 0, r = 0;
        int in = INT_MIN;
        
        for(int i = 0; i < n; i++){
            l+=A[i];
            left.push_back(l);
        }
        

        int tempL = left[0], retIN=INT_MIN, retOUT= INT_MIN;
        int tempR = left[0];
        
        for(int i = 1; i < n; i++){
            
            retIN = max(retIN, left[i] - tempL);
            retOUT = max(retOUT, left[n-1] - (left[i] - tempR));
            tempL = min(tempL, left[i]);
            tempR = max(tempR, left[i]);
            
        }
        retIN = max(retIN, retOUT);
        return max(retIN, left[n-1]);
    }
};
