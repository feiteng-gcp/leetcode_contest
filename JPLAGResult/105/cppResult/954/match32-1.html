const int MAXN = 30000 + 10;
int dp[MAXN], pre[MAXN];
class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        memset(dp, 0, sizeof(dp));
        int n = A.size();
        map<int, int> table;

        for (int i = 1; i <= n; i++) {
            dp[i] = A[i-1] + dp[i-1];
            pre[i] = max(dp[i], pre[i-1]);
            table[dp[i]] += 1;
            
        }
        int ret = INT_MIN;
        for (int i = 1; i <= n; i++) {
            ret = max(ret, table.rbegin()->first - dp[i-1]);
            table[dp[i]]--;
            if (table[dp[i]] == 0) {
                table.erase(dp[i]);
            }
            // cout << i << ' ' << ret;
            ret = max(ret, dp[n]-dp[i-1]+pre[i-1]);
            // cout << ' ' << ret << endl;
            // cout << dp[n] << ' ' << dp[i-1] << ' ' << pre[i-1] << endl;
        }
        return ret;
        
    }
};
