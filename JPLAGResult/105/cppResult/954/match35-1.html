class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int n = A.size();
        A.resize(n * 2);
        for (int i = 0; i < n; ++i) A[n + i] = A[i];
        int sum = 0, cnt = 0;
        int best = *max_element(A.begin(), A.begin() + n);
        for (int i = 0; i < n * 2; ++i) {
            sum += A[i];
            ++cnt;
            if (cnt > n) {
                sum -= A[i - n];
                --cnt;
                int cur = 0, k = -1, best2 = 0;
                for (int j = i - n + 1; j < i; ++j) {
                    cur += A[j];
                    if (best2 >= cur) {
                        best2 = cur;
                        k = j;
                    }
                }
                if (k >= 0) {
                    sum -= best2;
                    cnt -= k - (i - n + 1) + 1;
                }
            }
            //cout << sum << " " << cnt << " " << i << endl;
            if (sum < 0) {
                sum = 0;
                cnt = 0;
            }
            if (cnt) best = max(best, sum);
        }
        return best;
    }
};
