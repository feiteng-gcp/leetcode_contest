class Solution {
public:
int maxSubarraySumCircular(vector<int>& A) {
    int n = A.size();
    vector<int> sum2(n);
    sum2[0] = A[0];
    int total = A[0];
    for (int i = 1; i < n; i++) {
        total += A[i];
        sum2[i] = max(total, sum2[i - 1]);
    }
    
    vector<int> sum1(n);
    sum1[n - 1] = A[n - 1];
    total = A[n - 1];
    for (int i = n - 2; i >= 0; i--) {
        total += A[i];
        sum1[i] = max(total, sum1[i + 1]);
        // cout << sum1[i] << " ";
    }
    
    int ret = INT_MIN;
    total = 0;
    for (int i = 0; i < n; i++) {
        if (A[i] > 0) {
            total += A[i];
            ret = max(total, ret);
        } else {
            ret = max(A[i], ret);
            if (total + A[i] >= 0) {
                total += A[i];
            } else {
                total = 0;
            }
        }
    }
    
    if (n < 2) {
        return ret;
    }
    
    for (int i = 1; i < n; i++) {
        ret = max(sum1[n - i] + sum2[n - i - 1], ret);
    }
    return ret;
}
