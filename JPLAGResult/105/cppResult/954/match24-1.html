class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int n = A.size();
        if(!n) return 0;
        long res = A[0];
        vector<long> B, P(2*n + 1, 0L);
        for(int k: A) {
            B.push_back(long(k));
            res = max(res, long(k));
        }
        for(int k: A) B.push_back(long(k));
        for(int i=1; i<=2*n; ++i) P[i] = P[i-1] + B[i-1];
        deque<long> X;
        X.push_back(0);

        for(int i=1; i<=2*n; ++i){
            while(X.front() < i-n) X.pop_front();
            res = max(res, P[i] - P[X.front()]);
            while(!X.empty() && P[X.back()] >= P[i]) X.pop_back();
            X.push_back(i);
        }
        return res;
    }
};
