class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int res = A[0];
        for(int i = 1; i < A.size(); i++)
        {
            if(res < A[i])
                res = A[i];
        }
        
        int len = A.size();
        for(int i = 0; i < len - 1; i++)
            A.push_back(A[i]);
        
        int sum = 0;
        int st = 0;        
        for(int i = 0; i < A.size(); i++)
        {
            sum = sum + A[i];
            if(i - st + 1 > len)
            {
                int index = st;
                int tmp;
                sum -= A[st ++];
                tmp = sum;
                while(st < i)
                {
                    tmp -= A[st];
                    if(tmp > sum)
                    {
                        sum = tmp;
                        index = st;
                    }
                    st ++;
                }
                st = index + 1;
            }
            if(sum > res)
                res = sum;
            if(sum < 0)
            {
                st = i + 1;
                sum = 0;
            }
            
        }
        
        return res;
    }
};
