class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int sum_all = 0;
        int n = A.size();
        for (int i = 0;i < n;++i) sum_all += A[i];
        int ans = LONG_MIN;
        int dpmin = A[0], dpmax = A[0];
        bool all_negative = true;
        for (int i = 1;i < n;++i) {
            if(A[i] >= 0)
                all_negative = false;
            dpmin = min(A[i], dpmin + A[i]);
            ans =  max(ans, sum_all - dpmin);
            dpmax = max(A[i], dpmax + A[i]);
            ans =  max(ans, dpmax);
        }
        if (all_negative) {
            ans = INT_MIN;
            for (int i = 0;i < n;++i)
                ans = max(ans, A[i]);
        }
        return ans;
    }
};
