#define uniq(v) v.erase(unique(all(v)),v.end())


class Solution {
public:
	int maxSubarraySumCircular(vector<int>& A) {
		if (*max_element(A.begin(),A.end()) < 0) {
			return *max_element(A.begin(), A.end());
		}

		int n = A.size();
		vector<int> pre(n);
		int s = 0;

		for (int i = 0; i < n; i++) {
			s += A[i];
			pre[i] = s;
		}
		pre[0] = max(pre[0], 0);
		for (int i = 1; i < n; i++) {
			pre[i] = max(pre[i], pre[i - 1]);
		}

		int ret = pre[n - 1];
		s = 0;
		for (int i = n-1; i > 0; i--) {
			s += A[i];
			ret = max(ret, s + pre[i - 1]);
		}


		s = 0;
		for (int i = 0; i < n; i++) {
			if (s < 0) {
				s = 0;
			}
			s += A[i];
			ret = max(ret, s);
		}


		return ret;
	}
};
