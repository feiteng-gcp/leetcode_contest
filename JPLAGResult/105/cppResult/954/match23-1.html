class Solution {
public:
    struct Node{
        int pos;
        long long s;
    };
    int a[60000];
    long long sum[60000];
    int maxSubarraySumCircular(vector<int>& A) {

        int n=A.size();
        for(int i=0; i<n; i++) {
            a[i]=A[i];
            a[i+n]=A[i];
        }
        sum[0]=a[0];
        for(int i=1; i<n*2; i++) {
            sum[i]=sum[i-1]+a[i];
            // cout<<sum[i]<<' ';
        }
        deque<Node> st;
        st.push_back(Node{-1, 0});
        long long ans=-0x3f3f3f3f3f3f3f3f;
        for(int i=0; i<n*2; i++) {
            // cout<<endl;
            // cout<<sum[i]<<endl;
            // cout<<st.front().s<<endl;
            ans=max(ans, sum[i]-st.front().s);
            if (!st.empty()&&st.front().pos<i-n+1) st.pop_front();
            while(!st.empty()&&st.back().s>=sum[i]) st.pop_back();
            st.push_back(Node{i, sum[i]});
        }
        return ans;
    }
};
