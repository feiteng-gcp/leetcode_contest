class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int ret = A[0];
        int now = 0;
        for (int i = 0; i < A.size(); ++i) {
            now += A[i];
            ret = max(ret, now);
            if (now < 0) {
                now = 0;
            }
        }
        vector<int> left(A);
        now = 0;

        int maxv = A[0];
        for (int i = 0; i < A.size(); ++i) {
            now += A[i];
            maxv = max(maxv, now);
            left[i] = maxv;
        }
        
        vector<int> right(A);
        now = 0;
        maxv = A.back();
        for (int i = A.size() - 1; i >= 0; --i) {
            now += A[i];
            maxv = max(maxv, now);
            right[i] = maxv;
        }
        for (int i = 0; i < A.size() - 1; ++i) {
            ret = max(ret, left[i] + right[i + 1]);
        }
        return ret;
    }
};
