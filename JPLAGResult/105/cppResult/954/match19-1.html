class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        vector<int> b(A.size()+1);
        b[0] = 0;

        int imin = 0, imax = 0;
        int s1 = INT_MIN, s2 = INT_MAX;
        for(int i = 1; i <= A.size(); ++i){
            b[i] = b[i-1] + A[i-1];
            s1 = max(s1, b[i]-b[imin]);
            if(i==A.size())
                imax = max(imax, 1);
            s2 = min(s2, b[i]-b[imax]);
            if(b[i] < b[imin])
                imin = i;
            if(b[i] > b[imax])
                imax = i;
        }
        return max(s1, b.back()-s2);
    }
};
