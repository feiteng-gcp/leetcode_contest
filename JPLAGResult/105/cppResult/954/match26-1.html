class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int ans = 0;
        bool allneg = true;
        int maxneg = A[0];
        for(int& x : A) {
            if (x < 0) {
                maxneg = max(maxneg, x);
            }
            else {
                allneg = false;
            }
            ans += x;
        }
        
        if (allneg) {
            return maxneg;
        }
        
        int ans1 = maxSum(A);
        for(int& x : A) {
            x *= -1;
        }
        int ans2 = maxSum(A);
        return max(ans1, ans + ans2);
        
    }
private:
    int maxSum(vector<int>& nums) {

        int sum = nums[0];
        int max_sum = nums[0];
        int min_val = 0;
        for(int i = 1; i < nums.size(); ++i) {
            min_val = min(min_val, sum);
            sum += nums[i];
            max_sum = max(max_sum, sum - min_val);
        }
        return max_sum;        
    }
};
