class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int ans = A[0];
        int len = A.size();
        vector<int> endMax(len, 0);
        endMax[0] = A[0];
        
        for (int i = 1; i < len; i++) {
            endMax[i] = endMax[i - 1] > 0 ? A[i] + endMax[i - 1] : A[i];
            ans = max(endMax[i], ans);
        }
        vector<int> preMax(len, 0);
        preMax[0] = A[0];
        int preSum = A[0];
        for (int i = 1; i < len; i++) {
            preSum += A[i];
            preMax[i] = max(preMax[i - 1], preSum);
        }
        
        vector<int> postMax(len, 0);
        postMax[len - 1] = A[len - 1];

        int postSum = A[len - 1];
        for (int i = len - 2; i >= 0; i--) {
            postSum += A[i];
            postMax[i] = max(postMax[i + 1], postSum);
        }
        
        for (int i = 0; i < len - 1; i++) {
            ans = max(ans, preMax[i] + postMax[i + 1]);
        }
        
        return ans;
    }
        
};
