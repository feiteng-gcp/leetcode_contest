class Solution {
public:
	vector<int>x;
	int n;
	int maxSubarraySumCircular(vector<int>& A) {
		x = A;
		n = A.size();
		int maxx=solve();
		int ans = 0, i;
		int p = -1000000;

		for (i = 0; i < n; i++) {
			p = max(p, x[i]);
			ans += x[i];
			x[i] *= -1;
		}
		ans+=solve();
		if (max(ans, maxx) == 0) {
			return p;
		}
		else return max(ans, maxx);
	}
	int solve(){

		int maxx = 0, curr = 0;
		for (int i = 0; i < n; i++){
			curr+= x[i];
			curr = max(curr, 0);
			maxx = max(maxx, curr);
		}
		return maxx;
	}
};
