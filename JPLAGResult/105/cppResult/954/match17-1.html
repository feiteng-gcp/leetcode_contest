class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int sum = 0;
        int ret = 0;
        int minval = INT_MIN;
        for(int i = 0; i < A.size(); i++) {
            minval = max(A[i], minval);
            sum+=A[i];
            if(sum < 0) {
                sum = 0;
            } else {
                ret = max(sum, ret);
            }
        }
        int acc = 0, i; 

        for (i=0; i < A.size(); i++) { 
            acc += A[i]; 
            A[i] = -A[i];   
        }
        int ret2 = 0;
        sum = 0;
        for(int i = 0; i < A.size(); i++) {
            sum+=A[i];
            if(sum < 0) {
                sum = 0;
            } else {
                ret2 = max(sum, ret2);
            }
        }       
       acc = acc + ret2; 
       if(max(acc,ret) == 0) return minval;
       return max(acc, ret);
    }
};
