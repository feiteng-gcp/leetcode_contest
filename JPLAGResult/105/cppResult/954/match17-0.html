class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        if (A.empty())
            return 0;
        int maxSubarray = A[0], maxTmp = A[0];
        for (int i = 1; i < A.size(); ++i)
        {
            maxTmp = max(A[i], maxTmp + A[i]);
            maxSubarray = max(maxSubarray, maxTmp);
        }
        
        int minSubarray = A[0], minTmp = A[0];
        for (int i = 1; i < A.size(); ++i)
        {
            minTmp = min(A[i], minTmp + A[i]);
            minSubarray = min(minSubarray, minTmp);
        }
        
        int totalSum = accumulate(A.begin(), A.end(), 0);
        if (minSubarray == totalSum)
            return maxSubarray;
        else
            return max(maxSubarray, totalSum - minSubarray);
    }
};
