class Solution {
public:
    int maxSubarraySumCircular(vector<int>& A) {
        int length = A.size() ;  
        vector< int > dp1( A.size() , 0 ) ; 
        vector< int > dp2( A.size() , 0 ) ; 
        dp1[0] = A[0] ; 
        dp2[0] = A[0] ; 
        int sum = A[0] ;
        int ans1 = A[0] ;
        int ans2 = A[0] ;  
        for( int i=1  ; i<A.size() ; i++){
            dp1[i]  = max( dp1[i-1] + A[i] , A[i] ) ; 
            ans1 = max( dp1[i] ,ans1 ) ; 
            dp2[i] = min ( dp2[i-1] + A[i] , A[i] ) ; 
            ans2 = min ( ans2 , dp2[i] ) ;
            sum += A[i] ; 
        }
        sort(A.begin(),A.end()) ;
        if( A.back()< 0 ) 
            return ans1; 
        return max( ans1 , sum - ans2 ) ; 
    }
};
