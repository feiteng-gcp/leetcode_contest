class Solution {
public:
    string reverseOnlyLetters(string S) {
        int p1 = 0, p2 = S.size()-1;
        
        while(p1<p2){
            while(p1<p2 && (!isalpha(S[p1])))
                ++p1;
            while(p1<p2 && (!isalpha(S[p2])))
                --p2;
            if(p1<p2)
                swap(S[p1++],S[p2--]);
        }
        return S;
    }
};
