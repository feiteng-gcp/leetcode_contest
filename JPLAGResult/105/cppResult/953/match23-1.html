class Solution {
public:
    string reverseOnlyLetters(string S) {
        if (S.empty()) {
            return S;
        }

        int f = 0, b = S.length() - 1;
        while (f < b) {
            while (f < b && !isalpha(S[f])) {
                f++;
            }
            while (f < b && !isalpha(S[b])) {
                b--;
            }
            if (f < b) {
                swap(S[f], S[b]);
                f++;
                b--;
            }
        }
        return S;
    }
};
