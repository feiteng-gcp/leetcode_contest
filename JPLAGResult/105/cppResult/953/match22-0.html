class Solution {
public:
    
bool isLetter(char ch) {
    return (ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z');
}
string reverseOnlyLetters(string S) {
    int n = S.length();
    int i = 0, j = n - 1;
    while (i < j) {
        while (i < n && !isLetter(S[i])) {
            i++;
        }
        
        while (j >= 0 && !isLetter(S[j])) {
            j--;
        }
        
        if (i < j) {
            swap(S[i], S[j]);
            i++,
            j--;
        }
    }
    
    return S;
}
};
