class Solution {
public:
    string reverseOnlyLetters(string S) {

        int i = 0, j = S.size() - 1;
        
        while (i < j)
        {
            
            if (!isalpha(S[i])) {
                i++;
                continue;
            }
            
            
            if (!isalpha(S[j])) {
                j--;
                continue;
            }
            
            swap(S[i], S[j]);
            i++;
            j--;
        }
        
        return S;
    }
};
