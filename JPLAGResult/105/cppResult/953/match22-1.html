#include <string>
#include <cctype>
#include <algorithm>

using namespace std;

class Solution {
public:

    string reverseOnlyLetters(string S) {
        int n = S.size();
        int l = 0, r = n - 1;
        while(l < r){
            while(l < r && !isalpha(S[l])){
                l++;
            }
            while(r > l && !isalpha(S[r])){
                r--;
            }
            if(l < r){
                swap(S[l], S[r]);
                l++;
                r--;
            }
        }
        return S;
    }
};
