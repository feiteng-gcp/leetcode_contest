class Solution {
public:
    string reverseOnlyLetters(string S) {
        int l=0;
        int r = S.size()-1;
        while(l<r)
        {
            if(isalpha(S[l])==false){
             l++;continue;   
            }
            if(isalpha(S[r])==false)
            {
                r--;
                continue;
            }
            
            swap(S[l], S[r]);
            l++;r--;
        }
        return S;
    }
};
