class Solution {
public:
    string reverseOnlyLetters(string S) {
        if (S.size() < 2)
            return S;

        int i=0,j=S.size()-1;
        while (i < j) {
            while (i<j && !((S[i] >= 'a' && S[i] <= 'z') || (S[i] >= 'A' && S[i] <= 'Z')))
                i++;
            while (i<j && !((S[j] >= 'a' && S[j] <= 'z') || (S[j] >= 'A' && S[j] <= 'Z')))
                j--;
            swap(S[i], S[j]);
            i++; j--;
        }
        return S;
    }
};
