class Solution {
private:
    bool isAlpha(char c) {
        return ('A' <= c && c <= 'Z') || ('a' <= c && c <= 'z');
    }
    
public:
    string reverseOnlyLetters(string S) {
        int i = 0, j = S.length() - 1;
        while (i < j) {
            while (!isAlpha(S[i]) && i < j) i++;
            while (!isAlpha(S[j]) && i < j) j--;
            if (i >= j) break;
            swap(S[i], S[j]);
            i++, j--;
        }
        return S;
    }
};
