class Solution {
public:
    string reverseOnlyLetters(string S) {
        string s="";
        for (int i=0;i<S.size();i++)
        {
            if ((S[i]>=65 && S[i]<=90)||  (S[i]>=97 && S[i]<=122))
                s+=S[i];
        }
        reverse(s.begin(),s.end());
        int t=0;
        for (int i=0;i<S.size();i++)
        {
            if ((S[i]>=65 && S[i]<=90)||  (S[i]>=97 && S[i]<=122))
            {
                S[i]=s[t++];
            }
        }
        return S;
    }
};
