class Solution {
public:
    string reverseOnlyLetters(string S) {
        // S = "a-bC-dEf-ghIj";
        // S = "Test1ng-Leet=code-Q!";
        int len = S.length();

        int left = 0;
        int right = len-1;
        while(left < right){
            while(left<right && checkLetter(S[left])==false){
                left++;
            }
            while(left<right && checkLetter(S[right])==false){
                right--;
            }
            swap(S[left], S[right]);
            left++;
            right--;
        }
        return S;
    }
private:
    bool checkLetter(char a){
        int base = a-'a';
        if (base>=0 && base<26) {
            return true;
        }
        base = a-'A';
        if (base>=0 && base<26) {
            return true;
        }
        return false;
    }
};
