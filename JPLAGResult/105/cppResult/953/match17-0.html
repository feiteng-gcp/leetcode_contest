class Solution {
public:
    string reverseOnlyLetters(string S) {
        int x=0,y=S.size()-1;
        while(1){
            while(x<y && !isalpha(S[x]) )x++;
            while(x<y && !isalpha(S[y]) )y--;
            

            if(x>=y)break;
            swap(S[x],S[y]);
            x++;
            y--;
        }
        return S;
    }
};
