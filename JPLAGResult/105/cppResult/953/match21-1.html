class Solution {
public:
    string reverseOnlyLetters(string S) 
    {
        string T;

        for (int i=0; i<S.size(); i++)
        {
            if (isalpha(S[i]))
                T+=S[i];
        }
        reverse(T.begin(),T.end());
        int j=0;
        for (int i=0; i<S.size(); i++)
        {
            if (isalpha(S[i]))
            {
                S[i]=T[j];
                j++;    
            }                
        }
        return S;
        
    }
};
