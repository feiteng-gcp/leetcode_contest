class Solution {
public:
    string reverseOnlyLetters(string S) {
        int len=S.length();
        if (S.length()==0) return S;
        int i=0, j=len-1;
        while(i<j) {
            while(!isupper(S[i])&&!islower(S[i])&&i<=len-2) i++;
            while(!isupper(S[j])&&!islower(S[j])&&j>=1) j--;
            if (i<j) swap(S[i], S[j]);
            i++;
            j--;
        }
        return S;
    }
};
