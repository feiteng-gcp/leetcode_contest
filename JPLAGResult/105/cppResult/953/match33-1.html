class Solution {
public:
    string reverseOnlyLetters(string S) {
        if (S.size() <= 1) {return S;}

        int p1 = 0;
        int p2 = S.size()-1;
        while (p1 < p2) {
            while (p1 < p2 && !check(S[p1])) {p1++;}
            while (p1 < p2 && !check(S[p2])) {p2--;}
            swap(S[p1++], S[p2--]);
        }
        return S;
    }
                
    bool check(char c) {
        int asc = static_cast<int>(c);
        return (asc >= 65 && asc <= 90) or (asc >= 97 && asc <= 122);
    }
};
