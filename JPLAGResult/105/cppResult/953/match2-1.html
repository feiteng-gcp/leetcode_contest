class Solution {
public:
    string reverseOnlyLetters(string S) {
        if (S.size() == 0) return S;

        int startIndex = 0, endIndex = S.size() -1 ;
        while (startIndex < endIndex) {
            while  (startIndex < S.size() && (!((S[startIndex] >= 'a' && S[startIndex] <= 'z') || (S[startIndex] >= 'A' && S[startIndex] <= 'Z')))) startIndex++;
            while  (endIndex >= 0 && (!((S[endIndex] >= 'a' && S[endIndex] <= 'z') || (S[endIndex] >= 'A' && S[endIndex] <= 'Z')))) endIndex--;
            if (startIndex < endIndex) {
                swap(S[startIndex], S[endIndex]);
                startIndex++;
                endIndex--;
            }
        }
        return S;
    }
};
