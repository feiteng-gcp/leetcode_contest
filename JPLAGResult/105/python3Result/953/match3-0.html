class Solution:
    def reverseOnlyLetters(self, S):
        """
        :type S: str
        :rtype: str
        """
        if not S:
            return S
        
        S_list = list(S)
        left = 0
        right = len(S_list) - 1
        while left < right:
            while left < right and not S_list[left].isalpha():
                left += 1
            while left < right and not S_list[right].isalpha():
                right -= 1
            if left < right:
                S_list[left], S_list[right] = S_list[right], S_list[left]
                left += 1
                right -= 1
        return ''.join(S_list)
