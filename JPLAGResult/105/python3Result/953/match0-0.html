class Solution:
    def reverseOnlyLetters(self, S):
        """
        :type S: str
        :rtype: str
        """
        l = list(S)
        save = []
        for c in l:
            if c.isalpha():
                save.append(c)
        p = len(save) - 1
        for i in range(len(l)):
            if l[i].isalpha():
                l[i] = save[p]
                p -= 1
        return ''.join(l)
