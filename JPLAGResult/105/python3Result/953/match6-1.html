class Solution:
    def reverseOnlyLetters(self, S):
        """
        :type S: str
        :rtype: str
        """
        letters = [c for c in S
                   if 'a' <= c <= 'z' or 'A' <= c <= 'Z']
        letters.reverse()
        all_chars = list(S)
        j = 0
        for i,c in enumerate(all_chars):
            
if 'a' <= c <= 'z' or 'A' <= c <= 'Z':
                all_chars[i] = letters[j]
                j += 1
        return ''.join(all_chars)
