class Solution:
    def reverseOnlyLetters(self, S):
        x = list(filter(lambda c:('A'<=c<='Z') or ('a'<=c<='z'), [_ for _ in S]))
        x = list(reversed(x))
        idx = 0
        ans = [_ for _ in S]
        for i in range(len(ans)):
            c = ans[i]
            if ('A'<=c<='Z') or ('a'<=c<='z'):
                ans[i] = x[idx]
                idx += 1
        return ''.join(ans)
