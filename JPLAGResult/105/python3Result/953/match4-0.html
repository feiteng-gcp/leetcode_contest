class Solution:
    def reverseOnlyLetters(self, S):
        """
        :type S: str
        :rtype: str
        """
        res=list(S)
        i,j=0,len(res)-1
        while i<j:
            while i<j and not res[i].isalpha():
                i+=1
            while i<j and not res[j].isalpha():
                j-=1
            if i<j:
                res[i],res[j]=res[j],res[i]
                i+=1
                j-=1
        a=''
        for c in res:
            a+=c
        return a
