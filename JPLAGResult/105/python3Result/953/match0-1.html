class Solution:
    def reverseOnlyLetters(self, S):
        """
        :type S: str
        :rtype: str
        """
        l = []
        S = list(S)
        
for s in S:
            if s.isalpha():
                l.append(s)
        n = len(l) - 1
        for i in range(len(S)):
            if S[i].isalpha():
                S[i] = l[n]
                n -= 1
        return ''.join(S)
