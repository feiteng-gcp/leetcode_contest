class Solution:
    def reverseOnlyLetters(self, S):
        """
        :type S: str
        :rtype: str
        """
        stack = []
        result = []
        for c in S:
            if c.isalpha():
                stack.append(c)
        for c in S:
            if c.isalpha():
                result.append(stack.pop())
            else:
                result.append(c)
        return ''.join(result)
