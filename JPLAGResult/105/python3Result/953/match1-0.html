class Solution:
    def reverseOnlyLetters(self, S):
        """
        :type S: str
        :rtype: str
        """
        res = list(S)
        i, j = 0, len(S)-1
        while i < j:
            if not res[i].isalpha():
                i += 1
            elif not res[j].isalpha():
                j -= 1
            else:
                res[i], res[j] = res[j], res[i]
                i += 1
                j -= 1
        else:
            return ''.join(res)
