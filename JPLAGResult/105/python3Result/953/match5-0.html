class Solution:
    def reverseOnlyLetters(self, S):
        """
        :type S: str
        :rtype: str
        """
        l = 0
        r = len(S) - 1
        S = list(S)
        from string import ascii_letters
        letter = set(ascii_letters)
        while l < r:
            if S[l] not in letter:
                l += 1
            elif S[r] not in letter:
                r -= 1
            else:
                S[l], S[r] = S[r], S[l]
                l += 1
                r -= 1
        return ''.join(S)
