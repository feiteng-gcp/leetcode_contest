class Solution:
    def maxSubarraySumCircular(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        max_subArr = self.max_subarray(A) 
        if max_subArr < 0:
            return max_subArr
        max_wrap = 0
        for i in range(len
(A)): 
            max_wrap += A[i] 
            A[i] = -A[i] 
        max_wrap = max_wrap + self.max_subarray(A) 
        if max_wrap > max_subArr: 
            return max_wrap 
        else: 
            return max_subArr
        
    def max_subarray(self, A):
        max_ending_here, max_so_far = A[0], A[0]
        for x in A[1:]:
            max_ending_here = max(x, max_ending_here + x)
            max_so_far = max(max_so_far, max_ending_here)
        return max_so_far
