class Solution:
    
    def mSum(self, a): 
      n = len(a) 
      max_so_far = 0
      max_ending_here = 0
      for i in range(0, n): 
          max_ending_here = max_ending_here + a[i] 
          if (max_ending_here < 0): 
              max_ending_here = 0
          if (max_so_far < max_ending_here): 
              max_so_far = max_ending_here 
      return max_so_far
    
    def maxCircularSum(self, a): 
        n = len(a) 
        max_mSum = self.mSum(a) 
        max_wrap = 0
        for i in range(0, n): 
            max_wrap += a[i] 
            a[i] = -a[i] 
        max_wrap = max_wrap + self.mSum(a) 
        if max_wrap > max_mSum: 
            return max_wrap 
        else: 
            return max_mSum 
    def maxSubarraySumCircular(self, A):
        """
        :type A: List[int]
        :rtype: int
        """
        flag = False
        for a in A:
          if a>=0:
            flag = True
        if flag==True:
          return self.maxCircularSum(A)
        else:
          return sorted(A)[::-1][0]
