class Solution:
    def maxSubarraySumCircular(self, a):
        """
        :type A: List[int]
        :rtype: int
        """
        if all(n <= 0 for n in a):
            return max(a)
        def f(a):
            n = len(a) 
            max_so_far = 0
            max_ending_here = 0
            for i in range(0, n): 
                max_ending_here = max_ending_here + a[i] 
                if (max_ending_here < 0): 
                    max_ending_here = 0
                if (max_so_far < max_ending_here): 
                    max_so_far = max_ending_here 
            return max_so_far

        max_f = f(a)
        max_wrap = 0
        for i in range(0,len(a)): 
            max_wrap += a[i] 
            a[i] = -a[i]
        max_wrap = max_wrap + f(a)
        if max_wrap > max_f: 
            return max_wrap 
        else: 
            return max_f
