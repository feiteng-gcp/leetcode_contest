import collections

# Definition for a binary tree node.
class TreeNode:
    def __init__(self, x):
        self.val = x
        self.left = None
        self.right = None

class CBTInserter:

    def __init__(self, root):
        self.arr = []
        q = collections.deque()
        q.appendleft(root)
        while q:
            top = q[-1]
            q.pop()
            self.arr.append(top.val)
            
if top.left is not None: q.appendleft(top.left)
            if top.right is not None: q.appendleft(top.right)


    def insert(self, v):
        pos = len(self.arr) + 1
        self.arr.append(v)
        return self.arr[pos // 2 - 1]

    def get_root(self):
        return self.arr
