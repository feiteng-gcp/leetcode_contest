# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class CBTInserter:

    def __init__(self, root):
        """
        :type root: TreeNode
        """
        self.root = root
        
    
    def get_node_li(self):
      h = 0
      root = self.root
      x = root
      while x is not None:
        h += 1
        x = x.left
      root = self.root
      node_li = [[root]]
      for i in range(h):
        if len(node_li[-1])==2**i:
          node_li.append([])
          for x in node_li[i]:
            if x.left is not None:
              node_li[-1].append(x.left)
            if x.right is not None:
              node_li[-1].append(x.right)
      #len_li = [len(nodes) for nodes in node_li]
      #val_li = [[node.val for node in nodes] for nodes in node_li]
      #print(h)
      #print(len_li)
      #print(val_li)
      return node_li
      
      
    def insert(self, v):
        """
        :type v: int
        :rtype: int
        """
        node_li = self.get_node_li()
        if node_li[-1]==[]:
          parent = node_li[-2][0]
          parent.left = TreeNode(v)
          return parent.val
        else:
          l = len(node_li[-1])
          idx = l//2
          isLeft = (l%2==0)
          parent = node_li[-2][idx]
          if isLeft:
            parent.left = TreeNode(v)
          else:
            parent.right = TreeNode(v)
          return parent.val
        

    def get_root(self):
        """
        :rtype: TreeNode
        """
        return self.root
