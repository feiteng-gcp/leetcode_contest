# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class CBTInserter:

    def __init__(self, root):
        """
        :type root: TreeNode
        """
        self.node_arr = []
        def aux(i, node):
            if not node:
                return
            while i >= len(self.node_arr):
                self.node_arr.append(None)
            self.node_arr[i] = node
            aux(i * 2 + 1, node.left)
            aux(i * 2 + 2, node.right)
        aux(0, root)

    def insert(self, v):
        """
        :type v: int
        :rtype: int
        """
        node = TreeNode(v)
        i = len(self.node_arr)
        self.node_arr.append(node)
        parent_i = (i - 1) // 2
        if i % 2 == 1:
            self.node_arr[parent_i].left = node
        else:
            self.node_arr[parent_i].right = node
        return self.node_arr[parent_i].val

    def get_root(self):
        """
        :rtype: TreeNode
        """
        return self.node_arr[0]
