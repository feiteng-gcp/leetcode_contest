# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class CBTInserter:

    def __init__(self, root):
        """
        :type root: TreeNode
        """
        res=[]
        thislevel = [root]
        while thislevel:
            nextlevel = list()
            for n in thislevel:
                res.append(n.val)
                if n.left:
                    nextlevel.append(n.left)
                if n.right:
                    nextlevel.append(n.right)
            thislevel = nextlevel
        self.tree=res
        

    def insert(self, v):
        """
        :type v: int
        :rtype: int
        """
        self.tree.append(v)
        #print(self.tree)
        return self.tree[(len(self.tree)-2)//2]
        

    def get_root(self):
        """
        :rtype: TreeNode
        """
        return self.tree
