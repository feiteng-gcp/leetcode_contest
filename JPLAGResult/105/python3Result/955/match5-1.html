# class TreeNode:

#         self.val = x

import queue

class CBTInserter:
    def __init__(self, root):
        """
        :type root: TreeNode
        """
        self.root = root
        self.cnt = 1
        self.root_list = [TreeNode(-1)]
        q = queue.Queue()
        q.put(root)
        while not q.empty():
            cur = q
.get()
            if cur.left is not None:
                q.put(cur.left)
            if cur.right is not None:
                q.put(cur.right)
            self.root_list.append(cur)
            self.cnt += 1

    def insert(self, v):
        """
        :type v: int
        :rtype: int
        """
        self.root_list.append(TreeNode(v))
        par = self.root_list[self.cnt // 2]
        if par.left is None:
            par.left = self.root_list[-1]
        else:
            par.right = self.root_list[-1]
        self.cnt += 1
        return par.val

    def get_root(self):
        """
        :rtype: TreeNode
        """
        return self.root
