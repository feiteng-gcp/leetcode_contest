# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class CBTInserter:

    def __init__(self, root):
        """
        :type root: TreeNode
        """
        self.d = [None, root]
        i = 1
        while i < len(self.d):
            for node in [self.d[i].left, self.d[i].right]:
                if node:
                    self.d.append(node)
            i += 1

    def insert(self, v):
        """
        :type v: int
        :rtype: int
        """
        node = TreeNode(v)
        i = len(self.d)
        self.d.append(node)
        
if i % 2 == 0:
            self.d[i//2].left = node
        else:
            self.d[i//2].right = node
        return self.d[i//2].val

    def get_root(self):
        """
        :rtype: TreeNode
        """
        return self.d[1]
