# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class CBTInserter:

    def __init__(self, root):
        """
        :type root: TreeNode
        """
        self.root = root
        q = [root]
        self.arr = []
        while q:
            node = q.pop(0)
            self.arr.append(node)
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)

    def insert(self, v):
        """
        :type v: int
        :rtype: int
        """
        node = TreeNode(v)
        self.arr.append(node)
        parent = self.arr[len(self.arr) // 2 - 1]
        if parent.left: parent.right = node
        else: parent.left = node
        return parent.val

    def get_root(self):
        """
        :rtype: TreeNode
        """
        return self.root
