class Solution:
    def numMusicPlaylists(self, N, L, K):
        """
        :type N: int
        :type L: int
        :type K: int
        :rtype: int
        """
        def C(n, r):
            return factorial(n) // factorial(r) // factorial(n - r)

        def factorial(n, dic={0: 1}):
            k = dic.get(n, -1)
            if k == -1:
                res = n * factorial(n - 1)
                dic[n] = res
                return res
            else:
                return k

        def f(N, L, K):
            res 
= 1
            p = 0
            for i in range(L):
                res *= N - p
                if p < K:
                    p += 1
            return res

        def g(N, L, K):
            res = 0
            p = 1
            for i in range(N, K, -1):
                res += C(N, i) * f(i, L, K) * p
                p = -p
            return res % (10 ** 9 + 7)

        return g(N, L, K)
