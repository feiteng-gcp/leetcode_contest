from math import factorial

def comb(n, r):
    mod = 10 ** 9 + 7
    num = factorial(n) % mod
    denom = factorial(n-r) * factorial(r) % mod
    return num * pow(denom, mod-2, mod) % mod

class Solution:
    def x(self, N, L, K):
        res = 1
        tot = N
        mod = 10 ** 9 + 7
        for i in range(L):
            res = res * tot % mod
            if i < K: tot -= 1
        return res

    def numMusicPlaylists(self, N, L, K):
        """
        :type N: int
        :type L: int
        :type K: int
        :rtype: int
        """
        mod = 10 ** 9 + 7
        res = self.x(N, L, K)
        for n in range(N-1, K-1, -1):
            res += self.x(n, L, K) * comb(N, n) * (-1) ** (N - n)
            res = (res % mod + mod) % mod
        return res
