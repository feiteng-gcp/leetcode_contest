class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        return watched(watchedVideos, friends, id, level)

from collections import defaultdict
def watched(videos, friends, id, level):
    # process a list of graph to get the friends list and bfs for all the friends at level k
    
    f = [id]
    visited = set()
    visited.add(id)
    for i in range(level):
        next_f = []
        while f:
            curr = f.pop()
            for friend in friends[curr]:
                if friend not in visited:
                    visited.add(friend)
                    next_f.append(friend)
        f = next_f
    print(f)
    
    watched = defaultdict(int)
    for fri in f:
        for v in videos[fri]:
            watched[v] += 1
    
    res = sorted(watched.items(), key=lambda x: (x[1], x[0]))
    return [x[0] for x in res]
