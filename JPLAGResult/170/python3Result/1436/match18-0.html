class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        queue = collections.deque()
        queue.append(id)
        distance = {}
        distance[id] = 0
        while queue:
            v = queue.popleft()
            if distance[v] == level:
                continue
            for u in friends[v]:
                if u not in distance:
                    distance[u] = distance[v] + 1
                    queue.append(u)
        video_counts = collections.defaultdict(int)
        for v in range(len(friends)):
            if v in distance and distance[v] == level:
                for video in watchedVideos[v]:
                    video_counts[video] += 1
        output = list(video_counts.items())
        output.sort(key=lambda e: (e[1], e[0]))
        return [e[0] for e in output]
