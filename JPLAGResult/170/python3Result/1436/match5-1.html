class Solution:
    def watchedVideosByFriends(self, wv, fs, id, level):
        if level == 0:
            return sorted(wv[id])
        st = [id]
        n = len(fs)
        ff = [[0x3f3f3f3f] * n for _ in range(n)]
        for i, row in enumerate(fs):
            ff[i][i] = 0
            for j in row:
                ff[i][j] = 1
        
        for i in range(n):
            for j in range(i + 1, n):
                for k in range(1, n):
                    ff[i][j] = min(ff[i][j], ff[i][k] + ff[k][j])
                    ff[j][i] = min(ff[i][j], ff[j][i])
        
        cc = defaultdict(int)
        for i, d in enumerate(ff[id]):
            if d == level:
                for v in wv[i]:
                    cc[v] += 1
        res = [(c,n) for c, n in cc.items()]
        res.sort(key=lambda x: (x[1], x[0]))
        return [x[0] for x in res ]
