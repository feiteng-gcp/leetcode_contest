class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        dist = [-1] * len(friends)
        dist[id] = 0
        q = [id]
        d = 0
        while True:
            newQ = []
            d += 1
            for c in q:
                for nex in friends[c]:
                    if dist[nex] == -1:
                        dist[nex] = d
                        newQ.append(nex)
            q = newQ
            if d==level:
                break
        vids = {}
        for f in q:
            for v in watchedVideos[f]:
                if v in vids:
                    vids[v] += 1
                else:
                    vids[v] = 1
        ans = [k for k, _ in sorted(vids.items(), key=lambda item: (item[1], item[0]))]
        return ans
