class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        seen = set([id])
        queue = collections.deque([id])
        for _ in range(level):
            for _ in range(len
(queue)):
                curr = queue.popleft()
                for f in friends[curr]:
                    if f not in seen:
                        seen.add(f)
                        queue.append(f)
        freq 
= collections.Counter()
        for i in queue:
            for c in watchedVideos[i]:
                freq[c] += 1
        ans = []
        for key in freq:
            ans.append((freq[key], key))
        ans.sort()
        return [key for _, key in ans]
