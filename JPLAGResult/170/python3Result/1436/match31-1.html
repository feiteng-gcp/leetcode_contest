class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], ID: int, level: int) -> List[str]:
        frontier = collections.deque([ID])
        levels = {ID : 0}
        nlevel = 0
        while frontier:
            nlevel += 1
            if level == 0:
                break
            level -= 1
            sz = len(frontier)
            for i in range(sz):
                u = frontier.popleft()
                for v in friends[u]:
                    if v not in levels:
                        levels[v] = nlevel
                        frontier.append(v)

        freqs = collections.Counter()
        for u in frontier:
            for v in watchedVideos[u]:
                freqs[v] += 1
        return sorted([u for u in freqs], key=lambda x : (freqs[x], x))
