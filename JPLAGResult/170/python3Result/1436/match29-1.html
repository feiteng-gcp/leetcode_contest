class Solution:
    def watchedVideosByFriends(self,
                               watchedVideos: List[List[str]],
                               friends: List[List[int]],
                               myId: int,
                               level: int) -> List[str]:
        
        queue = collections.deque([myId])
        seen = {myId}
        
        for _ in range(level):
            for _ in range(len(queue)):
                id_ 
= queue.popleft()
                for friend in friends[id_]:
                    if friend not in seen:
                        seen.add(friend)
                        queue.append(friend)
        
        # print(queue)
        counter = collections.Counter
()
        while queue:
            movies = watchedVideos[queue.popleft()]
            for movie in movies:
                counter[movie] += 1
        
        items = sorted(list(counter.items()), key=lambda x: (x[1], x[0]))
        return [item[0] for item in items]
