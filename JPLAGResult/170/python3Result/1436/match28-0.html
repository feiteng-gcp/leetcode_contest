class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        # get friends with the given level
        queue = [id]
        visited = {id}
        step = 0
        while queue and step < level:
            lenq = len(queue)
            for _ in range(lenq):
                now = queue.pop(0)
                for i in friends[now]:
                    if i not in visited:
                        queue.append(i)
                        visited.add(i)
            
            step += 1
        
        # get videos and sort
        videos = collections.Counter(sum([watchedVideos[a] for a in queue], []))
        
        def myCmp(a: str, b: str) -> int:
            if videos[a] < videos[b]:
                return -1
            if videos[a] > videos[b]:
                return 1
            elif a < b:
                return -1
            elif a > b:
                return 1
            return 0
        
        from functools import cmp_to_key
        return sorted(list(videos.keys()), key = cmp_to_key(myCmp))
