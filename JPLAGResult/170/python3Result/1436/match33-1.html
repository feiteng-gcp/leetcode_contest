class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        seen = set()
        seen.add(id)
        cur = [id]
        while level:
            new_cur = []
            while cur:
                temp 
= cur.pop()
                for i in friends[temp]:
                    if i not in seen:
                        new_cur.append(i)
                        seen.add(i)
            cur = new_cur
            level -= 1
