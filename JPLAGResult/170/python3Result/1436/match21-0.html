class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], idd: int, level: int) -> List[str]:
        layer = [idd]
        cur_lvl = 0
        seen = set({idd})
        while cur_lvl != level and len(layer) != 0:
            next_layer = []
            for i in layer:
                
                for next_i in friends[i]:
                    if next_i in seen:
                        continue
                    seen.add(next_i)

                    next_layer.append(next_i)
            layer = next_layer
            cur_lvl += 1
        import collections
        freq = collections.defaultdict(int)
        for i in layer:
            for v in watchedVideos[i]:
                freq[v] += 1
        videos = list(freq)
        videos.sort(key=lambda x: (freq[x], x))
        return videos
