class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        cl=0
        q=[id]
        qnew=[id]
        visited=[0]*len(friends)
        visited[id]=1
        while(cl<level and len(qnew)>0):
            q=qnew[:]
            qnew
=[]
            cl+=1
            for item in q:
                for fr in friends[item]:
                    if not(visited[fr]):
                        visited[fr]=1
                        qnew.append(fr)

        movies={}
        for p in qnew:
            for mov in watchedVideos[p]:
                if mov in movies:
                    movies[mov]+=1
                else:
                    movies[mov]=1
        #print(movies)
        movies ={k: v for k, v in sorted(movies.items(), key=lambda item: item[0])}
        
        #print(movies)
        movies ={k: v for k, v in sorted(movies.items(), key=lambda item: item[1])}
        
        #print(movies)
        return movies.keys()
