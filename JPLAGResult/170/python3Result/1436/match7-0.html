class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        
        
        seen = {id}
        curr = {id}
        
        while curr and level:
            new_curr = set()
            
            for next_id in curr:
                for nn in friends[next_id]:
                    if nn not in seen:
                        new_curr.add(nn)
                        seen.add(nn)
            
            curr = new_curr
            level -= 1
        
        #print(curr)
        if not curr: return []
        out = collections.defaultdict(int)
        for id in curr:
            for v in watchedVideos[id]:
                out[v] += 1
            
        return sorted(out.keys(), key=lambda x: (out[x],x))
