class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        q = [id]
        seen = {id}
        for i in range(level):
            if not q:
                return []
            nq = []
            for c in q:
                for d in friends[c]:
                    if d not in seen:
                        nq.append(d)
                        seen.add(d)
            q = nq[:]

= collections.defaultdict(int)

        return ans
