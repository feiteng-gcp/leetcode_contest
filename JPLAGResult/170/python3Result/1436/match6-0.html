class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        level += 1
        seen = set([id])
        q = [id]
        while q:
            nq = []
            d = collections.defaultdict(int)
            
            for p in q:
                for v in watchedVideos[p]:
                    d[v] += 1
                    
                for np in friends[p]:
                    if np not in seen:
                        seen.add(np)
                        nq.append(np)
                        
            level -= 1
            q = nq
            
            if not level:
                heap = []
                for v, cnt in d.items():
                    heapq.heappush(heap, (cnt, v))
                
                res = []
                while heap:
                    res.append(heapq.heappop(heap)[1])
                return res
            
        return []
