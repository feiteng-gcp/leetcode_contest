from queue import Queue
from collections import defaultdict

class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], _id: int, level: int) -> List[str]:
        q = Queue()
        l = 0
        q.put(_id)
        visited = set()
        visited.add(_id)
        while not q.empty():
            # print(q.queue)
            n = q.qsize()
            if l == level:
                res = defaultdict(lambda: 0)
                for i in range(n):
                    cur = q.get()
                    for v in watchedVideos[cur]:
                        res[v] += 1
                return list(map(lambda y: y[0], sorted(res.items(), key = lambda x: (x[1], x[0]))))
            
            
for i in range(n):
                cur = q.get()
                for friend in friends[cur]:
                    if friend not in visited:
                        visited.add(friend)
                        q.put(friend)
            
            l += 1
