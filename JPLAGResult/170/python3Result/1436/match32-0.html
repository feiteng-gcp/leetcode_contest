class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        n = len(friends)
        q = collections.deque()
        q.append(id)
        seen = set()
        seen.add(id)
        while level:
            level -= 1
            steps = len(q)
            while steps:
                steps -= 1
                id = q.popleft()
                for friend in friends[id]:
                    if friend not in seen:
                        q.append(friend)
                        seen.add(friend)
        ans = collections.defaultdict(int)
        print(q)
        while q:
            videos = watchedVideos[q.popleft()]
            for vid in videos:
                ans[vid] += 1
        ans = sorted(ans.items(), key=lambda x: (x[1], x[0]))
        return [k for k,v in ans]
