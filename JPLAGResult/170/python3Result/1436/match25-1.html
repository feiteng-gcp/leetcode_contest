class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        visited={id}
        q=collections.deque([id])
        k=0
        while q and k<level:
            l=len(q)
            
for i in range(l):
                f=q.popleft()
                for p in friends[f]:
                    if p not in visited:
                        q.append(p)
                        visited.add(p)
            k+
=1
        if k<level:
            return []
        c=collections.Counter()
        for p in q:
            for v in watchedVideos[p]:
                c[v]+=1
        ans=sorted(list(c.keys()),key=lambda x: [c[x],x])
        return ans
