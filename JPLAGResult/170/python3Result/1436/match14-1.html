class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        visited = set([id])
        q = collections.deque([id])
        final = [id]
        while q and level > 0:
            level -= 1
            size = len(q)
            final = []
            
for _ in range(size):
                curr = q.popleft()
                for friend in friends[curr]:
                    if friend not in visited:
                        visited.add(friend)
                        q.append(friend)
                        final.append(friend)
        if level > 0:
            return []
        
        count = collections.defaultdict(int)
        for fr in final:
            for v in watchedVideos[fr]:
                count[v] += 1
                
        res = [v for v in sorted(count.keys(), key=lambda x:(count[x], x))]
        return res
