class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], uid: int, level: int) -> List[str]:
        q = collections.deque()
        dist = {}
        
        q.append(uid)
        dist[uid] = 0
        
        counter = collections.Counter()
        while len(q) > 0:
            now = q.popleft()
        
            if dist[now] == level:
                for watched in watchedVideos[now]:
                    counter[watched] += 1
            
            for friend in friends[now]:
                if friend not in dist:
                    dist[friend] = dist[now] + 1
                    q.append(friend)
                    
        return sorted(counter.keys(), key=lambda x: (counter[x], x))
