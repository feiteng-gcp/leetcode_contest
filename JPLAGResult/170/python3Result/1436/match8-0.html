class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:
        q = set()
        seen = set()
        q.add(id)
        seen.add(id)
        while level:
            nxtq = set()
            for node in q:
                for nxt in friends[node]:
                    if nxt in seen: continue
                    seen.add(nxt)
                    nxtq.add(nxt)
            
            level -= 1
            q = nxtq
            
        ctr = collections.defaultdict(int)
        for node in q:
            for v in watchedVideos[node]:
                ctr[v] += 1
                
        return sorted(ctr.keys(), key = lambda v:(ctr[v], v))
