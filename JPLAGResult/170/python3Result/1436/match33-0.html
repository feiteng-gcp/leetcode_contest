from collections import *

class Solution:
    def watchedVideosByFriends(self, watchedVideos: List[List[str]], friends: List[List[int]], id: int, level: int) -> List[str]:

        q = deque()
        q.append(id)

        watched = {}

        seen = {id}

        while q:
            if level == 0:
                while q:
                    pop = q.popleft()
                    for video in watchedVideos[pop]:
                        if video not in watched:
                            watched[video] = 0
                        watched[video] += 1
            else:
                size = len(q)
                for i in range(size):
                    pop = q.popleft()

                    for friend in friends[pop]:
                        if friend not in seen:
                            q.append(friend)
                            seen.add(friend)
                level -= 1


        arr = []
        for k, v in watched.items():
            arr.append((v, k))
        arr.sort()

        res = []
        for (v, k) in arr:
            res.append(k)
        return res
