class Solution:
    def xorQueries(self, arr: List[int], queries: List[List[int]]) -> List[int]:
        xor = [0]
        for num in arr:
            xor.append(xor[-1] ^ num)
        # print(xor)
        res = []
        for s, e in queries:
            res.append(xor[s] ^ xor[e + 1])        
        return res
