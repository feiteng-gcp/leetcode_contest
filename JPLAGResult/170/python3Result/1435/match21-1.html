class Solution:
    def xorQueries(self, arr: List[int], queries: List[List[int]]) -> List[int]:
        xor_arr = []
        xor_cur = 0
        for elem in arr:
            xor_cur ^= elem
            xor_arr.append(xor_cur)
        results = []
        for a, b in queries:
            if a == 0:
                results.append(xor_arr[b])
            else:
                results.append(xor_arr[a - 1] ^ xor_arr[b])
        return results
