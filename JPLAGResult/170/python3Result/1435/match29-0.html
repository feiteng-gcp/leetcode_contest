class Solution:
    def xorQueries(self, arr, queries):
        xs = [arr[0]]
        for i in range(1, len(arr)):
            xs.append(xs[-1] ^ arr[i])
        
        res = []
        for l, r in queries:
            if l == 0:
                res.append(xs[r])
            else:
                res.append(xs[r] ^ xs[l-1])
        return res
