class Solution:
    def xorQueries(self, arr: List[int], queries: List[List[int]]) -> List[int]:
        
        cur = 0
        pre = []
        for num in arr:
            cur = cur ^ num
            pre.append(cur)
            
        res = []
        for l, r in queries:
            if l == 0:
                res.append(pre[r])
            else:
                res.append(pre[l-1] ^ pre[r])
        return res
