class Solution:
    def xorQueries(self, arr: List[int], queries: List[List[int]]) -> List[int]:
        presums = [0]
        for a in arr:
            presums.append(presums[-1]^a)
        res = []
        for left, right in queries:
            res.append(presums[right+1]^presums[left])
        return res
