class Solution:
    def xorQueries(self, arr: List[int], queries: List[List[int]]) -> List[int]:
        pre = [0]
        
        for a in arr:
            pre.append(pre[-1] ^ a)
        
        res = []
        
        for a, b in queries:
            res.append(pre[b + 1] ^ pre[a])
        
        return res
