class Solution:
    def xorQueries(self, arr: List[int], queries: List[List[int]]) -> List[int]:
        tbl = [0]
        for e in arr:
            tbl.append(tbl[-1] ^ e)
        
        res = []
        for l, r in queries:
            res.append(tbl[r+1] ^ tbl[l])
        return res
