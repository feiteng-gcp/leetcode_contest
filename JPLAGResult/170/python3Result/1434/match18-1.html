class Solution:
    def freqAlphabets(self, s: str) -> str:
        n = len(s)
        i = 0
        ret = []
        while i < n:
            if i + 2 < n and s
[i + 2] == '#':
                ret.append(chr(ord('a') + int(s[i : i + 2]) - 1))
                i += 3
            else:
                ret.append(chr(ord('a') + int(s[i]) - 1))           
                i += 1
        return ''.join(ret)
