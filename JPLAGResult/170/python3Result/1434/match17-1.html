class Solution:
    def freqAlphabets(self, s: str) -> str:
        return convert(s)
    
    

def convert(s):
    """ 
    a-i maps to 0-9
    j - z maps to 10-26#
    if at i+2 is not # then map to a-i, otherwise map s[i,i+2] to j - z
    
    """
    res = []
    i = 0
    while i < len(s):
        if i+2 >= len(s) or s[i+2] != '#':
            res.append(chr(ord('a') + int(s[i])-1))
            i += 1
        else:
            res.append(chr(ord('j') + int(s[i:i+2]) - 10))
            i+=3
    return ''.join(res)
