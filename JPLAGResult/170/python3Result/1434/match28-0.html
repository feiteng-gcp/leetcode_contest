class Solution:
    def freqAlphabets(self, s: str) -> str:
        res = []
        i = len(s)-1
        while i >= 0:
            if s[i] == "#":
                res.append(int(s[i-2:i]))
                i -= 3
            else:
                res.append(int(s[i]))
                i -= 1
        return "".join(map(lambda x: chr(ord('a')+x-1), res))[::-1]
