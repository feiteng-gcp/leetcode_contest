class Solution:
    def freqAlphabets(self, s: str) -> str:
        result = []
        index = len(s) - 1

        while index >= 0:
            if s[index] in '123456789':
                result.append(chr(96 + int(s[index])))
                index -= 1
            else:
                result.append(chr(96 + int(s[index - 2: index])))
                index -= 3
        return "".join(result[::-1])
