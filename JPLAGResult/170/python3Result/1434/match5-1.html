class Solution:
    def freqAlphabets(self, s: str) -> str:
        n, ans, i = len(s), [], 0
        while i < n:
            if i >= n-2 or s[i+2] != '#':
                v = int(s[i])
                ans.append(chr(v-1+ord('a')))
                i += 1                
            else:
                v = int(s[i:i+2])
                ans.append(chr(v-10+ord('j')))
                i += 3
        return ''.join(ans)
