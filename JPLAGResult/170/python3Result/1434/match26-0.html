class Solution:
    def freqAlphabets(self, s: str) -> str:
        
        out = []
        i = len(s)-1
        
        while i >= 0:
            
            if s[i] == '#':
                num = int(s[i-2:i])
                out.append(chr(num - 10 + ord('j')))
                i = i - 3
            else:
                num = int(s[i])
                out.append(chr(num-1+ord('a')))
                i -= 1
        return ''.join(out[::-1])
