class Solution:
    def freqAlphabets(self, s: str) -> str:
        ans = []
        idx = 0
        while idx < len(s):
            if idx + 2 < len(s) and s[idx+2] == '#':
                ans.append(chr(ord('a') - 1 + int(s[idx:idx+2])))
                idx += 3
            else:
                ans.append(chr(ord('a') - 1 + int(s[idx])))
                idx += 1
        return ''.join(ans)
