class Solution:
    def freqAlphabets(self, s: str) -> str:
        i = len(s) - 1
        res = ''
        while i >= 0:
            if s[i] == '#':
                res += chr(int(s[i-2:i]) + ord('a') - 1)
                i -= 3
            else:
                res += chr(int(s[i]) + ord('a') - 1)
                i -= 1
        res = res[::-1]
        return res
