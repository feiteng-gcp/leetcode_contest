class Solution:
    def freqAlphabets(self, s: str) -> str:
        res = []
        n = len(s)

        i = n-1 
        while i >= 0: 
            c = s[i] 
            if c == '#': 
                num 
= int(s[i-2:i])
                res.append(chr(ord('a') + num - 1))
                i -= 3
            else: 
                num = int(s[i:i+1])
                res.append(chr(ord('a') + num - 1))
                i -= 1
        
        return ''.join(reversed(res))
