import string
chars = list(string.ascii_lowercase)

class Solution:
    def freqAlphabets(self, s: str) -> str:
        def num2char(n):
            return chars[int(n) - 1]
        
        ans 
= []
        i = len(s) - 1
        while i >= 0:
            if s[i] == '#':
                ans.append(num2char(s[i - 2: i]))
                i -= 3
            else:
                ans.append(num2char(s[i]))
                i -= 1
        
        return ''.join(reversed(ans))
