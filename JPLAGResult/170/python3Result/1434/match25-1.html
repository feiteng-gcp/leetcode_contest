class Solution:
    def freqAlphabets(self, s: str) -> str:
        res = ''
        n 
= len(s)
        idx = 0
        while idx < n:
            if idx+2 < n and s[idx+2] == '#':
                res += chr(int(s[idx:idx+2]) + ord('a') - 1)
                idx += 3
            else:
                res += chr(int(s[idx:idx+1]) + ord('a') - 1)
                idx += 1
        return res
