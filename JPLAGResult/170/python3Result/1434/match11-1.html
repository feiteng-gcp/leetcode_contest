class Solution:
    def freqAlphabets(self, s: str) -> str:
        n = len(s)
        i = 0
        ans = []
        
while i < n:
            if i + 2 < n and s[i + 2] == '#':
                num = int(s[i:i+2]) - 10
                ans.append(chr(ord('j') + num))
                i += 3
            else:
                num = int(s[i]) - 1
                ans.append(chr(ord('a') + num))
                i += 1
        return ''.join(ans)
