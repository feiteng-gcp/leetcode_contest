class Solution:
    def minInsertions(self, s: str) -> int:
        
        
        n = len(s)
        dp = [[0]*n for _ in range(n)]  # dp[i][j] 表示让 s[i:j+1] 变成回文所需的最少 insertion 次数
        for i in range(n-1):
            if s[i] != s[i+1]:
                dp[i][i+1] = 1
        for i in range(n-1, -1, -1):
            for j in range(i+2, n):
                if s[i] == s[j]:
                    dp[i][j] = dp[i+1][j-1]
                else:
                    dp[i][j] = 1 + min(dp[i+1][j], dp[i][j-1])
        return dp[0][-1]
