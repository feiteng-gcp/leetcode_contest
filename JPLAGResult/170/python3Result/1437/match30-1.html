class Solution:
    def minInsertions(self, s: str) -> int:
        n=len(s)
        res=[[-1]*n for k in range(n)]
        #print(res)
        for i in range(n):
            res[i][i]=0
        #print(res)
        for i in range(n-1):
            if s[i]==s[i+1]:
                res[i][i+1]=0
            else:
                res[i][i+1]=1
        #print(res)
        
for i in range(2,n):
            for k in range(n-i):
                # k  to k+i
                if s[k]==s[k+i]:
                    res[k][k+i]=res[k+1][k+i-1]
                else:
                    res[k][k+i] = 1 + min(res[k+1][k+i],res[k][k+i-1])
        
        #print("final",res)
        return res[0][n-1]
