from functools import lru_cache
class Solution:
    def minInsertions(self, s: str) -> int:
        @lru_cache(None)
        def helper(s):
            if len(s) <= 1: return 0
            if s[0] == s[-1]:
                return helper(s[1:-1])
            else:
                return 1 + min(helper(s[1:]), helper(s[:-1]))
        return helper(s)
