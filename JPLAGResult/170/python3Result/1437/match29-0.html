class Solution:
    def minInsertions(self, s: str) -> int:
        length = len(s)
        table = [[0 for _ in range(length)] for _ in range(length)]
        for gap in range(1, length):
            l = 0
            for h in range(gap, length):
                if s[l] == s[h]:
                    table[l][h] = table[l+1][h-1]
                else:
                    table[l][h] = min(table[l][h-1], table[l+1][h]) + 1
                l += 1
        return table[0][length-1]
