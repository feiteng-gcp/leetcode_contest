import functools

class Solution:
    def minInsertions(self, s: str) -> int:
        @functools.lru_cache(maxsize=None)
        
def dp(p, q):
            if p >= q:
                return 0
            if s[p] == s[q]:
                return dp(p + 1, q - 1)
            return min(dp(p + 1, q), dp(p, q - 1)) + 1
        
        return dp(0, len(s) - 1)
