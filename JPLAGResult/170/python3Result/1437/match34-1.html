class Solution:
    def minInsertions(self, s: str) -> int:
        n=len(s)
        dp=[[0]*n for _ in range(n)]
        
for k in range(2,n+1):
            for i in range(n-k+1):
                if s[i]==s[i+k-1]:
                     dp[i][i+k-1]=dp[i+1][i+k-2]
                else:
                    dp[i][i+k-1]=min(dp[i+1][i+k-1],dp[i][i+k-2])+1
                #print(i,k,dp[i][i+k-1])
        #print(dp)
        return dp[0][-1]
