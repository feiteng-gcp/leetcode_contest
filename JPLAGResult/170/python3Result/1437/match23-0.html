class Solution:
    def minInsertions(self, s: str) -> int:
        # returns the min insertions to make s[i:j+1] palindrome
        from functools import lru_cache
        @lru_cache(None)
        def getMinInsert(i: int, j: int) -> int:
            if i >= j:
                return 0
            
            if s[i] == s[j]:
                return getMinInsert(i + 1, j - 1)
            else:
                return min(getMinInsert(i + 1, j), getMinInsert(i, j - 1)) + 1
        
        return getMinInsert(0, len(s) - 1)
