import functools

class Solution:
    def minInsertions(self, s: str) -> int:

        @functools.lru_cache(None)
        def dp(start, end):
            if start >= end:
                return 0
            
            if s[start] == s[end]:
                return min(dp(start + 1, end - 1), dp(start, end - 1) + 1, dp(
start + 1, end) + 1)
            
            return min(dp(start, end - 1) + 1, dp(start + 1, end) + 1)
        
        return dp(0, len(s) - 1)
