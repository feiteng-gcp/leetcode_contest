class Solution:
    def longestPalindromeSubseq(self, s):
        n = len(s)
        if s == s[::-1]: return n         
        cur = [0] * n

        for i in range(len(s))[::-1]:
            pre = cur[:]
            cur[i] = 1
            for j in range(i+1, n):
                if s[i] == s[j]:
                    cur[j] = 2 + pre[j-1]
                else:
                    cur[j] = max(cur[j-1], pre[j])
        return cur[-1]

    def minInsertions(self, s: str) -> int:
        if s == s[::-1]: return 0
        return len(s) - self.longestPalindromeSubseq(s)
