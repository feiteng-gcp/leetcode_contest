class Solution:
    def minInsertions(self, s: str) -> int:
        n = len(s)
        dp = [[0] * n for _ in range(n)]  
        l, h, gap = 0, 0, 0

        
for gap in range(1, n): 
            l = 0
            for h in range(gap, n): 
                if s[l] == s[h]: 
                    dp[l][h] = dp[l + 1][h - 1] 
                else: 
                    dp[l][h] = (min
(dp[l][h - 1],  dp[l + 1][h]) + 1) 
                l += 1
        return dp[0][n - 1];
