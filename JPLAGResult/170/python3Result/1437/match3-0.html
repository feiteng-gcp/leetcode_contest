class Solution:
    def minInsertions(self, s: str) -> int:
        import numpy as np
        n = len(s)
        dp = np.zeros((n, n+1), dtype=int)
    
        for l in range(2, n+1):
            for i in range(n + 1 - l):
                if s[i] == s[i + l - 1]:
                    dp[i][l] = dp[i+1][l-2]
                else:
                    dp[i][l] = min(dp[i+1][l-1], dp[i][l-1]) + 1
                    
        return dp[0][n]
