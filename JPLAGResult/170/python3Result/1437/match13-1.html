import functools

class Solution:
    def minInsertions(self, s: str) -> int:
        @functools.lru_cache(maxsize=None)
        
def rec(a,b):
            if a > b:
                return 0
            if s[a] == s[b]:
                return rec(a+1,b-1)
            return 1 + min(rec(a,b-1),rec(a+1,b))
        return rec(0,len(s)-1)
