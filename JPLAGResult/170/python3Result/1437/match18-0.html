class Solution:
    def minInsertions(self, s: str) -> int:
        n = len(s)
        dp = [[0]*n for _ in range(n)]
        for sz in range(2, n+1):
            for l in range(n-sz+1):
                r = l+sz-1
                dp[l][r] = float('inf')
                if s[l]==s[r]:
                    dp[l][r] = min(dp[l][r], dp[l+1][r-1])
                else:
                    dp[l][r] = min(dp[l][r], 1 + min(dp[l + 1][r], dp[l][r-1]))
        return dp[0][-1]
