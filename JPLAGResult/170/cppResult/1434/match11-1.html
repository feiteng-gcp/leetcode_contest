class Solution {
public:
    string freqAlphabets(string s) {
        string t;

        for (int i = 0; i < s.size(); ) {
            if (i + 2 < s.size() and s[i + 2] == '#') {
                int x = (s[i] - '0') * 10 + (s[i + 1] - '0');
                cout << x << endl;
                t += x - 1 + 'a';
                i += 3;
            } else {
                t += s[i] - '0' - 1 + 'a';
                i++;
            }
        }
        return t;
    }
};
