class Solution {
public:
    string freqAlphabets(string s) {
        string res = "";
        res.reserve(s.length());
        for(int i = s.length() - 1 ; i >= 0 ; --i) {
            int num = 0;
            if(s[i] == '#') {
                num = stoi(s.substr(i-2, 2));
                i = i - 2;
            } else {
                num = s[i] - '0';
            }
            res += ('a' + num - 1);
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
