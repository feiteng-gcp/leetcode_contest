class Solution {
public:
    string freqAlphabets(string s) {
        int ls = s.size();
        string ans;
        
        
        
        for (int i = 0; i < ls;i++) {
            if (i+2 < ls && s[i+2] == '#') {
                int cnt = (s[i]-'0')*10 + s[i+1]-'1';
                ans.push_back(cnt + 'a');
                i+= 2;
            } else {
                ans.push_back('a' + s[i]-'1');
            }
        }
        
        return ans;
    }
};
