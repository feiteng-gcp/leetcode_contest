class Solution {
public:
    string freqAlphabets(string s) {
        int len = s.length();
        string ans = "";
        for (int i = len - 1; i >= 0; i--) {
            if (s[i] == '#') {
                int num = 0;
                --i;
                --i;
                num = (s[i] - '0') * 10 + s[i + 1] - '0' + 96;
                ans += char(num);
            } else {
                ans += char(s[i] - '0' + 96);
            }
        }
        reverse(ans.begin(),ans.end());
        return ans;
    }
};
