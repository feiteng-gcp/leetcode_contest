class Solution {
public:
    string freqAlphabets(string s) {
        string ans;
        int pos = s.size()-1;
        while (pos >= 0) {
            if (s[pos] == '#') {
                int digit = (s[pos-2]-'0')*10 + (s[pos-1]-'0');
                ans += (digit-1+'a');
                pos -= 3;
            } else {
                ans += (s[pos]-'1'+'a');
                pos --;
            }
        }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
