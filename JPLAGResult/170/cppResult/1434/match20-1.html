class Solution {
public:
    string freqAlphabets(string s) {
        string result;
        for (int i = 0; i < s.size(); ) {
            if (i + 2 < s.size() && s[i + 2] == '#' && (s[i] == '1' || s[i] == '2' && s[i + 1] <= '6')) {
                result.push_back('j' + 10 * (s[i] - '0') + (s[i + 1] - '0') - 10);
                i += 3;
            } else {
                result.push_back('a' + s[i] - '1');
                ++i;
            }
        }
        return result;
    }
};
