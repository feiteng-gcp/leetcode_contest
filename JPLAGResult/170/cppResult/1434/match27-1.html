class Solution {
public:
    string freqAlphabets(string s) {
        string res;
        int n = s.size();
        for (int i = n - 1; i >= 0; i--){
            if(s[i] == '#'){

                res = res + char('a' - 1 + (s[i-1] - '0' + 10 *(s[i-2] - '0')));
                i-= 2;
            }
            else {
                res = res + char('a' - 1 + s[i] - '0');
            }
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
