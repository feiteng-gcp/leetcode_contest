class Solution {
public:
    string freqAlphabets(string s) 
    {
        string ret;
        for (int i=0; i<s.size(); i++)
        {
            if (i+2<s.size() && s[i+2]=='#')
            {
                int num = stoi(s.substr(i,2));
                //cout<<num<<endl;
                ret.push_back('a'+num-1);
                i+=2;
            }
            else
            {
                int num = stoi(s.substr(i,1));
                //cout<<num<<endl;
                ret.push_back('a'+num-1);
            }
            //cout<<ret<<endl;
        }
        return ret;
    }
};
