class Solution {
public:
  string freqAlphabets(string s) {
    reverse(s.begin(),s.end());
    int n=s.size();
    string a;
    for(int i=0;i<n;i++){
      if(s[i]=='#'){
        int res=0;
        res=(s[i+2]-'0')*10+(s[i+1]-'0');
        a+=char('a'+res-1);
        i+=2;
      }else{
        a+=char('a'+(s[i]-'1'));
      }
    }
    reverse(a.begin(),a.end());
    return a;
  }
};
