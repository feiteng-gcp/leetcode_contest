class Solution {
public:
    string freqAlphabets(string s) {
        string ans;
        int n = s.size();
        for(int i=n-1;i>=0;--i){
            if(s[i] == '#'){
                int x = (s[i-1]-'0') + 10 * (s[i-2]-'0') - 1;
                ans.push_back('a'+x);
                i -= 2;
            }
            else{
                ans.push_back('a'+((s[i]-'0')-1));
            }
        }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
