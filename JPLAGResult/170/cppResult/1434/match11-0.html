class Solution {
public:
    string freqAlphabets(string s) {
        string res = "";
        for (int i = 0; i < s.length();)
        {
            if (i + 2 < s.length() && s[i + 2] == '#')
            {
                int num = stoi(s.substr(i, 2));
                res += 'j' + num - 10;
                i += 3;
            }
            else
            {
                res += 'a' + s[i++] - '1';
            }
        }
        return res;
    }
};
