class Solution {
public:
   string freqAlphabets(string s) {
		string res;

		for (int i = s.size() - 1; i >= 0; --i) {
			if (s[i] == '#') {
				char c = (s[i - 2] - '0') * 10 + (s[i - 1] - '0') - 1 + 'a';
				res += c;
				i -= 2;
			}
			else {
				char c = (s[i] - '0' - 1) + 'a';
                res += c;
			}
		}
		reverse(res.begin(), res.end());
		return res;
	}
};
