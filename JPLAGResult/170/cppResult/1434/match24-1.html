class Solution {
public:
    string freqAlphabets(string s) {
        string res = "";
        for (int i = s.size() - 1; i>=0; --i)
        {
            if (s[i] == '#')
            {

                string t = s.substr(i-2, 2); 
                res += stoi(t) - 10 + 'j';
                i -= 2;
            }
            else
            {
                res += s[i] - '1' + 'a';
            }
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
