class Solution {
public:
    string freqAlphabets(string s) {
        string res;
       // string s = "abcdefghij";
        for(int i=0; i<s.size();) {
            if (i+2<s.size() && s[i+2] == '#') {
                int val = stoi(s.substr(i, 2));
                res += ('j' + val - 10);
                i += 3;
            } else {
                res += 'a' + s[i] - '1';
                i+= 1;
            }
        }
        return res;
    }
};
