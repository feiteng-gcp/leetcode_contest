class Solution {
public:
    string freqAlphabets(string s) {
        string res;
        int n = s.size();
        for(int i = 0; i < n;) {
            if(i < n - 2 && s[i + 2] == '#') {
                res.push_back('a' + (stoi(s.substr(i, 2)) - 1));
                i += 3;
            }
            else {
                res.push_back('a' + (s[i] - '1'));
                i++;
            }
        }
        return res;
    }
};
