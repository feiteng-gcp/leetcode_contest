class Solution {
public:
    string freqAlphabets(string s) {
        string ans;
        for(int i = 0 ; i < s.size() ; i++) {
            if(i+2<s.size() && s[i+2] == '#') {
                int t = (s[i]-'0') * 10 + s[i+1]-'0'-10;
                ans.push_back(t+'j');
                i += 2;
            }
            else {
                ans.push_back(s[i]-'1'+'a');
            }
        }
        return ans;
    }
};
