class Solution {
public:
    string freqAlphabets(string s) {
        string res = "";
        int n = s.length();
        for (int i = n - 1; i >= 0; --i) {
            if (s[i] == '#') {
                char c = ((s[i-2] - '0') * 10 + (s[i-1] - '0') + 'a' - 1);
                res = c + res;
                i -= 2;
            } else {
                char c = ((s[i] - '1') + 'a');
                res = c + res;
            }
        }
        return res;
    }
};
