class Solution {
public:
    string freqAlphabets(string s) {
        int n = s.size();
        string ans;
        for(int i=0;i<n;) {
            if(i+2 < n && s[i+2] == '#') {
                int num = 10*(s[i]-'0') + s[i+1]-'0';
                assert(num >= 10);
                ans += 'j'+(num-10);
                i += 3;
            } else {
                ans += 'a'+s[i]-'1';
                i++;
            }
        }
        return ans;
    }
};
