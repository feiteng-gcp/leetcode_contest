class Solution {
public:
    string freqAlphabets(string s) {
        string ret;

        for (int i = s.size()-1; i >= 0; i--) {
            if (s[i] == '#') {
                int val = (s[i-2] - '0') * 10 + s[i-1] - '0';
                ret += 'a' + val - 1;
                i -= 2;
            }
            else {
                ret += 'a' + s[i] - '1';
            }
        }
        reverse(ret.begin(), ret.end());
        return ret;
    }
};
