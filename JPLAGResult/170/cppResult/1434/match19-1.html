class Solution {
public:
    string freqAlphabets(string s) {
        int i, n = s.size();
        stack<char>a;
        char t;
        for (i = 0; i < n; i++) {
            if (s[i] == '#') {
                int p = 0;
                t = a.top(); a.pop();
                p = t - '0';
                t = a.top(); a.pop();
                p += (t - '0') * 10;
                a.push(p + 'a' - 1);
            }
            else
                a.push(s[i]);
        }
        string res = "", res2 = "";
        while(!a.empty()) {
            t = a.top(); a.pop();
            if (t >= '0' && t <= '9')
                res.push_back(t - '0' + 'a' - 1);
            else
                res.push_back(t);
        }
        for (i = res.size() - 1; i >= 0; i--) {
            res2.push_back(res[i]);
        }
        return res2;
    }
};
