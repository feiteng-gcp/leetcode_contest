class Solution {
public:
    string freqAlphabets(string s) {
        string res;
        int i = s.length()-1;
        while (i >= 0) {
            if (s[i] == '#') {
                res.push_back(stoi(s.substr(i-2, 2)) - 10 + 'j');
                i = i-3;
            } else {
                res.push_back(s[i]-'1'+'a');
                i--;
            }
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
