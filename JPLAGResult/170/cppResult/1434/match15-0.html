class Solution {
public:
    string freqAlphabets(string s) {
        string res;
        for(int i = 0; i < s.size(); ++i)
        {
            if(i + 2 < s.size() && s[i + 2] == '#')
            {
                int curNumber = (s[i] - '0') * 10 + s[i + 1] - '0';
                res.push_back(curNumber - 10 + 'j');
                i += 2;
            }
            else
            {
                res.push_back('a' + s[i] - '1');
            }
        }
        return res;
    }
};
