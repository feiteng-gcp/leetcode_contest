class Solution {
public:
    string freqAlphabets(string s) {
        string res;
        int i = 0;
        while (i < s.size()) {
            if (i + 2 < s.size() && s[i + 2] == '#') {
                res.push_back((s[i] - '0') * 10 + s[i + 1] - '0' + 'a' - 1);
                i += 3;
            }
            else {
                res.push_back(s[i] - '0' + 'a' - 1);
                i++;
            }
        }
        return res;
    }
};
