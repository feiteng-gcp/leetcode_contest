class Solution {
public:
    string freqAlphabets(string s) {
        string ans;
        int len = s.size();
        for (int i = 0; i < len; ++i) {
            if (i+2 < len && s[i+2] == '#') {
                int num = (s[i]-'0')*10+(s[i+1]-'0');
                // printf("%d %d \n", i, num);
                ans += (num-1+'a');
                i+=2;
            } else {
                ans += ((s[i]-'0')-1+'a');
            }
        }
        return ans;
    }
};
