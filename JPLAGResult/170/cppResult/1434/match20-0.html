class Solution {
public:
    string freqAlphabets(string s) {
        string ans;
        for(int i = 0; i < s.size();){
            if(i + 2 < s.size() && s[i+2] == '#' && ((s[i] == '1') || (s[i] == '2' && s[i+1] <= '6'))){
                ans.push_back('a' + stoi(s.substr(i, 2)) - 1);
                i += 3;
            }else{
                ans.push_back('a' + s[i] - '1');
                i++;
            }
        }
        return ans;
    }
};
