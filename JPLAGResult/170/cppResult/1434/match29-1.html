class Solution {
public:
    string freqAlphabets(string s) {

        int n = s.size();
        string res;
        for(int i = n-1; i >= 0; ) {
            if(s[i] == '#') {
                int x = stoi(s.substr(i-2, 2));
                res += 'a'+x-1;
                i-=3;
            } else {
                res += 'a' + (s[i] - '1');
                i--;
            }
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
