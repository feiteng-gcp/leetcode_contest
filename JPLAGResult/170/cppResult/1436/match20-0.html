class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        queue<int> q; q.push(id);
        set<int> used; used.insert(id);
        for(int i=0; i<level; ++i) {
            int size = q.size();
            for(int j=0; j<size; ++j) {
                int node = q.front(); q.pop();
                for(auto k: friends[node]) {
                    if (used.find(k) == used.end()) {
                        q.push(k);
                        used.insert(k);
                    }
                }
            }
        }
        unordered_map<string, int> cc;
        while(!q.empty()) {
            int node = q.front(); q.pop();
            for(auto i: watchedVideos[node]) {
                cc[i] += 1;
            }
        }
        vector<pair<int, string> > items;
        for(auto i: cc) items.push_back(make_pair(i.second, i.first));
        sort(items.begin(), items.end());
        vector<string> res;
        for(auto i: items) res.push_back(i.second);
        return res;
    }
};
