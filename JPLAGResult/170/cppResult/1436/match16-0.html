class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        queue<int>Q;
        Q.push(id);
        unordered_set<int>visited;
        visited.insert(id);
        for (int z = 0; z < level && !Q.empty(); ++z) {
            int qsize = Q.size();
            while (qsize--) {
                int cur = Q.front();
                Q.pop();
                for (auto& nid : friends[cur]) {
                    if (visited.find(nid) == visited.end()) {
                        Q.push(nid);
                        visited.insert(nid);
                    }
                }
            }
        }
        map<string, int>dic;
        while (!Q.empty()) {
            for (auto& video : watchedVideos[Q.front()]) {
                ++dic[video];
            }
            Q.pop();
        }
        vector<string>strs;
        vector<pair<string, int>>freq(dic.begin(), dic.end());
        auto cmp = [&](pair<string, int>& lhs, pair<string, int>& rhs) {
            return lhs.second != rhs.second ? lhs.second < rhs.second : lhs.first < rhs.first;
        };
        sort(freq.begin(), freq.end(), cmp);
        for (auto& p : freq) {
            strs.emplace_back(p.first);
        }
        return strs;
    }
};
