class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level)   {
       int N = watchedVideos.size();
        queue<int> qq;
        qq.push(id);
        vector<int> dis(N,INT_MAX);
        dis[id]=0;
        int cnt=1;
        while(cnt<=level){
            int len=qq.size();
            while(len--){
                int p=qq.front();
                qq.pop();
                for(int f:friends[p]){
                    if(dis[f]>cnt){
                        dis[f]=cnt;
                        qq.push(f);
                    }
                }
            }
            cnt++;
        }
        
        unordered_map<string,int> mm;
        for(int i=0;i<N;i++){
            if(dis[i]==level){
                for(string w:watchedVideos[i]){
                    mm[w]++;
                }
            }
        }
        vector<pair<int,string>> pp;
        for(auto p:mm){
            pp.push_back(make_pair(p.second,p.first));
        }
        sort(pp.begin(),pp.end());
        vector<string> ans;
        for(auto p:pp){
            ans.push_back(p.second);
        }
        return ans;
    }
};
