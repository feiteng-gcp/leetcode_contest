class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& vv, vector<vector<int>>& g, int id, int level) {
        int n = g.size();
        map<string, int> mp;
        vector<pair<int, string>> res;
        

        vector<int> best(n, n);
        best[id] = 0;
        queue<int> q;
        q.push(id);
        
        while(!q.empty()){
            int x = q.front();
            q.pop();
            
            for(auto &y : g[x]){
                if(best[y] > best[x] + 1){
                    best[y] = best[x] + 1;
                    q.push(y);
                }
            }
        }
        

        for(int i = 0; i < n; ++i){
            if(best[i] == level){
                for(auto &y : vv[i]){
                    mp[y]++;
                }
            }
        }
        
        for(auto &p : mp){
            res.push_back({p.second, p.first});
        }
        sort(res.begin(), res.end());
        vector<string> fin;
        for(auto &x : res) fin.push_back(x.second);
        return fin;
    }
};
