class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        queue<int> q;
        q.push(id);
        vector<bool> visited(watchedVideos.size(), false);
        visited[id] = true;
        while(level--){
            int size = q.size();
            while(size--){
                int cur = q.front(); q.pop();
                for(int i : friends[cur]){
                    if(!visited[i]){
                        q.push(i);
                        visited[i] = true;
                    }
                }
            }
        }
        unordered_map<string, int> movies;
        while(!q.empty()){
            int cur = q.front();
            q.pop();
            for(const string& str : watchedVideos[cur]){
                movies[str]++;
            }
        }
        vector<pair<int, string>> ps;
        for(auto it = movies.begin(); it != movies.end(); ++it){
            ps.emplace_back(it->second, it->first);
        }
        sort(ps.begin(), ps.end());
        vector<string> ans;
        for(int i = 0; i < ps.size(); i++){
            ans.push_back(ps[i].second);
        }
        return ans;
    }
};
