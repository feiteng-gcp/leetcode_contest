class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int n = friends.size();
        vector<bool> vis(n, false);
        
        queue<int> q;
        q.push(id);
        vis[id] = true;
        int step = 0;

        while(!q.empty()) {
            int size = q.size();
            if(step == level) break;
            while(size--) {
                int cur = q.front(); q.pop();
                for(int fre: friends[cur]) {
                    if(!vis[fre]) {
                        q.push(fre);
                        vis[fre] = true;
                    }
                }
            }
            ++step;
        }
        
        unordered_map<string, int> m;
        while(!q.empty()) {
            int cur = q.front(); q.pop();
            for(string video: watchedVideos[cur]) {
                m[video]++;
            }
        }
        
        map<int, vector<string> > m2;
        for(auto &it: m) {
            m2[it.second].push_back(it.first);
        }
        
        vector<string> res;
        for(auto &it: m2) {
            vector<string> cur = it.second;
            sort(cur.begin(), cur.end());
            for(string tt: cur) res.push_back(tt);
        }
        return res;
    }
};
