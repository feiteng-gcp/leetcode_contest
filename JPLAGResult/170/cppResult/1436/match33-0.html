class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int n = friends.size();
        vector<int> dp(n, 1<<30);
        dp[id] = 0;
        queue<int> q;
        q.push(id);
        map<string,int> cnt;
        while(!q.empty()){
            int x = q.front();  q.pop();
            for(int y : friends[x]){
                if(dp[y] == 1<<30){
                    dp[y] = dp[x]+1;
                    q.push(y);
                }
            }
            if(dp[x] == level){
                for(string s: watchedVideos[x]) cnt[s]++;
            }
        }
        vector<pair<int,string>> vp;
        for(auto p : cnt)   vp.push_back({p.second, p.first});
        sort(vp.begin(), vp.end());
        vector<string> ret;
        for(auto p : vp)    ret.push_back(p.second);
        return ret;
    }
};
