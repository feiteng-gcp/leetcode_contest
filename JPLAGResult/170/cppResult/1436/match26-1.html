class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& v, vector<vector<int>>& f, int id, int level) {
        const int n = f.size();
        
        vector<bool> used(n, false);
        

        queue<int> q;
        q.push(id);
        used[id] = true;
        
        while(level-->0)
        {
            int cnt = q.size();
            
            while(cnt-->0)
            {
                int cur = q.front(); q.pop();
                
                for(int nxt: f[cur])
                {
                    if(!used[nxt])
                    {
                        used[nxt] = true;
                        q.push(nxt);
                    }
                }
            }
        }
        
        vector<string> res;
        
        map<string, int> m;
        
        while(!q.empty())
        {
            int cur = q.front(); q.pop();
            
            for(string& s: v[cur])
                m[s]++;
        }
        
        vector<pair<int, string>> resCnt;
        
        for(auto& p: m)
            resCnt.push_back({p.second, p.first});
        
        sort(resCnt.begin(), resCnt.end());
        
        for(auto& p: resCnt)
            res.push_back(p.second);
        
        return res;
    }
};
