class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) 
    {        
        queue<int>q({id});
        unordered_set<int>Set;
        Set.insert(id);
        int l = 0;
        
        vector<int>ff;
        while (q.size()>0)
        {
            int len = q.size();
            l++;
            while (len--)
            {
                int p = q.front();
                q.pop();                
                
                for (auto next: friends[p])
                {
                    if (Set.find(next)!=Set.end())
                        continue;
                    Set.insert(next);
                    q.push(next);
                    
                    if (l==level)
                        ff.push_back(next);                    
                }
            }
            if (l==level)
                break;
        }
        // for (auto x:ff)
        //     cout<<x<<" ";
        // cout<<endl;
        
        unordered_map<string,int>Map;
        for (auto x: ff)
        {
            for (auto y: watchedVideos[x])
                Map[y]++;
        }
        
        vector<pair<int,string>>rets;
        for (auto x:Map)
            rets.push_back({x.second,x.first});
        
        sort(rets.begin(),rets.end());
        
        vector<string>ans;
        for (auto x:rets)
            ans.push_back(x.second);
        
        return ans;
        
    }
};
