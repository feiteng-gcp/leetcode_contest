class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int n = watchedVideos.size();
        vector<int> dp(n, -1);
        queue<int> q;
        q.push(id);
        dp[id] = 0;
        vector<int> goods;
        while (!q.empty()) {
            int v = q.front(); q.pop();
            if (dp[v] == level) goods.push_back(v);
            for (int to : friends[v]) {
                if (dp[to] == -1) {
                    dp[to] = dp[v] + 1;
                    q.push(to);
                }
            }
        }
        map<string, int> cnt;
        for (int x : goods) {
            for (string s : watchedVideos[x]) {
                cnt[s]++;
            }
        }
        vector<pair<int, string> > tmp;
        for (__typeof(cnt.begin()) it = cnt.begin(); it != cnt.end(); it++) {
            tmp.push_back(make_pair((it->second), it->first));
        }
        sort(tmp.begin(), tmp.end());
        vector<string> result;
        for (auto it : tmp) {
            result.push_back(it.second);
        }
        return result;
    }
};
