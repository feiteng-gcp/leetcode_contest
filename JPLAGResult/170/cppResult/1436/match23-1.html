class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        map<string,int> m;
        vector<int> vis(friends.size(),-1);
        vis[id]=0;
        queue<int> q;
        q.push(id);

        while(!q.empty()){
            int x=q.front();
            q.pop();
            for(auto it:friends[x]){
                if(vis[it]==-1){
                    vis[it]=vis[x]+1;
                    q.push(it);
                }
            }
        }

        for(int i = 0;i<friends.size();i++){
            if(vis[i]==level){
                for(auto it:watchedVideos[i])
                        m[it]++;
            }
        }
        vector<pair<int,string> > v;
        for(auto it:m)
            v.push_back(make_pair(it.second,it.first));
        sort(v.begin(),v.end());
        vector<string> res;
        for(auto it:v)
            res.push_back(it.second);
        return res;
    }
};
