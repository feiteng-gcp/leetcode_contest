class Solution {
public:
    struct St
    {
        string str;
        int val;
        St(const string &str, int val) : str(str), val(val){}
        bool operator<(const St&other)
        {
            if (val != other.val)
                return val < other.val;
            return str < other.str;
        }
    };

    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int size = watchedVideos.size();
        queue<int> que;
        que.push(id);
        vector<bool> visited(size, false);
        visited[id] = true;
        for (int i = 0; i < level; i++)
        {
            int qs = que.size();
            for (int j = 0; j < qs; j++)
            {
                int cur = que.front();
                que.pop();
                for (int n : friends[cur])
                    if (!visited[n])
                    {
                        visited[n] = true;
                        que.push(n);
                    }
            }
        }
        if (que.empty())
            return vector<string>();
        map<string, int> mmap;
        while (que.size())
        {
            int cur = que.front();
            que.pop();
            for (string &str : watchedVideos[cur])
                mmap[str]++;
        }
        
        vector<string> res;
        vector<St> sts;
        for (auto iter = mmap.begin(); iter != mmap.end(); iter++)
            sts.push_back(St(iter->first, iter->second));
        sort(sts.begin(), sts.end());
        for (St &ss : sts)
            res.push_back(ss.str);
        
        return res;
    }
};
