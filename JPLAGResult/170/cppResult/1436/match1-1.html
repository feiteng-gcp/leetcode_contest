class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>> &watchedVideos,
                                          vector<vector<int>> &friends, int id,
                                          int level) {
        int n = watchedVideos.size();
        int vis[n];
        memset(vis, -1, sizeof(vis));
        vis[id] = 0;
        queue<int> sk;
        sk.push(id);
        map<string, int> ap;
        while (!sk.empty()) {
            int front = sk.front();
            sk.pop();
            if (vis[front] == level) {
                for (auto s : watchedVideos[front]) {
                    ap[s]++;
                }
                continue;
            }
            for (auto s : friends[front]) {
                if (vis[s] == -1) {
                    vis[s] = vis[front] + 1;
                    sk.push(s);
                }
            }
        }
        vector<pair<string, int>> app;
        for (auto s : ap) {
            app.push_back(make_pair(s.first, s.second));
        }
        sort(app.begin(), app.end(),
             [](const pair<string, int> &x, const pair<string, int> &y) {
                 if (x.second != y.second) {
                     return x.second < y.second;
                 }
                 return x.first < y.first;
             });
        vector<string> ans;
        for (auto s : app) {
            ans.push_back(s.first);
        }
        return ans;
    }
};
