class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int n = static_cast<int>(friends.size());
        vector<int> levels(n, -1);
        levels[id] = 0;
        queue<int> q;
        q.push(id);
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (auto v : friends[u]) {
                if (levels[v] == -1) {
                    levels[v] = levels[u] + 1;
                    q.push(v);
                }
            }
        }
        map<string, int> videoCounts;
        for (int i = 0; i < n; ++i) {
            if (levels[i] == level) {
                for (auto v : watchedVideos[i]) {
                    ++videoCounts[v];
                }
            }
        }
        vector<pair<int, string>> videos;
        for (auto it : videoCounts) {
            videos.push_back({it.second, it.first});
        }
        sort(videos.begin(), videos.end());
        vector<string> result;
        for (auto it : videos) {
            result.push_back(it.second);
        }
        return result;
    }
};
