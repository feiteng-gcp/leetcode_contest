class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int N = watchedVideos.size();
        vector<int> dis(N, INT_MAX);
        vector<int> bfs;
        dis[id] = 0;
        bfs.push_back(id);
        while (bfs.size()) {
            vector<int> next;
            for (auto now : bfs) {
                // printf("bfs %d \n", now);
                for (auto f : friends[now])
                    if (dis[f] == INT_MAX) {
                        dis[f] = dis[now] + 1;
                        next.push_back(f);

                    }
            }
            bfs = next;
        }
        map<string, int> cnt;
        for (int i = 0; i < N; ++i) {
            if (dis[i] == level) {
                // printf("target %d \n", i);
                for (auto w : watchedVideos[i]) {

                    cnt[w]++;                    
                }
            }
        }
        map<int, vector<string>> ri;
        for (auto pair : cnt) {
            // printf("%s %d\n", pair.first.c_str(), pair.second);
            ri[pair.second].push_back(pair.first);
        }
        vector<string> ans;
        for (auto pair : ri) {
            auto v = pair.second;
            sort(v.begin(), v.end());
            for (auto w : v)
                ans.push_back(w);
        }
        return ans;
    }
};
