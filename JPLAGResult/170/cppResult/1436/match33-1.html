class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int N = watchedVideos.size();
        vector<int> dist(N);
        dist[id] = 0;
        queue<int> bfsQ;
        bfsQ.push(id);
        int depth = 0;
        while (!bfsQ.empty()) {
            int size = bfsQ.size();
            while (size--) {
                int cur = bfsQ.front(); bfsQ.pop();
                for (auto next: friends[cur]) {
                    if (dist[next] || next == id) continue;
                    bfsQ.push(next);
                    dist[next] = depth+1;
                }
            }
            depth ++;
        }
        map<string, int> freq;
        for (int i = 0; i < N; i++) {
            if (dist[i] == level) {
                for (auto v: watchedVideos[i]) {
                    freq[v] ++;
                }

            }
        }
        vector<pair<int, string>> tmp;
        for (auto kv: freq)
            tmp.push_back({kv.second, kv.first});
        sort(tmp.begin(), tmp.end());
        vector<string> res;
        for (auto p: tmp)
            res.push_back(p.second);
        return res;
    }
};
