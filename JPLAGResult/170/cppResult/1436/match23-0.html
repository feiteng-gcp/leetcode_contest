class Solution {
public:
	vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
		int n = friends.size(), i, j, k;
		vector<int> depth(n, -1);
		queue<int> q;
		q.push(id);
		depth[id] = 0;

		while (!q.empty()) {
			int id = q.front();
			q.pop();

			for (auto p : friends[id]) {
				if (depth[p] == -1) {
					depth[p] = depth[id] + 1;
					q.push(p);
				}
			}
		}

		map<string, int> ct;
		for (i = 0; i < n; i++) {
			if (level == depth[i]) {
				for (auto p : watchedVideos[i]) ct[p]++;
			}
		}
		vector<pair<int, string>> ss;
		for (auto p : ct) ss.push_back({ p.second, p.first });
		sort(ss.begin(), ss.end());
		vector<string> ans;
		for (auto p : ss) ans.push_back(p.second);
		return ans;
	}
};
