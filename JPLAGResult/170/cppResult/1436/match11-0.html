class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level)
    {
        int n = friends.size();
        vector<int> people(n, -1);
        people[id] = 0;
        queue<int> q;
        q.push(id);
        while (q.size() > 0) {
            int p = q.front(); q.pop();
            for (int i : friends[p]) {
                if (people[i] != -1) continue;
                people[i] = people[p] + 1;
                q.push(i);
            }
        }
        unordered_map<string, int> videos;
        for (int j = 0; j < n; j++) {
            if (people[j] != level) continue;
            for (string& v : watchedVideos[j])
                videos[v]++;
        }
        vector<pair<int, string>> order;
        for (auto kv : videos)
            order.push_back(make_pair(kv.second, kv.first));
        std::sort(order.begin(), order.end());
        vector<string> answer;
        for (auto& pair : order)
            answer.push_back(pair.second);
        return answer;
    }
};
