class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int n = friends.size();
        vector<int> vis(n, -1);
        queue<int> que;
        que.push(id);
        vis[id] = 0;
        unordered_map<string, int> hm;
        while(!que.empty()) {
            int x = que.front(); que.pop();
            for(int y : friends[x]) {
                if(vis[y] < 0) {
                    vis[y] = vis[x] + 1;
                    if(vis[y] == level) {
                        for(auto& video : watchedVideos[y]) {
                            hm[video]++;
                        }
                    } else {
                        que.push(y);
                    }

                }
            }
        }
        vector<pair<int, string>> prs;
        for(auto& pr : hm) {
            prs.push_back({pr.second, pr.first});
        }
        sort(prs.begin(), prs.end());
        vector<string> res;
        for(auto& pr : prs) {
            res.push_back(pr.second);
        }
        return res;
    }
};
