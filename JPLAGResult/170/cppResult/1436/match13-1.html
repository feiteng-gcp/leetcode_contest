class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& w, vector<vector<int>>& f, int id, int l) {
        vector<string> res;
        vector<int> v(w.size(), 0);
        queue<int> q;
        q.push(id);
        v[id] = 1;
        
        for(int i = 0; i < l; i++) {
            int len = q.size();

            for(int j = 0; j < len; j++) {
                int h = q.front();
                q.pop();
                for(int k : f[h]) {
                    if(v[k] == 0) {
                        v[k] = 1; 
                        q.push(k);
                    }
                }
            }

        }
        
        unordered_map<string, int> a;
        while(!q.empty()) {
            int h = q.front();
            q.pop();
            for(auto& t : w[h]) {
                a[t]++;
            }
        }
        
        vector<pair<int, string>> b;
        for(auto& t : a) {
            b.push_back({t.second, t.first});
        }
        std::sort(b.begin(), b.end());
        for(auto& t : b) {
            res.push_back(t.second);
        }
        return res;
    }
};
