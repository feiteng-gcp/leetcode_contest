class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        queue<int> q;
        set<int> s;
        q.push(id);
        s.insert(id);
        for(int i = 0; i < level; i++) {
            for(int j = q.size(); j > 0; j--) {
                auto tmp = q.front(); q.pop();

                for(int f: friends[tmp]) {
                    if(s.count(f) == 0) {
                        s.insert(f);
                        q.push(f);
                    }
                }
            }
        }
    map<string, int> m;
    while(q.size() > 0) {
        int i = q.front(); q.pop();
        for(auto v: watchedVideos[i]) {
            m[v] = m[v] + 1;
        }
    }
    
    vector<string> ret;
    for(const auto& mm: m) {
        ret.push_back(mm.first);
    }
    sort(ret.begin(), ret.end(), [&](const auto a, const auto b) {
        if(m[a] == m[b]) return a < b;
        else return m[a] < m[b];
    });
    return ret;
        
    }
};
