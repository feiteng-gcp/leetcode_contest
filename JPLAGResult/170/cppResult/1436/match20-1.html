class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        queue<int> qe;
        qe.push(id);
        
        unordered_set<int> visited;
        visited.insert(id);
        
        while(!qe.empty()) {
            if(level == 0) {
                break;
            }
            

            int size = qe.size();
            for(int i = 0; i < size; i++) {
                int person = qe.front();
                qe.pop();
                
                for(auto fd : friends[person]) {
                    if(visited.count(fd) > 0) {
                        continue;
                    }
                    
                    visited.insert(fd);
                    qe.push(fd);
                }
            }
            
            level--;

        }
        
        unordered_map<string, int> mp;
        while(!qe.empty()) {
            int person = qe.front();
            qe.pop();
            
            for(auto video : watchedVideos[person]) {
                mp[video]++;
            }
        }
        
        set<pair<int, string>> st;
        for(auto p : mp) {
            st.insert({p.second, p.first});
        }
        
        vector<string> res;
        for(auto p : st) {
            res.push_back(p.second);
        }
        
        return res;
    }
};
