class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int n = friends.size();

        const int INF = 1 << 20;
        vector<int> dis(n, INF);
        dis[id] = 0;
        
        queue<int> que;
        que.push(id);
        
        while (!que.empty()) {
            int u = que.front();
            que.pop();
            
            for (auto v : friends[u]) {
                if (dis[v] > dis[u] + 1) {
                    dis[v] = dis[u] + 1;
                    que.push(v);
                }
             }
        }
        
        map<string,int> freqs;
        
        for (int i = 0; i < n; ++i) {
            if (dis[i] == level) {
                for (auto& s : watchedVideos[i]) {
                    ++freqs[s];
                }
            }
        }
        
        vector<pair<int,string>> V;
        for (auto& p : freqs) {
            V.emplace_back(p.second, p.first);
        }
        sort(V.begin(), V.end());
        
        vector<string> res;
        for (auto& p : V) {
            res.push_back(p.second);
        }
        
        return res;
    }
};
