class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        const int n = friends.size();
        vector<int> dist(n, -1);
        queue<int> q;
        q.push(id);
        dist[id] = 0;
        while (!q.empty()) {
            const int current = q.front();
            q.pop();
            for (int i = 0; i < friends[current].size(); ++i) {
                if (dist[friends[current][i]] == -1) {
                    dist[friends[current][i]] = dist[current] + 1;
                    q.push(friends[current][i]);
                }
            }
        }
        unordered_map<string, int> count;
        for (int i = 0; i < n; ++i) {
            if (dist[i] == level) {
                for (int j = 0; j < watchedVideos[i].size(); ++j) count[watchedVideos[i][j]]++;
            }
        }
        vector<pair<int, string>> candidates;
        for (auto it = count.begin(); it != count.end(); ++it) {
            candidates.push_back(make_pair(it->second, it->first));
        }
        sort(candidates.begin(), candidates.end());
        vector<string> ans;
        for (int i = 0; i < candidates.size(); ++i) ans.push_back(candidates[i].second);
        return ans;
    }
};
