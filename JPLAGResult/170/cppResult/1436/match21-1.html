class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int n=friends.size();
        vector<int> gr[n];
        for(int i=0; i<n; i++){
            for(auto y : friends[i]){
                gr[i].push_back(y);
                gr[y].push_back(i);
            }
        }
        

        vector<int> fl(n,-1);
        fl[id]=0;
        queue<int> q;
        q.push(id);
        while(!q.empty()){
            int j=q.front();
            q.pop();
            for(auto x : gr[j]){
                if(fl[x]==-1){
                    fl[x]=fl[j]+1;
                    q.push(x);
                }
            }
        }
        
        unordered_map<string,int> vds;
        for(int i=0; i<n; i++){
            if(fl[i]==level){
                for(auto x : watchedVideos[i]){
                    vds[x]++;
                }
            }
        }
        
        vector<pair<int,string>> ans;
        for(auto x : vds){
            ans.push_back({x.second,x.first});
        }
        sort(ans.begin(), ans.end());
        vector<string> ret;
        for(auto x : ans){
            ret.push_back(x.second);
        }
        return ret;
    }
};
