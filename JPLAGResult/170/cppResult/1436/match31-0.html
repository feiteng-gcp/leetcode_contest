bool cmp(const pair<int, string>& A, const pair<int, string>& B)
{
    if(A.first!=B.first) return A.first > B.first;
    return A.second < B.second;
}

class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& vids, vector<vector<int>>& friends, int id, int level) {
        int N=vids.size();
        const int INF=10001;
        vector<int> dist(N, INF);
        
        dist[id]=0;
        queue<int> Q; Q.push(id);
        while(!Q.empty())
        {
            int curr=Q.front(); Q.pop();
            for(int next: friends[curr])
            {
                if(dist[next]==INF)
                {
                    dist[next]=dist[curr]+1;
                    Q.push(next);
                }
            }
        }
        
        unordered_map<string, int> freq;
        for(int i=0; i<N; i++)
        {
            if(dist[i]==level)
            {
                for(auto s: vids[i]) freq[s]++;
            }
        }
        
        //int sz=freq.size();
        vector<pair<int, string>> PP;
        for(auto xx: freq)
        {
            PP.pb({xx.second, xx.first});
        }
        sort(PP.begin(), PP.end());
        vector<string> result;
        for(auto xx: PP)
        {
            result.pb(xx.second);
        }
        return result;
    }
};
