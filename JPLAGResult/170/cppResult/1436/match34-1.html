class Solution {
    //vector<int> visited;
 

public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int n=friends.size();
        vector<int> depth=vector<int>(n);
        vector<int> visited=vector<int>(n);
        for(int i=0;i<n;i++){
            depth[i]=-1;
            visited[i]=0;
        }
        queue<int> bfs;
        bfs.push(id);
        depth[id]=0;
        while(bfs.size()){
            int u=bfs.front();
            bfs.pop();
            if(visited[u]!=0) continue;
            visited[u]=1;
            for(int i=0;i<friends[u].size();i++){
                int v=friends[u][i];

                if(depth[v]==-1){
                    depth[v]=depth[u]+1;
                    bfs.push(v);
                }
            }
        }
        map<string,int> cnt;
        for(int i=0;i<n;i++){
            if(depth[i]==level){
                for(int j=0;j<watchedVideos[i].size();j++){
                    string s=watchedVideos[i][j];
                    if(cnt.find(s)==cnt.end()) cnt[s]=1;
                    else cnt[s]++;
                }

            }
        }
        vector<pair<int,string>> ans;
        for(auto it=cnt.begin();it!=cnt.end();it++){
            ans.push_back(make_pair(it->second,it->first));
        }
        sort(ans.begin(),ans.end());
        vector<string> res;
        for(int i=0;i<ans.size();i++) res.push_back(ans[i].second);
        return res;
    }
};
