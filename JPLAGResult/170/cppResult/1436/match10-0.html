class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        unordered_set<int> visited;
        queue<int> q;
        q.push(id);
        visited.insert(id);
        while (!q.empty()) {
            int sz = q.size();
            if (0 == level--) break;
            for (int i=0; i<sz; i++) {
                int cur = q.front();
                q.pop();
                //visited.insert(cur);
                for (int f : friends[cur]) {
                    if (visited.count(f) == 0) {
                        visited.insert(f);
                        q.push(f);
                    }
                }
            }
        }
        map<string,int> cnt;
        while (!q.empty()) {
            int cur = q.front();
            q.pop();
            for (string s : watchedVideos[cur]) {
                cnt[s]++;
            }
        }
        map<int, set<string>> reverse;
        for (auto p : cnt) {
            reverse[p.second].insert(p.first);
        }
        vector<string> res;
        for (auto p : reverse) {
            res.insert(res.end(), p.second.begin(), p.second.end());
        }
        return res;
    }
};
