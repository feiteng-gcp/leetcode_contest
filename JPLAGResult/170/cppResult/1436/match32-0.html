bool cmp(pair<string, int> &a, pair<string, int> &b) {
    if (a.second == b.second) {
        return a.first < b.first;
    }
    return a.second < b.second;
}
class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        int len = friends.size();
        unordered_map<int, unordered_set<int>> graph;
        for (int i = 0; i < len; i++) {
            for (auto f : friends[i]) {
                graph[i].insert(f);
            }
        }
        vector<int> visited(len, false);
        queue<int> todo;
        todo.push(id);
        visited[id] = true;
        int d = 0;
        while(!todo.empty() && d != level) {
            int num = todo.size();
            for (int i = 0; i < num; i++) {
                int t = todo.front();
                todo.pop();
                for (auto f : graph[t]) {
                    if (!visited[f]) {
                        todo.push(f);
                        visited[f] = true;
                    }
                }
            }
            d++;
        }
        if (todo.empty()) {
            return {};
        }
        unordered_map<string, int> freq;
        while(!todo.empty()) {
            int t = todo.front();
            cout<<t<<endl;
            todo.pop();
            for (auto v :watchedVideos[t]) {
                freq[v]++;
            }
        }
        vector<pair<string, int>> pairs;
        for (auto p: freq) {
            pairs.push_back(p);
        }
        sort(pairs.begin(), pairs.end(), cmp);
        vector<string> res;
        for (auto p : pairs) {
            res.push_back(p.first);
        }
        return res;
    }
};
