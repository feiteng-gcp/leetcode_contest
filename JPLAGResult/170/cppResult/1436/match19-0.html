class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& watchedVideos, vector<vector<int>>& friends, int id, int level) {
        queue<int> Q;
        Q.push(id);
        vector<bool> visited(friends.size(), false);
        visited[id] = true;
        
        for (int i = 1;i <= level;i++) {
            int n = Q.size();
            
            for (int j = 0;j < n;j++) {
                int x = Q.front();
                Q.pop();
                for (auto& y: friends[x]) {
                    if (visited[y]) {
                        continue;
                    }
                    visited[y] = true;
                    Q.push(y);
                }
            }
        }
        
        map<string, int> videosmap;
        int n = Q.size();
        for (int i = 0;i < n;i++) {
            int x = Q.front();
            Q.pop();
            for (auto& y: watchedVideos[x]) {
                videosmap[y]++;
            }
        }
        
        vector<pair<int, string>> videos;
        for (auto it = videosmap.begin();it != videosmap.end();it++) {
            videos.push_back(make_pair(it->second, it->first));
        }
        sort(videos.begin(), videos.end());
        
        vector<string> ret;
        for (auto& x: videos) {
            ret.push_back(x.second);
        }
        return ret;
    }
};
