class Solution {
public:
    vector<string> watchedVideosByFriends(vector<vector<string>>& v, vector<vector<int>>& g, int s, int k) {
        int n = g.size();
        vector<int> dp(n, -1);
        dp[s] = 0;
        queue<int> Q;
        Q.push(s);
        while (!Q.empty()) {
            int x = Q.front();
            Q.pop();
            for (auto& y : g[x]) {

                if (dp[y] >= 0) continue;
                dp[y] = dp[x] + 1;
                Q.push(y);
            }
        }
        map<string, int> cnt;
        for (int i = 0; i < n; ++i) {
            if (dp[i] != k) continue;
            for (auto& s : v[i]) {
                cnt[s]++;
            }
        }
        vector<pair<int, string>> a;
        for (auto& it : cnt) {
            a.push_back({it.second, it.first});
        }
        sort(a.begin(), a.end());
        vector<string> ret;
        for (auto& it : a) ret.push_back(it.second);
        return ret;
    }
};
