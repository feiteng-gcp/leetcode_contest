class Solution {
public:
    int minInsertions(string s) {
        int n = s.size();
        vector<vector<int>> dp(n+1, vector<int>(n+1, 0));
        for(int i = 2; i <= n; i++) {
            for(int j = 0; j+i-1 < n; j++) {
                if(s[j] == s[j+i-1]) {
                    dp[i][j] = dp[i-2][j+1];
                }
                else {
                    dp[i][j] = min(dp[i-1][j], dp[i-1][j+1])+1;
                }
            }
        }
        return dp[n][0];
    }
};
