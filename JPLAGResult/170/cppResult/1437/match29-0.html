class Solution {
public:
    int minInsertions(string s) {
        int n = s.length();
        vector<vector<int>> dp(n, vector<int>(n, 0x7fffffff));
        for (int i = n - 1; i>= 0; i--) {
            for (int j = i; j < n; j++) {
                if (j == i) {
                    dp[i][j] = 0;
                } else if (j == i + 1) {
                    if (s[i] == s[j]) {
                        dp[i][j] = 0;
                    } else {
                        dp[i][j] = 1;
                    }
                } else {
                    if (s[i] == s[j]) {
                        dp[i][j] = min(dp[i + 1][j - 1], min(dp[i + 1][j] + 1, dp[i][j - 1] + 1));
                    } else {
                        dp[i][j] = min(dp[i + 1][j - 1] + 2, min(dp[i + 1][j] + 1, dp[i][j - 1] + 1));
                    }
                }
            }
        }
        return dp[0][n - 1];
    }
};
