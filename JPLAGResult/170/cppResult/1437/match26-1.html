class Solution {
public:
    int minInsertions(string str) {
        int n = str.length();
//         if (str.length() <= 1) {
//             return 0;
//         }
        
//         if (str[0] == str.back()) {
//             return minInsertions(str.substr(1, n - 2));
//         }
        
//         int res1 = minInsertions(str.substr(0, n - 1)) + 1;
//         int res2 = minInsertions(str.substr(1, n - 1)) + 1;
//         return min(res1, res2);
        
        vector<vector<int>> dp(n, vector<int>(n, 0));  // len = 1
        for (int i = 0; i < n - 1; i++) {
            if (str[i] != str[i + 1]) {
                dp[i][i + 1] = 1;

            }
        }
        
        for (int len = 3; len <= n; len++) {
            for (int l = 0; l + len - 1 < n; l++) {

                int r = l + len - 1;
                if (str[l] == str[r]) {
                    dp[l][r] = dp[l + 1][r - 1];
                } else {
                    dp[l][r] = min(dp[l + 1][r], dp[l][r - 1]) + 1;
                }
            }
        }
        return dp[0][n - 1];
    }
};
