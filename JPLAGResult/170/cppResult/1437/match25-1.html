class Solution {
public:
    int minInsertions(string s) {
        int n = s.size();
        vector<vector<int>> dp(n, vector<int>(n, 0));
        for (int i = 0; i + 1 < n; ++i) {
            dp[i][i + 1] = s[i] != s[i + 1];

        }
        for (int sz = 3; sz <= n; ++sz) {
            for (int i = 0; i + sz <= n; ++i) {
                int j = i + sz - 1;
                if (s[i] == s[j]) {
                    dp[i][j] = dp[i + 1][j - 1];
                } else {
                    dp[i][j] = 1 + min(dp[i + 1][j], dp[i][j - 1]);
                }
            }
        }
        
        return dp[0][n - 1];
    }
};
