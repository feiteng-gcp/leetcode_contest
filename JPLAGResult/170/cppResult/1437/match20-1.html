class Solution {
public:
    int minInsertions(string s) {
        int n = s.size();
        vector<vector<int>> dp(n, vector<int>(n + 1));
        for (int len = 2; len <= n; ++len) {

            for (int i = 0; i + len <= n; ++i) {
                if (s[i] == s[i + len - 1]) {
                    dp[i][len] = dp[i + 1][len - 2];
                } else {
                    dp[i][len] = min(dp[i + 1][len - 1], dp[i][len - 1]) + 1;
                }
            }
        }
        return dp[0][n];
    }
};
