class Solution {
public:
    int minInsertions(string str) {
        int n=str.length();
        vector<vector<int>> table(n, vector<int>(n));
        int l, h, gap;  
        
        // Fill the table  
        for ( gap = 1; gap < n; ++gap)  
            for (l = 0, h = gap; h < n; ++l, ++h)  
                table[l][h] = (str[l] == str[h])?   table[l + 1][h - 1] : (min(table[l][h - 1], table[l + 1][h]) + 1);  

        // Return minimum number of insertions 
        // for str[0..n-1]  
        return table[0][n - 1];  
    }
};
