class Solution {
public:
    int c[510][510];
    int LCS(string a, string b) {
        int i, j, len = a.length();
        for (i = 1; i <= len; i++) {
            for (j = 1; j <= len; j++) {
                if (a[j - 1] == b[i - 1])
                    c[i][j] = c[i - 1][j - 1] + 1;
                else
                    c[i][j] = max(c[i - 1][j], c[i][j - 1]);
            }
        }
        return c[len][len];
    }
    int minInsertions(string s) {
        memset(c,0,sizeof(c));
        string s2 = s;
        reverse(s.begin(), s.end());
        return s.length() - LCS(s, s2);
    }
};
