class Solution {
public:
    int minInsertions(string s) {
        int n = s.size();
        vector<vector<int>> dp(n +1, vector<int>(n + 1, 1e9));
        
        for(int i = 1; i <= n;i ++){
            dp[i][i] = 0;
            dp[i][i-1] = 0;
        }
        
        for(int len = 2; len <= n; len ++){
            for(int i = 1; i<= n; i ++){
                int j = i + len - 1;
                if(j > n) break;
                if (s[i-1] == s[j-1]){
                    dp[i][j] = dp[i+1][j-1];
                } else{
                    dp[i][j] = min(dp[i+1][j], dp[i][j-1]) + 1;
                }
            }
        }
        
        return dp[1][n];
    }
};
