class Solution {
public:
    int minInsertions(string s) {  
    int n = s.size();
    vector<vector<int>> dp(n, vector<int>(n, 0));
    for (int gap = 1; gap < n; ++gap)  
        for (int l = 0, h = gap; h < n; ++l, ++h)  
            dp[l][h] = (s[l] == s[h]) ? dp[l + 1][h - 1] :  (min(dp[l][h - 1], dp[l + 1][h]) + 1);   
    return dp[0][n - 1];         
    }
};
