class Solution {
public:
    int minInsertions(string s) {
        int N = s.length();
        vector<vector<int>> dp(N, vector<int>(N));
        for (int i = N-1; i>= 0; i--) {
            for (int j = i; j < N; j++) {
                if (i == j)
                    dp[i][j] = 0;

                else {
                    if (s[i] == s[j]) {
                        dp[i][j] = dp[i+1][j-1];
                    } else {
                        dp[i][j] = 1+min(dp[i+1][j], dp[i][j-1]);
                    }
                }
            }
        }
        return dp[0][N-1];
    }
};
