class Solution {
public:
    int minInsertions(string s) {
        int n = s.size();
        vector<vector<int>> dp(n, vector<int>(n));
        for (int j = 0; j < n; j++) {
            for (int i = j-1; i >= 0; i--) {
                if (s[i] == s[j]) dp[i][j] = (j == i+1 ? 0 : dp[i+1][j-1]);
                else dp[i][j] = min(dp[i+1][j], dp[i][j-1]) + 1;
            }
        }
        return dp[0][n-1];
    }
};
