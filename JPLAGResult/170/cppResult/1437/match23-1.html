class Solution {
public:
    int minInsertions(string str) {
        int N=str.size();
        vector<vector<int>> dp(N+1,vector<int>(N));
        for(int i=0;i<N;i++){
            dp[0][i]=0;
            dp[1][i]=0;
        }
        for(int len=2;len<=N;len++){
            for(int s=0;s+len<=N;s++){

                if(str[s]==str[s+len-1]){
                    dp[len][s]=dp[len-2][s+1];
                }else{
                    dp[len][s]=min(dp[len-1][s],dp[len-1][s+1])+1;
                }
            }
        }
        return dp[N][0];
    }
};
