class Solution {
    map<pair<int,int>,int>Map;
public:
    int minInsertions(string s) 
    {
        return helper(s,0,s.size()-1);
    }
    
    int helper(string &str, int l, int h) 
    { 
        if (l > h) return INT_MAX; 
        if (l == h) return 0; 
        if (l == h - 1) return (str[l] == str[h])? 0 : 1; 
        if (Map.find({l,h})!=Map.end())
            return Map[{l,h}];
  
        int ans =  (str[l] == str[h])?  
                    helper(str, l + 1, h - 1): 
                    (min(helper(str, l, h - 1), 
                    helper(str, l + 1, h)) + 1); 
        Map[{l,h}] = ans;
        return ans;
    } 
};
