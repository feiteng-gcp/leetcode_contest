class Solution {
public:
    int minInsertions(string s) {
        int n = s.length();
        vector<vector<int>> dp(n, vector<int>(n, INT_MAX));
        for(int i = 0; i < n; i++) {
            dp[i][i] = 0;
            if(i < n - 1) {
                if(s[i] == s[i + 1]) {
                    dp[i][i + 1] = 0;
                }
                else {
                    dp[i][i + 1] = 1;
                }
            }
        }
        for(int len = 3; len <= n; len++) {
            for(int i = 0; i <= n - len; i++) {
                dp[i][i + len - 1] = min(dp[i][i + len - 1], dp[i][i + len - 2] + 1);
                dp[i][i + len - 1] = min(dp[i][i + len - 1], dp[i + 1][i + len - 1] + 1);
                if(s[i] == s[i + len - 1]) {
                    dp[i][i + len - 1] = min(dp[i][i + len - 1], dp[i + 1][i + len - 2]);
                }
                else {
                    dp[i][i + len - 1] = min(dp[i][i + len - 1], dp[i + 1][i + len - 2] + 2);
                }
            }
        }
        return dp[0][n - 1];
    }
};
