class Solution {
public:
    int minInsertions(string s) {
        int len = s.length();
        vector<vector<int>> dp(len + 1, vector<int>(len, 0));
        
        for(int l = 2; l <= len; l++) {
            for(int start = 0; start <= (len - l); start++) {
                if(s[start] == s[start + l - 1]) {
                    dp[l][start] = dp[l - 2][start + 1];
                } else {
                    dp[l][start] = min(dp[l - 1][start + 1], dp[l - 1][start]) + 1;
                }
            }
        }
        
        return dp[len][0];
    }
