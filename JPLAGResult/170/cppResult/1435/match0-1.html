class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) {
        vector<int> sum(1,0);
        for(int i:arr)
            sum.push_back(sum.back()^i);
        vector<int> ans;
        for(auto p:queries)
        {
            ans.push_back(sum[p[0]]^sum[p[1]+1]);
        }
        return ans;
    }
};
