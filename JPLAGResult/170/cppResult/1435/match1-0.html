class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) 
    {
        vector<int>q({0});
        for (int i=0; i<arr.size(); i++)
            q.push_back(q.back()^arr[i]);
        
        vector<int>rets;
        for (auto query:queries)
        {
            int a = query[0];
            int b = query[1];
            rets.push_back(q[a]^q[b+1]);            
        }
        return rets;
    }
};
