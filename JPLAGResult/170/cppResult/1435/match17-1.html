class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) {

        vector<int> sum = {0, arr[0]};
        for(int i = 1; i < arr.size(); i ++)
            sum.push_back(sum.back() ^ arr[i]);

        vector<int> res;
        for(const vector<int>& q: queries)
            res.push_back(sum[q[1] + 1] ^ sum[q[0]]);
        return res;
    }
};
