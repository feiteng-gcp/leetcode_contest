class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) {
        vector<int> dp(arr.size()+1, 0);
        for (int i=0; i<arr.size(); i++) {
            dp[i+1]=(dp[i]^arr[i]);
        }
        vector<int> ret;
        for (auto q:queries) {
            ret.push_back(dp[q[0]] ^ dp[1+q[1]]);
        }
        return ret;
    }
};
