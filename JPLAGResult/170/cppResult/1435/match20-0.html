class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) {
        int N = arr.size();
        vector<int>dp(N+1, 0);
        for(int i = 1; i <= N; ++i){
            dp[i] = (dp[i-1] ^ arr[i-1]);
        }
        vector<int>ans;
        for(auto &q:queries){
            ans.emplace_back(dp[q[0]] ^ dp[q[1] + 1]);
        }
        return ans;
    }
};
