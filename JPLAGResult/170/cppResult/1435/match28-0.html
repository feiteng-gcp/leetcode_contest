class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& que) {
        int L = arr.size();
        vector<int> dp(L+1,0);
        
        for(int i = 0; i < L; ++i){
            dp[i+1] = dp[i]^arr[i];
        }
        
        vector<int> rnt;
        
        for(auto q:que){
            int tmp = dp[q[1]+1]^dp[q[0]];
            rnt.emplace_back(tmp);
        }
        
        return rnt;
        
    }
};
