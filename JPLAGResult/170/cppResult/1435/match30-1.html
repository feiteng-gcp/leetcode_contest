class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) {
        vector<int> x = arr;
        for(int i = 1; i < x.size(); i++) {
            x[i] = x[i-1] ^ x[i];
        }
        vector<int> ret;
        for(auto& q: queries) {

            if(q[0] == q[1]) {
                ret.push_back(arr[q[0]]);
            } else if (q[0] == 0) {
                ret.push_back(x[q[1]]);
            }
            else {
                ret.push_back(x[q[1]] ^ x[q[0] - 1]);
            }
        }
        return ret;
    }
};
