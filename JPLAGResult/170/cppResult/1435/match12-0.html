class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) {
        vector<int> data(arr.size() + 1, 0);
        for (int i = 0;i < arr.size();i++) {
            data[i + 1] = arr[i] ^ data[i];
        }
        vector<int> ret;
        for (auto& query: queries) {
            ret.push_back(data[query[1] + 1] ^ data[query[0]]);
        }
        return ret;
    }
};
