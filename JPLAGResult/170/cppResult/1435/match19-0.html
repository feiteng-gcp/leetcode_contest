class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) {
        int n = arr.size();
        vector<int> sum(n + 1);
        for (int i = 0; i < n; ++i)
            sum[i+1] = sum[i] ^ arr[i];
        vector<int> ans;
        for (auto q : queries) {
            int l = q[0], r = q[1];
            ans.push_back(sum[l] ^ sum[r+1]);
        }        
        return ans;
    }
};
