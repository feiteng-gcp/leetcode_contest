class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) {
        vector<int> raw = arr;
        for(int i = 1; i < arr.size(); ++i)
        {
            arr[i] = (arr[i-1] ^ arr[i]);
        }
        
        vector<int> res;
        for(auto q : queries)
        {
            int start = q[0], end = q[1];
            if(start == end)
            {
                res.push_back(raw[start]);
            }
            else if(start == 0)
            {
                res.push_back(arr[end]);
            }
            else
            {
                res.push_back(arr[end] ^ arr[start - 1]);
            }
        }
        return res;
    }
};
