class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries)
    {
        vector<int> xors;
        xors.push_back(0);
        for (int num : arr)
            xors.push_back(xors.back() ^ num);
        vector<int> result;
        for (auto q : queries)
            result.push_back(xors[q[0]] ^ xors[q[1] + 1]);
        return result;
    }
};
