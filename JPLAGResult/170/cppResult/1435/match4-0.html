class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) {
        int n = arr.size();
        vector<int> dp(n + 1, 0);
        for (int i = 1; i <= n; ++i) {
            dp[i] = dp[i - 1] ^ arr[i - 1];
        }
        vector<int> res;
        for (const auto& query : queries) {
            res.push_back(dp[query[1] + 1] ^ dp[query[0]]);
        }
        return res;
    }
};
