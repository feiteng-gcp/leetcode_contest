class Solution {
public:
    vector<int> xorQueries(vector<int>& arr, vector<vector<int>>& queries) {
        int n = arr.size();
        vector<int> pre(n+1);
        for (int i = 1; i <= n; i++) {
            pre[i] = pre[i-1] ^ arr[i-1];
        }
        
        vector<int> ret;
        for (auto &q : queries) {
            ret.push_back(pre[q[1]+1] ^ pre[q[0]]);
        }
        return ret;
    }
};
