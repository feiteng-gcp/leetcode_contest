class Solution {
    public String freqAlphabets(String s) {
        StringBuilder sb = new StringBuilder();
        int i = s.length()-1;
        while(i >= 0) {
            if (s.charAt(i) == '#') {
                int temp = Integer.parseInt(s.substring(i-2, i));
                sb.append((char)(temp + 'a'-1));
                i -= 3;
            } else {
                int temp = s.charAt(i) - '0';
                sb.append((char)(temp + 'a'-1));
                i--;
            }
        }
        return sb.reverse().toString();
    }
}
