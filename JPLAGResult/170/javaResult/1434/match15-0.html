class Solution {
    public String freqAlphabets(String s) {
        int len = s.length() - 1;
        StringBuilder res = new StringBuilder();
        while (len >= 0) {
            if (s.charAt(len) == '#') {
                int index = Integer.valueOf(s.substring(len - 2, len));
                res.insert(0, (char)('a' + index - 1));
                len -= 3;
            } else {
                int index = Integer.valueOf("" + s.charAt(len));
                res.insert(0, (char)('a' + index - 1));
                len --;
            }
        }
        return res.toString();
    }
}
