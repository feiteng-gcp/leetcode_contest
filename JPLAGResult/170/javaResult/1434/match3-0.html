class Solution {
    public String freqAlphabets(String s) {
        StringBuilder sb = new StringBuilder();
        int i = 0;
        while (i < s.length()) {
            if (i + 2 < s.length() && s.charAt(i + 2) == '#') {
                char c = (char)((int)'a' + Integer.parseInt(s.substring(i, i + 2)) - 1);
                sb.append(c);
                i += 3;
            } else {
                char c = (char)((int)'a' + Integer.parseInt(s.substring(i, i + 1)) - 1);
                sb.append(c);
                i++;
            }
        }
        return sb.toString();
    }
}
