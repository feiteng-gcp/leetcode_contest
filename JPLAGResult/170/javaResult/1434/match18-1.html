class Solution {
    public String freqAlphabets(String s) {
        StringBuilder res=new StringBuilder();
        for(int i=0;i<s.length();i++){
            if(i+2<s.length()&&s.charAt(i+2)=='#'){
                Integer n=Integer.parseInt(s.substring(i,i+2));
                char c=(char)('j'+n-10);
                res.append(c);
                i+=2;
            }
            else{
                Integer n=Integer.parseInt(s.charAt(i)+"");
                
char c=(char)('a'+n-1);
                res.append(c);
            }
        }
        return res.toString();
    }
}
