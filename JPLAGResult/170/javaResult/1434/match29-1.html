class Solution {

	public String freqAlphabets(String s) {
		String ret = "";
		for (int i = 0; i < s.length();) {
			if (i + 2 < s.length() && s.charAt(i + 2) == '#') {
				String tmp = s.substring(i, i + 2);
				int j = Integer.parseInt(tmp);

				ret += (char) (j - 10 + 'j');
				i += 3;

			} else {
				ret += (char) (s.charAt(i) - '1' + 'a');
				i++;
			}
		}
		return ret;
	}
}
