class Solution {
    public String freqAlphabets(String s) {
        int n = s.length();
        String re = "";
        char[] chs = s.toCharArray();
        for(int i = 0; i < n; i++) {
            if(chs[i] == '#') continue;
            if(i + 2 < n && chs[i + 2] == '#') {
                int val = (chs[i] - '0') * 10 + chs[i + 1] - '0';
                re += (char)('a' + val - 1);
                i++;
            }
            else re += (char)('a' + chs[i] - '0' - 1);
        }
        return re;
    }
}
