class Solution {
    public String freqAlphabets(String s) {
        StringBuilder sb = new StringBuilder();
        int n = s.length();
        int index = 0;
        while(index < n){
            if(index + 2 < n && s.charAt(index + 2) == '#'){
                int sum = (s.charAt(index) - '0') * 10 + s.charAt(index + 1) - '0';
                // System.out.println(sum);
                sb.append((char)(sum - 10 + 'j'));
                index += 3;
            } else {
                sb.append((char)(s.charAt(index) + 'a' - '1'));
                index++;
            }
        }
        return sb.toString();
    }
}
