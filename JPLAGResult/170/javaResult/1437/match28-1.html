class Solution {
    public int minInsertions(String s) {
        int len = s.length();
        int[][] dp = new int[len][len];
        for (int i = 0; i < len; i ++) {
            dp[i][i] = 0;
        
}
        for (int l = 2; l <= len; l ++) {
            for (int i = 0; i < len - l + 1; i ++) {
                int j = i + l - 1;
                if (s.charAt(i) == s.charAt(j)) {
                    if (i + 1 == j) dp[i][j] = 0;
                    else dp[i][j] = dp[i + 1][j - 1];
                
} else {
                    dp[i][j] = Math.min(dp[i + 1][j], dp[i][j - 1]) + 1;
                }
            }
        }
        return dp[0][len - 1];
    }
}
