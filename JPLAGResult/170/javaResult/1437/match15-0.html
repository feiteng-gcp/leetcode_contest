class Solution {
    public int minInsertions(String s) {
        int l = s.length();
        
        int[][] dp = new int[l][l];
        for (int[] i : dp) {
            Arrays.fill(i, Integer.MAX_VALUE);
        }
        
        for (int len = 1; len <= l; len++) {
            for (int i = 0; i + len <= l; i++) {
                int j = i + len - 1;
                if (len <= 2) {
                    dp[i][j] = s.charAt(i) == s.charAt(j) ? 0 : 1;
                    continue;
                }
                
                if (s.charAt(i) == s.charAt(j)) {
                    dp[i][j] = dp[i + 1][j - 1];
                }
                else {
                    dp[i][j] = Math.min(dp[i + 1][j], dp[i][j - 1]) + 1;
                }
            }
        }
        return dp[0][l - 1];
    }
}
