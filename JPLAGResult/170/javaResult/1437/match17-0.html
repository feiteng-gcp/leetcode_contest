class Solution {
    public int minInsertions(String s) {
        // int[] f = new int[26];
        // for(char c : s.toCharArray()) {
        //     f[c - 'a']++;
        // }
        // int odd = 0;
        // for(int k : f) {
        //     if(k % 2 != 0) odd++;
        // }
        // if(odd == 1) return 0;
        // return odd - 1;
        int n = s.length(), re = 1 << 30;
        for(int i = 0; i < n; i++) {
            // expand at i
            String s1 = reverse(s.substring(0, i)),
                    s11 = reverse(s.substring(0, i + 1)),
                    s2 = s.substring(i + 1);
            int l1 = s1.length(), l2 = s2.length(), l3 = s11.length(),
                lis = LIS(s1, s2),
                lis2 = LIS(s11, s2);
            re = Math.min(re, l1 + l2 - 2 * lis);
            re = Math.min(re, l3 + l2 - 2 * lis2);
        }
        return re;
    }
    
    int LIS(String s1, String s2) {
        int m = s1.length(), n = s2.length();
        int[][] f = new int[m + 1][n + 1];
        for(int i = 0; i < m; i++) {
            for(int j = 0; j < n; j++) {
                if(s1.charAt(i) == s2.charAt(j)) {
                    f[i + 1][j + 1] = 1 + f[i][j];
                }
                else f[i + 1][j + 1] = Math.max(f[i][j + 1], f[i + 1][j]);
            }
        }
        return f[m][n];
    }
    
    String reverse(String str) {
        return new StringBuilder(str).reverse().toString();
    }    
}
