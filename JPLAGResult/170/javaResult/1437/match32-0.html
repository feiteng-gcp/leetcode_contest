class Solution {
    public int minInsertions(String s) {
        int n = s.length();
        char[] arr = s.toCharArray();
        int[][] dp = new int[n][n];
        for (int i = 0; i + 1 < n; i++) {
            if (arr[i] == arr[i + 1]) {
                dp[i][i + 1] = 0;
            } else {
                dp[i][i + 1] = 1;
            }
        }
        for (int gap = 2; gap < n; gap++) {
            for (int i = 0; i + gap < n; i++) {
                if (arr[i] == arr[i + gap]) {
                    dp[i][i + gap] = dp[i + 1][i + gap - 1];
                } else {
                    dp[i][i + gap] = Math.min(dp[i + 1][i + gap] + 1, Math.min(dp[i + 1][i + gap - 1] + 2, dp[i][i + gap - 1] + 1));
                }
            }
        }
        return dp[0][n - 1];
    }
}
