class Solution {
    public int minInsertions(String s) {
           int n = s.length();
           int[][] dp = new int[n][n];
            
           for (int len = 1; len <= n; len++) {
                   for (int i = 0; i < n; i++) {
                           int j = i + len - 1;
                           if (j >= n) {
                                   break;
                           }
                           if (len == 1) {
                                   dp[i][j] = 0;
                                   continue;
                           }
                           if (len == 2) {
                                   dp[i][j] = s.charAt(i) == s.charAt(j) ? 0 : 1;
                                   continue;
                           }
                           if (s.charAt(i) == s.charAt(j)) {
                                   dp[i][j] = dp[i+1][j-1];
                           }else {
                                   dp[i][j] = Math.min(dp[i+1][j], dp[i][j-1]) + 1;
                           }
                   } 
           }        
           return dp[0][n-1]; 
    }
}
