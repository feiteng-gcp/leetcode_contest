/*
1. def: int[][] dp where dp[i][j] is the min insertions to make s[i : j] palindrome
2. base case: dp[i][i] = 0, dp[i][i + 1] = s[i] == s[i + 1] ? 0 : 1
3. induction rule: dp[i][j] = s[i] == s[j] ? 
*/
class Solution {
    public int minInsertions(String s) {
        int len = s.length();
        int[][] dp = new int[len][len];
        for (int gap = 0; gap < len; gap++) {
            for (int i = 0; i + gap < len; i++) {
                int j = i + gap;
                if (gap == 0) {
                    dp[i][j] = 0;
                } else if (gap == 1) {
                    dp[i][j] = s.charAt(i) == s.charAt(j) ? 0 : 1;
                } else {
                    if (s.charAt(i) == s.charAt(j)) {
                        dp[i][j] = dp[i + 1][j - 1];
                    } else {
                        dp[i][j] = Math.min(dp[i + 1][j], dp[i][j - 1]) + 1;
                    }
                }
            }
        }
        return dp[0][len - 1];
    }
}
