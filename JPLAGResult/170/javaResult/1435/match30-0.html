class Solution {
    public int[] xorQueries(int[] arr, int[][] q) {
        int length = arr.length;
        int[] prexor = new int[length+1];
        for (int i = 0; i< length; i++) {
            prexor[i+1] = prexor[i] ^  arr[i];
        }
        int[] result = new int[q.length];
        int i = 0;
        for (int[] query: q) {
            result[i++] = prexor[query[1] + 1] ^ prexor[query[0]];
        }
        return result;
        
    }
}
