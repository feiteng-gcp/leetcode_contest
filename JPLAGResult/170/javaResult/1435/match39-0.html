class Solution {
    public int[] xorQueries(int[] arr, int[][] q) {
        int n = arr.length;
        int[] dp = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            dp[i] = dp[i - 1] ^ arr[i - 1];
        }
        int m = q.length;
        int[] res = new int[m];
        for (int i = 0; i < m; i++) {
            int l = q[i][0], r = q[i][1];
            res[i] = dp[r + 1] ^ dp[l];
        }
        return res;
    }
}
