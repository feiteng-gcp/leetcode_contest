class Solution {
    public int[] xorQueries(int[] arr, int[][] queries) {
        int len = queries.length;
        int[] res = new int[len];
        int[] preXOR = new int[arr.length];
        preXOR[0] = arr[0];
        for (int i = 1; i < arr.length; i++) {
            preXOR[i] = preXOR[i - 1] ^ arr[i];
        }
        for (int i = 0; i < len; i++) {
            int a = queries[i][0];
            int b = queries[i][1];
            res[i] = preXOR[b] ^ preXOR[a] ^ arr[a];
        }
        return res;
    }
}
