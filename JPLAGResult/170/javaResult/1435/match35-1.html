class Solution {
    public int[] xorQueries(int[] arr, int[][] queries) {
        final int N = arr.length;
        int[] sum = new int[N + 1];
        for (int i = 0; i < N; ++i) {
            sum[i + 1] = sum[i] ^ arr[i];
        }
        
        final int M = queries.length;
        int[] ans = 
new int[M];
        for (int i = 0; i < queries.length; ++i) {
            int L = queries[i][0];
            int R = queries[i][1] + 1;
            ans[i] = sum[R] ^ sum[L];
        }
        return ans;
    }
}
