class Solution {
    public int[] xorQueries(int[] arr, int[][] queries) {
        int n = arr.length;
        int[] sum = new int[n + 1];
        for (int i = 0; i < n; i++) {
            sum[i + 1] = sum[i] ^ arr[i];
        }
        
        int length = queries.length;
        int[] answer = new int[length];
        for (int i = 0; i < length; i++) {
            int[] query = queries[i];
            answer[i] = sum[query[1] + 1] ^ sum[query[0]];
        }
        return answer;
    }
}
