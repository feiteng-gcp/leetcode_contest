import java.util.Map.*;

class Solution {
    public int[] xorQueries(int[] arr, int[][] queries) {
        int n = arr.length;
        int[] preCal = new int[n + 1];
        for(int i = 1; i <= n; i++){
            preCal[i] = preCal[i-1] ^ arr[i - 1];
        }
        int m = queries.length;
        int[] res = new int[m];
        for(int i = 0 ; i < m; i++){
            int num1 = queries[i][0];
            int num2 = queries[i][1];
            res[i] = preCal[num2 + 1] ^ preCal[num1];
        }
        return res;
    }
}
