class Solution {
    public int[] xorQueries(int[] arr, int[][] queries) {
        int n = arr.length;
        int[] xor = new int[n+1];
        
        for (int i = 1; i <= n; i++) {
            xor[i] = xor[i-1] ^ arr[i-1];
        }
        
        int k = queries.length;
        int[] ans = new int[k];
        for (int i = 0; i < k; i++) {
            int s = queries[i][0], e = queries[i][1];
            ans[i] = xor[e+1] ^ xor[s];
        }
        return ans;
    }
}
