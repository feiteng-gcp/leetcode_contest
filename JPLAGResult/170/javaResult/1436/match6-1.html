class Solution {
    public List<String> watchedVideosByFriends(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
        Queue<Integer> queue = new LinkedList<>();
        HashSet<Integer> set = new HashSet<>();    
        queue.add(id);
        set.add(id);
        int curLevel = 0;
        
        while(curLevel < level) {
            int size = queue.size();
            while(size-- > 0) {
                int cur = queue.poll();
                
for(int f : friends[cur]) {
                    if(!set.contains(f)) {
                        set.add(f);
                        queue.add(f);
                    }
                }
            }
            curLevel++;
        
}
        
        TreeMap<String, Integer> map = new TreeMap<>();
        while(!queue.isEmpty()) {
            int cur = queue.poll();
            for(String video : watchedVideos.get(cur))
                map.put(video, map.getOrDefault(video, 0) + 1)
;
        }
        
        TreeMap<Integer, ArrayList<String>> treeMap = new TreeMap<>();
        for(String s : map.keySet()) {
            int fre = map.get(s);
            if(!treeMap.containsKey(fre))
                treeMap.put(fre, new ArrayList<>());
            treeMap.get(fre).add(s);
        }
        
        List<String> res = new ArrayList<>();
        for(int fre : treeMap.keySet()) {
            res.addAll(treeMap.get(fre));
        }
        return res;
    }
}
