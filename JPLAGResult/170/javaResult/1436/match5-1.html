class Solution {
    public List<String> watchedVideosByFriends(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
      Queue<Integer> q = new LinkedList<>();
      q.add(id);
      boolean[] vis = new boolean[friends.length];
      vis[id] = true;
      
while(level != 0) {
        int sz = q.size();
        for(int i=0; i<sz; i++) {
          int v = q.poll();
          int[] f = friends[v];
          for(int z : f) {
            if(!vis[z]) {
              vis[z] = true;
              
q.add(z);
            }
          }
        }
          level--;
      }
      Map<String, Integer> m = new HashMap<>();
      while (!q.isEmpty()) {
        int v = q.poll();
        for(String vid : watchedVideos.get(v)) 
{
          m.put(vid, m.getOrDefault(vid, 0)+1);
        }
      }
      List<String> res = new ArrayList<>(m.keySet());
      res.sort((a, b) -> {
        if(m.get(a).equals(m.get(b))) return a.compareTo(b);
        return Integer.compare(m.get(a), m.get(b));
      });
      return res;
    }
  }
