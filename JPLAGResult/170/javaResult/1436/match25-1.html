class Solution {
    public List<String> watchedVideosByFriends(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
        List<String> ans = new ArrayList<>();
        Queue<Integer> queue = new LinkedList<>();
        int N = friends.length;
        boolean[] visited = new boolean[N];
        queue.offer(id);
        visited[id] = true;
        int curl = 0;
        
Map<String, Integer> freq = new HashMap<>();
        while(!queue.isEmpty()) {
            int qsize = queue.size();
            for (int i = 0; i < qsize; i++) {
                int num = queue.poll();
                if (curl == level) {
                    
List<String> wv = watchedVideos.get(num);
                    for (String v : wv) {
                        freq.put(v, freq.getOrDefault(v, 0) + 1);
                    }
                } else {
                    
for (int j : friends[num]) {
                        if (!visited[j]) {
                            queue.offer(j);
                            visited[j] = true;
                        }
                    }
                }
            }
            curl++;
        }
        // System.out.println(freq);
        
        TreeMap<Integer, TreeSet<String>> tmp = new TreeMap<>();
        for (Map.Entry<String, Integer> entry : freq.entrySet()) {
            String v = entry.getKey();
            TreeSet<String> ts = tmp.getOrDefault(entry.getValue(), new TreeSet<>());
            ts.add(v);
            tmp.put(entry.getValue(), ts);
        }
        
        for (Map.Entry<Integer, TreeSet<String>> entry : tmp.entrySet()) {
            for (String v : entry.getValue()) {
                ans.add(0, v);
            }
        }
        // System.out.println(tmp);
        List<String> rans = new ArrayList<>();
        for (String s : ans) {
            rans.add(0, s);
        }
        return rans;
    }
}
