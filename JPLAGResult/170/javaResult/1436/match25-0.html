class Solution {
    public List<String> watchedVideosByFriends(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
        List<String> r = new ArrayList<String>();
        Queue<Integer> queue= new LinkedList<Integer>();
        queue.offer(id);
        boolean[] visited = new boolean[friends.length];
        visited[id] = true;
        Set<Integer> lf = new HashSet<Integer>();
        while(!queue.isEmpty()) {
            int size  = queue.size();
            for (int i = 0; i< size; i++) {
                int n = queue.poll();
                if (level == 0) {
                    lf.add(n);
                } else {
                    int[] f = friends[n];
                    for (int friend: f) {
                      if (!visited[friend]) {
                        queue.offer(friend);
                        visited[friend] = true;
                      }
                    }
                }
            }
            level--;
        }
        
        Map<String, Integer> videoCount = new HashMap<String, Integer>();
        for (int s: lf) {
            List<String> video = watchedVideos.get(s);
            for (String v: video) {
                videoCount.put(v, videoCount.getOrDefault(v, 0) +1);
            }
        }
        List<Map.Entry<String, Integer>> lv = new ArrayList<Map.Entry<String, Integer>>(videoCount.entrySet());
        Collections.sort(lv, (a,b) ->  {
          
            if (a.getValue() == b.getValue()) {
                return a.getKey().compareTo(b.getKey());
            }
            return a.getValue() - b.getValue();
            
        });
        
        return lv.stream().map(entry -> entry.getKey()).collect(Collectors.toList());
        
        
        
    }
    
    
    
}
