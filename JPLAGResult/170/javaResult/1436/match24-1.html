class Solution {
    public List<String> watchedVideosByFriends(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
        List<String> res=new ArrayList<>();
        Queue<Integer> queue=new LinkedList<>();
        int l=0;
        queue.add(id);
        HashSet<Integer> set=new HashSet<>();
        set.add(id);
        HashMap<String,Integer> freq=new HashMap<>();
        while(l<level&&!queue.isEmpty()){
            
int size=queue.size();
            for(int i=0;i<size;i++){
                int[] friend=friends[queue.poll()];
                for(int f:friend){
                    if(set.add(f)){
                        queue.add(f);
                    }
                }
            }
            l++;
        }
        
while(!queue.isEmpty())
        {
            int n=queue.poll();
            List<String> videos=watchedVideos.get(n);
            for(String v:videos){
                if(!freq.containsKey(v)){
                    res.add(v);
                    freq.put(v,1);
                }
                else freq.put(v,freq.get(v)+1);
            }
        }
        
Collections.sort(res,(a,b)->(freq.get(a)==freq.get(b)?a.compareTo(b):freq.get(a)-freq.get(b)));
        return res;
        
    }
}
