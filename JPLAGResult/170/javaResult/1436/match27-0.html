class Solution {
    public List<String> watchedVideosByFriends(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
        int n = friends.length;
        
        boolean[] visited = new boolean[n];
        Queue<Integer> queue = new LinkedList<>();
        
        queue.offer(id);
        visited[id] = true;
        
        while (!queue.isEmpty() && level-- > 0) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int curr = queue.poll();
                for (int next : friends[curr]) {
                    if (visited[next]) continue;
                    queue.offer(next);
                    visited[next] = true;
                }
            }
        }
        
        List<String> res = new ArrayList<>();
        if (queue.isEmpty()) return res;
        
        Map<String, Integer> map = new HashMap<>();
        while (!queue.isEmpty()) {
            int curr = queue.poll();
            for (String m : watchedVideos.get(curr)) {
                map.put(m, map.getOrDefault(m, 0) + 1);
            }
        }
        
        List<Map.Entry<String, Integer>> temp = new ArrayList<>(map.entrySet());
        Collections.sort(temp, new Comparator<Map.Entry<String, Integer>>() {
            @Override
            public int compare(Map.Entry<String, Integer> a, Map.Entry<String, Integer> b) {
                if (a.getValue() == b.getValue()) {
                    return a.getKey().compareTo(b.getKey());
                }
                
                return a.getValue() - b.getValue();
            }
        });
        
        for (Map.Entry<String, Integer> e : temp) {
            res.add(e.getKey());
        }
        
        return res;
    }
}
