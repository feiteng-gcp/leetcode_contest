class Solution {
    public List<String> watchedVideosByFriends(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
        int n = friends.length;

        int cur = 0;
        Queue<Integer> queue = new LinkedList();
        queue.offer(id);
        boolean[] visited = new boolean[n];
        
visited[id] = true;
        while (!queue.isEmpty() && cur < level) {
            cur++;
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int per = queue.poll();
                for (int fr : friends[per]) {
                    if (visited[fr]) continue;
                    visited[fr] = true;
                    queue.offer(fr);
                }
            }
        }
        
        
Map<String, Integer> cache = new HashMap();
        while (!queue.isEmpty()) {
            int per = queue.poll();
            for (String v : watchedVideos.get(per)) {
                cache.put(v, cache.getOrDefault(v, 0)+1);
            }
        }
        
        List<String> ans = new ArrayList(cache.keySet());
        Collections.sort(ans, (a, b) -> {
            if (cache.get(a) != cache.get(b)) return cache.get(a) - cache.get(b);
            return a.compareTo(b);
        });
        
        return ans;
    }
}
