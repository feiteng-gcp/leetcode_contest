class Solution {
    
    public List<String> watchedVideosByFriends(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
        // 1. BFS find the videos in level
        Map<String, Integer> videoToFreq = BFS(watchedVideos, friends, id, level);
        
        // 2. sort all videos by frequency
        return sortVideoByFreq(videoToFreq);
    }
    
    private Map<String, Integer> BFS(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
        Queue<Integer> queue = new ArrayDeque<>();
        boolean[] visited = new boolean[watchedVideos.size()];
        Map<String, Integer> videoToFreq = new HashMap<>();
        
        // 1. init
        queue.offer(id);
        visited[id] = true;
        
        // 2. BFS
        int curLevel = 0;
        
while (!queue.isEmpty() && curLevel <= level) { // terminate condition!!
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                // expand
                int cur = queue.poll();
                if (curLevel == level) {
                    
for (String video : watchedVideos.get(cur)) {
                        videoToFreq.put(video, videoToFreq.getOrDefault(video, 0)+1);
                    }
                }
                // generate
                for (int nei : friends[cur]) {
                    if (visited[nei]) {
                        continue;
                    
}
                    queue.offer(nei);
                    visited[nei] = true;
                }
            }
            curLevel++;
        }
        
        return videoToFreq;
    }
    
    private List<String> sortVideoByFreq(Map<String, Integer> videoToFreq) {
        // 1. sort
        
List<Map.Entry<String, Integer>> entrys = new ArrayList<>(videoToFreq.entrySet());
        
        Collections.sort(entrys, new Comparator<Map.Entry<String, Integer>>() {
            @Override
            public int compare(Map.Entry<String, Integer> e1, Map.Entry<String, Integer> e2) {
                if (e1.getValue() != e2.getValue()) {
                    return Integer.compare(e1.getValue(), e2.getValue());
                } else {
                    return e1.getKey().compareTo(e2.getKey());
                }
            }
        });
        
        // 2. get result
        List<String> res = new ArrayList<>();
        
for (Map.Entry<String, Integer> entry : entrys) {
            res.add(entry.getKey());
        }
        
        return res;
    }
    
}
