class Solution {
    public List<String> watchedVideosByFriends(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
        // get the friends who are level distance away from id
        // then look up videos they watched
        Queue<Integer> q = new LinkedList<>();
        q.offer(id);
        Set<Integer> visited = new HashSet<>();
        visited.add(id);
        while(level-- > 0) {
            int size = q.size();
            while(size-- > 0) 
{
                int cur = q.poll();
                for(int friend : friends[cur]) {
                    if(!visited.contains(friend))
                    {
                        visited.add(friend);
                        q.offer(friend);
                    }
                }
            }
        }
        
Map<String, Integer> map = new HashMap<>();
        for(int friend : q) {
            for(String videos : watchedVideos.get(friend)) {
                map.put(videos, map.getOrDefault(videos, 0) + 1);
            }
        }
        
        Queue<item> qq = new PriorityQueue<>(); //(a, b)->(a.count == b.count ? a.name.compareTo(b.name) : a.count - b.count));
        
        
for(String v : map.keySet()) {
            qq.offer(new item(v, map.get(v)));
        }
        
        List<String> list = new ArrayList<>();
        
        while(!qq.isEmpty()) list.add(qq.poll().name);
        
        return list;
    }
    
    
class item implements Comparable<item> {
        String name;
        int count;
        public item(String n, int c) {
            name = n;
            count = c;
        }
        
        public int compareTo(item v) {
            if(this.count == v.count) return this.name.compareTo(v.name);
            return this.count - v.count;
        }
    }
}
