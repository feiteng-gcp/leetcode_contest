class Solution {
    public List<String> watchedVideosByFriends(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
        int n = friends.length;
        boolean[] seen = new boolean[n];
        Deque<Integer> queue = new ArrayDeque<>();
        Map<String, Integer> cntMap = new HashMap<>();
        seen[id] = true;
        queue.add(id);
        while(!queue.isEmpty()){
            int size = queue.size();
            while(size-- > 0){
                if(level == 0){
                    for(String vedio: watchedVideos.get(queue.poll())){
                        cntMap.put(vedio, cntMap.getOrDefault(vedio, 0) + 1);
                    }
                } else {
                    for(int next: friends[queue.poll()])
{
                        if(!seen[next]){
                            seen[next] = true;
                            queue.add(next);
                        }
                    }
                }
            }
            if(level > 0){
                level--;
            }
        }
        if(level != 0){
            return new ArrayList<>();
        }
        
List<String> res = new ArrayList<>(cntMap.keySet());
        Collections.sort(res, new Comparator<String>(){
           @Override
            public int compare(String a, String b){
                int cnt1 = cntMap.get(a);
                int cnt2 = cntMap.get(b);
                if(cnt1 != cnt2){
                    return cnt1 - cnt2;
                } else {
                    return a.compareTo(b);
                }
            }
        });
        return res;
    }
}
