class Solution {
    public List<String> watchedVideosByFriends(List<List<String>> w, int[][] friends, int id, int level) {
        Queue<Integer> q = new ArrayDeque<>();
        q.add(id);
        Set<Integer> v = new HashSet<>();
        v.add(id);
        for (int i = 0; i < level; i++) {
            int size = q.size();
            while (size-- > 0) {
                int cur = q.poll();
                for (int f : friends[cur]) {
                    if (!v.contains(f)) {
                        v.add(f);
                        q.add(f);
                    }
                }
            }
        }
        
        Map<String, Integer> map = new HashMap<>();
        for (int i : q) {
            for (String s : w.get(i)) {
                map.put(s, map.getOrDefault(s, 0) + 1);
            }
        }
        
        List<String> res = new ArrayList<>();
        for (String s : map.keySet()) res.add(s);
        
        Collections.sort(res, (a, b) -> (map.get(a) == map.get(b)) ? a.compareTo(b) : map.get(a) - map.get(b));
        return res;
    }
}
