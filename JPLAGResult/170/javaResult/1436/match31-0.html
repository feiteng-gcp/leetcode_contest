class Solution {
    class Node {
        String name;
        int freq;
        
        public Node(String name, int freq) {
            this.name = name;
            this.freq = freq;
        }
    }
    
    public List<String> watchedVideosByFriends(List<List<String>> watchedVideos, int[][] friends, int id, int level) {
        int N = watchedVideos.size();
        List<String> res = new ArrayList<>();
        
        boolean[] visited = new boolean[N];
        Queue<Integer> q = new LinkedList<>();
        q.offer(id);
        visited[id] = true;
        int l = 0;
        while (l < level && !q.isEmpty()) {
            for (int k = q.size(); k > 0; k--) {
                int cur = q.poll();
                
                for (int f : friends[cur]) {
                    if (visited[f]) {
                        continue;
                    }
                    visited[f] = true;
                    q.offer(f);
                }
            }
            l++;
        }
        
        // System.out.println(q + " " + l);
        
        if (l != level) {
            return res;
        }
        
        Map<String, Integer> counter = new HashMap<>();
        while (!q.isEmpty()) {
            int cur = q.poll();
            for (String video : watchedVideos.get(cur)) {
                counter.put(video, counter.getOrDefault(video, 0) + 1);
            }
        }
        
        // System.out.println(counter);

        PriorityQueue<Node> pq = new PriorityQueue<>((a, b) -> a.freq == b.freq ? a.name.compareTo(b.name) : Integer.compare(a.freq, b.freq));
        
        for (String key : counter.keySet()) {
            pq.offer(new Node(key, counter.get(key)));
        }
        
        while (!pq.isEmpty()) {
            res.add(pq.poll().name);
        }
        
        return res;
    }
}
