class Solution {
    
public String removeDuplicates(String s, int k) {
        Stack<Node> stack = new Stack<>();
        
        for (char c : s.toCharArray()) {
            if (stack.isEmpty()) {
                stack.push(new Node(c, 1));
            } else {
                Node prev = stack.peek();
                if (prev.val == c) {
                    if (prev.count + 1 == k) {
                        for (int __ = 1; __ < k; __++) {
                            stack.pop();
                        }
                    } else {
                        stack.push(new Node(c, prev.count + 1));
                    
}
                } else {
                    stack.push(new Node(c, 1));
                }
            }
        }
        
        StringBuilder builder = new StringBuilder();
        while (!stack.isEmpty()) {
            builder.append(stack.pop().val);
        }
        
        return builder.reverse().toString();
    }
    
    
private class Node {
        char val;
        int count;
        
        public Node(char val, int count) {
            this.val = val;
            this.count = count;
        }
    }
}
