class Solution {
    public String removeDuplicates(String s, int k) {
        Deque<Character> chars = new LinkedList();
        Deque<Integer> cnts = new LinkedList();
        
        for (char c : s.toCharArray()) {
            if (chars.size() == 0 || chars.getFirst() - c != 0) {
                chars.addFirst(c);
                cnts.addFirst(1);
            }
            else {
                int n = cnts.removeFirst() + 1;
                if (n > k) {
                    cnts.addFirst(n - k);
                }
                else if (n == k) {
                    chars.removeFirst();
                }
                else {
                    cnts.addFirst(n);
                }
            }
        }
        
        StringBuilder sb = new StringBuilder();
        while (!chars.isEmpty()) {
            char c = chars.removeLast();
            int n = cnts.removeLast();
            for (int i = 0; i < n; i++)
                sb.append(c);
        }
        return sb.toString();
    }
}
