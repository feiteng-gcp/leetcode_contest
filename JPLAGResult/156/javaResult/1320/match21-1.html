class Solution {
    public String removeDuplicates(String s, int k) {
        Stack<Node> stack = new Stack<>();
        for(char c : s.toCharArray())
        {
            if(stack.empty())
            {
                stack.push(new Node(c, 1));
            }
            else
            {
                if(stack.peek().c == c)
                {
                    Node cur = stack.pop();
                    cur.count++;
                    if(cur.count != k)
                        stack.push(cur);
                }
                else
                    stack.push(new Node(c, 1));
            
}
        }
        StringBuilder sb = new StringBuilder();
        while(!stack.empty())
        {
            Node cur = stack.pop();
            while(cur.count-- > 0)
                sb.append(cur.c);
        }
        return sb.reverse().toString();
    }
    
    
class Node
    {
        char c;
        int count;
        public Node(char c, int count)
        {
            this.c = c;
            this.count = count;
        }
    }
}
