class Element {
    int count;
    char letter;
    public Element(int c, char ch) {
        count = c;
        letter = ch;
    }
}
class Solution {
    
public String removeDuplicates(String s, int k) {
        Deque<Element> stack = new ArrayDeque<>();
        for (char c:s.toCharArray()) {
            if (stack.isEmpty() || stack.peekFirst().letter != c) {
                stack.offerFirst(new Element(1, c));
                continue;
            }
            Element cur = stack.peekFirst();
            if (cur.count + 1 == k) {
                stack.pollFirst();
            } else {
                cur.count++;
            
}
        }
        StringBuilder builder = new StringBuilder();
        while (!stack.isEmpty()) {
            Element cur = stack.pollFirst();
            for (int i = 0; i < cur.count; i++)
                builder.append(cur.letter)
;
        }
        return builder.reverse().toString();
    }
}
