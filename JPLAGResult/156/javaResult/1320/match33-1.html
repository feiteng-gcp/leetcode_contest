class Solution {
    public String removeDuplicates(String s, int k) {
        return sln1(s,k);
    }
    
    
private String sln1(String s, int k){
        Deque<Character> st1 = new LinkedList<>();
        Deque<Integer> st2 = new LinkedList<>();
        char[] cs = s.toCharArray();
        for(char c: cs){
            
if(st1.size() > 0 && st1.getLast() == c){
                st2.addLast(st2.removeLast()+1);
            }else{
                st1.addLast(c);
                st2.addLast(1);
            }
            while(st2.size() > 0 && st2.getLast() >=k){
                int count = st2.removeLast();
                if(count == k){ 
                    st1.removeLast();
                }else{
                    st2.addLast(count-k);
                }
            }
        
}
        StringBuilder sb = new StringBuilder();
        while(st1.size() >0){
            char c = st1.removeFirst();
            int l = st2.removeFirst();
            for(int i = 0;i < l; i++) sb.append(c);
        }
        return sb.toString();
    }
}
