class Solution {
    public String removeDuplicates(String s, int k) {
        ArrayDeque<Character> stack1 = new ArrayDeque<>();
        ArrayDeque<Integer> stack2 = new ArrayDeque<>();
        
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            
            
if (stack1.isEmpty()) {
                stack1.push(c);
                stack2.push(1);
            } else {
                if (c != stack1.peek()) {
                    stack1.push(c);
                    stack2.push(1);
                } else {
                    int num = stack2.pop();
                    if (num + 1 == k) {
                        stack1.pop();
                    } else {
                        stack2.push(num + 1);
                    }
                
}
            }
        }
        
        StringBuilder sb = new StringBuilder();
        
        while (!stack2.isEmpty()) {
            int num = stack2.pop();
            char c = stack1.pop();
            for (int i = 0; i < num; i++) {
                sb.append(c);
            }
        }
        
        return sb.reverse().toString();
    }
}
