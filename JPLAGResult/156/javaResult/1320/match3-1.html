class Solution {
    public String removeDuplicates(String s, int k) {
        Stack<Cnt> stk = new Stack<>();
        for (char cur : s.toCharArray()) {
            if (stk.empty() || stk.peek().ch != cur) {
                Cnt cnt = new Cnt(cur, 1);
                stk.push(cnt);
            } else {
                stk.peek().cnt += 1;
                if (stk.peek().cnt == k) {
                    stk.pop();
                
}
            }
        }
        StringBuilder sb = new StringBuilder();
        while (!stk.empty()) {
            Cnt cnt = stk.pop();
            for (int i = 0; i < cnt.cnt; i++) {
                sb.insert(0, cnt.ch);
            }
        }
        
return sb.toString();
    }
    
    class Cnt {
        char ch;
        int cnt;
        Cnt(char ch, int cnt) {
            this.ch = ch;
            this.cnt = cnt;
        }
    }
}
