class Solution {
    class Info {
        char ch;
        int cnt;
        Info(char ch, int cnt) {
            this.ch = ch;
            this.cnt = cnt;
        }
        
        void inc() {
            cnt++;
        
}
    }
    public String removeDuplicates(String s, int k) {
        LinkedList<Info> stk = new LinkedList<>();
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            Info last = stk.peekLast();
            if (last != null && last.ch == ch) {
                last.inc();
                if (last.cnt >= k) {
                    stk.removeLast();
                }
            } else {
                stk.add(new Info(ch, 1));
            
}
        }
        StringBuilder sb = new StringBuilder();
        while(stk.size() > 0) {
            Info frt = stk.removeFirst();
            for (int i = 0; i < frt.cnt; i++) {
                sb.append(frt.ch);
            }
        }
        return sb.toString();
    }
}
