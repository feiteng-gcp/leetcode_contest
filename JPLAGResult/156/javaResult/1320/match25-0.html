class Solution {
    class Pojo {
        char c;
        int cnt;
        Pojo(char c, int cnt) {
            this.c = c;
            this.cnt = cnt;
        }
    }
    
    public String removeDuplicates(String s, int k) {
        Stack<Pojo> stack = new Stack<>();
        for(char c : s.toCharArray()) {
            if(stack.isEmpty()) {
                stack.push(new Pojo(c,1));
            }else {
                if(stack.peek().c == c) {
                    stack.peek().cnt++;
                    if(stack.peek().cnt == k) stack.pop();
                }else {
                    stack.push(new Pojo(c,1));
                }
            }
        }
        StringBuilder sb = new StringBuilder();
        while(!stack.isEmpty()) {
            Pojo cur = stack.pop();
            for(int i = 0 ; i < cur.cnt;i++) {
                sb.append(cur.c);
            }
        }
        return sb.reverse().toString();
    }
}
