class Element
{
    char ch;
    int count;
    
    public Element(char ch, int count)
    {
        this.ch = ch;
        this.count = count;
    }
}

class Solution {
    public String removeDuplicates(String s, int k) {
        if (s == null || s.length() <= 1 || k == 0)
        {
            return s;
        }
        
        LinkedList<Element> queue = new LinkedList<Element>();
        
        for (int i = 0; i < s.length(); i++)
        {
            char ch = s.charAt(i);
            
            if (
queue.isEmpty())
            {
                Element newEle = new Element(ch, 1);
                
                queue.add(newEle);
            }
            else
            {
                Element peekElement = queue.peekLast();
                
                if (queue.peekLast().ch == ch)
                {
                    int curCount = queue.peekLast().count;
                    
                    if (curCount + 1 >= k)
                    {
                        curCount = curCount + 1 - k;
                        
                        if (curCount <= 0)
                        {
                            queue.pollLast();
                        }
                        else
                        {
                            queue.peekLast().count = curCount;
                        }
                    }
                    else
                    {
                        queue.peekLast().count = curCount + 1;
                    }
                }
                else
                {
                    queue.add(new Element(ch, 1));
                
}
            }
        }
        
        StringBuilder result = new StringBuilder("");
        
        while (!queue.isEmpty())
        {
            Element element = queue.poll();
            
            for (int i = 0; i < element.count; i++)
            {
                result.append(element.ch);
            }
                
        }
        
        return result.toString();
    }
}
