class Solution {
    public String removeDuplicates(String s, int k) {
        char cs[] = s.toCharArray();
        int l = cs.length;
        int f[][] = new int[l][2];
        int p = 0;
        
        for(char c:cs){
            if(p==0||c!=f[p-1][0]){
                f[p][0] = c;
                f[p][1] = 1;
                p++;
            }else{
                f[p-1][1]++;
            }
            if(f[p-1][1]==k){
                --p;
            }
        }
        StringBuilder sb = new StringBuilder();
        for(int i=0;i<p;++i){
            for(int j=0;j<f[i][1];++j){
                sb.appendCodePoint(f[i][0]);
            }
        }
        return sb.toString();
        
        
    }
}
