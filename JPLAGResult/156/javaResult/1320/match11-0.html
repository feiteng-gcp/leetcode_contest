class Solution {
    public String removeDuplicates(String s, int k) {
        Stack<Character> st = new Stack<>();
        Stack<Integer> cnt = new Stack<>();
        for (char c : s.toCharArray()) {
            if (st.isEmpty()) {
                st.push(c);
                cnt.push(1);
            } else {
                if (c == st.peek()) {
                    cnt.push(cnt.pop() + 1);
                    if (cnt.peek() == k) {
                        cnt.pop();
                        st.pop();
                    }
                } else {
                    st.push(c);
                    cnt.push(1);
                }
            }
        }
        StringBuilder sb = new StringBuilder();
        while (!st.isEmpty()) {
            char c = st.pop();
            int count = cnt.pop();
            while (count-- > 0) {
                sb.append(c);
            }
        }
        return sb.reverse().toString();
    }
}
