class Solution {
    public String removeDuplicates(String s, int k) {
        Stack<Point> stack = new Stack<>();
        int i = 0;
        while (i < s.length()) {
            char c = s.charAt(i);
            
if (stack.isEmpty()) {
                stack.push(new Point(c, 1));
            } else {
                Point p = stack.peek();
                if (p.c == c) {
                    p.cnt++;
                    if (p.cnt == k) stack.pop()
;
                } else {
                    stack.push(new Point(c, 1));
                }
            }
            i++;
        }
        
        String res = "";
        while (!stack.isEmpty()) {
            Point p = stack.pop();
            int cnt = p.cnt;
            while (cnt > 0) {
                cnt--;
                res = p.c + res;
            }
        }
        return res;
    
}
    
    class Point {
        char c;
        int cnt;
        public Point(char c, int cnt) {
            this.c = c;
            this.cnt = cnt;
        }
    }
}
