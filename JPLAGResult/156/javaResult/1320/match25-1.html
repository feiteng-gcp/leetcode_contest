class Solution {
    public String removeDuplicates(String s, int k) {
        
        Stack<Pair> st = new Stack<>(); 
        char[] cs = s.toCharArray(); 
        for (int i = 0; i < cs.length; i++) 
{
            if (st.isEmpty()) {
                st.push(new Pair(cs[i], 1));
            } else {
                Pair tmp = st.peek(); 
                if (cs[i] == tmp.c) {
                    tmp.cnt += 1; 
                    if (tmp.cnt == k) {
                        st.pop();
                    
}
                } else {
                    st.push(new Pair(cs[i], 1));
                }
            }
        }
        StringBuilder sb = new StringBuilder(); 
        while (!st.isEmpty()) {
            Pair p = st.pop();
            while (p.cnt != 0) {
                sb.append(p.c);
                p.cnt -= 1; 
            }
        }
        return sb.reverse().toString(); 
    }
    
public class Pair {
        char c; 
        int cnt; 
        public Pair(char c, int cnt) {
            this.c = c;
            this.cnt = cnt; 
        }
    }
}
