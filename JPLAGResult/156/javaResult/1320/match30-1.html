import java.util.ArrayList;
import java.util.List;
import java.util.Stack;

/**
 * created by vidyut on 2019-09-29
 **/

public class Solution {
    public String removeDuplicates(String s, int k) {
        List<CharFreq> freqList = new ArrayList<>();
        for (int i = 0; i < s.length(); i++) {
            int j = i;
            while(j < s.length() && s.charAt(i) == s.charAt(j)) {
                j++;
            }
            freqList.add(new CharFreq(s.charAt(i), j - i));
            i = j - 1;
        }
        Stack<CharFreq> st = new Stack<>();
        for (int i = 0; i < freqList.size(); i++) {
            int totalFreq = freqList.get(i).freq;
            while(!st.isEmpty() && st.peek().c == freqList.get(i).c) {
                totalFreq += st.pop().freq;
            }
            if (totalFreq % k != 0) {
                st.push(new CharFreq(freqList.get(i).c, totalFreq % k));
            
}
        }
        StringBuilder ans = new StringBuilder();
        while(!st.isEmpty()) {
            CharFreq cf = st.pop();
            for (int i = 0; i < cf.freq; i++) {
                ans.append(cf.c);
            }
        }
        return ans.reverse().toString();
    }
    public static void main(String[] args) {
        System.out.println(new Solution().removeDuplicates("deeedbbcccbdaa", 3));
        System.out.println(new Solution().removeDuplicates("pbbcggttciiippooaais", 2));
    }
    class CharFreq {
        char c;
        int freq;

        @Override
        public String toString() {
            return "CharFreq{" +
                    "c=" + c +
                    ", freq=" + freq +
                    '}';
        }

        
public CharFreq(char c, int freq) {
            this.c = c;
            this.freq = freq;
        }
    }
}
