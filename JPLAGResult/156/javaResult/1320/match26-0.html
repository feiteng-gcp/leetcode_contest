class Solution {
    public String removeDuplicates(String s, int k) {
        Stack<E> stack = new Stack<>();
        for (char c : s.toCharArray()) {
            if (stack.isEmpty() || stack.peek().c != c) {
                E e = new E(c, 1);
                stack.push(e);
            } else {
                E peek = stack.peek();
                peek.count++;
                peek.count %= k;
                    if (peek.count == 0) stack.pop();                
            }
        }
        
        StringBuilder sb = new StringBuilder();
        while (!stack.isEmpty()) {
            E e = stack.pop();
            for (int i = 0; i < e.count; i++) sb.append(e.c);
        }
        
        sb.reverse();
        return sb.toString();
    }
    
    private static class E {
        char c;
        int count;
        public E(char c, int count) {
            this.c = c;
            this.count = count;
        }
    }
}
