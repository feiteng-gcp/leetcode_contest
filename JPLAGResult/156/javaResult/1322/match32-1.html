class Solution {
    public int minimumMoves(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        //int[][][] dp = new int[m][n][2];
        //pos -> tail head
        
int[][] pos = new int[][]{{0,0}, {0,1}};
        String target = (m-1) + "," + (n-2) + "," + (m-1) + "," + (n-1);
        HashSet<String> set = new HashSet<>();
        Deque<int[][]> que = new LinkedList<>();
        
que.add(pos);
        int step = 0;
        while(!que.isEmpty()) {
            int size = que.size();
            for(int i = 0; i < size; i++){
                pos = que.poll();
                int[] tail = pos[0];
                int[] head = pos[1];
                String curr = tail[0] + "," + tail[1] + "," + head[0] + "," + head[1];
                if(curr.equals(target)) return step;
                if(!set.add(curr)) continue;
                if(head[1] + 1 < n && grid[head[0]][head[1] + 1] != 1 &&
                  tail[1] + 1 < n && grid[tail[0]][tail[1] + 1] != 1
                  ) 
                    que.add(
new int[][]{{tail[0], tail[1]+1}, {head[0], head[1]+1}});
                
                if(head[0] + 1 < m && grid[head[0]+1][head[1]] != 1 &&
                  tail[0] + 1 < m && grid[tail[0]+1][tail[1]] != 1)
                    que.add(
new int[][]{{tail[0]+1, tail[1]}, {head[0]+1, head[1]}});
                //clockwise
                if(head[0] == tail[0] && 
                  head[0] + 1 < m && grid[head[0]+1][head[1]] != 1 &&
                  tail[0] + 1 < m && grid[tail[0]+1][tail[1]] != 1
                  ) 
                    que.add(new int[][]{tail, {tail[0]+1, tail[1]}});
                //counterclockwise
                if(head[1] == tail[1] &&
                    head[1] + 1 < n && grid[head[0]][head[1] + 1] != 1 &&
                    tail[1] + 1 < n && grid[tail[0]][tail[1] + 1] != 1
                  ) 
                    que.add(new int[][]{tail, {tail[0], tail[1]+1}});
            }
            step++;
        }
        return -1;
    }
}
