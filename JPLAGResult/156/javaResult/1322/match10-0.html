class Solution {
    public String getKey(int[] p) {
        return p[0]+":"+p[1]+":"+p[2]+":"+p[3];
    }
    
    public int minimumMoves(int[][] grid) {
        int n = grid.length;
        String destKey = (n-1)+":"+(n-2)+":"+(n-1)+":"+(n-1);
        HashSet<String> visited = new HashSet<>();
        Queue<int[]> q = new LinkedList<>();
        q.add(new int[]{0,0,0,1});
        visited.add("0:0:0:1");
        int steps = 0;
        while(!q.isEmpty()) {
            //System.out.println(steps);
            int size = q.size();
            while(size-- > 0) {
                int[] cur = q.remove();
                //System.out.println(getKey(cur));
                if(cur[0] == cur[2]) { // horizontal oriented
                    if(cur[3] < n-1 && grid[cur[0]][cur[3]+1] != 1) { // move right
                        int[] next = new int[4];
                        next[0] = cur[0];
                        next[1] = cur[1]+1;
                        next[2] = cur[2];
                        next[3] = cur[3]+1;
                        String nextKey = getKey(next);
                        if(nextKey.equals(destKey)) {
                            return steps+1;
                        }
                        if(!visited.contains(nextKey)) {
                            visited.add(nextKey);
                            q.add(next);
                        }
                    }
                    if(cur[0] < n-1 && grid[cur[0]+1][cur[1]] != 1 && grid[cur[2]+1][cur[3]] != 1) { // move down
                        int[] next = new int[4];
                        next[0] = cur[0]+1;
                        next[1] = cur[1];
                        next[2] = cur[2]+1;
                        next[3] = cur[3];
                        String nextKey = getKey(next);
                        if(nextKey.equals(destKey)) {
                            return steps+1;
                        }
                        if(!visited.contains(nextKey)) {
                            visited.add(nextKey);
                            q.add(next);
                        }
                    }
                    if(cur[0] < n-1 && grid[cur[0]+1][cur[1]] != 1 && grid[cur[2]+1][cur[3]] != 1) { // rotate clockwise 
                        int[] next = new int[4];
                        next[0] = cur[0];
                        next[1] = cur[1];
                        next[2] = cur[0]+1;
                        next[3] = cur[1];
                        String nextKey = getKey(next);
                        if(!visited.contains(nextKey)) {
                            visited.add(nextKey);
                            q.add(next);
                        }
                    }
                } else if(cur[1] == cur[3]) { // vertical oriented
                    if(cur[1] < n-1 && grid[cur[0]][cur[1]+1] != 1 && grid[cur[2]][cur[3]+1] != 1) { // move right
                        int[] next = new int[4];
                        next[0] = cur[0];
                        next[1] = cur[1]+1;
                        next[2] = cur[2];
                        next[3] = cur[3]+1;
                        String nextKey = getKey(next);
                        if(nextKey.equals(destKey)) {
                            return steps+1;
                        }
                        if(!visited.contains(nextKey)) {
                            visited.add(nextKey);
                            q.add(next);
                        }
                    }
                    if(cur[2] < n-1 && grid[cur[2]+1][cur[3]] != 1) { // move down
                        int[] next = new int[4];
                        next[0] = cur[0]+1;
                        next[1] = cur[1];
                        next[2] = cur[2]+1;
                        next[3] = cur[3];
                        String nextKey = getKey(next);
                        if(!visited.contains(nextKey)) {
                            visited.add(nextKey);
                            q.add(next);
                        }
                    }
                    if(cur[1] < n-1 && grid[cur[0]][cur[1]+1] != 1 && grid[cur[2]][cur[3]+1] != 1) { // rotate counterclockwise
                        int[] next = new int[4];
                        next[0] = cur[0];
                        next[1] = cur[1];
                        next[2] = cur[0];
                        next[3] = cur[1]+1;
                        String nextKey = getKey(next);
                        if(!visited.contains(nextKey)) {
                            visited.add(nextKey);
                            q.add(next);
                        }
                    }
                }
            }
            steps++;
        }
        return -1;
    }
}
