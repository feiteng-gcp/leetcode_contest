class Solution {
    public int minimumMoves(int[][] grid) {
        Queue<Integer> queue = new LinkedList<>();
        Set<Integer> visited = new HashSet<>();
        
        queue.offer(0);
        visited.add(0);
        
        int m = grid.length;
        int n = grid[0].length;
        
        // row << 12, col << 2, dir  
        int dist = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            for (int i = 0; i < size; i++) {
                int temp = queue.poll();
                
                int row = temp >> 24;
                int col = (temp >> 12) & 0xfff;
                int dir = temp & 1;
                
                if (row == m - 1 && col == n - 2) {
                    return dist;
                }
                
                if (dir == 0) {
                    if (col + 2 < n && grid[row][col + 2] == 0) {
                        
int state = getState(row, col + 1, 0);
                        if (!visited.contains(state)) {
                            visited.add(state);
                            queue.offer(state);
                        }
                    }
                    if (row + 1 < m && grid[row + 1][col] == 0 && grid[row + 1][col + 1] == 0) {
                        int state = getState(row + 1, col, 0);
                        
if (!visited.contains(state)) {
                            visited.add(state);
                            queue.offer(state);
                        }
                    }
                    if (row + 1 < m && grid[row + 1][col] == 0 && grid[row + 1][col + 1] == 0) {
                        
int state = getState(row, col, 1);
                        if (!visited.contains(state)) {
                            visited.add(state);
                            queue.offer(state);
                        }
                    }
                } else {
                    if (row + 2 < m && grid[row + 2][col] == 0) {
                        int state = getState(row + 1, col, 1);
                        
if (!visited.contains(state)) {
                            visited.add(state);
                            queue.offer(state);
                        }
                    }
                    
                    if (col + 1 < n && grid[row][col + 1] == 0 && grid[row + 1][col + 1] == 0) {
                        int state = getState(row, col + 1, 1);
                        
if (!visited.contains(state)) {
                            visited.add(state);
                            queue.offer(state);
                        }
                    }
                    
                    if (col + 1 < n && grid[row][col + 1] == 0 && grid[row + 1][col + 1] == 0) {
                        
int state = getState(row, col, 0);
                        if (!visited.contains(state)) {
                            visited.add(state);
                            queue.offer(state);
                        }
                    }
                }
            }
            dist++;
        }
        
        return -1;
    }
    
    public int getState(int row, int col, int dir) {
        return (row << 24) | (col << 12) | dir;
    }
        
    
    public boolean reachFinal(int temp, int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        return temp == (((m - 1) << 24) | ((n - 1) << 12) | 0);
    }
}
