class Solution {
    public int minimumMoves(int[][] g) {
        int n = g.length;
        
        Queue<int[]> q = new ArrayDeque<int[]>();
        q.offer(new int[]{0,1,0,0,0});
       // int sp = 0;
        Set<String> st = new HashSet<>();
        
        while(q.size()>0){
            int sz = q.size();
            while(sz-->0){
                int c[] = q.poll();
                int x1 = c[0];
                int y1 = c[1];
                int x2 = c[2];
                int y2 = c[3];
                int sp = c[4];
                
                if(x1==n-1&&x2==n-1&&y1==n-1&&y2==n-2){
                    return sp;
                }
                
                String s = x1+","+y1+";"+x2+","+y2;
                if(st.contains(s)){
                    continue;
                }
                st.add(s);
                
                // heng
                if(x1==x2&&x1+1<n&&g[x1+1][y1]==0&&g[x1+1][y2]==0){
                    
                    q.offer(new int[]{x1+1,y2,x2,y2,sp+1});
                    q.offer(new int[]{x1+1,y1,x2+1,y2,sp+1});
                }
                if(x1==x2&&y1+1<n&&g[x1][y1+1]==0){
                    q.offer(new int[]{x1,y1+1,x1,y1,sp+1});
                }
                
                if(y1==y2&&y1+1<n&&g[x1][y1+1]==0&&g[x2][y2+1]==0){
                    
                    q.offer(new int[]{x1,y2+1,x2,y2+1,sp+1});
                    q.offer(new int[]{x1-1,y2+1,x2,y2,sp+1});
                    
                }
                if(y1==y2&&x1+1<n&&g[x1+1][y1]==0){
                    q.offer(new int[]{x1+1,y1,x2+1,y2,sp+1});
                }
                
                
                
                
                
            }
            
        }
        return -1;
        
        
        
    }
}
