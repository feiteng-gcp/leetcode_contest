class Solution {
    public int minimumMoves(int[][] grid) {
        int[] dx = {0, 1};
        int[] dy = {1, 0}; 
        Set<String> set = new HashSet<>(); 
        
int[][] pos = {{0, 0}, {0, 1}};
        int n = grid.length;
       
        Queue<int[][]> q = new LinkedList<>(); 
        q.add(pos); 
        set.add("0/0/0/1");
        int cnt = 0; 
        while (!q.isEmpty()) {
            
            int size = q.size(); 
            while (size > 0) {
                int[][] tmp = q.poll(); 
                int tx = tmp[0][0];
                int ty = tmp[0][1];
                int hx = tmp[1][0];
                int hy = tmp[1][1];
                if (tx == n - 1 && ty == n - 2 && hx  == n - 1 && hy == n - 1) {
                    return cnt; 
                }
                // System.out.println(tx + " " + ty + " " + hx + "  " + hy);
                if (tx == hx) { // h
                    if (hy + 1 < n && grid[hx][hy + 1] != 1 
                        && !set.contains(tx + "/" + (ty + 1) + "/" + hx + "/" + (hy + 1))) {
                        set.add(tx + "/" + (ty + 1) + "/" + hx + "/" + (hy + 1));
                        
q.add(new int[][]{{tx, ty + 1}, {hx, hy + 1}});
                    }
                    if (tx + 1 < n && grid[tx + 1][ty] != 1 && grid[hx + 1][hy] != 1) {
                        if (!set.contains((tx + 1) + "/" + ty + "/" + (hx + 1) + "/" + hy)) {
                            set.add((tx + 1) + "/" + ty + "/" + (hx + 1) + "/" + hy);
                            
q.add(new int[][]{{tx + 1, ty}, {hx + 1, hy}});
                        }
                        if (!set.contains(tx + "/" + ty + "/" + (hx + 1) + "/" + (hy - 1))) {
                            set.add(tx + "/" + ty + "/" + (hx + 1) + "/" + (hy - 1)); 
                            
q.add(new int[][]{{tx, ty}, {hx + 1, hy - 1}});
                        }
                        
                    }
                    
                    
                } else { // v
                    if (hx + 1 < n && grid[hx + 1][hy] != 1 
                        && !set.contains((tx + 1) + "/" + ty + "/" + (hx + 1) + "/" + hy)) {
                        set.add((tx + 1) + "/" + ty + "/" + (hx + 1) + "/" + hy);
                        q.add(
new int[][]{{tx + 1, ty}, {hx + 1, hy}});
                    }
                    if (ty + 1 < n && hy + 1 < n && grid[tx][ty + 1] != 1 && grid[hx][hy + 1] != 1) {
                        if (!set.contains(tx + "/" + (ty + 1) + "/" + hx + "/" + (hy + 1))) {
                            set.add(tx + "/" + (ty + 1) + "/" + hx + "/" + (hy + 1));
                            q.add(new int[][]{{tx, ty + 1}, {hx, hy + 1}});
                        }
                        if (!set.contains(tx + "/" + ty + "/" + (hx - 1) + "/" + (hy + 1))) {
                            set.add(tx + "/" + ty + "/" + (hx - 1) + "/" + (hy + 1)); 
                            
q.add(new int[][]{{tx, ty}, {hx - 1, hy + 1}});
                        }
                        
                    }
                }
                
                size--; 
            }
            cnt++; 
        }
        return -1; 
        
    }
}
