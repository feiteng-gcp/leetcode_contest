class Solution {
    public int minimumMoves(int[][] grid) {
        int step = 0;
        int n = grid.length;
        
ArrayDeque<int[]> queue = new ArrayDeque<>();
        
        Set<String> vp = new HashSet<>();
        vp.add(0 + " " + 0 + " " + 0 + " " + 1); 
        queue.offer(new int[]{0, 0, 0, 1});
        
        
        while (!queue.isEmpty()) {
            
            ArrayDeque<int[]> queue2 = new ArrayDeque<>();
            while(!queue.isEmpty()) {
                int[] p = queue.poll();
                int a = p[0];
                int b = p[1];
                int c = p[2];
                int d = p[3];
                
                if ( a == n - 1 && b == n - 2 && c == n - 1 && d == n - 1)  {
                    return step;
                }
                
                if (a == c && b + 1 == d) { // shui ping 
                    if (d + 1 < n && grid[a][d + 1] == 0) {
                        if (!vp.contains(a + " " + d + " " + a + " " + (d + 1))) 
{
                            vp.add(a + " " + d + " " + a + " " + (d + 1));
                            queue2.offer(new int[]{a, d, a, d + 1});
                        }
                    }
                    
                    if (a + 1 < n && grid[a + 1][b] == 0 && grid[a + 1][d] == 0) {
                        if (!vp.contains(a + 1 + " " + b + " " + (a + 1) + " " + d)) {
                            vp.add(a + 1 + " " + b + " " + (a + 1) + " " + d);
                            queue2.offer(new int[]{a + 1, b, a + 1, d});
                        }
                        
                        if (!vp.contains(a + " " + b + " " + (a + 1) + " " + b)) 
{
                            vp.add(a  + " " + b + " " + (a + 1) + " " + b);
                            queue2.offer(new int[]{a, b, a + 1, b});
                        }
                        
                    }
                    
                } else if (a + 1 == c && b == d) { // shu zhi
                    if (d + 1 < n && grid[a][d + 1] == 0 && grid[c][d + 1] == 0) {
                        if (!vp.contains(a + " " + (d + 1) + " " + c + " " + (d + 1))) {
                            vp.add(a + " " + (d + 1) + " " + c + " " + (d + 1));
                            queue2.offer(new int[]{a, d + 1, c, d + 1});
                        }
                        
                        if (!vp.contains(a + " " + b + " " + a + " " + (b + 1))) 
{
                            vp.add(a + " " + b + " " + a + " " + (b + 1));
                            queue2.offer(new int[]{a, b, a, b + 1});
                        }
                    }
                    
                    if (c + 1 < n && grid[c + 1][d] == 0) {
                        if (!vp.contains(c + " " + d + " " + (c + 1) + " " + d)) 
{
                            vp.add(c + " " + d + " " + (c + 1) + " " + d);
                            queue2.offer(new int[]{c, d, c + 1, d});
                        }   
                    }
                }
                
            }
            step++;
            queue = queue2;
        }
        
        return -1;
        
        
    }
}
