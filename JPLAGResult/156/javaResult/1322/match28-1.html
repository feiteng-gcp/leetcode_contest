class Solution {
    public int minimumMoves(int[][] grid) {
        int n = grid.length;
        Queue<Pos> queue = new LinkedList<>();
        queue.add(new Pos(new int[]{0,1}, new int[]{0,0}, false));
        Set<String> visited = new HashSet<>();
        visited.add(queue.peek().toString());
        int step = 0;
        
        if (grid[n-1][n-1] == 1 || grid[n-1][n-2] == 1) return -1;
        
        
while (!queue.isEmpty()) {
            int size = queue.size();
            
            for (int i = 0; i < size; i++) {
                Pos p = queue.poll();
                if (!p.shu && p.head[0] == n-1 && p.head[1] == n-1 && p.tail[0] == n-1 && p.tail[1] == n-2) return step;
                if (!p.shu) { // heng
                    int dx = p.head[0];
                    int dy = p.head[1] + 1;
                    if (dx < n && dy < n && grid[dx][dy] != 1) {
                        Pos tmp = new Pos(
new int[]{dx, dy}, p.head, false);
                        if (!visited.contains(tmp.toString())) {
                             queue.add(tmp);
                            visited.add(tmp.toString());
                        }
                           
                    }
                    
                    int dx1 = p.head[0] + 1;
                    int dy1 = p.head[1];
                    int dx2 = p.tail[0] + 1;
                    int dy2 = p.tail[1];
                    if (dx1 < n && dy1 < n && dx2 < n && dy2 < n && grid[dx1][dy1] != 1 && grid[dx2][dy2] != 1) {
                        Pos tmp = new Pos(new int[]{dx1, dy1}, 
new int[]{dx2, dy2}, false);
                        if (!visited.contains(tmp.toString())) {
                             queue.add(tmp);
                            visited.add(tmp.toString());
                        }
                    }
                    
                    dx = p.head[0] + 1;
                    dy = p.head[1] - 1;
                    if (dx < n && dy < n && grid[dx][dy] != 1 && grid[p.tail[0] + 1][p.tail[1] + 1] != 1) {
                        Pos tmp = new Pos(
new int[]{dx, dy}, p.tail, true);
                        if (!visited.contains(tmp.toString())) {
                             queue.add(tmp);
                            visited.add(tmp.toString());
                        }
                    }
                } else { // shu de 
                    int dx = p.head[0] + 1;
                    int dy = p.head[1];
                    if (dx < n && dy < n && grid[dx][dy] != 1) {
                        Pos tmp = new Pos(
new int[]{dx, dy}, p.head, true);
                        if (!visited.contains(tmp.toString())) {
                             queue.add(tmp);
                            visited.add(tmp.toString());
                        }
                    }
                    
                    int dx1 = p.head[0];
                    int dy1 = p.head[1] + 1;
                    int dx2 = p.tail[0];
                    int dy2 = p.tail[1] + 1;
                    if (dx1 < n && dy1 < n && dx2 < n && dy2 < n && grid[dx1][dy1] != 1 && grid[dx2][dy2] != 1) {
                        Pos tmp = new Pos(new int[]{dx1, dy1}, new int[]{dx2, dy2}, true);
                        if (!visited.contains(tmp.toString())) {
                             queue.add(tmp);
                            visited.add(tmp.toString());
                        }
                    }
                    
                    dx = p.head[0] - 1;
                    dy = p.head[1] + 1;
                    // System.out.println(dx + " " + dy);
                    if (dx < n && dy < n && grid[dx][dy] != 1 && grid[p.tail[0] + 1][p.tail[1] + 1] != 1) {
                        Pos tmp = new Pos(new int[]{dx, dy}, p.tail, false);
                        if (!visited.contains(tmp.toString())) {
                             queue.add(tmp);
                            visited.add(tmp.toString());
                        }
                    }
                }
                
            }
            step++;
        }
        
        return -1;
    }
    
    class Pos {
        int[] head;
        int[] tail;
        boolean shu;
        
        public Pos(int[] h, int[] t, boolean s) {
            this.head = h;
            this.tail = t;
            this.shu = s;
        }
        
        public String toString() {
            return head[0] + "," + head[1] + "," + tail[0] + "," + tail[1];
        }
    }
}
