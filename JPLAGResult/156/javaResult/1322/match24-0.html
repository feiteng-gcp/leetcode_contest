class Solution {
    public int minimumMoves(int[][] grid) {
        int[] start = new int[]{0,0,0,1};
        Set<Integer> seen = new HashSet<>();
        seen.add(index(start));
        Queue<int[]> q = new LinkedList<>();
        q.offer(start);
        int len = 0, n = grid.length;
        while(!q.isEmpty()){
            int sz = q.size();
            for(int t=0;t<sz;t++){
                int[] cur = q.poll();
                //System.out.print(cur[0]+" "+cur[1]+" "+cur[2]+" "+cur[3]);
                if(index(cur)==index(new int[]{n-1,n-2,n-1,n-1})) return len;
                for(int i=0;i<4;i++){
                    if(i==0){
                        //right
                        int[] next = new int[]{cur[0], cur[1]+1, cur[2], cur[3]+1};
                        if(cur[1]+1<n && cur[3]+1<n && grid[cur[0]][cur[1]+1]==0 && grid[cur[2]][cur[3]+1]==0 && !seen.contains(index(next))){
                            q.offer(next);
                            seen.add(index(next));
                        }
                    }else if(i==1){
                        //down
                        int[] next = new int[]{cur[0]+1, cur[1], cur[2]+1, cur[3]};
                        if(cur[0]+1<n && cur[2]+1<n && grid[cur[0]+1][cur[1]]==0 && grid[cur[2]+1][cur[3]]==0 && !seen.contains(index(next))){
                            q.offer(next);
                            seen.add(index(next));
                        }
                    }else if(i==2 && cur[0]==cur[2] && cur[1]+1==cur[3]){
                        int[] next = new int[]{cur[0], cur[1], cur[0]+1, cur[1]};
                        if(cur[0]+1<n && grid[cur[0]+1][cur[1]]==0  && grid[cur[0]+1][cur[1]+1]==0 && !seen.contains(index(next))){
                            q.offer(next);
                            seen.add(index(next));
                        }
                    }else if(i==3 && cur[0]+1==cur[2] && cur[1]==cur[3]){
                        int[] next = new int[]{cur[0], cur[1], cur[0], cur[1]+1};
                        if(cur[1]+1<n && grid[cur[0]][cur[1]+1]==0 && grid[cur[0]+1][cur[1]+1]==0 && !seen.contains(index(next))){
                            q.offer(next);
                            seen.add(index(next));
                        }
                    }
                }
            }
            len++;
            //System.out.println();
        }
        return -1;
    }
    
    public int index(int[] a){
        return a[0]+a[1]*100+a[2]*10000+a[3]*1000000;
    }
}
