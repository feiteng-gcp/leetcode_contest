import java.util.*;

class Solution {

    int[][] grid;
    int m;
    int n;
    Map<Integer, Map<Integer, Integer>> dis = new HashMap<>();

    int minimumMoves(int[][] grid) {
        this.grid = grid;
        this.m = grid.length;
        this.n = grid[0].length;
        if (grid[0].length < 2) return 0;
        if (grid[0][0] == 1 || grid[0][1] == 1) return 0;
//        int[][] distance = new int[m * n][m * n];
        
//        for (int i = 0; i < m * n; i++) {
//            Arrays.fill(distance[i], Integer.MAX_VALUE);
//        }
//        distance[0][1] = 0;
       put(0, 1, 0);
        
        Queue<int[]> q = new ArrayDeque<>();
        q.add(new int[]{0, 1});
        while (!q.isEmpty()) {
            int[] node = q.poll();
            int[] tail = getXY(node[0]);
            int[] head = getXY(node[1]);
            
if (head[0] == tail[0] && head[1] + 1 < n &&
                    grid[head[0]][head[1] + 1] == 0) {
                int[] next = new int[]{node[1], node[1] + 1};
                if (getValue(node[0], node[1]) + 1 < getValue(next[0], next[1])) {
                    put(next[0], next[1], getValue(node[0], node[1]) + 1 );
                    q.add(next);
                }
            }
            
if (head[0] == tail[0] && head[0] + 1 < m &&
                    grid[head[0] + 1][head[1]] == 0 &&  grid[tail[0] + 1][tail[1]] == 0) {
                int[] next = new int[]{node[0] + n, node[1] + n};
                if (getValue(node[0], node[1])  + 1 < getValue(next[0], next[1])) {
                    put(next[0], next[1], getValue(node[0], node[1]) + 1 );
                    q.add(next);
                }
            }
            
if (head[0] == tail[0] && head[0] + 1 < m &&
                    grid[head[0] + 1][head[1]] == 0 &&  grid[tail[0] + 1][tail[1]] == 0) {
                int[] next = new int[]{node[0], node[0] + n};
                if (getValue(node[0], node[1])  + 1 < getValue(next[0], next[1])) {
                    put(next[0], next[1], getValue(node[0], node[1]) + 1 );
                    q.add(next);
                }
            }
            
if (head[1] == tail[1] && head[0] + 1 < m &&
                    grid[head[0] + 1][head[1]] == 0) {
                int[] next = new int[]{node[1], node[1] + n};
                if (getValue(node[0], node[1]) + 1 < getValue(next[0], next[1])) {
                    put(next[0], next[1], getValue(node[0], node[1]) + 1 );
                    q.add(next);
                }
            }
            if (head[1] == tail[1] && head[1] + 1 < n &&
                    grid[head[0]][head[1] + 1] == 0 && grid[tail[0]][tail[1] + 1] == 0) {
                int[] next = new int[]{node[0] + 1, node[1] + 1};
                if (getValue(node[0], node[1]) + 1 < getValue(next[0], next[1])) {
                    put(next[0], next[1], getValue(node[0], node[1]) + 1 );
                    q.add(next);
                }
            }
            if (head[1] == tail[1] && head[1] + 1 < n &&
                    grid[head[0]][head[1] + 1] == 0 && grid[tail[0]][tail[1] + 1] == 0) {
                int[] next = new int[]{node[0], node[0] + 1};
                if (getValue(node[0], node[1])  + 1 < getValue(next[0], next[1])) {
                    put(next[0], next[1], getValue(node[0], node[1]) + 1 );
                    q.add(next);
                }
            }
        }
        return getValue(m * n - 2, m * n  - 1) == Integer.MAX_VALUE ? -1 : getValue(m * n - 2, m * n  - 1)  ;
    }
    
    private int getValue(int a, int b) {
        if (dis.containsKey(a) && dis.get(a).containsKey(b)) {
            return dis.get(a).get(b);
        }
        return Integer.MAX_VALUE;
    }

    private void put(int a, int b, int value) {
        dis.putIfAbsent(a, new HashMap<>());
        dis.get(a).put(b, value);
    }

    private int[] getXY(int x) {
        return new int[]{x / n, x % n};
    }

    public static void main(String[] args) {
        System.out.println(new Solution().minimumMoves(new int[][]{{0,0},{0,0}}));
    }
}
