class Solution {
    public int minimumMoves(int[][] grid) {
        int rows = grid.length, cols = grid[1].length;
        boolean[][][] viewed = new boolean[rows][cols][2];
        if(grid[0][0] == 1 || grid[0][1] == 1) return -1;
        
        Deque<int[]> dq = new LinkedList();
        
        viewed[0][1][0] = true;
        dq.add(new int[]{0,1,0});
        
        int step = 0;
        while(!dq.isEmpty()) {
            
            for(int size = dq.size(); size>0; size--) {
                int[] cur = dq.poll();
                int r = cur[0], c = cur[1], h = cur[2];
                //System.out.println(r+" "+c+" "+h);
                if(r == rows-1 && c==cols-1 && h==0)
                    return step;
                if(h == 0) {
                    if(c+1<cols && grid[r][c+1] == 0 && !viewed[r][c+1][0]) {
                        viewed[r][c+1][0] = true;
                        dq.add(new int[]{r, c+1, 0});
                    }
                        
                    if(r+1<rows && grid[r+1][c-1]==0 && grid[r+1][c] == 0 ) {
                        if(!viewed[r+1][c-1][1]) {
                            viewed[r+1][c-1][1] = true;
                            dq.add(new int[]{r+1, c-1, 1});
                        }
                        
                        if(!viewed[r+1][c][0]) {
                            viewed[r+1][c][0] = true;
                            dq.add(new int[]{r+1, c, 0});
                        }
                        
                    }
                        
                }
                
                if(h == 1) {
                    if(r+1<rows && grid[r+1][c] == 0 && !viewed[r+1][c][1]) {
                        viewed[r+1][c][1] = true;
                        dq.add(new int[]{r+1, c, 1});
                    }
                        
                    if(c+1<cols && grid[r][c+1]==0 && grid[r-1][c+1] == 0 ) {
                        if(!viewed[r-1][c+1][0]) {
                            viewed[r-1][c+1][0] = true;
                            dq.add(new int[]{r-1, c+1, 0});
                        }
                        
                        if(!viewed[r][c+1][1]) {
                            viewed[r][c+1][1] = true;
                            dq.add(new int[]{r, c+1, 1});
                        }
                        
                    }
                        
                }
            }
            
            step++;
        }
        
        return -1;
    }
}
