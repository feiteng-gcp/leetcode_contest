class Solution {
    public int minimumMoves(int[][] grid) {
        LinkedList<int[]> queue = new LinkedList<>();
        queue.offer(new int[]{0,0,0,1});
        int step = 0;
        Set<String> set = new HashSet<>();
        while(!queue.isEmpty()) {
            int size = queue.size();
            for(int i = 0 ; i < size;i++) {
                int[] cur = queue.poll();
                if(cur[0] == grid.length-1 && cur[1] == grid[0].length-2 && cur[2] == grid.length - 1 && cur[3] == grid[0].length-1) {
                    return step;
                }
                // right
                int[] right = new int[]{cur[0],cur[1]+1,cur[2],cur[3]+1};
                if(right[0] < grid.length && right[1] < grid[0].length && right[2] < grid.length && right[3] < grid[0].length && grid[right[0]][right[1]] == 0 && grid[right[2]][right[3]] ==0) {
                    String str = set2String(right);
                    if(!set.contains(str)) {
                        set.add(str);
                        queue.offer(right);
                    }
                }
                
                // down
                int[] down = new int[]{cur[0]+1, cur[1],cur[2]+1,cur[3]};
                if(down[0] < grid.length && down[1] < grid[0].length && down[2] < grid.length && down[3] < grid[0].length && grid[down[0]][down[1]] == 0 && grid[down[2]][down[3]] == 0) {
                    String str = set2String(down);
                    if(!set.contains(str)) {
                        set.add(str);
                        queue.offer(down);
                    }
                }
                
                //clock
                if(cur[0]+1 == cur[2]) {
                    int[] tmp = new int[]{cur[0], cur[1]+1, cur[2],cur[3]+1};
                    if(tmp[0] < grid.length && tmp[1] < grid[0].length && tmp[2] < grid.length && tmp[3] < grid[0].length && grid[tmp[0]][tmp[1]] == 0 && grid[tmp[2]][tmp[3]] == 0) {
                        int[] clock = new int[]{cur[0],cur[1],cur[2]-1, cur[3]+1};
                        String str = set2String(clock);
                        if(!set.contains(str)) {
                            set.add(str);
                            queue.offer(clock);
                        }
                    }
                }
                
                // ant-clock
                if(cur[1]+1 == cur[3]) {
                    int[] tmp = new int[]{cur[0]+1, cur[1],cur[2]+1,cur[3]};
                    System.out.println(set2String(tmp));
                    if(tmp[0] < grid.length && tmp[1] < grid[0].length && tmp[2] < grid.length && tmp[3] < grid[0].length && grid[tmp[0]][tmp[1]] == 0 && grid[tmp[2]][tmp[3]] == 0) {
                        int[] clock = new int[]{cur[0],cur[1],cur[2]+1,cur[3]-1};
                        String str = set2String(clock);
                        if(!set.contains(str)) {
                            set.add(str);
                            queue.offer(clock);
                        }
                    }
                }
            }
            step++;
        }
        return -1;
    }
    
    String set2String(int[] arr) {
        return arr[0] + "," + arr[1] + "," + arr[2] + "," + arr[3];
    }
}
