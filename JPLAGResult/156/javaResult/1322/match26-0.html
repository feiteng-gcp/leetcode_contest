class Solution {
    public int minimumMoves(int[][] grid) {
        Queue<Integer> q = new LinkedList<>();
        Set<Integer> v = new HashSet<>();
        q.add(0);
        v.add(0);
        int res = 0;
        int cnst = 100000, cnst2 = 10;
        int n = grid.length;
        while (!q.isEmpty()) {
            int size = q.size();
            for (int i = 0; i < size; i++) {
                int cur = q.poll();
                int x0 = cur/cnst, y0 = (cur%cnst)/cnst2, x1 = x0, y1 = y0, dir = cur %10;
                // System.out.println(""+x0+":"+y0+":"+dir);
                if (x0 == n-1 && y0 == n-2 && dir == 0) {
                    // System.out.println("end");
                    return res;
                }
                if (dir == 0) {
                    y1 = y0+1;
                } else {
                    x1 = x0+1;
                }

                // // Left.
                // if (y0 > 0 && y1 > 0 && grid[x0][y0-1] == 0 && grid[x1][y1-1] == 0) {
                //     int next = (x0) * cnst + (y0-1) * cnst2 + dir;
                //     if (!v.contains(next)) {
                //         v.add(next);
                //         q.add(next);
                //     }
                // }
                // Right.
                if (y0 < n-1 && y1 < n-1 && grid[x0][y0+1] == 0 && grid[x1][y1+1] == 0) {
                    int next = (x0) * cnst + (y0+1) * cnst2 + dir;
                    if (!v.contains(next)) {
                        v.add(next);
                        q.add(next);
                    }
                }
                // // Up.
                // if (x0 > 0 && x1 > 0 && grid[x0-1][y0] == 0 && grid[x1-1][y1] == 0) {
                //     int next = (x0-1) * cnst + (y0) * cnst2 + dir;
                //     if (!v.contains(next)) {
                //         v.add(next);
                //         q.add(next);
                //     }
                // }
                // Down.
                if (x0 < n-1 && x1 < n-1 && grid[x0+1][y0] == 0 && grid[x1+1][y1] == 0) {
                    int next = (x0+1) * cnst + (y0) * cnst2 + dir;
                    if (!v.contains(next)) {
                        v.add(next);
                        q.add(next);
                    }
                }
                // Rotate clockwise.
                if (x0 < n-1 && y0 < n-1 && grid[x0+1][y0] == 0 && grid[x0][y0+1] == 0 && grid[x0+1][y0+1] == 0){
                    if (dir == 0) {
                        int next = cur + 1;
                        if (!v.contains(next)) {
                            v.add(next);
                            q.add(next);
                        } 
                    }else {
                        int next = cur - 1;
                        if (!v.contains(next)) {
                            v.add(next);
                            q.add(next);
                        }
                    }
                }
            }
            res++;
        }
        return -1;
    }
}
