class Solution {
    public int minimumMoves(int[][] grid) {
        int N = grid.length;
        Set<Integer> visited = new HashSet<>();
        Queue<int[][]> q = new LinkedList<>();
        
int[][] pos = new int[][]{{0, 0}, {0, 1}};
        q.offer(pos);
        addVisited(visited, pos);
        int dist = 0;
        while (!q.isEmpty()) {
            int size = q.size();
            for (int i = 0; i < size; i++) {
                pos = q.poll();
                if (reachTarget(N, pos)) {
                    return dist;
                }
                List<int[][]> nextPos = 
nextMove(grid, visited, pos);
                q.addAll(nextPos);
            }
            dist++;
        }
        return -1;
    }
    
    private List<int[][]> nextMove(int[][] grid, Set<Integer> visited, int[][] pos) {
        int N = grid.length;
        List<int[][]> res = new ArrayList<>();
        boolean isVert = pos[0][0] != pos[1][0];
        // if (pos[0][0] > pos[1][0] || pos[0][1] > pos[1][1]) {
        //     int[] t = pos[0];
        //     pos[0] = pos[1];
        //     pos[1] = t;
        // }
        int headR = pos[1][0];
        int headC = pos[1][1];
        int tailR = pos[0][0];
        int tailC = pos[0][1];
        if (isVert) {
            if (headC + 1 < N && grid[headR][headC + 1] == 0 && grid[tailR][tailC + 1] == 0) {
                int[][] nextPos = 
new int[][]{{tailR, tailC + 1}, {headR, headC + 1}};
                if (!isVisited(visited, nextPos)) {
                    addVisited(visited, nextPos);
                    res.add(nextPos);
                }
                
nextPos = new int[][]{{tailR, tailC}, {tailR, tailC + 1}};
                if (!isVisited(visited, nextPos)) {
                    addVisited(visited, nextPos);
                    res.add(nextPos);
                }
            }
            if (headR + 1 < N && grid[headR + 1][headC] == 0) {
                int[][] nextPos = 
new int[][]{{tailR + 1, tailC}, {headR + 1, headC}};
                if (!isVisited(visited, nextPos)) {
                    addVisited(visited, nextPos);
                    res.add(nextPos);
                }
            }
            
        } else {
            if (headR + 1 < N && grid[headR + 1][headC] == 0 && grid[tailR + 1][tailC] == 0) {
                int[][] nextPos = 
new int[][]{{tailR + 1, tailC}, {headR + 1, headC}};
                if (!isVisited(visited, nextPos)) {
                    addVisited(visited, nextPos);
                    res.add(nextPos);
                }
                nextPos = 
new int[][]{{tailR, tailC}, {tailR + 1, tailC}};
                if (!isVisited(visited, nextPos)) {
                    addVisited(visited, nextPos);
                    res.add(nextPos);
                }
            }
            if (headC + 1 < N && grid[headR][headC + 1] == 0) {
                int[][] nextPos = new int[][]{{tailR, tailC + 1}, {headR, headC + 1}};
                if (!isVisited(visited, nextPos)) {
                    addVisited(visited, nextPos);
                    res.add(nextPos);
                }
            
}
        }
        return res;
    }
    
    private boolean reachTarget(int n, int[][] pos) {
        //return (pos[0][0] == n - 1 && pos[0][1] == n - 1 && pos[1][0] == n - 1 && pos[1][1] == n - 2) || (pos[0][0] == n - 1 && pos[0][1] == n - 2 && pos[1][0] == n - 1 && pos[1][1] == n - 1));
        return (pos[0][0] == n - 1 && pos[0][1] == n - 2 && pos[1][0] == n - 1 && pos[1][1] == n - 1);
    }
    
    private void addVisited(Set<Integer> visited, int[][] pos) {
        int head = pos[1][0] * 100 + pos[1][1];
        int tail = pos[0][0] * 100 + pos[0][1];
        visited.add(head * 10000 + tail);
        //visited.add(tail * 10000 + head);
    }
    
    private boolean isVisited(Set<Integer> visited, int[][] pos) {
        int head = pos[1][0] * 100 + pos[1][1];
        int tail = pos[0][0] * 100 + pos[0][1];
        return visited.contains(head * 10000 + tail); //|| visited.contains(tail * 10000 + head);
    }
}
