import java.util.*;

class Solution {
    public int minimumMoves(int[][] grid) {
        return bfs(grid);
    }

    public int bfs(int[][] grid) {
        int y = grid.length, x = grid[0].length;
        if (grid[y - 1][x - 2] == 1 || grid[y - 1][x - 1] == 1) return -1;

        Set<String> set = new HashSet<>();
        Queue<int[]> queue = new LinkedList<>();

        queue.add(new int[]{0, 0, 0, 0});
        while (!queue.isEmpty()) {
            int[] tri = queue.poll();
            int a = tri[0], b = tri[1], s = tri[2], c = tri[3];
            if (a == x - 2 && b == y - 1 && s == 0) return c;
            String key = a + "," + b + "," + s;
            if (set.contains(key)) {
                continue;
            }
            set.add(key);
            if (s == 0) {
                if (a + 2 < x && grid[b][a + 2] == 0) {
                    queue.add(new int[]{a + 1, b, 0, c + 1});
                }
                if (b + 1 < y && grid[b + 1][a] == 0 && grid[b + 1][a + 1] == 0) {
                    queue.add(new int[]{a, b, 1, c + 1});
                    queue.add(new int[]{a, b + 1, 0, c + 1});
                }
            } else {
                if (b + 2 < y && grid[b + 2][a] == 0) {
                    queue.add(new int[]{a, b + 1, 1, c + 1});
                }
                if (a + 1 < y && grid[b][a + 1] == 0 && grid[b + 1][a + 1] == 0) {
                    queue.add(new int[]{a, b, 0, c + 1});
                    queue.add(new int[]{a + 1, b, 1, c + 1});
                }
            }
        }
        return -1;
    }
}
