class Solution {
    int n;
    public int minimumMoves(int[][] grid) {
        n = grid.length;
        Queue<int[]> queue = new LinkedList<>();
        TreeSet<int[]> visited = new TreeSet<>(new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                if (o1[0] == o2[0]) return o1[1] - o2[1];
                return o1[0] - o2[0];
            }
        });
        queue.add(new int[] {0, 1});
        visited.add(new int[] {0, 1});
        int res = 0;
        while (!queue.isEmpty()) {
            int sz = queue.size();
            res++;
            for (int i = 0; i < sz; i++) {
                int[] cur = queue.poll();
                int x0 = cur[0] / n;
                int y0 = cur[0] % n;
                int x1 = cur[1] / n;
                int y1 = cur[1] % n;

                if (x0 == x1) {
                    if (y1 + 1 < n && grid[x0][y1+1] == 0) {
                        int x2 = x0;
                        int y2 = y1+1;
                        if (x2 == n - 1 && y2 == n - 1) return res;
                        if (!visited.contains(new int[] {cur[1], x2 * n + y2})) {
                            visited.add(new int[] {cur[1], x2 * n + y2});
                            queue.add(new int[] {cur[1], x2 * n + y2});
                        }
                    }
                    if (x0 + 1 < n && grid[x0+1][y0] == 0 && grid[x0+1][y1] == 0) {
                        if (x0 + 1 == n - 1 && y1 == n - 1) return res;
                        if (!visited.contains(new int[]{index(x0+1, y0), index(x1+1, y1)})) {
                           visited.add(new int[]{index(x0+1, y0), index(x1+1, y1)});
                           queue.add(new int[]{index(x0+1, y0), index(x1+1, y1)});
                        }
                        int x2 = x0+1;
                        int y2 = y0;
                        if (!visited.contains(new int[]{cur[0], index(x2, y2)})) {
                            visited.add(new int[]{cur[0], index(x2, y2)});
                            queue.add(new int[]{cur[0], index(x2, y2)});
                        }
                    }
                } else {
                    if (x1 + 1 < n && grid[x1+1][y0] == 0) {
                        int x2 = x1 + 1;
                        int y2 = y0;
                        if (!visited.contains(new int[] {cur[1], x2 * n + y2})) {
                            visited.add(new int[] {cur[1], x2 * n + y2});
                            queue.add(new int[] {cur[1], x2 * n + y2});
                        }
                    }
                    if (y0 + 1 < n && grid[x0][y0+1] == 0 && grid[x1][y1+1] == 0) {
                        if (!visited.contains(new int[]{index(x0, y0+1), index(x1, y1+1)})) {
                            visited.add(new int[]{index(x0, y0+1), index(x1, y1+1)});
                            queue.add(new int[]{index(x0, y0+1), index(x1, y1+1)});
                        }
                        int x2 = x0;
                        int y2 = y0+1;
                        if (!visited.contains(new int[] {cur[0], x2 * n + y2})) {
                            visited.add(new int[] {cur[0], x2 * n + y2});
                            queue.add(new int[] {cur[0], x2 * n + y2});
                        }
                    }
                }
            }
        }
        return -1;
    }

    int index(int x, int y) {
        return x * n + y;
    }
}
