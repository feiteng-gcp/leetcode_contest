class Solution {
    public int minimumMoves(int[][] A) {
        Queue<List<Integer>> queue = new LinkedList<>();
        Set<List<Integer>> set = new HashSet<>();
        queue.offer(Arrays.asList(new Integer[]{0, 0, 0, 1}));
        set.add(Arrays.asList(new Integer[]{0, 0, 0, 1}));
        int n = A.length;
        int step = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            while (size-- > 0) {
                List<Integer> cur = queue.poll();
                int r0 = cur.get(0);
                int c0 = cur.get(1);
                int r1 = cur.get(2);
                int c1 = cur.get(3);
                if (r0 == n - 1 && c0 == n - 2 && r1 == n - 1 && c1 == n - 1) {
                    return step;
                }
                
                if (c0 + 1 < n && A[r0][c0 + 1] == 0 && c1 + 1 < n && A[r1][c1 + 1] == 0 ) {
                    if (set.add(Arrays.asList(new Integer[]{r0, c0 + 1, r1, c1 + 1}))) {
                        queue.offer(Arrays.asList(new Integer[]{r0, c0 + 1, r1, c1 + 1}));
                    }
                }
                
                if (r0 + 1 < n && A[r0 + 1][c0] == 0 && r1 + 1 < n && A[r1 + 1][c1] == 0 ) {
                    if (set.add(Arrays.asList(new Integer[]{r0 + 1, c0, r1 + 1, c1}))) {
                        queue.offer(Arrays.asList(new Integer[]{r0 + 1, c0, r1 + 1, c1}));
                    }
                }
                
                if (r0 == r1) {
                    if (r0 + 1 < n && A[r0 + 1][c0] == 0 && r1 + 1 < n && A[r1 + 1][c1] == 0 ) {
                        if (set.add(Arrays.asList(new Integer[]{r0, c0, r1 + 1, c1 - 1}))) {
                            queue.offer(Arrays.asList(new Integer[]{r0, c0, r1 + 1, c1 - 1}));
                        }
                    }
                } else {
                    if (c0 + 1 < n && A[r0][c0 + 1] == 0 && c1 + 1 < n && A[r1][c1 + 1] == 0 ) {
                        if(set.add(Arrays.asList(new Integer[]{r0, c0, r1 - 1, c1 + 1}))) {
                            queue.offer(Arrays.asList(new Integer[]{r0, c0, r1 - 1, c1 + 1}));
                        } 
                    }
                }
            }
            //System.out.println(queue);
            //System.out.println(set);
            step++;
        }
        
        return -1;
    }
}
