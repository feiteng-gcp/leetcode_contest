class Solution {
    public int minimumMoves(int[][] grid) {
        
        int n = grid.length, ans = 0, finish = (n - 1) * n + (n - 2);
        
        HashSet<Integer> pos = new HashSet<>();
        
        Queue<Integer> q = new LinkedList<>();
        
        q.add(0);
        
        while(!q.isEmpty()){
            
            int size = q.size();
            
            while(size-- > 0){
                
                int curr = q.poll();
                
                if(curr == finish)
                    return ans;
                
                int dir = curr / (n * n);
                
                curr = curr % (n * n);
                
                int r = curr / n, c = curr % n;
                
                if(dir == 0){
                    
                    if(c + 2 < n && grid[r][c + 2] == 0){
                        
                        int next = dir * (n * n) + r * n + (c + 1);
                        
                        if(pos.add(next))
                            q.add(next);
                    }
                    
                    if(r + 1 < n && grid[r + 1][c] == 0 && grid[r + 1][c + 1] == 0){
                        
                        int next = dir * (n * n) + (r + 1) * n + c;
                        
                        if(pos.add(next))
                            q.add(next);
                    }
                    
                    if(r + 1 < n && grid[r + 1][c] == 0 && grid[r + 1][c + 1] == 0){
                        int next = (n * n) + r * n + c;
                        if(pos.add(next))
                            q.add(next);
                    }
                }
                
                
                else {
                    
                    if(r + 2 < n && grid[r + 2][c] == 0){
                        
                        int next = dir * (n * n) + (r + 1) * n + c;
                        
                        if(pos.add(next))
                            q.add(next);
                    }
                    
                    if(c + 1 < n && grid[r][c + 1] == 0 && grid[r + 1][c + 1] == 0){
                        
                        int next = dir * (n * n) + r * n + (c + 1);
                        
                        if(pos.add(next))
                            q.add(next);
                    }
                    
                    if(c + 1 < n && grid[r][c + 1] == 0 && grid[r + 1][c + 1] == 0){
                        int next = r * n + c;
                        if(pos.add(next))
                            q.add(next);
                    }
                }
                
            }
            
            ans++;
        }
        
        return -1;
    }
}
