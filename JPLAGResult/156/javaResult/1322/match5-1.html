class Solution {
    public int minimumMoves(int[][] grid) {
        int ans=0,n=grid.length;
        Set<String> visited = new HashSet<>();
        visited.add("0,0,0,1");
        Queue<int[][]> que=new LinkedList<>();
        que.offer(new int[][]{{0,0},{0,1}});
        while(!que.isEmpty()){
            ans++;
            int size=que.size();
            for(int i=0;i<size;i++){
                int[][] cur = que.poll();
                if(cur[0][0]==cur[1][0]){
                    int[][] next2=new int[][]{cur[0].clone(),{cur[1][0]+1,cur[1][1]-1}};
                    if(next2[0][0]==n-1&&next2[0][1]==n-2&&next2[1][0]==n-1&&next2[1][1]==n-1){
                        return ans;
                    }else if(check(visited,next2,grid)&&grid[cur[0][0]+1][cur[0][1]]!=1&&grid[cur[1][0]+1][cur[1][1]]!=1){
                        que.offer(next2);
                    }
                }
                if(cur[0][1]==cur[1][1]){
                    int[][] next2=new int[][]{cur[0].clone(),{cur[0][0],cur[0][1]+1}};
                    if(next2[0][0]==n-1&&next2[0][1]==n-2&&next2[1][0]==n-1&&next2[1][1]==n-1){
                        return ans;
                    }else if(check(visited,next2,grid)&&grid[cur[0][0]][cur[0][1]+1]!=1&&grid[cur[1][0]][cur[1][1]+1]!=1){
                        que.offer(next2);
                    }
                }
                int[][] next3=new int[][]{{cur[0][0],cur[0][1]+1},{cur[1][0],cur[1][1]+1}};
                if(next3[0][0]==n-1&&next3[0][1]==n-2&&next3[1][0]==n-1&&next3[1][1]==n-1){
                    return ans;
                }else if(check(visited,next3,grid)){
                    que.offer(next3);
                }
                int[][] next4=new int[][]{{cur[0][0]+1,cur[0][1]},{cur[1][0]+1,cur[1][1]}};
                if(next4[0][0]==n-1&&next4[0][1]==n-2&&next4[1][0]==n-1&&next4[1][1]==n-1){
                    return ans;
                }else if(check(visited,next4,grid)){
                    que.offer(next4);
                }
            }
        }
        return -1;
    }
    private boolean check(Set<String> visited,int[][] next,int[][] grid){
        int r1=next[0][0],c1=next[0][1],r2=next[1][0],c2=next[1][1],n=grid.length;
        if(r1<0||r1>=n||c1<0||c1>=n||r2<0||r2>=n||c2<0||c2>=n){
            return false;
        }
        String key=""+next[0][0]+","+next[0][1]+","+next[1][0]+","+next[1][1];
        if(visited.contains(key) ||grid[next[0][0]][next[0][1]]==1||grid[next[1][0]][next[1][1]]==1){
            return false;
        }
        visited.add(key);
        return true;
    }
}
