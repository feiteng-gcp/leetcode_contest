class Solution {
    public int minimumMoves(int[][] grid) {
        int n = grid.length;
        Queue<Snake> q = new LinkedList<>();
        Snake origin = new Snake(0,0,0,1,0);
        q.offer(origin);
        Set<List<Integer>> visited = new HashSet<>();
        visited.add(snakeToList(origin));
        int res = -1;
        while(!q.isEmpty())
        {
            res++;
            int size = q.size();
            while(size-- > 0)
            {
                Snake cur = q.poll();
                if(cur.leftx == n-1 && cur.lefty == n-2 && cur.rightx == n-1 && cur.righty == n-1)
                    return res;
                if(cur.state == 0)
                {
                    if(cur.righty + 1 < n && grid[cur.leftx][cur.righty+1] != 1)
                    {
                        Snake temp = new Snake(cur.leftx, cur.lefty + 1, cur.rightx, cur.righty + 1, 0);
                        
if(!visited.contains(snakeToList(temp)))
                        {
                            q.offer(temp);
                            visited.add(snakeToList(temp));
                        }
                    }
                    if(cur.leftx + 1 < n && grid[cur.leftx+1][cur.lefty] != 1 && grid[cur.rightx+1][cur.righty] != 1)
                    {
                        Snake temp = new Snake(cur.leftx+1, cur.lefty, cur.rightx+1, cur.righty, 0);
                        
if(!visited.contains(snakeToList(temp)))
                        {
                            q.offer(temp);
                            visited.add(snakeToList(temp));
                        }
                        Snake tempr = new Snake(cur.leftx, cur.lefty, cur.leftx+1, cur.lefty, 1);
                        
if(!visited.contains(snakeToList(tempr)))
                        {
                            q.offer(tempr);
                            visited.add(snakeToList(tempr));
                        }
                    }
                }
                else
                {
                    if(cur.rightx + 1 < n && grid[cur.rightx+1][cur.righty] != 1)
                    {
                        Snake temp = new Snake(cur.leftx+1, cur.lefty, cur.rightx+1, cur.righty, 1);
                        
if(!visited.contains(snakeToList(temp)))
                        {
                            q.offer(temp);
                            visited.add(snakeToList(temp));
                        }
                    }
                    if(cur.lefty+1 < n && grid[cur.leftx][cur.lefty+1] != 1 && grid[cur.rightx][cur.righty+1] != 1)
                    {
                        Snake temp = new Snake(cur.leftx, cur.lefty+1, cur.rightx, cur.righty+1, 1);
                        
if(!visited.contains(snakeToList(temp)))
                        {
                            q.offer(temp);
                            visited.add(snakeToList(temp));
                        }
                        Snake tempr = new Snake(cur.leftx, cur.lefty, cur.leftx, cur.lefty + 1, 0);
                        
if(!visited.contains(snakeToList(tempr)))
                        {
                            q.offer(tempr);
                            visited.add(snakeToList(tempr));
                        }
                    }
                }
            }
            
        }
        return -1;
    }
    
    private List<Integer> snakeToList(Snake s)
    {
        List<Integer> list = new ArrayList<>();
        list.add(s.leftx);
        list.add(s.lefty);
        list.add(s.rightx);
        list.add(s.righty);
        list.add(s.state);
        return list;
    }
    
    class Snake
    {
        int leftx;
        int lefty;
        int rightx;
        int righty;
        int state;
        public Snake(int leftx, int lefty, int rightx, int righty, int state)
        {
            this.leftx = leftx;
            this.lefty = lefty;
            this.rightx = rightx;
            this.righty = righty;
            this.state = state;
        }
    }
}
