class Solution {
  public boolean uniqueOccurrences(int[] arr) {
    Map<Integer, Integer> m = new HashMap<>();
    for(int v : arr) {
      if(!m.containsKey(v)) m.put(v, 0);
      m.put(v, m.get(v)+1);
    }
    HashSet<Integer> l = new HashSet<>(m.values());
    return l.size() == m.size();
  }
}
