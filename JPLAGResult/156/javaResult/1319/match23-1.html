class Solution {
    public boolean uniqueOccurrences(int[] arr) {
        int v[] = new int[2001];
        for(int a:arr){
            v[a+1000]++;
        }
        boolean vis[] =new boolean[1001];
        for(int s:v){
            if(s>0){
                if(vis[s]) return false;
                vis[s] = true;
            }
        }
        return true;
        
    }
}
