class Solution {
    public boolean uniqueOccurrences(int[] arr) {
        Map<Integer, Integer> map = new HashMap();
        for(int num: arr){
            map.put(num, map.getOrDefault(num,0)+1);
        }
        Set<Integer> cnt = new HashSet();
        for(int num:map.values()){
            if(cnt.contains(num)) return false;
            cnt.add(num);
        }
        return true;
        
    }
}
