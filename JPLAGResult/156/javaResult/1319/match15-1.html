class Solution {
    public boolean uniqueOccurrences(int[] arr) {
        Map<Integer,Integer> map = new HashMap<>();
        for(int num:arr){
            Integer val = map.getOrDefault(num, 0);
            map.put(num,val+1);
        }
        Set<Integer> set = new HashSet<>();
        for(int num:map.values()){
            if(set.contains(num)){
                return false;
            }
            set.add(num);
        }
        return true;
    }
}
