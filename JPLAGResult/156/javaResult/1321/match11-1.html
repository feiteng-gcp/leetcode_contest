class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        int cost = 0;
        int prev = 0;
        int ans = 0;
        for (int i = 0; i < s.length(); i++) {
            cost += Math.abs(s.charAt(i)-t.charAt(i));
            while(prev <= i && cost > maxCost) {
                int dt = 
Math.abs(s.charAt(prev)-t.charAt(prev));
                cost -= dt;
                prev++;
            }
            ans = Math.max(i-prev+1, ans);
        }
        return ans;
    }
}
