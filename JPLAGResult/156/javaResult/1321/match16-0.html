class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        int N = s.length();
        int[] cost = new int[N];
        for (int i = 0; i < N; i++) {
            cost[i] = Math.abs(s.charAt(i) - t.charAt(i));
        }
        int res = 0;
        int currCost = 0;
        for (int head = 0, tail = 0; head < N; head++) {
            currCost += cost[head];
            while (currCost > maxCost) {
                currCost -= cost[tail++];
            }
            res = Math.max(res, head - tail + 1);
        }
        return res;
    }
}
