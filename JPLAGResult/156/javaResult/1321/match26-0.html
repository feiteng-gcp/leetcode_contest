class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        int start = 0, end = 0, cost = 0, res = 0;
        while (end < s.length()) {
            cost += Math.abs(s.charAt(end) - t.charAt(end));
            while (cost > maxCost) {
                cost -= Math.abs(s.charAt(start) - t.charAt(start));
                start++;
            }
            res = Math.max(res, end - start + 1);
            end++;
        }
        return res;
    }
}
