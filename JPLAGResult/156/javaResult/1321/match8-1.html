class Solution {
    public int equalSubstring(String s, String t, int c) {
        char cs[] = s.toCharArray();
        char ts[] = t.toCharArray();
        
        int l = cs.length;
        
int sum = 0;
        int st  = 0;
        int r =0;
        for(int e=0;e<l;++e){
            sum += Math.abs(cs[e]-ts[e]);
            while(sum>c){
                sum -= Math.abs(cs[st]-ts[st]);
                st++;
            }
            r = Math.max(r,e-st+1);
        }
        return r;
        
    }
}
