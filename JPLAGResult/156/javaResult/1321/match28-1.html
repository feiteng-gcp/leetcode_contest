class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        
        if(maxCost < 0)
            return 0;
        
        
int n = s.length();
        int[] cost = new int[n];
        
        for(int i = 0; i < n; i++)
        {
            cost[i] = Math.abs(s.charAt(i) - t.charAt(i));
        }
        
        int res = 0;
        
        int i = 0;
        int sum = 0;
        for(
int j = 0; j < n; j++)
        {
            sum += cost[j];
            while(sum > maxCost)
            {
                sum -= cost[i++];
            }
            
            res = Math.max(res, j - i + 1);
        }
        
        return res;
    }
}
