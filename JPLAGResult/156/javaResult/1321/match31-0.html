class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        char[] cs = s.toCharArray(), ct = t.toCharArray();
        int n = cs.length;
        int[] prefix = new int[n + 1];
        for (int i = 0; i < n; ++i) {
            prefix[i + 1] = prefix[i] + Math.abs(cs[i] - ct[i]);
        }
        int max = 0;
        for (int i = 1; i <= n; ++i) {
            int target = prefix[i] - maxCost;
            int lo = 0, hi = i + 1;
            while (lo < hi) {
                int mid = lo + ((hi - lo) >> 1);
                if (target <= prefix[mid]) hi = mid;
                else lo = mid + 1;
            }
            max = Math.max(max, i - lo);
        }
        return max;
    }
}
