class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        
        int l = 0, r = 0;
        int max = 0;
        while(r < 
s.length()) {
            maxCost -= Math.abs(s.charAt(r) - t.charAt(r));
            
            while(maxCost < 0 && l <= r) {
                maxCost += Math.abs(s.charAt(l) - t.charAt(l));
                l++;
            }
            
            max = Math.max(r - l + 1, max);
            r++;
        }
        return max;
    }
}
