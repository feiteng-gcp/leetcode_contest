class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        int len = 0;
        int l = 0;
        int cost = 0;
        for(int r = 0;r < t.length();r++) {
            cost += Math.abs(t.charAt(r) - s.charAt(r));
            while(cost > maxCost) {
                cost -= Math.abs(t.charAt(l) - s.charAt(l));
                l++;
            }
            len = Math.max(len,r - l+1);
        }
        return len;
    }
}
