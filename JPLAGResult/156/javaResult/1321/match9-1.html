class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        int n = s.length();
        int[] cost = new int[n];
        for (int i = 0; i < n; i++) {
            cost[i] = Math.abs((int)s.charAt(i) - (int)t.charAt(i));
        }
        int start = 0;
        int end = 0;
        int sum = 0;
        int output = 0;
        while (end < n) {
            sum += cost[end];
            end++;
            while (sum > maxCost) {
                sum -= cost[start];
                start++;
            }
            output = Math.max(output, end - start);
        }
        return output;
    }
}
