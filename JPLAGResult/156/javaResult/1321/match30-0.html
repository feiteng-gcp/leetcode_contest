class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        
        char[] cs = s.toCharArray(); 
        char[] ct = t.toCharArray(); 
        int ret = 0; 
        int cost = 0; 
        int j = 0; 
        for (int i = 0; i < cs.length; i++) {
            cost += Math.abs(cs[i] - ct[i]);
            while (cost > maxCost) {
                cost -= Math.abs(ct[j] - cs[j]); 
                j++; 
            }
            ret = Math.max(ret, i - j + 1);
        }
        return ret;
    }
}
