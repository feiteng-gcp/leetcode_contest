class Solution {
    public int equalSubstring(String s, String t, int maxCost) {
        int n = s.length();
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = Math.abs((int)(s.charAt(i) - 
t.charAt(i)));
        }
        
        // Map<Integer, Integer> map = new HashMap<>();
        int sum = 0, left = 0;
        int max = 0;
        // map.put(0, -1);
        for (int i = 0; i < n; i++) {
            sum += arr[i];
            while (sum > maxCost) {
                sum -= arr[left++];
            }
            max = Math.max(max, i - left + 1);
        }
        return max;
    }
}
