class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        unordered_map<int, int> m;
        for(auto num : arr){
            m[num] += 1;
        }
        unordered_set<int> m2;
        for(auto k_v : m){
            if(m2.find(k_v.second) != m2.end()) return false;
            m2.insert(k_v.second);
        }
        return true;
    }
};
