class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        unordered_map<int, int> freq;
        for(auto a : arr)
            freq[a]++;
        unordered_set<int> visited;
        for(auto f : freq){
            if(visited.count(f.second))
                return false;
            visited.insert(f.second);
        }
        return true;
    }
};
