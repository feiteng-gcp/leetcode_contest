class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
		map<int, int> R;
		for (auto x : arr) ++R[x];

		set<int> S;
		for (auto p : R) {
			if (S.count(p.second)) return false;
			S.insert(p.second);
		}
		return true;
    }
};
