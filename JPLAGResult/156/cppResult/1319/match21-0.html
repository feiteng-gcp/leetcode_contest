class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        unordered_map<int,int> freq;
        unordered_set<int> count;
        for (int a : arr) ++freq[a];
        
        for (auto e : freq) {
            if (count.find(e.second) != count.end()) return false;
            count.insert(e.second);
        }
        
        return true;
    }
};
