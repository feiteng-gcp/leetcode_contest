class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        map<int, int> v;
        for (auto& x: arr)
            ++v[x];
        
        set<int> q;
        for (auto& y: v)
        {
            if (q.find(y.second) != q.end())
                return false;
            
            q.insert(y.second);
        }
        
        return true;
    }
};
