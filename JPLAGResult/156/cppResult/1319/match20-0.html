class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        unordered_map<int, int> mm;
        unordered_set<int> ms;
        for (int x : arr) ++mm[x];
        for (auto& p : mm) {
            if (ms.count(p.second)) return false;
            ms.insert(p.second);
        }
        return true;
    }
};
