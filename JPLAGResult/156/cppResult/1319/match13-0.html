class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        unordered_map<int,int> m;
        for(auto a: arr)
            m[a]++;
        
        unordered_set<int> s;
        for(auto _m:m){
            if(s.find(_m.second) != s.end())
                return false;
            s.insert(_m.second);
        }
        return true;
    }
};
