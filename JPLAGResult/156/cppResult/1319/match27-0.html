class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        map<int, int> mp;
        for (int x : arr) {
            mp[x]++;
        }
        set<int> seen;
        for(auto x : mp) {
            if (seen.count(x.second)) {
                return false;
            }
            seen.insert(x.second);
        }
        return true;
    }
};
