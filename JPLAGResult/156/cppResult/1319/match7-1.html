class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        unordered_map<int, int> cnt;
        for (int a : arr) cnt[a] ++;
        unordered_set<int> s;
        for (auto& p : cnt) {
            if (s.count(p.second))
                return false;
            s.insert(p.second);
        }
        return true;
    }
};
