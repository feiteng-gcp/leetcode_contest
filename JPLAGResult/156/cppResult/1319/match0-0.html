class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        map<int,int> counts;
        for(int x : arr)
            counts[x]++;
        set<int> unique_counts;
        for(auto x : counts) {
            if (unique_counts.find(x.second) != unique_counts.end())
                return false;
            unique_counts.insert(x.second);
        }
        return true;
    }
};
