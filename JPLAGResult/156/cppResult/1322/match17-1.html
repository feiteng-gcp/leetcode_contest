class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        int rows = grid.size();
        int cols = grid[0].size();
        vector<vector<int>> hor(rows, vector<int>(cols-1, -1));
        hor[0][0] = 0;
        vector<vector<int>> ver(rows-1, vector<int>(cols, -1));
        set<vector<int>> currHor;
        currHor.insert({0,0});
        set<vector<int>> currVer;
        
        helper(grid, hor, ver, currHor, currVer, 0);
        return hor[rows-1][cols-2]; 
    }
    
    void helper(vector<vector<int>>& grid, vector<vector<int>>& hor, vector<vector<int>>& ver, 
                set<vector<int>> currHor, set<vector<int>> currVer, int m) {
        // cout << "m =" << m;
        // for(auto it = currHor.begin(); it!= currHor.end(); ++it) {
        //     cout << "\thor["<<(*it)[0]<<","<<(*it)[1]<<"]";  
        // }
        // for(auto it = currVer.begin(); it!= currVer.end(); ++it) {
        //     cout << "\tver["<<(*it)[0]<<","<<(*it)[1]<<"]";  
        // }
        // cout << endl;
        int rows = grid.size();
        int cols = grid[0].size();
        if(hor[rows-1][cols-2] != -1) {
            return;
        }
        if(currHor.size() == 0 && currVer.size() == 0) {
            return;
        }
        ++m;
        set<vector<int>> nextHor;
        set<vector<int>> nextVer;
        for(auto it = currHor.begin(); it!= currHor.end(); ++it) {
            int row = (*it)[0];
            int col = (*it)[1];

            if(col+2 < cols && grid[row][col+2] == 0) {
                nextHor.insert({row, col+1});
                hor[row][col+1] = m;
            }
            if(row+1 < rows && grid[row+1][col] == 0 && grid[row+1][col+1] == 0) {
                nextHor.insert({row+1, col});
                hor[row+1][col] = m;
            }
            if(row+1 < rows && col+1 < cols && grid[row+1][col] == 0 && grid[row+1][col+1] == 0 && ver[row][col] == -1) {
                nextVer.insert({row, col});
                ver[row][col] = m;
            }
        }
        for(auto it = currVer.begin(); it!= currVer.end(); ++it) {
            int row = (*it)[0];
            int col = (*it)[1];

            if(row+2 < rows && grid[row+2][col] == 0) {
                nextVer.insert({row+1, col});
                ver[row+1][col] = m;
            }
            if(col+1 < cols && grid[row][col+1] == 0 && grid[row+1][col+1] == 0) {
                nextVer.insert({row, col+1});
                ver[row][col+1] = m;
            }
            if(row+1 < rows && col+1 < cols && grid[row][col+1] == 0 && grid[row+1][col+1] == 0 && hor[row][col] == -1) {
                nextHor.insert({row, col});
                hor[row][col] = m;
            }
        }
        helper(grid, hor, ver, nextHor, nextVer, m);
        
    }
};
