class Solution {
public:
    int minimumMoves(vector<vector<int>>& g) {
        queue<array<int, 4>> q;
        
        int n = g.size();
        
        array<int, 4> dest = {n-1, n-2,n-1, n-1};
        
        
        q.push({0, 0, 0, 1});
        
        set<array<int, 4>> ex;
        
        
        int step = 0;
        while (!q.empty()) {
            
            for (int s = q.size(); s > 0; s--) {
                auto r = q.front(); q.pop();
                
                int y1 = r[0], x1 = r[1];
                int y2 = r[2], x2 = r[3];
                
                if (ex.count(r)) continue;
                if (r == dest) return step;
                ex.insert(r);
                

                if (y1==y2) { // 横
                    if (x2 < n-1 && g[y2][x2+1] == 0) { // 向右
                        q.push({y1, x1+1, y2, x2+1});
                    }
                    if (y1 < n-1 && g[y1+1][x1] == 0 && g[y2+1][x2] == 0) { 
                        q.push({y1+1, x1, y2+1, x2});// 向下
                        q.push({y1, x1, y2+1, x2-1});//顺时针
                    } 
                } else { // 竖
                    if (y2 < n-1 && g[y2+1][x2] == 0) {
                        q.push({y1+1, x1, y2+1, x2});
                    }
                    if (x2 < n-1 && g[y1][x1+1] == 0 && g[y2][x2+1] == 0) {
                        q.push({y1, x1+1, y2, x2+1}); // 向右
                        q.push({y1, x1, y2-1, x2+1});
                    }
                }
                
            }
            step++;
        }
        return -1;
        
    }
};
