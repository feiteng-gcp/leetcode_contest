const int dx[] = { -1, 0, 1, 0 };
const int dy[] = { 0, 1, 0, -1 };

class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
		int n = grid.size(), m = grid[0].size();
		vector<vector<int> > h(n, vector<int>(m, INT_MAX));
		vector<vector<int> > v(n, vector<int>(m, INT_MAX));
		h[0][1] = 0;	

		queue<tuple<int, int, int> > que;
		que.push(make_tuple(0, 0, 1));	
		while (!que.empty()) {

			int d, x, y;
			tie(d, x, y) = que.front();
			que.pop();
			if (d == 0) {
				// right
				if (y + 1 < m && !grid[x][y + 1] && h[x][y + 1] == INT_MAX) {
					h[x][y + 1] = h[x][y] + 1;
					que.push(make_tuple(0, x, y + 1));
				}
				// down
				if (x + 1 < n && !grid[x + 1][y - 1] && !grid[x + 1][y] &&
						h[x + 1][y] == INT_MAX) {
					h[x + 1][y] = h[x][y] + 1;
					que.push(make_tuple(0, x + 1, y));	
				}
				if (y > 0 && x + 1 < n && !grid[x + 1][y - 1] && !grid[x + 1][y]
						&& v[x + 1][y - 1] == INT_MAX) {
					v[x + 1][y - 1] = h[x][y] + 1;
					que.push(make_tuple(1, x + 1, y - 1));
				}
			} else {
				// down
				if (x + 1 < n && !grid[x + 1][y] && v[x + 1][y] == INT_MAX) {
					v[x + 1][y] = v[x][y] + 1;
					que.push(make_tuple(1, x + 1, y));
				}
				// right
				if (y + 1 < m && !grid[x][y + 1] && !grid[x - 1][y + 1]
						&& v[x][y + 1] == INT_MAX) {
					v[x][y + 1] = v[x][y] + 1;
					que.push(make_tuple(1, x, y + 1));
				}
				if (x > 0 && y + 1 < m && !grid[x - 1][y + 1] && !grid[x][y + 1]
						&& h[x - 1][y + 1] == INT_MAX) {
					h[x - 1][y + 1] = v[x][y] + 1;
					que.push(make_tuple(0, x - 1, y + 1));
				}
			}
		}
		return h[n - 1][n - 1] < INT_MAX ? h[n - 1][n - 1] : -1;	
    }
};
