class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        int n=grid.size();
        queue<pair<pair<int,int>, int>> q;
        q.emplace(make_pair(0,0), 0); //hor
        vector<vector<int>> vis(n,vector<int>(n,-1)); // hor
        vector<vector<int>> vis2(n,vector<int>(n,-1));
        vis[0][0]=0;
        while(!q.empty()){
            auto [it, ang]=q.front();q.pop();
            int x=it.first,y=it.second;
            // cout<<x<<' '<<y<<endl;
            switch(ang){
                case 0://hor
                    if(y<n-2 && grid[x][y+2]==0 && vis[x][y+1]==-1){
                        vis[x][y+1] = vis[x][y]+1;
                        q.emplace(make_pair(x,y+1), 0);
                    }
                    if(x<n-1 && grid[x+1][y]==0 && grid[x+1][y+1]==0 && vis[x+1][y]==-1){
                        vis[x+1][y] = vis[x][y]+1;
                        q.emplace(make_pair(x+1,y), 0);
                        
                    }
                    if(x<n-1 && grid[x+1][y]==0 && grid[x+1][y+1]==0 && vis2[x][y]==-1){
                        vis2[x][y] = vis[x][y]+1;
                        q.emplace(make_pair(x,y), 1);
                    }
                    break;
                case 1:
                    if(x<n-2 && grid[x+2][y]==0 && vis2[x+1][y]==-1){
                        vis2[x+1][y] = vis2[x][y]+1;
                        q.emplace(make_pair(x+1,y), 1);
                    }
                    if(y<n-1 && grid[x][y+1]==0 && grid[x+1][y+1]==0 && vis2[x][y+1]==-1){
                        vis2[x][y+1] = vis2[x][y]+1;
                        q.emplace(make_pair(x,y+1), 1);
                        
                    }
                    if(y<n-1 && grid[x][y+1]==0 && grid[x+1][y+1]==0 && vis[x][y]==-1){
                        vis[x][y] = vis2[x][y]+1;
                        q.emplace(make_pair(x,y), 0);
                    }
                    break;
            }
        }
        if(vis[n-1][n-2]==-1) return -1;
        // if(vis[n-1][n-2]==-1 || vis2[n-2][n-1]==-1) return max(vis[n-1][n-2] , vis2[n-2][n-1]);
        return vis[n-1][n-2];
        
    }
};
