class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        int n = grid.size();
        queue<pair<pair<int, int>, pair<int, int>>> q;
        q.push({{0, 0},{0, 1}});
        map<pair<pair<int, int>, pair<int, int>>, int> ds;
        ds[q.front()] = 0;
        auto exit = make_pair(make_pair(n - 1, n - 2), make_pair(n - 1, n - 1));
        while(!q.empty()) {
            auto v = q.front();
            q.pop();
            if(v == exit) return ds[v];
            bool hor = v.first.first == v.second.first;
            if(hor) {
                if(v.second.second + 1 < n && grid[v.second.first][v.second.second + 1] == 0) {
                    auto to = v;
                    to.first.second++;
                    to.second.second++;
                    if(!ds.count(to)) {
                        q.push(to);
                        ds[to] = ds[v] + 1;
                    }
                }
                if(v.first.first + 1 < n && grid[v.first.first + 1][v.first.second] == 0 && grid[v.second.first + 1][v.second.second] == 0) {
                    auto to = v;
                    to.first.first++;
                    to.second.first++;
                    if(!ds.count(to)) {
                        q.push(to);
                        ds[to] = ds[v] + 1;
                    }
                    to = v;
                    to.second.first = to.first.first + 1;
                    to.second.second = to.first.second;
                    if(!ds.count(to)) {
                        q.push(to);
                        ds[to] = ds[v] + 1;
                    }
                }
            } else {
                if(v.second.first + 1 < n && grid[v.second.first + 1][v.second.second] == 0) {
                    auto to = v;
                    to.first.first++;
                    to.second.first++;
                    if(!ds.count(to)) {
                        q.push(to);
                        ds[to] = ds[v] + 1;
                    }
                }
                if(v.first.second + 1 < n && grid[v.first.first][v.first.second + 1] == 0 && grid[v.second.first][v.second.second + 1] == 0) {
                    auto to = v;
                    to.first.second++;
                    to.second.second++;
                    if(!ds.count(to)) {
                        q.push(to);
                        ds[to] = ds[v] + 1;
                    }
                    to = v;
                    to.second.second = to.first.second + 1;
                    to.second.first = to.first.first;
                    if(!ds.count(to)) {
                        q.push(to);
                        ds[to] = ds[v] + 1;
                    }
                }
            }
        }
        return -1;
    }
};
