int dp[105][105][2]; // 0 横着 , 1 竖着
class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        int n = grid.size();
        for(int i=0;i<105;++i)for(int j=0;j<105;++j)for(int k=0;k<2;++k)dp[i][j][k] = 1e4;
        
        if(grid[0][0]==0 && grid[0][1]==0){
            dp[0][1][0] = 0;
        }
        
        for(int i=0;i<n;++i){
            for(int j=0;j<n;++j){ //col
                
                if(i==0&&j==0) continue;
                if(i==0&&j==1) continue;
                
                if(grid[i][j]==0){
                    
                    // 蛇横 横走
                    if(j>0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i][j-1][0] + 1);
                    }
                    
                    // 蛇竖 竖走
                    if(i>0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i-1][j][1] + 1);
                    }
                    
                    // 蛇竖 横走
                    if(j>0 && i>0 && grid[i-1][j]==0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i][j-1][1] + 1);
                    }
                    
                    // 蛇横 竖走
                    if(i>0 && j>0 && grid[i][j-1]==0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i-1][j][0] + 1);                        
                    }

                    // 蛇顺时针转动
                    if(i>0 && j+1<n && grid[i][j+1]==0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i-1][j+1][0] + 1);
                    }
                    
                     // 蛇逆时针转动
                    if(j>0 && i+1<n && grid[i+1][j]==0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i+1][j-1][1] + 1); 
                    }

                }
                
            }
        }
        for(int i=0;i<n;++i){
            for(int j=0;j<n;++j){ //col
                
                if(i==0&&j==0) continue;
                if(i==0&&j==1) continue;
                
                if(grid[i][j]==0){
                    
                    // 蛇横 横走
                    if(j>0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i][j-1][0] + 1);
                    }
                    
                    // 蛇竖 竖走
                    if(i>0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i-1][j][1] + 1);
                    }
                    
                    // 蛇竖 横走
                    if(j>0 && i>0 && grid[i-1][j]==0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i][j-1][1] + 1);
                    }
                    
                    // 蛇横 竖走
                    if(i>0 && j>0 && grid[i][j-1]==0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i-1][j][0] + 1);                        
                    }

                    // 蛇顺时针转动
                    if(i>0 && j+1<n && grid[i][j+1]==0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i-1][j+1][0] + 1);
                    }
                    
                     // 蛇逆时针转动
                    if(j>0 && i+1<n && grid[i+1][j]==0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i+1][j-1][1] + 1); 
                    }

                }
                
            }
        }
        for(int i=0;i<n;++i){
            for(int j=0;j<n;++j){ //col
                
                if(i==0&&j==0) continue;
                if(i==0&&j==1) continue;
                
                if(grid[i][j]==0){
                    
                    // 蛇横 横走
                    if(j>0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i][j-1][0] + 1);
                    }
                    
                    // 蛇竖 竖走
                    if(i>0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i-1][j][1] + 1);
                    }
                    
                    // 蛇竖 横走
                    if(j>0 && i>0 && grid[i-1][j]==0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i][j-1][1] + 1);
                    }
                    
                    // 蛇横 竖走
                    if(i>0 && j>0 && grid[i][j-1]==0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i-1][j][0] + 1);                        
                    }

                    // 蛇顺时针转动
                    if(i>0 && j+1<n && grid[i][j+1]==0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i-1][j+1][0] + 1);
                    }
                    
                     // 蛇逆时针转动
                    if(j>0 && i+1<n && grid[i+1][j]==0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i+1][j-1][1] + 1); 
                    }

                }
                
            }
        }
        for(int i=0;i<n;++i){
            for(int j=0;j<n;++j){ //col
                
                if(i==0&&j==0) continue;
                if(i==0&&j==1) continue;
                
                if(grid[i][j]==0){
                    
                    // 蛇横 横走
                    if(j>0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i][j-1][0] + 1);
                    }
                    
                    // 蛇竖 竖走
                    if(i>0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i-1][j][1] + 1);
                    }
                    
                    // 蛇竖 横走
                    if(j>0 && i>0 && grid[i-1][j]==0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i][j-1][1] + 1);
                    }
                    
                    // 蛇横 竖走
                    if(i>0 && j>0 && grid[i][j-1]==0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i-1][j][0] + 1);                        
                    }

                    // 蛇顺时针转动
                    if(i>0 && j+1<n && grid[i][j+1]==0){
                        dp[i][j][1] = min(dp[i][j][1],dp[i-1][j+1][0] + 1);
                    }
                    
                     // 蛇逆时针转动
                    if(j>0 && i+1<n && grid[i+1][j]==0){
                        dp[i][j][0] = min(dp[i][j][0],dp[i+1][j-1][1] + 1); 
                    }

                }
                
            }
        }
        /*
        for(int i=0;i<n;++i){
            for(int j=0;j<n;++j){
                cout<<dp[i][j][0]<<" ";
            }
            cout<<endl;
        }
        cout<<"==="<<endl;
        for(int i=0;i<n;++i){
            for(int j=0;j<n;++j){
                cout<<dp[i][j][1]<<" ";
            }
            cout<<endl;
        }  
        */ 
            

        if(dp[n-1][n-1][0]>=1e4){
            return -1;
        }else{
            return dp[n-1][n-1][0];
        }
    }
};
