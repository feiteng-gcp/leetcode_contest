class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        int n = grid.size();
        
        queue<vector<int>> q;
        map<vector<int>, int> dist;
        
        vector<int> st = {0, 0, 0, 1};
        q.push(st);
        dist[st] = 0;
        
        vector<int> end = {n - 1, n - 2, n - 1, n - 1};
        while (!q.empty()) {
            vector<int> cur = q.front();
            q.pop();
            
            if (cur == end) {
                return dist[cur];
            }
            
            // horizontal
            if (cur[0] == cur[2]) {
                if (cur[3] + 1 < n && grid[cur[0]][cur[3] + 1] == 0) {
                    vector<int> temp = cur;
                    temp[1] = temp[1] + 1;
                    temp[3] = temp[3] + 1;
                    if (dist.find(temp) == dist.end()) {
                        q.push(temp);
                        dist[temp] = dist[cur] + 1;
                    }
                }
                
                if (cur[0] + 1 < n && grid[cur[0] + 1][cur[1]] == 0 && grid[cur[2] + 1][cur[3]] == 0) {

                    vector<int> temp = cur;
                    temp[2] = temp[0] + 1;
                    temp[3] = temp[1];
                    if (dist.find(temp) == dist.end()) {
                        q.push(temp);
                        dist[temp] = dist[cur] + 1;
                    }
                    
                    temp = cur;
                    temp[0] = temp[0] + 1;
                    temp[2] = temp[2] + 1;
                    if (dist.find(temp) == dist.end()) {
                        q.push(temp);
                        dist[temp] = dist[cur] + 1;
                    }
                }
            }
            
            //vertical
            if (cur[1] == cur[3]) {
                if (cur[2] + 1 < n && grid[cur[2] + 1][cur[1]] == 0) {
                    vector<int> temp = cur;
                    temp[0] = temp[0] + 1;
                    temp[2] = temp[2] + 1;
                    if (dist.find(temp) == dist.end()) {
                        q.push(temp);
                        dist[temp] = dist[cur] + 1;
                    }
                }
                
                if (cur[1] + 1 < n && grid[cur[0]][cur[1] + 1] == 0 && grid[cur[2]][cur[3] + 1] == 0) {

                    vector<int> temp = cur;
                    temp[2] = temp[0];
                    temp[3] = temp[1] + 1;
                    if (dist.find(temp) == dist.end()) {
                        q.push(temp);
                        dist[temp] = dist[cur] + 1;
                    }
                    
                    temp = cur;
                    temp[1] = temp[1] + 1;
                    temp[3] = temp[3] + 1;
                    if (dist.find(temp) == dist.end()) {
                        q.push(temp);
                        dist[temp] = dist[cur] + 1;
                    }
                }
            }
        }
        
        return -1;
    }
};
