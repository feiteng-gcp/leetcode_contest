class Solution {
public:
  int minimumMoves(vector<vector<int>>& grid) {
    int n = grid.size();
    unordered_set<int> visited;
    queue<vector<int>> pos_queue;
    vector<int> target = {n - 1, n - 2, n - 1, n - 1};
    vector<int> start = {0, 0, 0, 1};
    pos_queue.push(start);
    int steps = 0;
    while (!pos_queue.empty()) {
      int queue_size = pos_queue.size();
      for (int i = 0; i < queue_size; i++) {
        auto pos = pos_queue.front();
        if (pos == target) {
          return steps;
        }
        pos_queue.pop();
        int id = getID(pos, n);
        if (visited.find(id) != visited.end()) {
          continue;
        }
        visited.insert(id);
        int r = pos[2], c = pos[3];
        if (pos[0] == pos[2]) {
          if (c < n - 1 && grid[r][c + 1] == 0) {
            pos_queue.push({r, c, r, c + 1});
          }
          if (r < n - 1 && grid[r + 1][c] == 0 && grid[r + 1][c - 1] == 0) {
            pos_queue.push({r, c - 1, r + 1, c - 1});
            pos_queue.push({r + 1, c - 1, r + 1, c});
          }
        } else {
          if (r < n - 1 && grid[r + 1][c] == 0) {
            pos_queue.push({r, c, r + 1, c});
          }
          if (c < n - 1 && grid[r][c + 1] == 0 && grid[r - 1][c + 1] == 0) {
            pos_queue.push({r - 1, c, r - 1, c + 1});
            pos_queue.push({r - 1, c + 1, r, c + 1});
          }
        }
      }
      steps++;
    }
    return -1;
  }
  int getID(vector<int> &pos, int n) {
    return (pos[0] * n + pos[1]) * n * n + pos[2] * n + pos[3];
  }
};
