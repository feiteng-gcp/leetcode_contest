struct Position {
    int x, y, step;
    bool v;
};

class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        int n = grid.size();
        bool used[100][100][2]{};
        queue<Position> bfs;
        bfs.push(Position{1, 0, 0, 1});
        used[0][1][1] = true;
        while (!bfs.empty()) {
            auto front = bfs.front();
            bfs.pop();
            int x = front.x;
            int y = front.y;
            bool v = front.v;
            // cout << x << " " << y << " " << v << endl;
            if (x == n-1 && y == n-1 && v) return front.step;

            if (v) {
                if (x < n - 1 && !grid[y][x+1] && !used[y][x+1][1]) {
                    bfs.push(Position{x+1, y, front.step+1, 1});
                    used[y][x+1][1] = true;
                }
                if (y < n - 1 && !grid[y+1][x] && !grid[y+1][x-1] && !used[y+1][x][1]) {
                    bfs.push(Position{x, y+1, front.step+1, 1});
                    used[y+1][x][1] = true;
                }
                if (y < n - 1 && !grid[y+1][x-1] && !grid[y+1][x] && !used[y+1][x-1][0]) {
                    bfs.push(Position{x-1, y+1, front.step+1, 0});
                    used[y+1][x-1][0] = true;
                }
            } else {
                if (x < n - 1 && !grid[y][x+1] && !grid[y-1][x+1] && !used[y][x+1][0]) {
                    bfs.push(Position{x+1, y, front.step+1, 0});
                    used[y][x+1][0] = true;
                }
                if (y < n - 1 && !grid[y+1][x] && !used[y+1][x][0]) {
                    bfs.push(Position{x, y+1, front.step+1, 0});
                    used[y+1][x][0] = true;
                }
                if (x < n - 1 && !grid[y-1][x+1] && !grid[y][x+1] && !used[y-1][x+1][1]) {
                    bfs.push(Position{x+1, y-1, front.step+1, 1});
                    used[y-1][x+1][1] = true;
                }
            }
        }
        return -1;
    }
};
