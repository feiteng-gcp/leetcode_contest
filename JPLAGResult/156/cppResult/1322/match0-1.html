class Solution {
public:
    struct node{
        int x,y,head,cnt;
        bool operator <(const node &tmp)const{
            return cnt>tmp.cnt;
        }
    };
    bool vis[105][105][2];
    int minimumMoves(vector<vector<int>>& grid) {
        int n=grid.size();
        priority_queue<node>q;
        q.push({0,0,1,0});
        vis[0][0][1]=1;
        while(!q.empty()){
            node s = q.top(); q.pop();
            if(s.x==n-1&&s.y==n-2&&s.head==1)return s.cnt;
            if(s.head==1){
                if(s.y+2<n&&grid[s.x][s.y+2]==0&&!vis[s.x][s.y+1][s.head]){
                    vis[s.x][s.y+1][s.head]=1;
                    q.push({s.x,s.y+1,s.head,s.cnt+1});
                }    
                if(s.x+1<n&&grid[s.x+1][s.y]==0&&grid[s.x+1][s.y+1]==0&&!vis[s.x+1][s.y][s.head]){
                    vis[s.x+1][s.y][s.head]=1;
                    q.push({s.x+1,s.y,s.head,s.cnt+1});
                }
                if(s.x+1<n&&grid[s.x+1][s.y]==0&&grid[s.x+1][s.y+1]==0&&!vis[s.x][s.y][0]){
                    vis[s.x][s.y][0]=1;
                    q.push({s.x,s.y,0,s.cnt+1});
                }
            }
            else if(s.head==0){
                if(s.y+1<n&&grid[s.x][s.y+1]==0&&grid[s.x+1][s.y+1]==0&&!vis[s.x][s.y+1][0]){
                    vis[s.x][s.y+1][0]=1;
                    q.push({s.x,s.y+1,s.head,s.cnt+1});
                }    
                if(s.x+2<n&&grid[s.x+2][s.y]==0&&grid[s.x+2][s.y]==0&&!vis[s.x+1][s.y][0]){
                    vis[s.x+1][s.y][s.head]=1;
                    q.push({s.x+1,s.y,s.head,s.cnt+1});
                }
                if(s.y+1<n&&grid[s.x][s.y+1]==0&&grid[s.x+1][s.y+1]==0&&!vis[s.x][s.y][1]){
                    vis[s.x][s.y][1]=1;
                    q.push({s.x,s.y,1,s.cnt+1});
                }
            }
        }
        return -1;
    }
};
