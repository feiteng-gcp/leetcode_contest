class Solution {
    //position, heading(right: 1, down: 0)
    set<pair<int, int>> visited;
    queue<pair<int, int>> toExplore;
    int n;
public:
    int minimumMoves(vector<vector<int>>& grid) {
        n = grid.size();
        visited.insert(make_pair(0, 1));
        toExplore.push(make_pair(0, 1));
        pair<int, int> target = make_pair((n - 1) * n + n - 2, 1);
        int steps = 0;
        while(!toExplore.empty()){
            int cnt = toExplore.size();
            while(cnt--){
                auto curr = toExplore.front(); toExplore.pop();
                if(curr == target)
                    return steps;
                int r = curr.first / n;
                int c = curr.first % n;

                if(curr.second == 1){//heading right
                    if(c + 2 < n && grid[r][c + 2] == 0 && visited.count(make_pair(r * n + c + 1, 1)) == 0){//no blocking right
                        visited.insert(make_pair(r * n + c + 1, 1));
                        toExplore.push(make_pair(r * n + c + 1, 1));
                    }//moving right
                    if(r + 1 < n && grid[r + 1][c] == 0 && grid[r + 1][c + 1] == 0){
                        if(visited.count(make_pair((r + 1) * n + c, 1)) == 0){
                            visited.insert(make_pair((r + 1) * n + c, 1));
                            toExplore.push(make_pair((r + 1) * n + c, 1));
                        }//moving down
                        if(visited.count(make_pair(r * n + c, 0)) == 0){
                            visited.insert(make_pair(r * n + c, 0));
                            toExplore.push(make_pair(r * n + c, 0));
                        }//rotate down
                    }
                }else{ //heading down

                    if(r + 2 < n && grid[r + 2][c] == 0 && visited.count(make_pair((r + 1) * n + c, 0)) == 0){//no blocking
                        visited.insert(make_pair((r + 1)  * n + c, 0));
                        toExplore.push(make_pair((r + 1)  * n + c, 0));
                    }//moving down
                    if(c + 1 < n && grid[r][c + 1] == 0 && grid[r + 1][c + 1] == 0){
                        if(visited.count(make_pair(r * n + c + 1, 0)) == 0){//moving right
                            visited.insert(make_pair(r * n + c + 1, 0));
                            toExplore.push(make_pair(r * n + c + 1, 0));
                        }
                        if(visited.count(make_pair(r * n + c, 1)) == 0){//rotate up
                            visited.insert(make_pair(r * n + c, 1));
                            toExplore.push(make_pair(r * n + c, 1));
                        }
                    }
                }
                
            }
            steps++;
        }
        return -1;
    }
};
