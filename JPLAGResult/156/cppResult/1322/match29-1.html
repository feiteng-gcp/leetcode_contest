class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        // memo_right[r][c], memo_down[r][c];
        n = grid.size();
        
        
        vector<vector<int>> a(n, vector<int>(n, -1));
        vector<vector<int>> b(n, vector<int>(n, -1));
        memo_right = a;
        memo_down = b;
        int res = minimumMove(grid, 0, 0, true, false);
        return (res >= 1000) ? -1 : res;
    }
    
    int minimumMove(vector<vector<int>>& grid, int r, int c, bool isRight, bool hasRotated){
        //cout << r << ", " << c << endl;
        if (r == n-1 && c == n-2){
            return 0;
        }
        if (isRight && memo_right[r][c] != -1){
            return memo_right[r][c];
        }
        if (!isRight && memo_down[r][c] != -1){
            return memo_down[r][c];
        }
        int res = 1000;
        // move right

        if (isRight && c+2 < n && grid[r][c+2]==0){
            res = min(res, minimumMove(grid, r, c+1, isRight, false) + 1);
        }
        if (!isRight && c+2 < n && grid[r][c+1] == 0 && grid[r+1][c+1] == 0){
            res = min(res, minimumMove(grid, r, c+1, isRight, false) + 1);
        }
        // move down
        if (isRight && r+1 < n && grid[r+1][c]==0 && grid[r+1][c+1]==0){
            res = min(res, minimumMove(grid, r+1, c, isRight, false) + 1);
        }

        if (!isRight && r+2 < n && grid[r+2][c] == 0){
            res = min(res, minimumMove(grid, r+1, c, isRight, false) + 1);
        }
            
        // rotate
        if (!hasRotated && r+1<n && c+1<n && grid[r+1][c]==0 && grid[r][c+1]==0 && grid[r+1][c+1]==0){
            res = min(res, minimumMove(grid, r, c, !isRight, true) + 1);
        }
        
        if (isRight){
            memo_right[r][c] = res;
        }
        else{
            memo_down[r][c] = res;
        }
        return res;
        
    }
    
private:
    vector<vector<int>> memo_right;
    vector<vector<int>> memo_down;
    int n;
};
