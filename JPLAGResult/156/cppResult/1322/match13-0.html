class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        vector<vector<vector<int>>> hv(grid.size(), vector<vector<int>>(grid[0].size(), vector<int>(2, 0)));
        deque<vector<int>> queue;
        queue.push_back({0, 0, 0});
        hv[0][0][0] = 1;
        while (!queue.empty()) {
            const auto& f = queue.front();
            if (f[2] == 0) {
                // horizontal
                if (f[1] < grid[0].size() - 2 && grid[f[0]][f[1] + 2] == 0 && hv[f[0]][f[1] + 1][0] == 0) {
                    // right
                    hv[f[0]][f[1] + 1][0] = hv[f[0]][f[1]][0] + 1;
                    queue.push_back({f[0], f[1] + 1, 0});
                }
                if (f[0] < grid.size() - 1 && grid[f[0] + 1][f[1]] == 0 && grid[f[0] + 1][f[1] + 1] == 0 && hv[f[0] + 1][f[1]][0] == 0) {
                    // down
                    hv[f[0] + 1][f[1]][0] = hv[f[0]][f[1]][0] + 1;
                    queue.push_back({f[0] + 1, f[1], 0});
                }
                if (f[0] < grid.size() - 1 && grid[f[0] + 1][f[1]] == 0 && grid[f[0] + 1][f[1] + 1] == 0 && hv[f[0]][f[1]][1] == 0) {
                    // rotate
                    hv[f[0]][f[1]][1] = hv[f[0]][f[1]][0] + 1;
                    queue.push_back({f[0], f[1], 1});
                }
            } else {
                // vertical
                if (f[1] < grid[0].size() - 1 && grid[f[0]][f[1] + 1] == 0 && grid[f[0] + 1][f[1] + 1] == 0 && hv[f[0]][f[1] + 1][1] == 0) {
                    // right
                    hv[f[0]][f[1] + 1][1] = hv[f[0]][f[1]][1] + 1;
                    queue.push_back({f[0], f[1] + 1, 1});
                }
                if (f[0] < grid.size() - 2 && grid[f[0] + 2][f[1]] == 0 && hv[f[0] + 1][f[1]][1] == 0) {
                    // down
                    hv[f[0] + 1][f[1]][1] = hv[f[0]][f[1]][1] + 1;
                    queue.push_back({f[0] + 1, f[1], 1});
                }
                if (f[1] < grid[0].size() - 1 && grid[f[0]][f[1] + 1] == 0 && grid[f[0] + 1][f[1] + 1] == 0 && hv[f[0]][f[1]][0] == 0) {
                    // rotate
                    hv[f[0]][f[1]][0] = hv[f[0]][f[1]][1] + 1;
                    queue.push_back({f[0], f[1], 0});
                }
            }
            queue.pop_front();
        }
        // for (int i = 0; i < grid.size(); i++) {
        //     for (int j = 0; j < grid.size(); j++) {
        //         cout << hv[i][j][0] << "\t";
        //     }
        //     cout << "\n";
        // }
        // cout << "\n";
        // for (int i = 0; i < grid.size(); i++) {
        //     for (int j = 0; j < grid.size(); j++) {
        //         cout << hv[i][j][1] << "\t";
        //     }
        //     cout << "\n";
        // }
        // cout << "\n";
        // cout << "\n";
        if (hv.back()[grid[0].size() - 2][0] == 0) {
            return -1;
        }
        return hv.back()[grid[0].size() - 2][0] - 1;
    }
};
