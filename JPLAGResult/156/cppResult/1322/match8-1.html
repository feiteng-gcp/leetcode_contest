class Solution
{
public:
    int minimumMoves(vector<vector<int>>& grid)
    {
        int n = grid.size();
        vector<vector<int>> hdist(n, vector<int>(n, -1));
        vector<vector<int>> vdist(n, vector<int>(n, -1));
        hdist[0][0] = 0;
        queue<int> rq, cq, dq;
        rq.push(0);
        cq.push(0);
        dq.push(0);
        while (rq.size() > 0) {
            int r = rq.front(); rq.pop();
            int c = cq.front(); cq.pop();
            int d = dq.front(); dq.pop();

            if (d == 0) {
                if (c + 2 < n && grid[r][c + 2] == 0 && hdist[r][c + 1] == -1) {
                    hdist[r][c + 1] = hdist[r][c] + 1;
                    rq.push(r);
                    cq.push(c + 1);
                    dq.push(0);
                }
                if (r + 1 < n && grid[r + 1][c] == 0 && grid[r + 1][c + 1] == 0 && hdist[r + 1][c] == -1)
                {
                    hdist[r + 1][c] = hdist[r][c] + 1;
                    rq.push(r + 1);
                    cq.push(c);
                    dq.push(0);
                }
                if (r + 1 < n && grid[r + 1][c] == 0 && grid[r + 1][c + 1] == 0 && vdist[r][c] == -1)
                {
                    vdist[r][c] = hdist[r][c] + 1;
                    rq.push(r);
                    cq.push(c);
                    dq.push(1);
                }
            }
            else {
                if (r + 2 < n && grid[r + 2][c] == 0 && vdist[r + 1][c] == -1) {
                    vdist[r + 1][c] = vdist[r][c] + 1;
                    rq.push(r + 1);
                    cq.push(c);
                    dq.push(1);
                }
                if (c + 1 < n && grid[r][c + 1] == 0 && grid[r + 1][c + 1] == 0 && vdist[r][c + 1] == -1)
                {
                    vdist[r][c + 1] = vdist[r][c] + 1;
                    rq.push(r);
                    cq.push(c + 1);
                    dq.push(1);
                }
                if (c + 1 < n && grid[r][c + 1] == 0 && grid[r + 1][c + 1] == 0 && hdist[r][c] == -1)
                {
                    hdist[r][c] = vdist[r][c] + 1;
                    rq.push(r);
                    cq.push(c);
                    dq.push(0);
                }
            }
        }
        return hdist[n - 1][n - 2];
    }
};
