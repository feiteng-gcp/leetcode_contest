class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        if (grid.size() == 0 || grid[0].size() <= 1) return -1;
        if (grid[0][0] || grid[0][1]) return -1;
        int n = grid.size(), m = grid[0].size();
        deque<vector<int>> q;
        vector<vector<vector<int>>> v (n, vector<vector<int>> (m, vector<int> (2, -1)));
        q.push_back({0, 0, 0});
        v[0][0][0] = 0;
        while (!q.empty()) {
            int x = q.front()[0], y = q.front()[1], z = q.front()[2];
            q.pop_front();
            switch (z) {
                case (0): // --
                    // go right
                    if (y + 2 < m && !grid[x][y + 2] && !~v[x][y + 1][0]) {
                        v[x][y + 1][0] = v[x][y][z] + 1;
                        q.push_back({x, y + 1, 0});
                    }
                    // go down
                    if (x + 1 < n && !grid[x + 1][y] && !grid[x + 1][y + 1] && !~v[x + 1][y][0]) {
                        v[x + 1][y][0] = v[x][y][z] + 1;
                        q.push_back({x + 1, y, 0});
                    }
                    // rotate
                    if (x + 1 < n && !grid[x + 1][y] && !grid[x + 1][y + 1] & !~v[x][y][1]) {
                        v[x][y][1] = v[x][y][z] + 1;
                        q.push_back({x,y,1});
                    }
                    break;
                case (1): // ||
                    // go right
                    if (y + 1 < m && !grid[x][y + 1] && !grid[x + 1][y + 1] && !~v[x][y + 1][1]) {
                        v[x][y + 1][1] = v[x][y][z] + 1;
                        q.push_back({x, y + 1, 1});
                    }
                    // go down
                    if (x + 2 < n && !grid[x + 2][y] && !~v[x + 1][y][1]) {
                        v[x + 1][y][1] = v[x][y][z] + 1;
                        q.push_back({x + 1, y, 1});
                    }
                    // rotate
                    if (y + 1 < m && !grid[x][y + 1] && !grid[x + 1][y + 1] & !~v[x][y][0]) {
                        v[x][y][0] = v[x][y][z] + 1;
                        q.push_back({x,y,0});
                    }
                    break;
            }
        }
        return v[n - 1][n - 2][0];
    }
};
