class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        int n=grid.size();
        set<pair<int,int>> s1;
        queue<pair<int,int>> q1;
        queue<int> q12;
        set<pair<int,int>> s2;
        queue<pair<int,int>> q2;
        q1.push(make_pair(0,1));
        q12.push(0);
        q2.push(make_pair(n*n-2,n*n-1));
        s1.insert(q1.front());
        s2.insert(q2.front());
        while(!q1.empty() && !q2.empty()){
            int x1=q1.front().first/n;
            int y1=q1.front().first%n;
            int x2=q1.front().second/n;
            int y2=q1.front().second%n;
            //cout<<x1<<" "<<y1<<" "<<x2<<" "<<y2<<endl;
            int step=q12.front();
            if(q1.front().first==n*n-2 && q1.front().second==n*n-1)
                return step;
            q12.pop();
            q1.pop();
            if(x1==x2){
                if(y2!=n-1 && grid[x1][y2+1]==0 && s1.find(make_pair(x1*n+y2,x1*n+y2+1))==s1.end()){
                    q1.push(make_pair(x1*n+y2,x1*n+y2+1));
                    s1.insert(make_pair(x1*n+y2,x1*n+y2+1));
                    q12.push(step+1);
                }
                if(x1!=n-1 && grid[x1+1][y1]==0 && grid[x2+1][y2]==0){
                    if(s1.find(make_pair((1+x1)*n+y1,(1+x1)*n+y2))==s1.end()){
                    q1.push(make_pair((1+x1)*n+y1,(1+x1)*n+y2));
                    s1.insert(make_pair((1+x1)*n+y1,(1+x1)*n+y2));
                        q12.push(step+1);
                    }
                    if(s1.find(make_pair(x1*n+y1,(1+x1)*n+y1))==s1.end()){
                        q1.push(make_pair(x1*n+y1,(1+x1)*n+y1));
                        s1.insert(make_pair(x1*n+y1,(1+x1)*n+y1));
                        q12.push(step+1);
                    }
                }
            }
            else if(y1==y2){
                if(x2!=n-1 && grid[x2+1][y2]==0 && s1.find(make_pair(x2*n+y2,(1+x2)*n+y2))==s1.end()){
                    q1.push(make_pair(x2*n+y2,(1+x2)*n+y2));
                    q12.push(step+1);
                    s1.insert(make_pair(x2*n+y2,(1+x2)*n+y2));
                }
                if(y1!=n-1 && grid[x1][y1+1]==0 && grid[x2][y2+1]==0){
                    if(s1.find(make_pair(x1*n+y1+1,x2*n+y2+1))==s1.end()){
                        q1.push(make_pair(x1*n+y1+1,x2*n+y2+1));
                        s1.insert(make_pair(x1*n+y1+1,x2*n+y2+1));
                        q12.push(step+1);
                    }
                    if(s1.find(make_pair(x1*n+y1,x1*n+y1+1))==s1.end()){
                        q1.push(make_pair(x1*n+y1,x1*n+y1+1));
                        s1.insert(make_pair(x1*n+y1,x1*n+y1+1));
                        q12.push(step+1);
                    }
                }
            }
        }
        return -1;
    }
};
