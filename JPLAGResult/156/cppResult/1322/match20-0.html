class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        int n = grid.size();
        if (grid[n - 1][n - 2] == 1 || grid[n - 1][n - 1] == 1) return -1;

        vector<vector<vector<int>>> vis(n, vector<vector<int>>(n, vector<int>(2, 0)));
        queue<pair<int, int>> q;
        q.push({ 0 * n + 1, 0 });
        vis[0][1][0] = 1;

        int cnt = 0;
        while (!q.empty()) {
            int size = q.size();
            for (int i = 0; i < size; i++) {
                int cur_x = q.front().first / n;
                int cur_y = q.front().first % n;
                int cur_d = q.front().second;

                // cout << cur_x << " " << cur_y << " " << cur_d << endl;

                q.pop();
                if (cur_d == 0) {
                    if (cur_y + 1 < n && grid[cur_x][cur_y + 1] == 0 && !vis[cur_x][cur_y + 1][cur_d]) {
                        if (cur_x == n - 1 && cur_y + 1 == n - 1) return cnt + 1;
                        vis[cur_x][cur_y + 1][cur_d] = 1;
                        q.push({ cur_x * n + cur_y + 1, cur_d });
                    }
                    if (cur_x + 1 < n && grid[cur_x + 1][cur_y - 1] == 0 && grid[cur_x + 1][cur_y] == 0 && !vis[cur_x + 1][cur_y][cur_d]) {
                        if (cur_x + 1 == n - 1 && cur_y == n - 1) return cnt + 1;
                        vis[cur_x + 1][cur_y][cur_d] = 1;
                        q.push({ (cur_x + 1) * n + cur_y, cur_d });
                    }
                    if (cur_x + 1 < n && grid[cur_x + 1][cur_y] == 0 && grid[cur_x + 1][cur_y - 1] == 0 && !vis[cur_x + 1][cur_y - 1][1]) {
                        vis[cur_x + 1][cur_y - 1][1] = 1;
                        q.push({ (cur_x + 1) * n + cur_y - 1, 1 });
                    }
                }
                else {
                    if (cur_x + 1 < n && grid[cur_x + 1][cur_y] == 0 && !vis[cur_x + 1][cur_y][cur_d]) {
                        vis[cur_x + 1][cur_y][cur_d] = 1;
                        q.push({ (cur_x + 1) * n + cur_y, cur_d });
                    }
                    if (cur_y + 1 < n && grid[cur_x][cur_y + 1] == 0 && grid[cur_x - 1][cur_y + 1] == 0 && !vis[cur_x][cur_y + 1][cur_d]) {
                        vis[cur_x][cur_y + 1][cur_d] = 1;
                        q.push({ cur_x * n + cur_y + 1, cur_d });
                    }
                    if (cur_y + 1 < n && grid[cur_x - 1][cur_y + 1] == 0 && grid[cur_x][cur_y + 1] == 0 && !vis[cur_x - 1][cur_y + 1][0]) {
                        vis[cur_x - 1][cur_y + 1][0] = 1;
                        q.push({ (cur_x - 1) * n + cur_y + 1, 0 });
                    }
                }
            }
            cnt++;
        }

        return -1;
    }
};
