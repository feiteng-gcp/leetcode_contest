class Solution {
private:
    int has[105][105][2];
    int n, m;
    vector<vector<int>> mp;
    struct Node{
        int x, y, dir, step;
        Node(int a=0, int b=0, int c=0, int d=0):x(a), y(b), dir(c), step(d){}
    };
    
    bool ok(int x1, int y1, int x2, int y2, int step, int tag=0) {
        int dir = x1 != x2;
        if(x1 < n && y1 < m && x2 < n && y2 < m && has[x1][y1][dir] > step && mp[x1][y1] != 1 && mp[x2][y2] != 1) {
            if(tag && mp[x1 + 1][y1 + 1] == 1) return false; 
            else return true;
        } else return false;
    }
    
public:
    int minimumMoves(vector<vector<int>>& grid) {
        mp = grid;
        n = grid.size();
        m = grid[0].size();
        for(int i = 0; i < n; ++i) {
            for(int j = 0; j < m; ++j) {
                has[i][j][0] = has[i][j][1] = 1000000000;
            }
        }
        
        queue<Node> Q;
        Q.push(Node(0, 0, 0, 0));
        has[0][0][0] = 0;
        
        while(!Q.empty()) {
            Node tmp = Q.front(); Q.pop();
            // printf("%d %d %d %d\n", tmp.x, tmp.y, tmp.dir, tmp.step);
            
            if(tmp.x == n-1 && tmp.y == m-2 && tmp.dir == 0) {
                return tmp.step;
            }
            
            if(tmp.dir == 0) {
                if( ok(tmp.x, tmp.y + 1, tmp.x, tmp.y + 2, tmp.step + 1) ) {
                    has[tmp.x][tmp.y + 1][tmp.dir] = tmp.step + 1;
                    Q.push(Node(tmp.x, tmp.y + 1, tmp.dir, tmp.step + 1));
                }
                if( ok(tmp.x + 1, tmp.y, tmp.x + 1, tmp.y + 1, tmp.step + 1) ) {
                    has[tmp.x + 1][tmp.y][tmp.dir] = tmp.step + 1;
                    Q.push(Node(tmp.x + 1, tmp.y, tmp.dir, tmp.step + 1));
                }
                if( ok(tmp.x, tmp.y, tmp.x + 1, tmp.y, tmp.step + 1, 1) ) {
                    has[tmp.x][tmp.y][tmp.dir ^ 1] = tmp.step + 1;
                    Q.push(Node(tmp.x, tmp.y, tmp.dir ^ 1, tmp.step + 1));
                }
            } else {
                if( ok(tmp.x, tmp.y + 1, tmp.x + 1, tmp.y + 1, tmp.step + 1) ) {
                    has[tmp.x][tmp.y + 1][tmp.dir] = tmp.step + 1;
                    Q.push(Node(tmp.x, tmp.y + 1, tmp.dir, tmp.step + 1));
                }
                if( ok(tmp.x + 1, tmp.y, tmp.x + 2, tmp.y, tmp.step + 1) ) {
                    has[tmp.x + 1][tmp.y][tmp.dir] = tmp.step + 1;
                    Q.push(Node(tmp.x + 1, tmp.y, tmp.dir, tmp.step + 1));
                }
                if( ok(tmp.x, tmp.y, tmp.x, tmp.y + 1, tmp.step + 1, 1) ) {
                    has[tmp.x][tmp.y][tmp.dir ^ 1] = tmp.step + 1;
                    Q.push(Node(tmp.x, tmp.y, tmp.dir ^ 1, tmp.step + 1));
                }
            }
            
        }
        return -1;
    }
};
