class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<vector<int>> dv(n, vector<int>(n));
        vector<vector<int>> dh(n, vector<int>(n));
        vector<vector<int>> uv(n, vector<int>(n));
        vector<vector<int>> uh(n, vector<int>(n));
        
        uh[0][0] = uh[0][1] = 1;
        queue<int> q;
        q.push(0);
        q.push(1);
        q.push(0);

        while (!q.empty())
        {
            int cr = q.front();
            q.pop();
            int cc = q.front();
            q.pop();
            int cp = q.front();
            q.pop();
            
            if (cp == 0) // horz
            {
                if (cc < n - 1 && uh[cr][cc + 1] == 0 && grid[cr][cc + 1] == 0)
                {
                    q.push(cr);
                    q.push(cc + 1);
                    q.push(0);
                    dh[cr][cc + 1] = dh[cr][cc + 1] == 0 ? dh[cr][cc] + 1 : min(dh[cr][cc + 1], dh[cr][cc] + 1);
                    uh[cr][cc + 1] = 1;
                }
                
                if (cr < n - 1 && uv[cr + 1][cc - 1] == 0 && grid[cr + 1][cc - 1] == 0 && grid[cr + 1][cc] == 0)
                {
                    q.push(cr + 1);
                    q.push(cc - 1);
                    q.push(1);
                    dv[cr + 1][cc - 1] = dv[cr + 1][cc - 1] == 0 ? dh[cr][cc] + 1 : min(dv[cr + 1][cc - 1], dh[cr][cc] + 1);
                    uv[cr + 1][cc - 1] = 1;
                }

                if (cr < n - 1 && uh[cr + 1][cc] == 0 && grid[cr + 1][cc - 1] == 0 && grid[cr + 1][cc] == 0)
                {
                    q.push(cr + 1);
                    q.push(cc);
                    q.push(0);
                    dh[cr + 1][cc] = dh[cr + 1][cc] == 0 ? dh[cr][cc] + 1 : min(dh[cr + 1][cc], dh[cr][cc] + 1);
                    uh[cr + 1][cc] = 1;
                }
            }
            else
            {
                if (cr < n - 1 && uv[cr + 1][cc] == 0 && grid[cr + 1][cc] == 0)
                {
                    q.push(cr + 1);
                    q.push(cc);
                    q.push(1);
                    dv[cr + 1][cc] = dv[cr + 1][cc] == 0 ? dv[cr][cc] + 1 : min(dv[cr + 1][cc], dv[cr][cc] + 1);
                    uv[cr + 1][cc] = 1;
                }
                
                if (cc < n - 1 && uh[cr - 1][cc + 1] == 0 && grid[cr - 1][cc + 1] == 0 && grid[cr][cc + 1] == 0)
                {
                    q.push(cr - 1);
                    q.push(cc + 1);
                    q.push(0);
                    dh[cr - 1][cc + 1] = dh[cr - 1][cc + 1] == 0 ? dv[cr][cc] + 1 : min(dh[cr - 1][cc + 1], dv[cr][cc] + 1);
                    uh[cr - 1][cc + 1] = 1;
                }

                if (cc < n - 1 && uv[cr][cc + 1] == 0 && grid[cr][cc + 1] == 0 && grid[cr - 1][cc + 1] == 0)
                {
                    q.push(cr);
                    q.push(cc + 1);
                    q.push(1);
                    dv[cr][cc + 1] = dv[cr][cc + 1] == 0 ? dv[cr][cc] + 1 : min(dv[cr][cc + 1], dv[cr][cc] + 1);
                    uv[cr][cc + 1] = 1;
                }
            }
        }
        
        if (dh[n - 1][n - 1] == 0)
            return -1;
        
        return dh[n - 1][n - 1];
    }
};
