class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        int n = grid.size();
        memset(vis, -1, sizeof vis);
        queue<P> que;
        que.push(P{0, 1, 1});
        vis[0][1][1] = 0;
        while (que.size()) {
            int x = que.front().x;
            int y = que.front().y;
            int d = que.front().d;
            que.pop();
            if (d == 1) {
                if (y + 1 < n && grid[x][y+1] == 0 && !~vis[x][y+1][1]) {
                    que.push(P{x, y+1, 1});
                    vis[x][y+1][1] = vis[x][y][1]+1;
                }
                if (x + 1 < n && grid[x+1][y] == 0 && grid[x+1][y-1] == 0 && !~vis[x+1][y][1]) {
                    que.push(P{x+1, y, 1});
                    vis[x+1][y][1] = vis[x][y][1]+1;
                }
                if (x + 1 < n && grid[x+1][y-1] == 0 && grid[x+1][y] == 0 && !~vis[x+1][y-1][2]) {
                    que.push(P{x+1, y-1, 2});
                    vis[x+1][y-1][2] = vis[x][y][1]+1;
                }
            } else {
                if (y + 1 < n && grid[x][y+1] == 0 && grid[x-1][y+1] == 0 && !~vis[x][y+1][2]) {
                    que.push(P{x, y + 1, 2});
                    vis[x][y+1][2] = vis[x][y][2]+1;
                }
                if (x + 1 < n && grid[x+1][y] == 0 && !~vis[x+1][y][2]) {
                    que.push(P{x + 1, y, 2});
                    vis[x+1][y][2] = vis[x][y][2]+1;
                }
                if (y + 1 < n && grid[x][y+1] == 0 && grid[x-1][y+1] == 0 && !~vis[x-1][y+1][1]) {
                    que.push(P{x - 1, y + 1, 1});
                    vis[x-1][y+1][1] = vis[x][y][2]+1;
                }
            }
        }
        return vis[n-1][n-1][1];
    }
private:
    int vis[109][109][4];
    struct P {
        int x, y, d;
    };
};
