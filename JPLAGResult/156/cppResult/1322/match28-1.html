class Solution {
 public:
  int minimumMoves(vector<vector<int>>& grid) {
    int n = (int)grid.size();
    vector<vector<vector<bool>>> seen(n,
                                      vector<vector<bool>>(n, vector<bool>(2)));
    constexpr int RIGHT = 0, DOWN = 1;
    seen[0][1][RIGHT] = true;
    queue<tuple<int, int, int>> cur;
    cur.emplace(0, 1, RIGHT);
    int ans = 0;
    while (!cur.empty()) {
      queue<tuple<int, int, int>> nxt;
      while (!cur.empty()) {
        int r = get<0>(cur.front());
        int c = get<1>(cur.front());
        int dir = get<2>(cur.front());
        cur.pop();
        // cout << r << ' ' << c << ' ' << dir << endl;
        if (r == n - 1 && c == n - 1 && dir == RIGHT) {
          return ans;
        }

        if (dir == RIGHT && c + 1 < n && grid[r][c + 1] == 0 &&
            !seen[r][c + 1][RIGHT]) {
          seen[r][c + 1][RIGHT] = true;
          nxt.emplace(r, c + 1, RIGHT);
        }
        if (dir == DOWN && c + 1 < n && grid[r][c + 1] == 0 &&
            grid[r - 1][c + 1] == 0 && !seen[r][c + 1][DOWN]) {
          seen[r][c + 1][DOWN] = true;
          nxt.emplace(r, c + 1, DOWN);
        }
        if (dir == DOWN && r + 1 < n && grid[r + 1][c] == 0 &&
            !seen[r + 1][c][DOWN]) {
          seen[r + 1][c][DOWN] = true;
          nxt.emplace(r + 1, c, DOWN);
        }

        if (dir == RIGHT && r + 1 < n && grid[r + 1][c] == 0 &&
            grid[r + 1][c - 1] == 0 && !seen[r + 1][c][RIGHT]) {
          seen[r + 1][c][RIGHT] = true;
          nxt.emplace(r + 1, c, RIGHT);
        }
        if (dir == RIGHT && r + 1 < n && grid[r + 1][c - 1] == 0 &&
            grid[r + 1][c] == 0 && !seen[r + 1][c - 1][DOWN]) {
          seen[r + 1][c - 1][DOWN] = true;
          nxt.emplace(r + 1, c - 1, DOWN);
        }
        if (dir == DOWN && c + 1 < n && grid[r - 1][c + 1] == 0 &&
            grid[r][c + 1] == 0 && !seen[r - 1][c + 1][RIGHT]) {
          seen[r - 1][c + 1][RIGHT] = true;
          nxt.emplace(r - 1, c + 1, RIGHT);
        }
      }
      cur = move(nxt);
      ++ans;
    }
    return -1;
  }
};
