class Solution {
public:
    int minimumMoves(vector<vector<int>>& grid) {
        int n=grid.size();
        if(n==0) return -1;
        int m=grid[0].size();
        if(grid[n-1][m-1] || grid[n-1][m-2]) return -1;
        vector<vector<int>> heng(n,vector<int>(m,100000));
        vector<vector<int>> shu=heng;
        heng[0][1]=0;
        queue<vector<int>> q;
        q.push({0,0,1});
        while(!q.empty()){

            auto cur=q.front();
            q.pop();
            if(!cur[0]){
                if(cur[2]+1<m && grid[cur[1]][cur[2]+1]==0 && heng[cur[1]][cur[2]+1]>heng[cur[1]][cur[2]]+1){
                    heng[cur[1]][cur[2]+1]=heng[cur[1]][cur[2]]+1;
                    q.push({0,cur[1],cur[2]+1});
                }
                if(cur[1]+1<n && grid[cur[1]+1][cur[2]]==0 && grid[cur[1]+1][cur[2]-1]==0 && shu[cur[1]+1][cur[2]-1]>heng[cur[1]][cur[2]]+1){
                    shu[cur[1]+1][cur[2]-1]=heng[cur[1]][cur[2]]+1;
                    q.push({1,cur[1]+1,cur[2]-1});
                }
                if(cur[1]+1<n && grid[cur[1]+1][cur[2]]==0 && grid[cur[1]+1][cur[2]-1]==0 && heng[cur[1]+1][cur[2]]>heng[cur[1]][cur[2]]+1){
                    heng[cur[1]+1][cur[2]]=heng[cur[1]][cur[2]]+1;
                    q.push({0,cur[1]+1,cur[2]});
                }
            }else{
                if(cur[1]+1<n && grid[cur[1]+1][cur[2]]==0 && shu[cur[1]+1][cur[2]]>shu[cur[1]][cur[2]]+1){
                    shu[cur[1]+1][cur[2]]=shu[cur[1]][cur[2]]+1;
                    q.push({1,cur[1]+1,cur[2]});
                }
                if(cur[2]+1<m && grid[cur[1]][cur[2]+1]==0 && grid[cur[1]-1][cur[2]+1]==0 && heng[cur[1]-1][cur[2]+1]>shu[cur[1]][cur[2]]+1){
                    heng[cur[1]-1][cur[2]+1]=shu[cur[1]][cur[2]]+1;
                    q.push({0,cur[1]-1,cur[2]+1});
                }
                if(cur[2]+1<m && grid[cur[1]][cur[2]+1]==0 && grid[cur[1]-1][cur[2]+1]==0 && shu[cur[1]][cur[2]+1]>shu[cur[1]][cur[2]]+1){
                    shu[cur[1]][cur[2]+1]=shu[cur[1]][cur[2]]+1;
                    q.push({1,cur[1],cur[2]+1});
                }
            }
            if(heng[n-1][m-1]!=100000) return heng[n-1][m-1];
            //cout<<q.size()<<endl;
        }
        
        return -1;
    }
};
