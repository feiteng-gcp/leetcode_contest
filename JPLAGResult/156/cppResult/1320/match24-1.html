class Solution {
public:
    string removeDuplicates(string s, int k)
    {
        vector<char> chars;
        vector<int> counts;
        for (char c : s) {
            if (chars.empty() || chars.back() != c) {
                chars.push_back(c);
                counts.push_back(1);
            }
            else if (++counts.back() == k) {
                chars.pop_back();
                counts.pop_back();
            }
        }
        string str;
        for (int i = 0; i < chars.size(); i++)
            for (int j = 0; j < counts[i]; j++)
                str.push_back(chars[i]);
        return str;
    }
};
