class Solution {
public:
    string removeDuplicates(string s, int k) {
        deque<pair<char, int>> q;
        for (int i = 0; i < s.size(); ++i) {
            char c = s[i];
            if (q.empty() || (!q.empty() && q.back().first != c)) {
                q.push_back(make_pair(c, 1));
            } else if (!q.empty() && q.back().first == c){
                if (++q.back().second == k) {
                    q.pop_back();
                }
            }
        }
        string res;
        while (!q.empty()) {
            res.insert(res.end(), q.front().second, q.front().first);
            q.pop_front();
        }
        return res;
    }
};
