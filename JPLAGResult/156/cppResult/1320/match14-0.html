class Solution {
public:
    string removeDuplicates(string s, int k) {
        stack<char> stc;
        stack<int> stn;
        for(char c:s)
        {
            if(stc.empty()||c!=stc.top())
            {
                stc.push(c);
                stn.push(1);
            }
            else
            {
                int n=stn.top(); stn.pop();
                n++;
                if(n==k) stc.pop();
                else stn.push(n);
            }
        }
        string ans;
        while(!stc.empty())
        {
            char c=stc.top(); stc.pop();
            int n=stn.top(); stn.pop();
            for(int i=0;i<n;i++)
                ans+=c;
        }
        reverse(ans.begin(),ans.end());
        return ans;
    }
};
