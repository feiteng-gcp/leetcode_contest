class Solution {
public:
    string removeDuplicates(string s, int k) {
        vector<pair<char, int>> data;
        for (char c:s) {
            if (!data.empty() && data.back().first==c) {
                if (data.back().second==k-1) {
                    data.pop_back();
                } else {
                    data.back().second++;
                }
            } else {
                data.push_back(make_pair(c, 1));
            }
        }
        string ret="";
        for (auto p:data) {
            for (int i=0; i<p.second; i++) {
                ret+=p.first;
            }
        }
        return ret;
    }
};
