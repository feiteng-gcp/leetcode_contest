class Solution {
public:
    string removeDuplicates(string s, int k) {
        stack<pair<char,int>> ss;
        for (char& c: s) {
            if (ss.empty() || ss.top().first != c) ss.push({c, 1});
            else if (ss.top().second == k-1) ss.pop();
            else {
                int t = ss.top().second + 1;
                ss.pop();
                ss.push({c, t});
            }
        }
        string r;
        while (ss.size()) {
            r = string(ss.top().second, ss.top().first) + r;
            ss.pop();
        }
        return r;
    }
};
