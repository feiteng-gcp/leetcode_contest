class Solution {
public:
    string removeDuplicates(string s, int k) {
        stack<pair<char, int>> sta;
        for(char c: s)
        {
            if(sta.empty())
            {
                sta.push({c, 1});
                continue;
            }
            if(c != sta.top().first)
            {
                sta.push({c, 1});
            }
            else
            {
                sta.top().second++;
                if(sta.top().second  == k)
                    sta.pop();
            }
        }
        string res;
        while(!sta.empty())
        {
            string curr(sta.top().second, sta.top().first);
            res += curr;
            sta.pop();
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
