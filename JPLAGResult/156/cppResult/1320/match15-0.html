class Solution {
public:
    string removeDuplicates(string s, int k) {
        vector<pair<char, int>> st;
        for (char c: s) {
            if (st.empty() || st.back().first != c) {
                st.push_back({c, 1});
            }
            else {
                st.back().second += 1;
            }
            
            if (st.back().second == k) st.pop_back();
        }
        
        string ret;
        for (pair<char, int>& p: st) {
            ret += string(p.second, p.first);
        }
        return ret;
    }
};
