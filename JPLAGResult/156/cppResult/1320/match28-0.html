class Solution {
public:
string removeDuplicates(string s, int k) {
    vector<char> vs;
    vector<int> freq;
    for (char c : s) {
        if (vs.size() && c == vs.back()) {
            freq.push_back(freq.back() + 1);
        } else {
            freq.push_back(1);
        }
        vs.push_back(c);
        if (freq.size() && freq.back() == k) {
            char to_pop = vs.back();
            while (vs.size() && vs.back() == to_pop) {
                vs.pop_back();
                freq.pop_back();
            }
        }
    }
    
    string r = "";
    for (char c : vs)
        r.push_back(c);
    return r;
}
};
