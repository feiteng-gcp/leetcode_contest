class Solution {
public:
    string removeDuplicates(string s, int k) {
        stack<pair<char, int>> t;
        for (char &c : s) {
            if (!t.empty() && t.top().first == c) {
                if (t.top().second == k - 1) {
                    t.pop();
                } else {
                    t.top().second++;
                }
            } else {
                t.emplace(c, 1);
            }
        }
        string ans;
        while (!t.empty()) {
            ans += string(t.top().second, t.top().first);
            t.pop();
        }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
