class Solution {
public:
    string removeDuplicates(string s, int k) {
        while(1){
            int n=s.size();
            bool flag=0;
            string tmp;
            for(int i=0;i<n;){
                bool ok=0;
                if(i+k-1<n){
                    for(int j=i+1;j<i+k;j++){
                        if(s[i]!=s[j]){ok=1;break;}
                    }    
                }else ok=1;
                if(ok)tmp.push_back(s[i]),i++;
                else i=i+k,flag=1;
            }
            cout<<tmp<<endl;
            if(!flag)break;
            s=tmp;
        }
        return s;
    }
};
