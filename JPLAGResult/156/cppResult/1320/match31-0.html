class Solution {
public:
    string removeDuplicates(string s, int k) {
        stack<pair<char, int> > st;	// letter, count
        for (int i = 0; i < s.size(); ++i) {
        	char c = s[i];
        	if (st.empty()) {
        		st.push(make_pair(c, 1));
        	} else {
        		if (c == st.top().first) {
        			if (st.top().second + 1 == k) {
        				st.pop();
        			} else {
        				st.top().second++;
        			}
        		} else {
        			st.push(make_pair(c, 1));
        		}
        	}
        }
        string ans = "";
        while (!st.empty()) {
        	char c = st.top().first;
        	int count = st.top().second;
        	for (int i = 0; i < count; ++i) {
        		ans += c;
        	}
        	st.pop();
        }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};
