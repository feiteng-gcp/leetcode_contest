class Solution {
public:
    string removeDuplicates(string s, int k) {
        vector<pair<char,int>> p;
        for(char c: s) {
            if(p.empty()) p.emplace_back(c, 1);
            else if(p.back().first == c) p.emplace_back(c, p.back().second + 1);
            else p.emplace_back(c, 1);
            if(p.back().second == k) {
                for(int i = 0; i < k; i++) 
                    p.pop_back();
            }
        }
        string ans;
        for(auto e: p) {
            ans.push_back(e.first);
        }
        return ans;
    }
};
