class Solution {
public:
    string removeDuplicates(string s, int k) {
        stack<pair<char, int>> stk;
        
        for (char c : s) {
            if (stk.empty() || stk.top().first != c) {
                stk.push({c, 1});
            } else {
                stk.top().second++;
                if (stk.top().second == k) {
                    stk.pop();
                }
            }
        }
        
        string str;
        while (!stk.empty()) {
            char c = stk.top().first;
            int n = stk.top().second;
            while (n--) {
                str.push_back(c);
            }
            stk.pop();
        }
        
        reverse(str.begin(), str.end());
        return str;
    }
};
