class Solution {
public:
    string removeDuplicates(string s, int k) {
        vector<char> st;
        vector<int> cnt;
        for (char c : s) {
            if (st.size() > 0 && st.back() == c) {
                cnt.back()++;
            } else {
                st.push_back(c);
                cnt.push_back(1);
            }
            
            while (st.size() > 0 && cnt.back() == k) {
                st.pop_back();
                cnt.pop_back();
            }
        }
        
        string ret = "";
        for (int i = 0; i < st.size(); ++i)
            ret += string(cnt[i], st[i]);
        return ret;
    }
};
