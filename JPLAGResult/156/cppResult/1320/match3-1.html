class Solution {
public:
    string removeDuplicates(string str, int k) {
        string A;
        vector<int> B;

        for (auto ch: str) {
            if (A.empty() || A.back() != ch) {
                B.push_back(1);
            } else {
                B.push_back(B.back()+1);
            }
            A.push_back(ch);
            if (B.back() == k) {
                for (int i = 0; i < k; i++) {
                    B.pop_back();
                    A.pop_back();
                }
            }
        }
        return A;
    }
};
