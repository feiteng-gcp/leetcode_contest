class Solution {
public:
    string removeDuplicates(string s, int k) {
        vector<pair<char,int> > v;
        for(auto c:s){
            if(v.size()==0){
                v.push_back({c,1});
            }else{
                auto p  = v.back();
                if(c==p.first){
                    v.pop_back();
                    p.second++;
                    if(p.second<k){
                        v.push_back(p);
                    }
                }else{
                    v.push_back({c,1});
                }
            }
        }
        string ret = "";
        for(auto p:v){
            for(int i=0;i<p.second;i++){
                ret+=p.first;
            }
        }
        return ret;
        
    }
};
