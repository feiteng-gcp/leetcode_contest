class Solution {
public:
string removeDuplicates(string s, int k) {
	stack<pair<char, int>> st;
	for (char ch : s) {
		if (st.size() && st.top().first == ch) {
			st.top().second++;
		} else {
			st.push({ch, 1});
		}
		if (st.top().second == k){
			st.pop();
		}
	}
	string ret;
	while (st.size()) {
		ret += string(st.top().second, st.top().first);
		st.pop();
	}
	reverse(ret.begin(), ret.end());
	
	return ret;
}

};
