class Solution {
public:
    string removeDuplicates(string s, int k) {
        stack<pair<char, int> > t;
        for(auto x : s){
            if(t.empty() || t.top().first != x){
                t.push({x, 1});
            }
            else {
                t.top().second += 1;
            }
            if(t.top().second == k) t.pop();
        }
        string res = "";
        while(!t.empty()) {
            int cnt = t.top().second;
            while(cnt--) res += t.top().first;
            t.pop();
        }
        reverse(res.begin(), res.end());
        cout<<res<<endl;
        return res;
    }
};
