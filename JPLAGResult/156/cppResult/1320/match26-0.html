class Solution {
public:
    string removeDuplicates(string s, int k) {
        stack<pair<char, int>> st;
        for(auto c : s) {
            if(st.empty()) st.emplace(c, 1);
            else {
                if(c == st.top().first) st.emplace(c, st.top().second + 1);
                else st.emplace(c, 1);
            }
            if(st.top().second >= k) {
                for(int i = 0; i < k; i++)
                    st.pop();
            }
        }
        string res;
        while(st.size()) {
            res += st.top().first;
            st.pop();
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
