class Solution {
public:
    string removeDuplicates(string s, int k) {
        string res;
        vector<int> cnt;
        
        for (auto& ch : s) {
            if (res.empty()) {
                res.push_back(ch);
                cnt.push_back(1);
            } else {
                if (ch == res.back()) {
                    res.push_back(ch);
                    cnt.push_back(cnt.back() + 1);
                } else {
                    res.push_back(ch);
                    cnt.push_back(1);
                }

            }
            
            if (cnt.back() == k) {
                for (int i = 0; i < k; ++i) {
                    res.pop_back();
                    cnt.pop_back();
                }
            }
        }
        
        return res;
    }
};
