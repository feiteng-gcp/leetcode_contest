class Solution {
public:
    string removeDuplicates(string s, int k) {
        vector<pair<char, int>> stk;
        for (const char& c : s) {
            if (stk.empty() || stk.back().first != c)
                stk.push_back({c, 1});
            else {
                if (stk.back().second + 1 == k)
                    stk.pop_back();
                else
                    ++stk.back().second;
            }
        }
        
        string ans;
        for (const auto& p : stk)
            ans.insert(ans.end(), p.second, p.first);
        return ans;
    }
};
