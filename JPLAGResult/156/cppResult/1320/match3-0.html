class Solution {
public:
    string removeDuplicates(string s, int k) {
        string r;
        vector<int> repeated;
        for (char c : s) {
            if (r.empty() || r.back() != c) {
                repeated.push_back(1);
            } else {
                repeated.push_back(repeated.back() + 1);
            }
            r.push_back(c);
            if (repeated.back() >= k) {
                for (int i = 0; i < k; i++) {
                    repeated.pop_back();
                    r.pop_back();
                }
            }
        }
        return r;
    }
};
