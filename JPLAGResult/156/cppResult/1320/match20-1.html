class Solution {
public:
    string removeDuplicates(string s, int k) {
        vector<pair<char, int>> st;
        for (char c : s) {
            if (st.empty() || st.back().first != c)
                st.push_back({c, 1});
            else
                st.back().second ++;
            while (!st.empty() && st.back().second >= k)
                st.pop_back();
        }
        string t = "";
        for (auto& p : st) t += string(p.second, p.first);
        return t;
    }
};
