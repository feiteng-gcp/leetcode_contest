class Solution {
public:
    string removeDuplicates(string s, int k) {
        stack<pair<char, int>> st;
        
        for (auto c : s) {
            if (st.empty() || c != st.top().first)
                st.push(make_pair(c, 1));
            else {
                if (st.top().second + 1 == k)
                    st.pop();
                else
                    st.top().second++;
            }
        }
        
        string ans;
        ans.reserve(s.size());
        while (!st.empty()) {
            auto curr = st.top();
            st.pop();
            
            for (auto i=0; i<curr.second; i++)
                ans.push_back(curr.first);
        }
        
        std::reverse(ans.begin(), ans.end());
        return ans;
    }
};
