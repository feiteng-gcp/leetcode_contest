class Solution {
 public:
  int equalSubstring(string s, string t, int maxCost) {
    int cost = 0, start = 0, ans = 0;
    for (int i = 0; i < (int)s.size(); ++i) {
      cost += abs(s[i] - t[i]);
      while (cost > maxCost) {
        cost -= abs(s[start] - t[start]);
        ++start;
      }
      ans = max(ans, i - start + 1);
    }
    return ans;
  }
};
