class Solution {
public:
    int pre[100010];
    bool judge(int x,int t,int n){
        for(int i=1;i+x-1<=n;i++){
            int r=i+x-1;
            if(t>=pre[r]-pre[i-1])return true;
        }
        return false;
    }
    int equalSubstring(string s, string t, int maxCost) {
        int n=s.size();
        for(int i=1;i<=n;i++)pre[i]=pre[i-1]+abs(s[i-1]-t[i-1]);
        int l=0,r=n,mid;
        while(l<r){
            mid=l+(r-l+1)/2;
            if(judge(mid,maxCost,n))l=mid;
            else r=mid-1;
        }
        return l;
    }
};
