class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        vector<int> cost(s.size(), 0);
        for(int i = 0; i<s.size(); i++){
            cost[i]=abs(s[i]-t[i]);
        }

        int max_ = 0, start = 0, sum = 0;
        for(int end = 0; end<cost.size(); end++){
            sum+=cost[end];
            while(sum>maxCost){
                sum-=cost[start++];
            }
            max_ = max(max_, end-start+1);
        }
        return max_;
    }
};
