class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        vector<int> costs;
        for (int i = 0; i < s.size(); ++i){
            costs.push_back(abs(s[i] - t[i]));
        }
        int maxLength = 0, left = 0, right = 0, curCost = 0;

        while (right < s.size()){
            curCost += costs[right];
            while (curCost > maxCost && left <= right){
                curCost -= costs[left];
                left++;
            }
            right++;
            maxLength = max(maxLength, right - left);
            
        }
        return maxLength;
    }
};
