class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int curr_cost = 0;
        int i = 0;
        int ans = 0;
        for(int j = 0; j < s.size(); j++){
            curr_cost += abs(s[j] - t[j]);
            if(curr_cost <= maxCost){
                ans = max(ans, j - i + 1);
            }
            else{
                while(i <= j && curr_cost > maxCost){
                    curr_cost -= abs(s[i]-t[i]);
                    i++;
                }
                ans = max(ans, j - i + 1);
            }
        }
        return ans;
    }
};
