class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int n = s.length();
        vector<int> d(n);
        for (int i = 0; i < n; ++i)
            d[i] = abs(s[i] - t[i]);
        
        int ans = 0, st = 0, cost = 0;
        for (int en = 0; en < n; ++en) {
            cost += d[en];
            while (st <= en && cost > maxCost) {
                cost -= d[st];
                ++st;
            }
            
            ans = max(ans, en - st + 1);
        }
        
        return ans;
    }
};
