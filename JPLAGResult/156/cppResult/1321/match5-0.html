class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        vector<int> diff;
        for (int i = 0; i < s.size(); i++) {
            diff.push_back(abs(s[i] - t[i]));
            //printf("%d, ", diff[i]);
        }
        int si = 0;
        int ei = 0;
        int tot = 0;
        int res = 0;
        while (ei < diff.size()) {
            tot += diff[ei];
            while(tot > maxCost) {
                tot -= diff[si];
                si++;
            }
            res = max(res, (ei-si)+1);
            ei++;
        }
        //printf("\n");
        return res;
    }
};
