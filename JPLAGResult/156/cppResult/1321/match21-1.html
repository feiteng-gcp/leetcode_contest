class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int n = s.size();
        vector<int> d(n, 0);
        for(int i = 0; i < n; ++i) {
            d[i] = abs(s[i] - t[i]);
        }
        int sum = 0;
        int res = 0;

        for(int i = 0, j = -1; i < n; ++i) {
            sum += d[i];
            while(j < i && sum > maxCost) {
                sum -= d[++j];
            }
            res = max(res, i - j);
        }
        return res;
    }
};
