class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        vector<int> v;
        for(int i=0;i<s.length();i++){
            v.push_back(abs((int)s[i]-(int)t[i]));
        }
        int ret = 0;
        int start = 0;
        int end = 0;
        int sum = 0;
        while(end<s.length()){
            sum+=v[end];
            while(sum>maxCost){
                sum-=v[start++];
            }
            ret = max(ret, end-start+1);
            end++;
        }
        return ret;
    }
};
