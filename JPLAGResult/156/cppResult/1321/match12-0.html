class Solution {
public:
int equalSubstring(string s, string t, int maxCost) {
    vector<int> cost;
    for (int i = 0; i < s.size(); i++) {
        cost.push_back(abs(s[i] - t[i]));
    }
    int max_length = 0;
    int p1 = 0;
    int p2 = 0;
    int cur = 0;
    while (p2 < s.size()) {
        cur += cost[p2];
        while (cur > maxCost && p1 <= p2) {
            cur -= cost[p1];
            p1++;
        }
        max_length = max(max_length, p2 - p1 + 1);
        p2++;
    }
    return max_length;
}
};
