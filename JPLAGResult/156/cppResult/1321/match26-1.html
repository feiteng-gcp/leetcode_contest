class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int n = s.size();
        int diff[n] = {};

        for(int i = 0; i < n; i++) {
            diff[i] = abs(s[i] - t[i]);
        }
        int ls = 0, sm = 0, ans = 0;
        for(int i = 0; i < n; i++) {
            sm += diff[i];
            while(sm > maxCost && ls <= i) {
                sm -= diff[ls++];
            }
            ans = max(ans, i - ls + 1);
        }
        return ans;
    }
};
