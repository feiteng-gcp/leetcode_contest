class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int n=s.size();
        vector<int> diff(n,0);
        for(int i=0;i<n;i++)
            diff[i]=abs(s[i]-t[i]);
        int sum=0,start=0,res=0;
        for(int i=0;i<n;i++){
            sum+=diff[i];
            while(sum>maxCost){
                sum-=diff[start++];
            }
            res=max(res,i-start+1);
        }
        return res;
    }
};
