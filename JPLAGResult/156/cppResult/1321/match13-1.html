class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int size = s.size();
        vector<int> diffs(size, 0);
        for(int i = 0; i < size; ++i) {
            diffs[i] = abs(s[i] - t[i]);
        }

        int res = 0;
        int left = -1;
        int total = 0;
        for(int right = 0; right < size; ++right) {
            total += diffs[right] ;
            while(total > maxCost) {
                ++left;
                total -= diffs[left];
            }
            res = max(res, right-left);
        }
        return res;
    }
};
