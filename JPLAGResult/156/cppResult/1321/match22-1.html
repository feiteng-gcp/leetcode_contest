class Solution {
public:
    int equalSubstring(string s, string t, int cost) {

        int ret = 0, N = s.size();
        vector<int> dp(N);
        for (int i = 0; i < N; i++) {
            dp[i] = abs(s[i] - t[i]);
        }
        int last = 0;
        for (int i = 0; i < N; i++) {

            cost -= dp[i];
            while (cost < 0) {
                cost += dp[last++];
            }
            ret = max(i - last + 1, ret);
        }
        return ret;
    }
};
