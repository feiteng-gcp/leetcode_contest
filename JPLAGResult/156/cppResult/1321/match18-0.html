class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int n = s.length();
        vector<int> arr(n);
        for ( int i = 0; i < n; ++i ) arr[i] = abs(s[i] - t[i]);
        int res = 0;
        int pt = 0;
        int sum = 0;
        for ( int i = 0; i < n; ++i ) {
            sum += arr[i];
            while ( pt <= i && sum > maxCost ) {
                sum -= arr[pt++];
            }
            res = max(res, i - pt + 1);
        }
        return res;
    }
};
