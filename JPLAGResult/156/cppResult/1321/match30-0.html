class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
        int N = s.size();
        vector<int> cost(N);
        for (int i = 0; i < N; ++i) {
            cost[i] = abs(s[i] - t[i]);
        }
        
        int ans = 0, sum = 0;
        for (int i = 0, j = 0; j < N;) {
            sum += cost[j];
            while (sum > maxCost && i < j) {
                sum -= cost[i++];
            }
            if (sum <= maxCost) ans = max(ans, j-i+1);
            ++j;
        }
        
        return ans;
    }
};
