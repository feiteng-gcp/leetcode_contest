class Solution {
public:
    int equalSubstring(string s, string t, int maxCost) {
    	int ret = 0, nowCost = 0, n = s.length();
    	for (int l = 0, r = 0; r < n; ++r) {

			nowCost += abs(s[r] - t[r]);
			while (nowCost > maxCost) {
				nowCost -= abs(s[l] - t[l]);
				++l;
			}
			ret = max(ret, r - l + 1);
		}
		return ret;
    }
};
