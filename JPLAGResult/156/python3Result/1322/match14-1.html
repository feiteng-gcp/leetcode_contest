class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        '''
        minmoves[ornt, row, col] = minimum moves required to reach a state where
        the snake has orientation 'ornt' and the top left cell of the snake is at the given row/column
        ornt must be 0 or 1 - 0 means horizontal snake, 1 means vertical snake
        '''
        n = len(grid)
        minmoves = []
        for i in range(2):
            sub = []
            for j in range(n):
                sub.append([None]*n)
            minmoves.append(sub)
        minmoves[0][0][0] = 0
        
        queue = [(0,0,0,0)] # ornt, row, col, #moves to get to that state
        while len(queue):
            next = queue.pop(0)
            ornt = next[0]
            row = next[1]
            col = next[2]
            numMoves = next[3]            
            # what are all the possible states?
            if not ornt: #horizontal
                # move down
                
if row+1 < n and minmoves[0][row+1][col] is None and not grid[row+1][col] and not grid[row+1][col+1]:
                    minmoves[0][row+1][col] = numMoves+1
                    queue.append((0,row+1,col,numMoves+1))
                # move right
                if col+2 < n and minmoves
[0][row][col+1] is None and not grid[row][col+2]:
                    minmoves[0][row][col+1] = numMoves+1
                    queue.append((0,row,col+1,numMoves+1))
                # flip
                if row+1 < n and col+1 < n and minmoves[1][row][col] is None and not grid[row+1][col] and not grid[row+1][col+1]:
                    minmoves[1][row][col] = numMoves+1
                    queue.append((1,row,col,numMoves+1))
            else: #vertical
                # move down
                if row+2 < n and minmoves[1][row+1][col] is None and not grid[row+2][col]:
                    minmoves[1][row+1][col] = numMoves+1
                    queue.append((1,row+1,col,numMoves+1))
                # move right
                if col+1 < n and minmoves[1][row][col+1] is None and not grid[row][col+1] and not grid[row+1][col+1]:
                    minmoves[1][row][col+1] = numMoves+1
                    queue.append((1,row,col+1,numMoves+1))
                # flip
                if row+1 < n and col+1 < n and minmoves[0][row]
[col] is None and not grid[row][col+1] and not grid[row+1][col+1]:
                    minmoves[0][row][col] = numMoves+1
                    queue.append((0,row,col,numMoves+1))
        #for row in minmoves[0]:
        #    print(row)
        #print("---")
        #for row in minmoves[1]:
        #   print(row)
        min = minmoves[0][n-1][n-2]
        if min is None:
            return -1
        return min
