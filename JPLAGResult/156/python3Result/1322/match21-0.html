class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        def ishor(ax,ay,bx,by):
            return ax==bx and ay+1==by
        visited={(0,0,0,1)}
        q=collections.deque()
        q.append((0,0,0,1))
        ans=0
        n=len(grid)
        tax,tay,tbx,tby=n-1,n-2,n-1,n-1
        while q:
            for _ in range(len(q)):
                ax,ay,bx,by=q.popleft()
                if (ax,ay,bx,by)==(tax,tay,tbx,tby):
                    return ans
                if ishor(ax,ay,bx,by):
                    if by+1<n and grid[bx][by+1]==0:
                        if (bx,by,bx,by+1) not in visited:
                            visited.add((bx,by,bx,by+1))
                            q.append((bx,by,bx,by+1))
                    if ax+1<n and grid[ax+1][ay]==0 and grid[bx+1][by]==0:
                        me=(ax+1,ay,bx+1,by)
                        if me not in visited:
                            visited.add(me)
                            q.append(me)
                        me=(ax,ay,ax+1,ay)
                        if me not in visited:
                            visited.add(me)
                            q.append(me)
                else:
                    if bx+1<n and grid[bx+1][by]==0:
                        me=(bx,by,bx+1,by)
                        if me not in visited:
                            visited.add(me)
                            q.append(me)
                    if ay+1<n and grid[ax][ay+1]==0 and grid[bx][by+1]==0:
                        me=(ax,ay+1,bx,by+1)
                        if me not in visited:
                            visited.add(me)
                            q.append(me)
                        me=(ax,ay,ax,ay+1)
                        if me not in visited:
                            visited.add(me)
                            q.append(me)
            ans+=1
        return -1
