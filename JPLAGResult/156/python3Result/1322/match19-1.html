from collections import deque
class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        
        q = deque([(1, 0, 1)]) #0,1,2,3 => north, east, south, west and i, j
        visited = set([(1, 0, 1)])
        n = len(grid)
        moves = 0
        while q:
            for _ in range(len(q)):
                direction, i, j= q.popleft()
                if direction==1 and i == n-1 and j == n-1:
                    return moves
                
                #right
                if direction == 1:
                    if j+1 < n and grid
[i][j+1] == 0:
                        state = (direction, i, j+1)
                        if state not in visited:
                            visited.add(state)
                            q.append(state)
                else:
                    if j+1 < n and grid
[i][j+1] == 0 and grid[i-1][j+1] == 0:
                        state = (direction, i, j+1)
                        if state not in visited:
                            visited.add(state)
                            q.append(state)
                
                #down
                if direction == 1:
                    if i+1<n and grid[i+1]
[j] == 0 and grid[i+1][j-1] == 0:
                        state = (direction, i+1, j)
                        if state not in visited:
                            visited.add(state)
                            q.append(state)
                else:
                    if i+1<n and grid[i+1][j] == 0:
                        state = (direction, i+1, j)
                        if state not in visited:
                            visited.add(state)
                            q.append(state)
                            
                            
                #clockwise
                if direction == 1:
                    if i+1 < n and grid
[i+1][j] == 0 and grid[i+1][j-1] == 0:
                        state = (0, i+1, j-1)
                        if state not in visited:
                            visited.add(state)
                            q.append(state)
                            
                else:
                    if j+1 < n and grid[i][j+1] == 0 and grid[i-1][j+1] == 0:
                        state 
= (1, i-1, j+1)
                        if state not in visited:
                            visited.add(state)
                            q.append(state)
                
            moves += 1
            
        return -1
