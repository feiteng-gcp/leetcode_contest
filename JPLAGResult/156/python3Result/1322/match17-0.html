class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        n = len(grid)
        if grid[n-1][n-1]==1 or grid[n-1][n-2]==1:
            return -1
        q = collections.deque()
        visited = set()
        ct = 0
        q.append((0,0,0,1))
        visited.add((0,0,0,1))
        while q:
            ct+=1
            l = len(q)
            for i in range(l):
                x1,y1,x2,y2 = q.popleft()
                #right
                newx1,newy1,newx2,newy2 = x1,y1+1,x2,y2+1
                if newy2<n and grid[newx1][newy1]==0 and grid[newx2][newy2]==0:
                    if (newx1,newy1,newx2,newy2)==(n-1,n-2,n-1,n-1):
                        return ct
                    if (newx1,newy1,newx2,newy2) not in visited:
                        q.append((newx1,newy1,newx2,newy2))
                        visited.add((newx1,newy1,newx2,newy2))
                #down
                newx1,newy1,newx2,newy2 = x1+1,y1,x2+1,y2
                if newx2<n and grid[newx1][newy1]==0 and grid[newx2][newy2]==0:
                    if (newx1,newy1,newx2,newy2)==(n-1,n-2,n-1,n-1):
                        return ct
                    if (newx1,newy1,newx2,newy2) not in visited:
                        q.append((newx1,newy1,newx2,newy2))
                        visited.add((newx1,newy1,newx2,newy2))
                #clock-wise
                if x1==x2:
                    newx1,newy1,newx2,newy2 = x1,y1,x1+1,y1
                    if newx2<n and grid[newx2][newy2]==0 and grid[x1+1][y1+1]==0:
                        if (newx1,newy1,newx2,newy2) not in visited:
                            q.append((newx1,newy1,newx2,newy2))
                            visited.add((newx1,newy1,newx2,newy2))
                #anti-clock-wise
                if y1==y2:
                    newx1,newy1,newx2,newy2 = x1,y1,x1,y1+1
                    if newy2<n and grid[newx2][newy2]==0 and grid[x1+1][y1+1]==0:
                        if (newx1,newy1,newx2,newy2) not in visited:
                            q.append((newx1,newy1,newx2,newy2))
                            visited.add((newx1,newy1,newx2,newy2))
        return -1
