class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        
        
        N = len(grid)
        if grid[0][0]==1 or grid[0][1]==1 or grid[-1][-2]==1 or grid[-1][-1]==1:
            return -1
        steps = 0
        level = {(0,0,'h')}  # 尾巴坐标x，尾巴坐标y，方位是水平h还是竖直v
        visited = {(0,0,'h')}
        while level != set():
            steps += 1
            nextlevel = set()
            for px,py,pdir in level:
                if pdir == 'h' and py+2 < N and grid[px][py+2] == 0:
                    curr = (px,py+1,pdir)
                    if curr == (N-1,N-2,'h'):
                        return steps
                    
if curr not in visited:
                        nextlevel.add(curr)
                        visited.add(curr)
                if pdir == 'v' and px+2 < N and grid[px+2][py] == 0:
                    curr = (px+1,py,pdir)
                    if curr not in visited:
                        nextlevel.add(curr)
                        visited.add(curr)
                if pdir == 'h' and px+1 < N and grid[px+1][py] == 0 and grid[px+1][py+1] == 0:
                    curr = (px,py,'v')
                    if curr not in visited:
                        nextlevel.add(curr)
                        visited.add(curr)
                    curr = (px+1,py,'h')
                    if curr == (N-1,N-2,'h'):
                        return steps
                    if curr not in visited:
                        nextlevel.add(curr)
                        visited.add(
curr)
                if pdir == 'v' and py+1 < N and grid[px][py+1] == 0 and grid[px+1][py+1] == 0:
                    curr = (px,py,'h')
                    if curr not in visited:
                        nextlevel.add(curr)
                        visited.add(curr)
                    curr = (px,py+1,'v')
                    if curr not in visited:
                        nextlevel.add(curr)
                        visited.add(curr)
            level = nextlevel
        return -1
