class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        n = len(grid)
        has = set()
        
        bfs = collections.deque()
        bfs.append((0, 1, True))
        has.add((
0, 1, True))
        
        step = 0
        while bfs:
            l = len(bfs)
            for _ in range(l):
                i, j, d = bfs.popleft()
                if i == n - 1 and j == n - 1 and d:
                    return step
                if d == True:
                    if j < n - 1 and grid
[i][j + 1] == 0:
                        if (i, j + 1, d) not in has:
                            has.add((i, j + 1, d))
                            bfs.append((i, j + 1, d))
                    if i < n - 1 and grid
[i + 1][j] == 0 and grid[i + 1][j - 1] == 0:
                        if (i + 1, j, d) not in has:
                            has.add((i + 1, j, d))
                            bfs.append((i + 1, j, d))
                        if (i + 1, j - 1, False) not in has:
                            has.add((i + 1, j - 1, False))
                            bfs.append((i + 1, j - 1, False))
                else:
                    if i < n - 1 and grid
[i + 1][j] == 0:
                        if (i + 1, j, d) not in has:
                            has.add((i + 1, j, d))
                            bfs.append((i + 1, j, d))
                    if j < n - 1 and grid
[i][j + 1] == 0 and grid[i - 1][j + 1] == 0:
                        if (i, j + 1, d) not in has:
                            has.add((i, j + 1, d))
                            bfs.append((i, j + 1, d))
                        if (i - 1, j + 1, True) not in has:
                            has.add((i - 1, j + 1, True))
                            bfs.append((i - 1, j + 1, True))
            step += 1
        
        return -1
