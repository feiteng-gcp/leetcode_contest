class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        res = 0
        q = {(0, 1, 0)}
        n = len(grid)
        visited = set()
        while q:
            nextq = set()
            for i, j, d in q:
                if (i, j) == (n-1, n-1) and d == 0:
                    return res
                if d == 0:
                    if j+1 < n and grid[i][j+1] == 0:
                        if (i, j+1, 0) not in visited:
                            visited.add((i, j+1, 0))
                            nextq.add((i,
 j+1, 0))
                    if i+1 < n and grid[i+1][j] == 0 and grid[i+1][j-1] == 0:
                        if (i+1, j, 0) not in visited:
                            visited.add((i+1, j, 0))
                            nextq.add((i+1, j, 0))
                        if (i+1, j-1, 1) not in visited:
                            visited.add((i+1, j-1, 1))
                            nextq.add((i+1, j-1, 1))
                else:
                    if i+1 < n and grid[i+1][j] == 0:
                        if (i+1, j, 1) not in visited:
                            visited.add((i+1, j, 1))
                            nextq.add((i+
1, j, 1))
                    if j+1 < n and grid[i-1][j+1] == 0 and grid[i][j+1] == 0:
                        if (i, j+1, 1) not in visited:
                            visited.add((i, j+1, 1))
                            nextq.add((i, j+1, 1))
                        if (i-1, j+1, 0) not in visited:
                            visited.add((i-1, j+1, 0))
                            nextq.add((i-1, j+1, 0))
                q = nextq
            res += 1
        return -1
