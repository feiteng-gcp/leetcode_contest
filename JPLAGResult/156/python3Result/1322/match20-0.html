class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        n = len(grid)
        if n == 2:
            return 1 if sum(grid[1]) == 0 else -1
        start = (0, 0, 0, 1)
        finish = (n - 1, n - 2, n - 1, n - 1)
        moves = {}
        level = [start]
        visited = set()
        steps = 0
        while level:
            next_level = []
            for r1, c1, r2, c2 in level:
                cur_pos = (r1, c1, r2, c2)
                moves[cur_pos] = steps
                # try right
                if max(c1, c2) < n - 1 and sum([grid[r1][c1 + 1], grid[r2][c2 + 1]]) == 0:
                    new_pos = (r1, c1 + 1, r2, c2 + 1)
                    if new_pos not in moves and new_pos not in visited:
                        next_level.append(new_pos)
                        visited.add(new_pos)
                # try down
                if max(r1, r2) < n - 1 and sum([grid[r1 + 1][c1], grid[r2 + 1][c2]]) == 0:
                    new_pos = (r1 + 1, c1, r2 + 1, c2)
                    if new_pos not in moves and new_pos not in visited:
                        next_level.append(new_pos)
                        visited.add(new_pos)
                # try rotation
                o = 'h' if c1 != c2 else 'v'
                if o == 'h' and r1 < n - 1 and sum([grid[r1 + 1][c1], grid[r2 + 1][c2]]) == 0:
                    new_pos = (r1, c1, r2 + 1, c2 - 1)
                    if new_pos not in moves and new_pos not in visited:
                        next_level.append(new_pos)
                        visited.add(new_pos)
                if o == 'v' and c1 < n - 1 and sum([grid[r1][c1 + 1], grid[r2][c2 + 1]]) == 0:
                    new_pos = (r1, c1, r2 - 1, c2 + 1)
                    if new_pos not in moves and new_pos not in visited:
                        next_level.append(new_pos)
                        visited.add(new_pos)
            level = next_level
            steps += 1
        if finish in moves:
            return moves[finish]
        return -1
