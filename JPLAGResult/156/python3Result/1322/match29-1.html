class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        n = len(grid)
        ans = [[[None]*n for _ in range(n)], [[None]*n for _ in range(n)]]
        
        ans[1][n-1][n-2] = 0
        stack = [(1, n-1, n-2)]
        while stack:
            p, i, j = stack.pop(0)
            if p == 0:
                if i - 1 >= 0 and grid
[i-1][j] == 0 and ans[p][i-1][j] == None:
                    ans[p][i-1][j] = ans[p][i][j] + 1
                    stack.append((p, i-1, j))
                if j - 1 >= 0 and grid[i]
[j-1] == 0 and grid[i+1][j-1] == 0 and ans[p][i][j-1] == None:
                    ans[p][i][j-1] = ans[p][i][j] + 1
                    stack.append((p, i, j - 1))
                if j + 1 < n and grid[i][j+1] == 0 and grid[i+1][j+1] == 0 and ans[p^1][i][j] == None:
                    ans[p^1][i][j] = ans[p][i][j] + 1
                    stack.append((p^1, i, j))
            if p == 1:
                if j - 1 >= 0 and grid[i][j-1] == 0 and ans[p]
[i][j - 1] == None:
                    ans[p][i][j - 1] = ans[p][i][j] + 1
                    stack.append((p, i, j-1))
                if i - 1 >= 0 and grid[i-1]
[j] == 0 and grid[i-1][j+1] == 0 and ans[p][i-1][j] == None:
                    ans[p][i-1][j] = ans[p][i][j] + 1
                    stack.append((p, i-1, j))
                if i + 1 < n and grid[i+1][j] == 0 and grid[i+1][j+1] == 0 and ans[p^1][i][j] == None:
                    ans[p^1][i][j] = ans[p][i][j] + 1
                    stack.append((p^1, i, j))
        if ans[1][0][0]!= None:
            return ans[1][0][0]
        else:
            return -1
