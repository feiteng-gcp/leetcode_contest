class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        n = len(grid)
        q = []
        vis = [[[True, True] for _ in range(n)] for _ in range(n)]
        if grid[0][0] == grid[0][1] == 0:
            q.append([0, 1, 0, 0])
            vis.append([0, 1, 0])
            vis[0][1][0] = False
        else:
            return -1
        while (q):
            x, y, d, s = q.pop(0)
            if x == y == n - 1 and d == 0:
                return s
            if d == 0:
                
if y + 1 < n and grid[x][y + 1] == 0 and vis[x][y+1][0]:
                    q.append([x, y + 1, 0, s + 1])
                    vis[x][y+1][0] = False
                if x + 1 < n and y > 0 and grid[x + 1][y] == 0 and grid[x + 1][y - 1] == 0:
                    if vis
[x+1][y-1][1]:
                        q.append([x + 1, y - 1, 1, s + 1])
                        vis[x+1][y-1][1] = False
                    if vis[x+1][y][0]:
                        q.append([x + 1, y, 0, s + 1])
                        vis[x+1][y][0] = False

            elif d == 1:
                if x+1 < n and grid[x + 1][y] == 0 and vis[x+1][y][1]:
                    q.append([
x + 1, y, 1, s + 1])
                    vis.append([x + 1, y, 1])
                    vis[x+1][y][1] = False
                if y + 1 < n and x > 0 and d == 1 and grid[x][y + 1] == 0 and grid[x - 1][y + 1] == 0:
                    if vis
[x-1][y+1][0]:
                        q.append([x - 1, y + 1, 0, s + 1])
                        vis[x-1][y+1][0] = False
                    if vis[x][y+1][1]:
                        q.append([x, y + 1, 1, s + 1])
                        vis[x][y+1][1] = False
        return -1
