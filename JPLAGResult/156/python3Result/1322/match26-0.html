class Solution:
    def minimumMoves(self, grid):
        MAX_VALUE = 0x7fffffff
        n = len(grid)
        ans = [[[MAX_VALUE, MAX_VALUE] for i in range(n)] for j in range(n)]
        ans[0][0][0] = 0
        if grid[1][0] == 0 and grid[1][1] == 0:
            ans[0][0][1] = 1

        for i in range(n):
            for j in range(n):
                if i == j == 0:
                    continue
                if grid[i][j] == 1:
                    continue
                # hori
                if j < n - 1 and grid[i][j + 1] == 0:
                    ans[i][j][0] = min(ans[i][j][0], ans[i - 1][j][0] + 1 if i > 0 else MAX_VALUE, ans[i][j - 1][0] + 1 if j > 0 else MAX_VALUE)

                # vert
                if i < n - 1 and grid[i + 1][j] == 0:
                    ans[i][j][1] = min(ans[i][j][1], ans[i - 1][j][1] + 1 if i > 0 else MAX_VALUE, ans[i][j - 1][1] + 1 if j > 0 else MAX_VALUE)
                    if j < n - 1 and grid[i + 1][j] == 0 and grid[i + 1][j + 1] == 0:
                        ans[i][j][1] = min(ans[i][j][1], ans[i][j][0] + 1)
                # hori
                if j < n - 1 and grid[i][j + 1] == 0:
                    if i < n - 1 and grid[i + 1][j] == 0 and grid[i + 1][j + 1] == 0:
                        ans[i][j][0] = min(ans[i][j][0], ans[i][j][1] + 1)

        # print("hori")
        # for i in range(n):
        #     print([ans[i][j][0] for j in range(n)])
        # 
        # print("vert")
        # for i in range(n):
        #     print([ans[i][j][1] for j in range(n)])
        return -1 if ans[-1][-2][0] == MAX_VALUE else ans[-1][-2][0]
