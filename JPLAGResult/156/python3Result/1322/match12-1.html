class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        # BFS
        n = len(grid)
        q = collections.deque([(0, 0, 0, 1)])
        seen = set([(0,0,0,1)])
    
        step = 0
        while q:
            size = len(q)
            for _ in range(size):
                r1, c1, r2, c2 = q.popleft()
                if r1 == n-1 and c1 == n-2 and r2 == n-1 and c2 == n-1:
                    return step
                if r1 == r2: # horizontal snake
                    self.horizontal_next(r1, c1, r2, c2, grid, seen, q)
                elif c1 == c2:
                    self.vertical_next(r1, c1, r2, c2, grid, seen, q)
            
            step += 1
        
        return -1
    
    def horizontal_next(self, r1, c1, r2, c2, grid, seen, q):
        # r1 == r2 and c1+1 == c2, snake is horizontal
        # 1) move right
        if 0 <= c2+1 < len(grid) and grid
[r2][c2+1] == 0:
            if (r1, c1+1, r2, c2+1) not in seen:
                q.append((r1, c1+1, r2, c2+1))
                seen.add((r1, c1+1, r2, c2+1))
            
        if 0 <= r1+1 < len(grid) and grid
[r1+1][c1] == 0 and grid[r2+1][c2] == 0:
            # 2) move down
            if (r1+1, c1, r2+1, c2) not in seen:
                q.append((r1+1, c1, r2+1, c2))
                seen.add((r1+1, c1, r2+1, c2))
            
            # 3) rotate clockwise
            if (r1, c1, r1+1, c1) not in seen:
                q.append((r1, c1, r1+1, c1))
                seen.add((r1, c1, r1+1, c1))
    
    def vertical_next(self, r1, c1, r2, c2, grid, seen, q):
        # c1 == c2 and r1+1 == r2, snake is vertical
        
        # 1) move down
        if 0 <= r2+1 < len(grid) and grid
[r2+1][c1] == 0:
            if (r1+1, c1, r2+1, c2) not in seen:
                q.append((r1+1, c1, r2+1, c2))
                seen.add((r1+1, c1, r2+1, c2))
        
        if 0 <= c1+1 < len(grid) and grid
[r1][c1+1] == 0 and grid[r2][c2+1] == 0:
        # 2) move right
            if (r1, c1+1, r2, c2+1) not in seen:
                q.append((r1, c1+1, r2, c2+1))
                seen.add((r1, c1+1, r2, c2+1))
                
        # 3) rotate counterclockwise
            if (r1, c1, r1, c2+1) not in seen:
                q.append((r1, c1, r1, c2+1))
                seen.add((r1, c1, r1, c2+1))
