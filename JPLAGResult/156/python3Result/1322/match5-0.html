class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        
        n = len(grid)
        seen = set()
        seen.add((0, 0, 0, 1))
        q = [(0, 0, 0, 1)]
        cnt = 1
        while q:
            nq = []
            for x1, y1, x2, y2 in q:
                
                if x1 == x2:
                    if y2 + 1 < n and grid[x2][y2 + 1] == 0:
                        if (x2, y2, x2, y2 + 1) == (n-1, n-2, n-1, n-1):
                            return cnt
                        if (x2, y2, x2, y2 + 1) not in seen:
                            seen.add((x2, y2, x2, y2 + 1))
                            nq.append((x2, y2, x2, y2 + 1))
                    if x2 + 1 < n and grid[x2 + 1][y1] == 0 and grid[x2 + 1][y2] == 0:
                        if (x2 + 1, y1, x2 + 1, y2) == (n-1, n-2, n-1, n-1):
                            return cnt
                        if (x2 + 1, y1, x2 + 1, y2) not in seen:
                            seen.add((x2 + 1, y1, x2 + 1, y2))
                            nq.append((x2 + 1, y1, x2 + 1, y2))
                        if (x1, y1, x1 + 1, y1) not in seen:
                            seen.add((x1, y1, x1 + 1, y1))
                            nq.append((x1, y1, x1 + 1, y1))
                else:
                    if x2 + 1 < n and grid[x2 + 1][y2] == 0:
                        if (x2, y2, x2 + 1, y2) not in seen:
                            seen.add((x2, y2, x2 + 1, y2))
                            nq.append((x2, y2, x2 + 1, y2))
                    if y2 + 1 < n and grid[x1][y2 + 1] == 0 and grid[x2][y2 + 1] == 0:
                        if (x1, y2 + 1, x2, y2 + 1) not in seen:
                            seen.add((x1, y2 + 1, x2, y2 + 1))
                            nq.append((x1, y2 + 1, x2, y2 + 1))
                        if (x1, y1, x1, y1 + 1) not in seen:
                            seen.add((x1, y1, x1, y1 + 1))
                            nq.append((x1, y1, x1, y1 + 1))
            q = nq
            cnt += 1
        return -1
