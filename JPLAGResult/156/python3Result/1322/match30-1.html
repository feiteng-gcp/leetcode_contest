from collections import deque
class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        n = len(grid)
        q = deque()
        q.append((0, 0, 0))
        mp = dict()
        mp[(0, 0, 0)] = 0
        while q:
            r, c, o = q.popleft()
            cost = mp[(r, c, o)]
            if (r, c, o) == (n-1, n-2, 0):
                return cost
            if o == 0:
                if c + 2 < n and grid
[r][c+2] == 0 and (r, c+1, o) not in mp:
                    mp[(r, c+1, o)] = cost + 1
                    q.append((r, c+1, o))
                if r + 1 < n and c + 1 < n and grid[r+1][c] == 0 and grid[r+1][c+1] == 0:
                    if (
r+1, c, o) not in mp:
                        mp[(r+1, c, o)] = cost + 1
                        q.append((r+1, c, o))
                    if (r, c, 1-o) not in mp:
                        mp[(r, c, 1-o)] = cost + 1
                        q.append((r, c, 1-o))
            else:
                if r + 2 < n and grid
[r+2][c] == 0 and (r+1, c, o) not in mp:
                    mp[(r+1, c, o)] = cost + 1
                    q.append((r+1, c, o))
                if r + 1 < n and c + 1 < n and grid[r][c+1] == 0 and grid[r+1][c+1] == 0:
                    if (
r, c+1, o) not in mp:
                        mp[(r, c+1, o)] = cost + 1
                        q.append((r, c+1, o))
                    if (r, c, 1-o) not in mp:
                        mp[(r, c, 1-o)] = cost + 1
                        q.append((r, c, 1-o))
        return -1
