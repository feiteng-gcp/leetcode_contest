class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        visited = set()
        
        queue = [(0, 0, False)]
        visited.add(tuple(queue[0]))
        
        count = 0
        while queue:
            level = []
            
            for r, c, isVertical in queue:
                if r == len(grid) - 1 and c == len(grid[0]) - 2 and not isVertical:
                    return count
            
                if isVertical:
                    #right
                    if r + 1 < len(grid) and c + 1 < len(grid[0]) and grid
[r][c + 1] != 1 and grid[r + 1][c + 1] != 1:
                        nextMove = (r, c + 1, isVertical)
                        if nextMove not in visited:
                            level.append(nextMove)
                            visited.add(nextMove)
                            
                        nextMove = (r, c, not isVertical)
                        if nextMove not in visited:
                            level.append(nextMove)
                            visited.add(nextMove)
                            
                    #down
                    if r + 2 < len(grid) and grid
[r + 2][c] != 1:
                        nextMove = (r + 1, c, isVertical)
                        if nextMove not in visited:
                            level.append(nextMove)
                            visited.add(nextMove)
                            
                else:
                    #right
                    if c + 2 < len(grid[0]) and grid[r][c + 2] != 1:
                        nextMove = (r, c + 1, isVertical)
                        if nextMove not in visited:
                            level.append(nextMove)
                            visited.add(nextMove)
                            
                    #down
                    if r + 1 < len(grid) and c + 1 < len(grid[0]) and grid
[r + 1][c] != 1 and grid[r + 1][c + 1] != 1:
                        nextMove = (r + 1, c, isVertical)
                        if nextMove not in visited:
                            level.append(nextMove)
                            visited.add(nextMove)
                        nextMove = (r, c, not isVertical)
                        if nextMove not in visited:
                            level.append(nextMove)
                            visited.add(nextMove)
            
            count += 1
            queue = level
            
        
        return -1
