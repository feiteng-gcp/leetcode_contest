class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        height = len(grid)
        width = len(grid[0])

        dp = []
        for i in range(height):
            row = []
            for j in range(width):
                row.append([1000, 1000])
            dp.append(row)

        if grid[0][0] == 1 or grid[0][1] == 1:
            return -1

        dp[0][0][0] = 0

        for i in range(height):
            for j in range(width):
                
if grid[i][j] == 1:
                    # pass if current cell is blocked
                    continue
                if 0 < j < width - 1 and grid[i][j+1] == 0:
                    dp[i][j][0] = min(dp
[i][j][0], dp[i][j-1][0]+1)
                if 0 < i < height - 1 and grid[i+1][j] == 0:
                    dp[i][j][1] = min(dp[i][j][1], dp[i-1][j][1] + 1)
                if i < height - 1 and j < width - 1:
                    if grid[i+1][j] == 0 and grid
[i][j+1] == 0 and grid[i+1][j+1] == 0:
                        dp[i][j][0] = min(dp[i][j][0], dp[i][j][1] + 1)
                        dp[i][j][1] = min(dp[i][j][1], dp[i][j][0] + 1)
                        dp[i+1][j][0] = min(dp[i+1][j]
[0], dp[i][j][0]+1)
                        dp[i][j+1][1] = min(dp[i][j+1][1], dp[i][j][1]+1)


        if dp[height-1][width-2][0] == 1000:
            return -1
        else:
            return dp[height-1][width-2][0]
