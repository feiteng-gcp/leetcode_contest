class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        mm = 1000000
        dp = [[[mm, mm] for _ in range(len(grid[0]))] for _ in range(len(grid))]
        # i,j,k: i, j and direction: 0=right 1=down
        
        # -2 is blocked
        # mm is unvisited
        m = len(grid)
        n = len(grid[0])
        dp[0][0][0] = 0
        
        for i in range(m):
            for j in range(n):
                if grid[i][j]:
                    # is wall
                    continue
                   
                if (i<m-1) and (j<n-1) and (grid[i][j]==0) and (grid[i+1][j]==0) and (grid[i+1][j+1]==0) and (grid
[i][j+1]==0):
                    dp[i][j][0] = min(dp[i][j][1]+1, dp[i][j][0])
                    dp[i][j][1] = min(dp[i][j][0]+1, dp[i]
[j][1])
                    dp[i+1][j][0] = min(dp[i][j][0]+1, dp[i+1][j][0])
                    dp[i][j+1][1] = min(dp[i][j][1]+1, dp[i][j+1][1])
                    
                
                
if (j < n-2) and (grid[i][j+1]==0) and (grid[i][j+2]==0):
                    dp[i][j+1][0] = min(dp[i][j][0] + 1, dp[i][j+1][0])
                
                
if (i < m-2) and (grid[i+1][j]==0) and (grid[i+2][j]==0):
                    dp[i+1][j][1] = min(dp[i][j][1] + 1, dp[i+1][j][1])
                    
        if dp[n-1][n-2][0] == mm:
            return -1
        else:
            return dp[n-1][n-2][0]
