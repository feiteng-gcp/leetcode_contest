class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        n = len(grid)
        
        visited = set([((0,1), (0,0))])
        # print(visited)
        
        q = [((0,1), (0,0))]
        
        directions = [(1,0),(0,1)]
        
        ans = 0
        while True:
            tmp = []
            while q:
                # print(q)
                head, tail = q.pop(0)
                xh,yh = head
                xt,yt = tail
                # if (head,tail) in [((n-1,n-2),(n-1,n-1)),((n-1,n-1),(n-1,n-2))]:
                if (head,tail) == ((n-1,n-1),(n-1,n-2)):
                    return ans

                for dx,dy in directions:
                    xxh,yyh = xh+dx, yh+dy
                    xxt,yyt = xt+dx, yt+dy
                    if 0<=xxh<n and 0<=yyh<n and 0<=xxt<n and 0<=yyt<n and ((xxh,yyh),(xxt,yyt)) not in visited \
                    and grid[xxh][yyh]==0 and grid[xxt][yyt]==0:
                        tmp.append(((xxh,yyh),(xxt,yyt)))
                        visited.add(((xxh,yyh),(xxt,yyt)))

                xxt,yyt = xt,yt
                #clockwise
                if xh==xt and yh==yt+1:
                    xxh,yyh = xh+1, yh-1
                    if 0<=xxh<n and 0<=yyh<n and 0<=xxt<n and 0<=yyt<n and ((xxh,yyh),(xxt,yyt)) not in visited \
                        and grid[xxh][yyh]==0 and grid[xxt][yyt]==0 and grid[xt+1][yt+1]==0:
                        tmp.append(((xxh,yyh),(xxt,yyt)))
                        visited.add(((xxh,yyh),(xxt,yyt)))

                # anticlockwise
                if xh==xt+1 and yh==yt:
                    xxh,yyh = xh-1, yh+1
                    if 0<=xxh<n and 0<=yyh<n and 0<=xxt<n and 0<=yyt<n and ((xxh,yyh),(xxt,yyt)) not in visited \
                        and grid[xxh][yyh]==0 and grid[xxt][yyt]==0 and grid[xt+1][yt+1]==0:
                        tmp.append(((xxh,yyh),(xxt,yyt)))
                        visited.add(((xxh,yyh),(xxt,yyt)))
            ans+=1
            q = tmp
            if not tmp:
                break
        # print(visited)
        # print(ans)
        return -1
