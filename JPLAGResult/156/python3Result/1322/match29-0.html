from collections import deque

class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        self.VER = 0
        self.HOR = 1
        r = len(grid)
        c = len(grid[0])
        
        final_state = ((r-1, c-1), self.VER)
        
        q = deque()
        visited = set()
        init_state = ((0, 1), self.VER)
        
        q.append(init_state)
        visited.add(init_state)
        
        result = 0
        while q:
            result += 1
            for _ in range(len(q)):
                state_prev = q.popleft()
                states = self.get_next_states(grid, state_prev)
                for state in states:
                    if state == final_state:
                        return result
                    if state not in visited:
                        visited.add(state)
                        q.append(state)
        return -1
                
    def get_next_states(self, grid, state):
        pos, dir = state[0], state[1]
        result = []
        r = len(grid)
        c = len(grid[0])
        if dir == self.VER:
            # right
            if pos[1] + 1 < c and grid[pos[0]][pos[1] + 1] == 0:
                next_right = (pos[0], pos[1] + 1)
                result.append((next_right, dir))
            # down
            if pos[0] + 1 < r and grid[pos[0] + 1][pos[1]] == 0 and grid[pos[0] + 1][pos[1] - 1] == 0:
                next_down = (pos[0] + 1, pos[1])
                result.append((next_down, dir))
            # clockwise
            if  pos[0] + 1 < r and grid[pos[0] + 1][pos[1] - 1] == 0 and grid[pos[0] + 1][pos[1]] == 0:
                next = (pos[0] + 1, pos[1] - 1)
                result.append((next, self.HOR))
        else:
            # right
            if pos[1] + 1 < c and grid[pos[0]][pos[1] + 1] == 0 and grid[pos[0] - 1][pos[1] + 1] == 0:
                next_right = (pos[0], pos[1] + 1)
                result.append((next_right, dir))
            # down
            if pos[0] + 1 < r and grid[pos[0] + 1][pos[1]] == 0:
                next_down = (pos[0] + 1, pos[1])
                result.append((next_down, dir))
            
            # counterclock
            if pos[1] + 1 < c and grid[pos[0] - 1][pos[1] + 1] == 0 and grid[pos[0]][pos[1] + 1] == 0:
                next = (pos[0] - 1, pos[1] + 1)
                result.append((next, self.VER))
        return result
