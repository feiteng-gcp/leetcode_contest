from collections import deque
class Solution(object):
    def minimumMoves(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        if not grid or not grid[0]:
            return -1

        n = len(grid)
        stack = deque([(0, 0, 0)])# (x, y, h/v)
        visited = [[[False, False] for _ in range(n)] for _ in range(n)]

        res = 0
        while stack:
            size = len(stack)
            while size > 0:
                x, y, direc = stack.popleft()

                if x == n - 1 and y == n - 2 and direc == 0:
                    return res



                if y + 2 < n and direc == 0 and not grid[x][y + 2] and not visited[x][y + 1][0]:
                    stack.append((x, y + 1, 0))
                    visited[x][y + 1][0] = True
                if y + 1 < n and direc == 1 and not grid[x][y + 1] and not grid[x + 1][y + 1] and not visited[x][y + 1][1]:
                    stack.append((x, y + 1, 1))
                    visited[x][y + 1][1] = True
                if x + 2 < n and direc == 1 and not grid[x + 2][y] and not visited[x + 1][y][1]:
                    stack.append((x + 1, y, 1))
                    visited[x + 1][y][1] = True
                if x + 1 < n and direc == 0 and not grid[x + 1][y] and not grid[x + 1][y + 1] and not visited[x + 1][y][0]:
                    stack.append((x + 1, y, 0))
                    visited[x + 1][y][0] = True
    
                if direc == 0 and x + 1 < n and not grid[x + 1][y] and not grid[x + 1][y + 1] and not visited[x][y][1]:
                    stack.append((x, y, 1))
                    visited[x][y][1] = True
                if direc == 1 and y + 1 < n and not grid[x][y + 1] and not grid[x + 1][y + 1] and not visited[x][y][0]:
                    stack.append((x, y, 0))
                    visited[x][y][0] = True

                size -= 1
        
            res += 1

        return -1
