class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        q = collections.deque()
        hsh = set()
        hsh.add((0, 0, 0))
        q.append((0, 0, 0, 0))
        m = len(grid)
        n = len(grid[0])
        while len(q) > 0:
            (x, y, d, st) = q.popleft()
            # print(x, y, d, st)
            if (x == m-1 and y == n-2 and d == 0):
                return st
            if d == 0 and y + 2 < n and grid[x][y+2] == 0 and (x, y+1, 0) not in hsh:
                q.append((x, y+1, 0, st+1))
                hsh.add((x, y+1, 0))
            if d == 0 and x + 1 < m and grid[x+1][y] == 0 and grid[x+1][y+1] == 0 and (x+1, y, 0) not in hsh:
                q.append((x+1, y, 0, st+1))
                hsh.add((x+1, y, 0))
            if d == 1 and x + 2 < m and grid[x+2][y] == 0 and (x+1, y, 1) not in hsh:
                q.append((x+1, y, 1, st+1))
                hsh.add((x+1, y, 1))
            if d == 1 and y + 1 < n and grid[x][y+1] == 0 and grid[x+1][y+1] == 0 and (x, y+1, 1) not in hsh:
                q.append((x, y+1, 1, st+1))
                hsh.add((x, y+1, 1))
            if d == 0 and x + 1 < m and grid[x+1][y] == 0 and grid[x+1][y+1] == 0 and (x, y, 1) not in hsh:
                q.append((x, y, 1, st+1))
                hsh.add((x, y, 1))
            if d == 1 and y + 1 < n and grid[x][y+1] == 0 and grid[x+1][y+1] == 0 and (x, y, 0) not in hsh:
                q.append((x, y, 0, st+1))
                hsh.add((x, y, 0))
        return -1
