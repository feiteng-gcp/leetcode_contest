class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        
        moves = [(0, 1), (1, 0)]
        # rotates = [(1, -1), (-1, 1)]
        
        N = len(grid)
        
        queue = collections.deque([(0, 0, 0, 1, 0)])
        step = 0
        seen = set([(0, 0, 0, 1)])
        
        while queue:
            # print(queue)
            for _ in range(len(queue)):
                tailR, tailC, headR, headC, state = queue.popleft()
                if (tailR, tailC, headR, headC) == (N - 1, N - 2, N - 1, N - 1): 
                    return step
                for m0, m1 in moves:
                    newTailR, newTailC = tailR + m0, tailC + m1
                    newHeadR, newHeadC = headR + m0, headC + m1
                    
if newHeadR < N and newHeadC < N and grid[newHeadR][newHeadC] != 1 and grid[newTailR][newTailC] != 1 and (newTailR, newTailC, newHeadR, newHeadC) not in seen:
                        queue.append((newTailR, newTailC, newHeadR, newHeadC, state))
                        seen.add((newTailR, newTailC, newHeadR, newHeadC))
                if state == 0:
                    newHeadR, newHeadC = headR + 1, headC - 1
                    
if newHeadR < N and newHeadC < N and grid[newHeadR][newHeadC] != 1 and grid[tailR+1][tailC+1] != 1 and (tailR, tailC, newHeadR, newHeadC) not in seen:
                        queue.append((tailR, tailC, newHeadR, newHeadC, 1))
                        seen.add((tailR, tailC, newHeadR, newHeadC))
                elif state == 1:
                    newHeadR, newHeadC = headR - 1, headC + 1
                    
if newHeadR < N and newHeadC < N and grid[newHeadR][newHeadC] != 1 and grid[tailR+1][tailC+1] != 1 and (tailR, tailC, newHeadR, newHeadC) not in seen:
                        queue.append((tailR, tailC, newHeadR, newHeadC, 0))
                        seen.add((tailR, tailC, newHeadR, newHeadC))
            step += 1
        
        return -1
