class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        cur = [(0,0,0,1)]
        cnt = 0
        seen = set(cur)
        n = len(grid)
        while cur and (n-1,n-2,0,1) not in cur:
            cnt += 1
            tmp = []
            for x,y,dx,dy in cur:
                if dx==0:
                    if y+2 < n and grid[x][y+2] == 0:
                        tmp += [(x,y+1,dx,dy)]
                    if x+1 < n and grid[x+1][y] == 0 and grid[x+1][y+1] == 0:
                        tmp += [(x,y,1,0)]
                        tmp += [(x+1,y,0,1)]
                if dy == 0:
                    if x+2 < n and grid[x+2][y] == 0:
                        tmp += [(x+1,y,dx,dy)]
                    if y+1 < n and grid[x][y+1] == 0 and grid[x+1][y+1] == 0:
                        tmp += [(x,y,0,1)]
                        tmp += [(x,y+1,1,0)]
            tmp = set(tmp)
            cur = tmp - seen
            seen |= tmp
        if cur:
            return cnt
        return -1
