class Solution:
    def minimumMoves(self, grid: List[List[int]]) -> int:
        n = len(grid)
        q = collections.deque([(0, 0, 0), (0, 0, None)])
        res = 0
        visited = {(0, 0, 0)}
        while q:
            i, j, k = q.popleft()
            if k is None:
                if q:
                    q.append((0, 0, None))
                res += 1
            else:
                if i == n - 1 and j == n - 2 and k == 0:
                    return res
                if k == 0:
                    if j + 2 < n and grid[i][j+2] == 0 and (i, j+1, 0) not in visited:
                        visited.add((i, j+1, 0))
                        q.append((i, j+1, 0))
                    if i + 1 < n and grid[i+1][j] == 0 and grid[i+1][j+1] == 0:
                        if (i+1, j, 0) not in visited:
                            visited.add((i+1, j, 0))
                            q.append((i+1, j, 0))
                        if (i, j, 1) not in visited:
                            visited.add((i, j, 1))
                            q.append((i, j, 1))
                else:
                    if i + 2 < n and grid[i+2][j] == 0 and (i+1, j, 1) not in visited:
                        visited.add((i+1, j, 1))
                        q.append((i+1, j, 1))
                    if j + 1 < n and grid[i][j+1] == 0 and grid[i+1][j+1] == 0:
                        if (i, j+1, 1) not in visited:
                            visited.add((i, j+1, 1))
                            q.append((i, j+1, 1))
                        if (i, j, 0) not in visited:
                            visited.add((i, j, 0))
                            q.append((i, j, 0))
        return -1
