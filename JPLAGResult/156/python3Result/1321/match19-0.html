class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        
        costs = []
        
        for i in range(len(s)):
            costs.append(abs(ord(s[i])-ord(t[i])))
        

        left = 0
        out, curr = 0, 0
        for right in range(len(s)):
            curr += costs[right]
            while curr > maxCost:
                curr -= costs[left]
                left += 1
            out = max(out, right-left+1)
        return out
