class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        costs = [abs(ord(a) - ord(b)) for a, b in zip(s, t)]
        curcost = 0
        left = -1
        ret = 0
        for i, c in enumerate(costs):
            curcost += c
            while curcost > maxCost:
                left += 1
                curcost -= costs[left]
            ret = max(ret, i - left)
        return ret
