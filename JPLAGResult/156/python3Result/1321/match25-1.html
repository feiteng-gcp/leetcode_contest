from bisect import bisect_left


class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        cost = [abs(ord(c1)-ord(c2)) for c1,c2 in zip(s, t)]
        ps = [0]
        s = 0
        ans = 0
        for i, c in enumerate(cost, 1):
            s += c
            j = bisect_left(ps, s - maxCost)
            ans = max(ans, i - j)
            ps.append(s)
        return ans
