class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        costs = []
        for x,y in zip(s,t):
            costs.append(abs(ord(x) - ord(y)))
        
        start, end = 0, 0
        cur_cost = 0
        max_len = 0
        while end < len(s):
            cur_cost += costs[end]
            while cur_cost > maxCost:
                cur_cost -= costs[start]
                start += 1

            max_len = max(max_len, end - start + 1)
            end += 1
            
        return max_len
