class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        cost_arr = []
        for i in range(0, len(s)):
            cost_arr.append(abs(ord(s[i]) - ord(t[
i])))
        l = 0
        r = 0
        ans = 0
        while r < len(cost_arr):
            maxCost -= cost_arr[r]
            while maxCost < 0:
                maxCost += cost_arr[l]
                l += 1
            ans = max(ans, r - l + 1)
            r += 1
        return ans
