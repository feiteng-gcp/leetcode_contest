class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        # a b c d
        # b c d f
        # 1 1 1 2 => maximum subarray sum = target
        if len(s) != len(t):
            return 0
        
        n = len(s)
        arr = [abs(ord(s[i]) - ord(t[i])) for i in range(n)]
        left, curSum 
= 0, 0
        maxLen = 0
        for i, x in enumerate(arr):
            curSum += x
            while curSum > maxCost:
                curSum -= arr[left]
                left += 1
            maxLen = max(maxLen, i - left + 1)
        return maxLen
