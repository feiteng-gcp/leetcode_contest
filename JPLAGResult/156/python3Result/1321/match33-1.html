class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        gaps = [abs(ord(c1)-ord(c2)) for c1, c2 in zip(s, t)]
        agg, prev, ans, i 
= 0, {}, 0, 0
        for j, x in enumerate(gaps):
            agg += x
            while agg > maxCost:
                agg -= gaps[i]
                i += 1
            ans = max(ans, j-i+1)
        return ans
