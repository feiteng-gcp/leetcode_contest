class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        n = len(s)
        left = 0
        cost = 0
        ans = 0
        for right in range
(n):
            cost += abs(ord(t[right]) - ord(s[right]))
            while cost > maxCost:
                cost -= abs(ord(t[left]) - ord(s[left]))
                left += 1
            ans = max(ans, right - left + 1)
        return ans
