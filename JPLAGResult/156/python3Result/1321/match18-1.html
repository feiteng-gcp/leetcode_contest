class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        costs = []
        for i in range(len(s)):
            costs.append(abs(ord(s[i]) - ord(t[i])))
        if min(costs) > maxCost:
            return 0
        window = costs[0]
        j 
= 0
        longest = 1
        for i in range(1, len(costs)):
            window += costs[i]
            while j < i and window > maxCost:
                window -= costs[j]
                j += 1
            if window <= maxCost:
                longest = max(longest, i - j + 1)
        return longest
