class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        costs = [abs(ord(s[i]) - ord(t[i])) for i in range(len(s))]
        l = r = 0
        curTotalCost 
= 0
        res = 0
        for r in range(len(s)):
            curTotalCost += costs[r]
            while curTotalCost > maxCost and l <= r:
                curTotalCost -= costs[l]
                l += 1
            res = max(res, r - l + 1)
        return res
