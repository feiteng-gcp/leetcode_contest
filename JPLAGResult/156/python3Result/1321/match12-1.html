class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        n = len(s)
        arr = []
        for i in range
(n):
            arr.append(abs(ord(s[i])-ord(t[i])))
        l, r = 0, 0
        cost = 0
        ans = 0
        while r < n:
            cost 
+= arr[r]
            while cost > maxCost:
                cost -= arr[l]
                l += 1
            ans = max(ans, r-l+1)
            r += 1
        return ans
