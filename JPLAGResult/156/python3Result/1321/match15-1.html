class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        res = 0
        p = 0
        cost = 0
        for q in range(len
(t)):
            cost += abs(ord(t[q]) - ord(s[q]))
            while cost > maxCost:
                cost -= abs(ord(t[p]) - ord(s[p]))
                p += 1
            res = max(res, q - p + 1)
        return res
