class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        begin = end = 0
        cur_cost 
= 0
        res = 0
        while end < len(t):
            cur_cost += abs(ord(s[end]) - ord(t[end]))
            while cur_cost > maxCost:
                cur_cost -= abs(ord(s[begin]) - ord(t[begin]))
                begin += 1
            res = max(res, end-begin+1)
            end += 1
        return res
