class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        diff = []
        for i in range(len(s)):
            diff.append(abs(ord(t[i]) - ord(s[i])))
        
        left = 0
        right = 0
        current_sum = 0
        longest_sub_size = 0
        while right < len(s):
            current_sum += diff[right]
            right += 1
            
            while current_sum > maxCost:
                current_sum -= diff[left]
                left += 1

            sub_size = right - left
            longest_sub_size = max(longest_sub_size, sub_size)
                
        return longest_sub_size
