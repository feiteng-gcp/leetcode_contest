class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        
        diff = []
        for i in range(len(s)):
            diff.append(abs(ord(s[i]) - ord(t[i]) ))
        
        p1, p2 = 0,0
        temp = 0
        res = 0
        while p1 < len(s) and p2 < len(s):
            temp += diff[p2]
            p2 += 1
            while temp > maxCost:
                
                temp -= diff[p1]
                p1 += 1
            res = max(res, p2-p1)
            
        return res
