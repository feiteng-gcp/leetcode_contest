class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        d = []
        n = len(s)
        for i in range(n):
            d.append(abs(ord(s[i]) - ord(t[i])))
        start = 0
        end = 0
        sc = 0
        ans = 0
        while end < n:
            sc += d[end]
            end += 1
            if sc <= maxCost:
                ans = max(ans, end - start)
            while sc > maxCost:
                sc -= d[start]
                start += 1
        return ans
