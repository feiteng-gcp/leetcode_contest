class Solution:
	def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
		n = len(s)
		costs = [abs(ord(s[j])-ord(t[j])) for j in range(n)]
		i = 0
		mx = 0
		cursum = 0
		for j in range(n):
			cursum += costs[j]
			while cursum > maxCost:
				cursum -= costs[i]
				i += 1
			mx = max(mx,j-i+1)
		return mx
