class Solution:
    def equalSubstring(self, s, t, maxCost):
        diff = [abs(ord(x) - ord(y)) for x, y in zip(s, t)]
        # print(s)
        # print(t)
        # print(diff)

        if all([val > maxCost for val in diff]):
            return 0

        p = 0
        ans 
= 1
        cur = 0
        for q in range(len(s)):
            cur += diff[q]
            while cur > maxCost:
                cur -= diff[p]
                p += 1
            ans = max(ans, q - p + 1)
        return ans
