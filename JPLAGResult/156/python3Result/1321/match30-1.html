class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        i, j = 0, 0
        ans = 0
        while j < len
(s):
            maxCost -= (abs(ord(s[j])-ord(t[j])))
            while i <= j and maxCost < 0:
                maxCost += (abs(ord(s[i])-ord(t[i])))
                i += 1
            ans = max(ans, j-i+1)
            j += 1
        return ans
