class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        a = []
        for i in range(len
(s)):
            a.append(abs(ord(s[i]) - ord(t[i])))
        c = 0
        l, r = 0, 0
        ans = 0
        # print(a)
        while (r < len(s)):
            # print(l, r, c)
            while (c <= maxCost and r < len(s)):
                ans = max(ans, r-l)
                # print(l, r, c)
                c += a[r]
                r += 1
            
if (c <= maxCost):
                ans = max(ans, r-l)
            while (c > maxCost and l < r):
                c -= a[l]
                l += 1
        return ans
