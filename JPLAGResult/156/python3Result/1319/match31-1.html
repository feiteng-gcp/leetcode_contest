class Solution:
    def uniqueOccurrences(self, arr: List[int]) -> bool:
        counter = {}
        for num in arr:
            if num not in counter:
                counter[num] = 1
            else:
                counter[num] += 1
        
        seen = set()
        
for num in counter:
            if counter[num] in seen:
                return False
            seen.add(counter[num])
        
        return True
