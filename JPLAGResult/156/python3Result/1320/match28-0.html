class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        stk = []
        for c in s:
            if not stk:
                stk.append((c, 1))
            else:
                if stk[-1][0] == c:
                    if stk[-1][1] + 1 == k:
                        for i in range(k-1):
                            stk.pop()
                    else:
                        stk.append((c, stk[-1][1]+1))
                else:
                    stk.append((c, 1))
        res = ''
        for s in stk:
            res += s[0]
        return res
