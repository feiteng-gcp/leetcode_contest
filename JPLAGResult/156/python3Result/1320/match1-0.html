class Solution:
    def removeDuplicates(self, s: str, K: int) -> str:
        tmp = []
        # print([(k, list(g)) for k, g in itertools.groupby(s) ])
        conti = False
        for k, g in itertools.groupby(s):
            g = list(g)
            # print(g)
            if len(g) >= K:
                tmp.extend([k] * (len(g) - K))
                conti = True
            else:
                tmp.extend(g)
        
        if conti:
            return self.removeDuplicates(''.join(tmp), K)
        else:
            return ''.join(tmp)
