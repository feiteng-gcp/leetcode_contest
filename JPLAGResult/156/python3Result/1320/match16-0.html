class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        stack = [['A', 0]]
        for c in s:
            if c == stack[-1][0]:
                stack[-1][1] += 1
                if stack[-1][1] == k:
                    stack.pop()
            else:
                stack.append([c, 1])
        res = ""
        for c, n in stack[1:]:
            res += c*n
        return res
