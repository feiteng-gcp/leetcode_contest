class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        cnt = [['-',1]]
        for i in s:
            if i != cnt[-1][0]:
                cnt += [[i,1]]
            else:
                cnt[-1][-1] += 1
                if cnt[-1][-1] >= k:
                    cnt[-1][-1] -= k
                if cnt[-1][-1] == 0:
                    cnt.pop()
        s = ''
        for pair in cnt[1:]:
            s += pair[0] * pair[1]
        return s
