class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        ans 
= []
        for c in s:
            if ans and ans[-1][0] == c:
                ans[-1][1] += 1
                if ans[-1][1] == k:
                    ans.pop()
            else:
                ans.append([c, 1])
        return ''.join([c * n for c, n in ans])
