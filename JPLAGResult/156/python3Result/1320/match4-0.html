class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        q, c = [], []
        for w in s:
            if q and q[-1] == w:
                c.append(c[-1] + 1)
            else:
                c.append(1)
            q.append(w)
            if c[-1] >= k:
                for _ in range(k):
                    q.pop()
                    c.pop()
        return ''.join(q)
