class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        stack=[]
        removed=True
        while removed:
            removed
=False
            for ch in s:
                if not stack or stack[-1][0]!=ch:
                    stack.append([ch,1])
                else:
                    stack[-1][1]+=1
                    if stack[-1][1]==k:
                        stack.pop()
                        removed=True
            s=''.join([ch*count for ch,count in stack])
            stack=[]
        return s
