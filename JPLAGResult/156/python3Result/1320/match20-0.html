class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        a = [[s[0], 1]]
        for c in s[1:]:
            if a and a[-1][0] == c:
                a[-1][1] += 1
                if a[-1][1] % k == 0:
                    a.pop()
            else:
                a.append([c, 1])
        
        return ''.join(c * n for c, n in a)
