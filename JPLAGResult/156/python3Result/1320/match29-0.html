class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        stack = []
        for _ in s:
            if not stack or stack[-1][0] != _:
                stack.append([_, 1])
            else:
                stack[-1][1] += 1
                if stack[-1][1] == k:
                    stack.pop()
        return ''.join(_*c for _, c in stack)
