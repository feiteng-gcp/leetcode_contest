class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        st = ['']
        count 
= [0]
        for c in s:
            if c == st[-1]:
                count.append(count[-1] + 1)
            else:
                count.append(1)
            st.append(c)
            if count[-1] == k:
                for i in range(k):
                    st.pop()
                    count.pop()
        return ''.join(st)
