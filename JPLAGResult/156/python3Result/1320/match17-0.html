class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        stack = [['', 0]]
        
        for i, char in enumerate(s):
            if char == stack[-1][0]:
                stack[-1][1] += 1
                if stack[-1][1] == k:
                    stack.pop()
            else:
                stack.append([char, 1])
                
            #print(stack)
             
        ret = []
        for pair in stack:
            ret += pair[0]*pair[1]
        return ''.join(ret)
