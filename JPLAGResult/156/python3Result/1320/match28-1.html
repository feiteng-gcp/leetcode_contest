class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        ret = [('', 1)]
        for c in s:
            
if c == ret[-1][0]:
                if ret[-1][1] == k - 1:
                    for _ in range(k - 1):
                        ret.pop()
                else:
                    ret.append((c, ret[-1][1] + 1))
            else:
                ret.append((c, 1))
        return ''.join([r[0] for r in ret])
