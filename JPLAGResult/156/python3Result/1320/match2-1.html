class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        stack = []
        for c in s:
            if not stack or stack[-1][0] != c:
                stack.append((c, 1))
            else:
                stack.append((c, stack[-1][1]+1))
            if stack and stack[-1][1] >= k:
                for _ in range(k):
                    stack.pop()
        return "".join(c for c, i in stack)
