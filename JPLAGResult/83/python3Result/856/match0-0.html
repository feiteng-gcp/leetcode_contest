class Solution:
    def consecutiveNumbersSum(self, N):
        """
        :type N: int
        :rtype: int
        """
        cnt=1
        l=1 # length of the sequence
        while l*(l+1)<2*N:
            sol = (1.0 * N - (l * (l + 1) ) / 2) / (l + 1)
            if sol-int(sol)==0.0:
                cnt+=1
            l+=1
        return cnt
