class Solution:
    def consecutiveNumbersSum(self, N):
        """
        :type N: int
        :rtype: int
        """
        allSums=0
        num=1
        while (num*(num+1))<(2*N):
            value=((1.0*N)-(num*(num+1))/2)/(num+1)
            if (value-int(value)==0.0):
                allSums+=1
            num+=1
        return (allSums+1) # +1 to add N
