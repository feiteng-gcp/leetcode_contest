class Solution {
    public String maskPII(String S) {

        S = S.toLowerCase();

        if (Character.isLetter(S.charAt(0))) {
            String[] strs = S.split("@");

            StringBuilder builder = new StringBuilder();
            builder = builder.append(String.valueOf(strs[0].charAt(0)))
                    .append("*****").append(strs[0].charAt(strs[0].length()-1))
                    .append("@")
                    .append(strs[1]);

            return builder.toString();
        } else {
            String stripped = S.replaceAll("[^\\d]", "");

            String last4 = stripped.substring(stripped.length()-4, stripped.length());

            if (stripped.length() == 10) {
                return "***-***-"+last4;
            } else if (stripped.length() == 11) {
                return "+*-***-***-"+last4;
            } else if (stripped.length() == 12) {
                return "+**-***-***-"+last4;
            } else {
                return "+***-***-***-"+last4;
            }
        }
    }
}
