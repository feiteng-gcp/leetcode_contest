class Solution {
    public String maskPII(String S) {
        StringBuilder sb = new StringBuilder();
        if (S.contains("@")) {
            S = S.toLowerCase();
            String[] split = S.split("@");
            sb.append(S.charAt(0));
            sb.append("*****");
            sb.append(split[0].charAt(split[0].length() - 1));
            sb.append("@");
            sb.append(split[1]);
        } else {
            S = S.replaceAll("[() +-]+","");
            if (S.length() == 10) {
                return "***-***-" + S.substring(6,10);
            } else {
                int len = S.length();
                sb.append("+");
                for (int i = 0; i < len - 10; i++) {
                    sb.append("*");
                }
                sb.append("-***-***-");
                sb.append(S.substring(len - 4, len));
            }
        }
        return sb.toString();
    }
}
