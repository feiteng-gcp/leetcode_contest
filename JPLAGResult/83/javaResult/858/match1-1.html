class Solution {
    public String maskPII(String S) {
        int flag = 1;
        char[] arr = S.toCharArray();
        for(char ch: arr) {
            if(ch == '@') {
                flag = 0;
                break;
            }
        }
        if(flag == 0) {
            String[] a = S.split("@");
            String left = a[0].toLowerCase();
            left = 
left.charAt(0) + "*****" + left.charAt(left.length() - 1);
            // String[] b = a[1].split(".");
            return left + "@" + a[1].toLowerCase();
        } else {
            int count = 0;
            StringBuilder sb = new StringBuilder();
            for(char ch: arr) {
                if(Character.isDigit(ch)) {
                    count++;
                    sb.append(ch);
                } 
            }
            if(count == 10)  return "***-***-" + sb.substring(6, 10);
            StringBuilder res = new StringBuilder();
            res.append("+");
            for(int i = 0; i < count - 10; i++)  res.append("*");
            res.append("-");
            return res.toString() + "***-***-" + sb.substring(count - 4, count);
        }
    }
}
