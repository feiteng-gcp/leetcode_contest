class Solution {
    public int consecutiveNumbersSum(int N) {
        int count = 1;
        int end = (int) Math.sqrt(2*N);
        for (int i=2; i<=end; i++) {
            if (i % 2 == 0) {
                if (N % i == i/2) {
                    count++;
                }
            } else {
                if (N % i == 0) {
                    count++;
                }
            }
        }
        return count;
    }
}
