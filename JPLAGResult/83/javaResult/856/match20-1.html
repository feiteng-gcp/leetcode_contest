class Solution {
        public int consecutiveNumbersSum(int N) {
int count = 1;
		
		for(int k = 1; k < Math.sqrt(N * 2); k ++) {
			double temp = (double)(2 * N - k * k - k) / (2 * k + 2);
			if (temp > 0.0 && (temp == Math.floor(temp)) && !Double.isInfinite(temp)) {

			    count ++;
			}
		}
		return count;
        }
}
