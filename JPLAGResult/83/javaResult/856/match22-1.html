class Solution {
	    public int consecutiveNumbersSum(int N) {
	    	long n = N;
	    	long upperLimit = (long)Math.sqrt(n*2);
	    	long n2 = 2*n;
	    	int ans = 0;
	    	for (int i
 = 1; i <= upperLimit; ++i) {
	    		if (i%2==1) {
	    			if ( n % i == 0) {
	    				ans++;
	    			}
	    		} else {
	    			if ( (n2 + i)%(i*2) == 0) {
	    				ans++;
	    			}
	    		}
	    	}
	    	return ans;
	    }
}
