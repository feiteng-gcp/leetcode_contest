class Solution {
    public int consecutiveNumbersSum(int N) {
        int ans = 1;
        for (int i = 2; i <= N; i++) {
            int base = (i * (i + 1)) / 2;
            if (base > N) {
                break;
            }
            int diff = (N - base) / i;
            if (diff >= 0 && base + diff * i == N) {
                ans++;
            }
        }
        return ans;
    }
}
