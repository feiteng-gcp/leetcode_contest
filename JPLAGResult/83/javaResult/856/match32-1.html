class Solution {
	public int consecutiveNumbersSum(int N) {
		int count = 1;
		for (long L = 1; L * (L + 1) < 2 * N; L++) {
			long a = (N - (L * (L + 1)) / 2) / (L + 1);
//			TestSolution.tr(a, L, (a + a + L) * (L + 1) / 2);
			if (N == (a + a + L) * (L + 1) / 2)
				count++;
		}
		return count;
	}
}
