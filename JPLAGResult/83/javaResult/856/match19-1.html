class Solution {
    public int consecutiveNumbersSum(int N) {
        int count=0;// it self;
        
        for(int i=1;i<=N;i++){
            int sum=(i-1)*(i)/2;
            if(sum>=N) break;// = means the 1st integer is 0. not fit
            
int na=N-sum;
            int a=na/i;
            if(a*i==na){
                // System.out.println(a);
                count++;
            }
        }
        return count;
    }
}
