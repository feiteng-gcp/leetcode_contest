class Solution {
    public int consecutiveNumbersSum(int n) {
        
        int count = 0;
        for(
int i = 0; i * (i + 1) < 2 * n; i++) {
            if((n - ((i + 1) * i / 2)) % (i + 1) == 0) {
                count++;
            }
        }
        return count;
    }
}
