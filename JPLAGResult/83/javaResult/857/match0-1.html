class Solution {
    public List<List<Integer>> largeGroupPositions(String S) {
        List<List<Integer>> ret = new LinkedList<>();
        int count = 0;
        int startPos = 0;
        char cur = 0;
        for (int i = 0; i < S.length(); i++) {
            if (cur == 
S.charAt(i)) {
                count++;
            } else {
                if (count >= 3) {
                    List<Integer> set = new LinkedList<>();
                    set.add(startPos);
                    set.add(i - 1);
                    ret.add(set);
                }
                cur = S.charAt(i);
                startPos = i;
                count = 1;
            
}
        }
        if (count >= 3) {
            List<Integer> set = new LinkedList<>();
            set.add(startPos);
            set.add(S.length() - 1);
            ret.add(set);
        }
        
        return ret;
    }
}
