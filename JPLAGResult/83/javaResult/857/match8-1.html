	class Solution {
	    public List<List<Integer>> largeGroupPositions(String S) {
	    	List<List<Integer>> ret = new ArrayList<>();
	        for(int i = 0;i < S.length();){
	        	int j = i;
	        	while(j < S.length() && S.charAt(i) == S.charAt(j))j++;

	        	
	        	if(j-i >= 3){
	        		List<Integer> item = new ArrayList<>();
	        		item.add(i);
	        		item.add(j-1);
	        		ret.add(item);
	        	}
	        	
	        	i = j;
	        }
	        return ret;
	    }
	}
