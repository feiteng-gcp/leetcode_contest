class Solution {
    public List<List<Integer>> largeGroupPositions(String S) {
        List<List<Integer>> res = new ArrayList<>();
        char[] arr = S.toCharArray();
        for(int i = 0; i < arr.length; i++) {
            char ch = arr[i];
            int count = 1;
            int start = i;
            while(i + 1 < arr.length && arr[i + 1] == ch) {
                i++;
                count++;
            }
            
if(count >= 3) {
                List<Integer> temp = new ArrayList<>();
                temp.add(start);
                temp.add(i);
                res.add(temp);
            }
        }
        return res;
    }
}
