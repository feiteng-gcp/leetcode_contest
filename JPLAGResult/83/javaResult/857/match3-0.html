import java.util.*;

class Solution {
    public List<List<Integer>> largeGroupPositions(String S) {

        List<List<Integer>> result = new ArrayList<List<Integer>>();
        
        int length = 1;
        for (int on = 1; on < S.length(); on++) {
            
            // elongation
            if (S.charAt(on) == S.charAt(on-1)) {
                length++;
            }
            
            // reset
            else {
                if (length >= 3) {
                    List<Integer> oneMore = new ArrayList<Integer>();
                    oneMore.add(on - length);
                    oneMore.add(on - 1);
                    result.add(oneMore);
                }
                length = 1;
            }
            
        }
        
        // finish with large group?
        if (length >= 3) {
            List<Integer> oneMore = new ArrayList<Integer>();
            oneMore.add(S.length()-length);
            oneMore.add(S.length()-1);
            result.add(oneMore);
        }
        return result;
    }
}
