class Solution {
    private final static long MOD = 1000000007;
    
public int uniqueLetterString(String S) {
        if (S == null || S.length() == 0) {
            return 0;
        }
        long result = 0;
        int[] count = new int[256];
        for (char c : S.toCharArray()) {
            count[c]++;
        }
        for (int i = 0; i < 256; ++i) {
            if (count[i] == 0) {
                continue;
            }
            char c = (char)i;
            result = (result + curResult(S, c)) % MOD;
        }
        return (int)result;
    }
    
    private long curResult(String S, char c) {
        int left = 0;
        int right = 0;
        long result = 0;
        int count = 0;
        int pivot = -1;
        while (right < S.length()) {
            if (S.charAt(right) != c) {
                if (count == 1) {
                    result = (result + pivot - left + 1) % MOD;
                }
                right++;
            } else {
                count++;
                if (count == 1) {
                    pivot = right;
                    result = (result + pivot - left + 1) % MOD;
                } else {
                    left = pivot + 1;
                    count--;
                    pivot = right;
                    result = (result + pivot - left + 1) % MOD;
                }
                right++;
            }
        }
        return result;
    }
}
