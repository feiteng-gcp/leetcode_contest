class Solution {
    public int uniqueLetterString(String S) {
        if (S == null || S.length() == 0)
            return 0;
        HashMap<Character, List<Integer>> positions = new HashMap<>();
        final int mod = 1000 * 1000 * 1000 + 7;
        long result = 0;
        for (int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            List<Integer> charPositions = positions.getOrDefault(c, new ArrayList<Integer>());
            charPositions.add(i);
            positions.put(c, charPositions);
            for (char key : positions.keySet()) {
                List<Integer> row = positions.get(key);
                int lastPosition = row.get(row.size() - 1);
                if (row.size() == 1)
                    result = add(result, lastPosition + 1, mod);
                else {
                    int before = row.get(row.size() - 2);
                    result = add(result, lastPosition - before, mod);
                }
            }
        }
        return (int) (result % mod);
    }
    private long add(long result, int x, int mod) {
        result = result + x;
        return result % mod;
    }
    
}
