class Solution {
    public int uniqueLetterString(String S) {
        long output = 0;
        for (int i = 0; i < S.length(); i++) {
            output += uniqueLetterStringStartsWithFirstCharacter(S.substring(i));
        }
        return (int) (output % (1e9 + 7));
    }

    private long uniqueLetterStringStartsWithFirstCharacter(String S) {
        long output = 0;
        int unique = 0;
        int[] count = new int[256];
        for (int i = 0; i < S.length(); i++) {
            char c = S.charAt(i);
            if (count[c] == 0) {
                unique++;
            } else if (count[c] == 1) {
                unique--;
            }
            count[c]++;
            output += unique;
        }
        return output;
    }

        System.out.println(s.uniqueLetterString("ABC"));

    }
}
