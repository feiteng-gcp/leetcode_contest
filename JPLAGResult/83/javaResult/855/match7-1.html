class Solution {
    private long count;
    private int mod = (int) Math.pow(10, 9) + 7;
    public int uniqueLetterString(String S) {
       if(S.length() == 0) return 0;
        count = S.length();
        int mod = (int) Math.pow(10, 9) + 7;
        
        Map<Character, List<Integer>> map = new HashMap<>();
        int index = 0;
        for(char c: S.toCharArray()){
            List<Integer> lst = map.getOrDefault(c, new ArrayList<>());
            lst.add(index);
            map.put(c, lst);
            ++ index;
        }
        
        for(char c: map.keySet()){
            List<Integer> lst = map.get(c);
            if(lst.size() == 1){
                countU(0, lst.get(0), S.length()-1);
                continue;
            }
            int start = 0;
            for(int i = 0; i < lst.size() - 1; ++i){
                countU(start, lst.get(i), lst.get(i+1) - 1);
                start = lst.get(i) + 1;
            }
            countU(start, lst.get(lst.size() - 1), S.length() - 1);
        
}
        
        return (int) count;
    }
    
    public void countU(int start, int pos, int end){
        int val = end - pos;
        val = val + val * (pos - start);
        val += (pos-start);
        count += val;
        if(count > mod) count %= mod;
    }
    
    
}
