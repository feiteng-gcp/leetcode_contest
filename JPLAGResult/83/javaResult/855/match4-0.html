class Solution {
    public int uniqueLetterString(String S) {
        if (S.length() == 0) {
            return 0;
        }
        
        long ans = 0;
        long[] dp = new long[S.length() + 1];
        
        int[][] pos = new int[128][2];
        
        for (int i = 0; i < 128; i++) {
            for (int j = 0; j < 2; j++) {
                pos[i][j] = -1;
            }
        }
        
        for (int i = 0; i < S.length(); i++) {
            int index = S.charAt(i);
            dp[i + 1] = (dp[i] + (i - pos[index][0]) - (pos[index][0] - pos[index][1])) % 1000000007;
            pos[index][1] = pos[index][0];
            pos[index][0] = i;
        }
        
        for (int i = 1; i < dp.length; i++) {
            ans = (ans + dp[i]) % 1000000007;
        }
        
        return (int) ans;
    }
}
