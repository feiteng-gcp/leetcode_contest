class Solution {

	
	private int upper(int index , char ch) {
		
		int low = 0 , high = map.get(ch).size();
		List<Integer> list = map.get(ch);
		while (low < high) {
			int mid = (low + high) / 2;
			if (list.get(mid) < index) {
				low = mid + 1;
			} else {
				high = mid;
			}
		}
		high --;
		if (high >= 0) {
			return list.get(high);
		} else {
			return - 1;
		}
		
	}
	
	private int lower(int index , char ch) {
		
		int low = 0 , high = map.get(ch).size();
		List<Integer> list = map.get(ch);
		while (low < high) {
			int mid = (low + high) / 2;
			if (list.get(mid) <= index) {
				low = mid + 1;
			} else {
				high = mid;
			}
		}
		if (low < list.size()) {
			return list.get(low);
		} else {
			return - 1;
		}
		
	}
	
	private final long MOD = 1000000007;
	private Map<Character , List<Integer>> map = new HashMap<>();
	
    public int uniqueLetterString(String S) {
     
    	long ans = 0;
    	int i , length = S.length();
    	for (i = 0;i < length;i ++) {
    		char ch = S.charAt(i);
    		if (!map.containsKey(ch)) {
    			map.put(ch , new ArrayList<>());    			
    		}
    		map.get(ch).add(i);
    	}
    	for (i = 0;i < length;i ++) {
    		char ch = S.charAt(i);
    		int low = lower(i , ch);
    		int up = upper(i , ch);
    		if (low < 0) {
    			low = length;
    		}
    		if (up < 0) {
    			up = - 1;
    		}
    		long left = i - up - 1;
    		long right = low - i - 1;
    		ans = (ans + 1) % MOD;
    		ans = (ans + left) % MOD;
    		ans = (ans + right) % MOD;
    		ans = (ans + left * right) % MOD;
    	}
    	return (int) ans;
    		
    }
	
}
