class Solution {
   private static final int MOD = 1000000007;
    public int uniqueLetterString(String S) {
        HashMap<Character, TreeSet<Integer>> map = new HashMap<>();
        for (int i = 0; i < S.length(); i++) {
            if (!map.containsKey(S.charAt(i))) {
                TreeSet<Integer> set = new TreeSet<>();
                set.add(-1);
                set.add(S.length());
                map.put(S.charAt(i), set);
            }
            TreeSet<Integer> set = map.get(S.charAt(i));
            set.add(i);
        }
        
long count = 0;
        for (int i = 0; i < S.length(); i++) {
            char ch = S.charAt(i);
            TreeSet<Integer> set = map.get(ch);
            int lower = set.lower(i);
            int higher = set.higher(i);
            count = (count % MOD + (1L * (i - lower) * (higher - i)) % MOD) % MOD;
            
        }
        return (int)count;
    }
}
