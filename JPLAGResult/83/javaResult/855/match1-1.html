class Solution {
    int LIMIT = 1000000007;
    
public int uniqueLetterString(String S) {
        
        int sum = 0;
        int uniCnt = 0;
        int [] map = new int[26];
        
        for (int i = 0; i < S.length(); i++) {
            map = new int[26];
            uniCnt = 0;
            for(int j = i; j < S.length(); j++) {
                char chr = S.charAt(j);
                if (map[chr -'A'] == 0) {
                    map[chr -'A'] = 1;
                    uniCnt+=1;
                } else {
                    map[chr -'A'] += 1;
                    if (map[chr -'A']==2) {
                        uniCnt -=1;
                    } else {
                        //return uniCnt;
                    }
                }
                sum += uniCnt;
                while (sum > LIMIT) {
                    sum %= LIMIT;
                }
            }
        }
        return sum;
    }
