class Solution {
public:
    string maskPII(string S) {
        int len = S.length();
        bool flag = false;
        int k;
        for(int i = 0; i < len; i++){
            if(S[i] <= 'Z' && S[i] >='A')
                S[i] = S[i] - 'A'+'a';
            if(S[i] == '@'){
                flag = true;
                k = i-1;
            }
        }
        if(flag){
            S.erase(1,k-1);
            S.insert(1,"*****");
            return S;

        }
        else{
            int num = 0;
            vector<string> v;
            for(int i = 0; i<len; i++){
                if(S[i] <= '9' && S[i] >= '0'){
                    v.push_back(S.substr(i,1));
                    num ++;
                }
            }
            string s;
            if(num > 10){
                s.insert(0,"+");
                for(int i = 0; i<num-10; i++)
                    s.insert(s.length(),"*");
                s.insert(s.length(),"-");
            }
            int t = num - 10;
            for(int i = 0; i < 2; i++){
                for(int j = 0; j<3; j++){
                    if(s.length() == 0)
                        s = "*";
                    else
                        s.insert(s.length(),"*");
                    t++;
                }
                s.insert(s.length(),"-");
            }
            for(int i = 0; i<4; i++){
                s.insert(s.length(), v[t]);
                t++;
            }
            return s;
        }
    }
};
