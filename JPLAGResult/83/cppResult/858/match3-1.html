class Solution {
public:
    string maskPII(string S) {
        string res;
        bool isPhone = false;
        for(int i = 0; i < S.length(); i++){
            if(S[i] >= '0' && S[i] <= '9'){
                isPhone = true;
                break;
            }
        }
        if(isPhone){
            string phone;
            for(int i = 0; i < S.length(); i++){
                if(S[i] >= '0' && S[i] <= '9'){
                    phone.push_back(S[i]);
                }
            }
            int len = phone.size();
            if(len > 10){
                res = "+";
                for(int i = 0; i < len-10; i++){
                    res.push_back('*');
                }
                res.push_back('-');
            }
            res = res + "***-***-" + phone.substr(len-4, len);
        }
        else{

            int at_pos = 0;
            for(int i = 0; i < S.length(); i++){
                if(S[i] == '@'){
                    at_pos = i;
                    break;
                }
            }
            // cout << at_pos << endl;;
            res.push_back(tolower(S[0]));
            // cout << res << endl;
            res += "*****";
            // cout << res << endl;
            // print(res);
            
            res.push_back(tolower(S[at_pos-1]));
            
            // cout << res << endl;
            // print(res);
            for(int i = at_pos; i < S.length(); i++){
                if(S[i] >= 'A' && S[i] <= 'Z'){
                    S[i] = S[i] - ('A' - 'a');
                }
                res.push_back(S[i]);
            }
            // res += S.substr(at_pos, -1);
            // cout << res << endl;
            // print(res);
        
        }
        return res;
    }
};
