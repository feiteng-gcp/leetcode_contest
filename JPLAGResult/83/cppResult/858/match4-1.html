class Solution {
public:
    string maskPII(string S) {

        if((S[0] <= '9' && S[0] >= '0')||(S[0] == '+')||(S[0] == '(')){
            string ans = "";
            int num = 0;
            for(int i = S.length()-1;i >= 0;--i){
                if(S[i] <= '9' && S[i] >= '0'){
                    if(num == 10)ans = '-'+ans;
                    if(num  < 4)ans = S[i] + ans;
                    else  ans = '*'+ans;
                    num++;
                    if(num == 4 || num == 7)ans = '-'+ans;
                }
            }
            if(num > 10)ans = '+'+ans;
            return ans;
        }
        else{
            string ans = "";
            char lastChar;
            bool atBe = true;
            for(int i = 0;i < S.length();++i){
                if(S[i] != '@' && atBe){
                    if(ans == "")ans += (S[i]>='a' && S[i]<='z')?S[i]:(S[i]-'A'+'a');
                    lastChar = S[i];
                }
                else if(atBe){
                    atBe = false;
                    ans += "*****";
                    ans += (lastChar>='a' && lastChar<='z')?lastChar:(lastChar-'A'+'a');
                    ans += S[i];
                    
                }
                else{
                    if(S[i]>='A' && S[i]<='Z')ans += S[i]-'A'+'a';
                    else ans += S[i];
                }
            }
            return ans;
        }
    }
};
