class Solution {
public:
    string maskPII(string S) {
        if (S.empty()) {
            return "";
        }
        string sol = "";
        if (isalpha(S[0])) {
            int pos = S.find('@');
            sol += tolower(S[0]);
            sol.append(5, '*');
            sol += tolower(S[pos - 1]);//不能直接把@加在后面
            sol += "@";
            for (int i = pos + 1; i < S.size(); ++i) {
                sol += tolower(S[i]);
            }
            return sol;
        } else {
            string tmp = "", sol = "";
            for (int i = 0; i < S.size(); ++i) {
                if (isdigit(S[i])) {
                    tmp += S[i];
                }
            }
            if (tmp.size() > 10) {
                sol += "+";
                sol.append(tmp.size() - 10, '*');//
                sol += "-";
            }
            sol += "***-***-" + tmp.substr(tmp.size() - 4);
            return sol;
        }
    }
};
