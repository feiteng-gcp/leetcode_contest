class Solution {
public:
    char tolower(char s) {
        if (s <= 'Z' && s >= 'A') {
                return s - 'A' + 'a';
            }
            else {
              return s;  
            }
    }
    string maskPII(string S) {
        int n = S.size();
        if ((S[0] >= 'a' && S[0] <= 'z') || (S[0] >= 'A' && S[0] <= 'Z')){
            //this is an email
            string res = "";
            int i = 0;
            res += tolower(S[i]);
            i++;
            while (i < n && S[i] != '@') {
                i++;
            }
            res += "*****";
            i--;
            res += tolower(S[i]);
            res += '@';
            i += 2;
            while (i < n) {
                res += tolower(S[i]);
                i++;
            }
            return res;
        }
        else {
            //this is a phone number
            int i = n - 1;
            int d = 0;
            string res1 = "1234";
            while (i >= 0 && d < 4) {
                if (S[i] >= '0' && S[i] <= '9') {
                    res1[3 - d] = S[i];
                    d++;
                }
                i--;
            }
            while (i >= 0) {
                if (S[i] >= '0' && S[i] <= '9') {
                    d++;
                }
                i--;
            }
            string res2;
            if (d > 10) {
                res2 += '+';
                int j = 0;
                while (j < d - 10) {
                    res2 += '*';
                    j++;
                }
                res2 += "-***-***-";
                return res2 + res1;
            }
            else {
                return "***-***-" + res1;
            }
        }
    }
};
