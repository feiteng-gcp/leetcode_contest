class Solution {
public:
    string maskPII(string S) {
        string res = "";
        int len = S.size();
        if(S.find('@') != string::npos){
            for(int i=0; i<len; ++i){
                if(S[i]>='A' && S[i]<='Z'){
                    S[i] = S[i]-'A'+'a';
                }
            }
            res += S.substr(0,1) +"*****"+S.substr(S.find('@')-1, string::npos);
        }
        else{
            int cnt = 0;
            string tmp = "";
            for(int i=0; i<len; ++i){
                if(S[i]>='0' && S[i]<='9'){
                    cnt++;
                    tmp += S[i];
                }
                    
            }
            if(cnt==10){
                res = "***-***-"+tmp.substr(tmp.size()-4, string::npos);
            }
            else{
                res += "+";
                for(int i=0; i<cnt-10; ++i){
                    res+="*";
                }
                res +="-***-***-"+tmp.substr(tmp.size()-4, string::npos);
            }
        }
        return res;
    }
};
