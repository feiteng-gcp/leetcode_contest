class Solution {
public:
    string maskPII(string S) {
        string res="";
	transform(S.begin(), S.end(), S.begin(),::tolower);
	if (S[0] >= 'a'&&S[0] <= 'z')
	{
		res = S[0];
		int id = 0;
		for (int i = 1; i < S.size(); i++)
		{
			if (S[i]=='@')
			{
				id = i;
				break;
			}
		}
		res = res+"*****"+S[id-1]+S.substr(id);
	}
	else
	{
		for (int i = S.size() - 1; i >= 0; i--)
		{
			if (S[i] >= '0'&&S[i] <= '9')
			{
				if (res.size() < 4)
					res = S[i] + res;
				else
				{
					if (res.size() == 4 || res.size() == 8 || res.size() == 12)
					{
						res = '-' + res;
					}
					res = '*' + res;
				}
			}
		}
		if (res.size()>13)
			res = '+' + res;
	}
	return res;
    }
};
