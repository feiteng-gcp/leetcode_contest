class Solution {
public:
    string maskPII(string S) {
        string ans = "";
        if (isalpha(S[0])) {
            transform(S.begin(), S.end(), S.begin(), ::tolower);
            ans += S[0];
            int slen = S.size();
            int i;
            for (i = 0;i < slen;i++) {
                if (S[i] == '@') break;
            }
            ans += "*****";
            ans += S[i-1];
            for (;i<slen;i++) {
                ans += S[i];
            }

        }
        else {
            int slen = S.size();
            string tmp = "";
            for (int i = 0;i < slen;i++) {
                if (isdigit(S[i])) {
                    tmp += S[i];
                }
            }
            if (tmp.size() <= 10) {
                ans = "***-***-";
            }
            else if (tmp.size() <= 11)ans = "+*-***-***-";
            else if (tmp.size() <= 12)ans = "+**-***-***-";
            else ans = "+***-***-***-";
            slen = tmp.size();
            for (int i = slen - 4;i < slen;i++) {
                ans +=tmp[i];
            }
        }
        return ans;
    }
};
