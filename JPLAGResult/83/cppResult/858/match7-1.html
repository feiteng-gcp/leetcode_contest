class Solution {
public:
    char toLower(char s){
        if('a'<=s&&'z'>=s){
                return s;
            }else{
                return s+32;
            }
    }
    string maskPII(string S) {
        string res;
        if(S.size()<=0)
            return res;
        
        if(('a'<=S[0]&&'z'>=S[0])||('A'<=S[0]&&'Z'>=S[0])){
            int len=S.size();
            res="";
            res.push_back(toLower(S[0]));
            for(int j=0;j<5;j++){
                res.push_back('*');
            }
            int flag=0;
            for(int i=0;i<len;i++){
                if(S[i]=='@'){
                    res.push_back(toLower(S[i-1]));
                    res.push_back('@');
                    flag=1;
                }
                if(flag&&(S[i])!='@'){
                    if(S[i]=='.'){
                        res.push_back('.');
                    }else
                    res.push_back(toLower(S[i]));
                }
            }
            return res;
        }else{
            string head="+";
            string med="***-***-";
            int len=S.size();
            int count=0;
            vector<int> index;
            for(int i=0;i<len;i++){
                if((S[i]>='0')&&(S[i]<='9')){
                    count++;
                    index.push_back(i);
                }
            }
            int in_len=index.size();
            string last="";
            for(int j=in_len-4;j<in_len;j++)
            {
                last.push_back(S[index[j]]);
            }
            res="";
            for(int k=0;k<(in_len-10);k++){
                head.push_back('*');
            }
            head.push_back('-');
            if(count>10){
                res=res+head+med+last;
            }else{
                res=res+med+last;
            }
            return res;
        }
    }
};
