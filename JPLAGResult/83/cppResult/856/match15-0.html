class Solution {
public:
    int consecutiveNumbersSum(int S) {
        int res = 0;
        for (int n = 1; n <= S; ++n) {
            double a = (S - n * (n - 1) / 2.0) / n;
            if (a < 1) break;
            if ((int)a * n + (n - 1) * n / 2 == S)
                res++;
        }
        return res;
    }
};
