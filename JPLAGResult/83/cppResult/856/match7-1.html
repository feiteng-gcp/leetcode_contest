class Solution {
public:
    int consecutiveNumbersSum(int N) {
        int res = 0;
        for(int i=1; i*i<2*N; ++i){
            if(i%2 == 1){
                if(N%i == 0 && N/i>i/2)
                    res++;
            }
            else{
                int k = i/2;
                if(N%k==0 && (N/k)%2==1 && (N/k)>=(2*k+1))
                    res++;
            }
        }
        return res;
    }
};
