class Solution {
public:
    int consecutiveNumbersSum(int N) {
        int num = 0;
        int f = sqrt(2 * N);
        for (int i = 1; i <= f; i++) {
            if ((2 * N) % i == 0) {
                int j = (2 * N) / i;
                if (j - i + 1 >= 2 && (j - i + 1) % 2 == 0)
                    num++;
            }
        }
        return num;
    }
};
