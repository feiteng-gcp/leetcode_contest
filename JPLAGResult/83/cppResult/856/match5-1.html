class Solution {
public:
    int consecutiveNumbersSum(int N) {
        int re=1;
        for(int i=2;i<=N;++i){
            int m=N%i;
            if(((i%2==0&&(double)m/(double)i==0.5)||(i%2==1&&m==0))&&(i*(i+1)/2<=N))
                re++;
            if(i*(i+1)/2>N)
                break;
        }
        return re;
    }
};
