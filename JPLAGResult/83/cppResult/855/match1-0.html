class Solution {
public:
typedef long long ll;
const int mod = 1e9 + 7;
int uniqueLetterString(string s) {
    int n = s.size();
    vector<int> a;
    ll res = 0;
    for (int i = 0; i < 26; i++) {
        a.clear();
        a.push_back(-1);
        for (int j = 0; j < n; j++) {
            if(s[j] == 'a' + i) {
                a.push_back(j);
            }
        }
        a.push_back(n);
        ll r = 0;
        for (int j = 1; j < a.size() - 1; j++) {
            int x = a[j] - a[j - 1] - 1, y = a[j + 1] - a[j] - 1;
            r = r + x + y + 1;
            r += x * y;
        }
        //cout << i << " "  << r << endl;
        res  = (res + r) % mod;
    }
    //upper
    for (int i = 0; i < 26; i++) {
        a.clear();
        a.push_back(-1);
        for (int j = 0; j < n; j++) {
            if(s[j] == 'A' + i) {
                a.push_back(j);
            }
        }
        a.push_back(n);
        ll r = 0;
        for (int j = 1; j < a.size() - 1; j++) {
            int x = a[j] - a[j - 1] - 1, y = a[j + 1] - a[j] - 1;
            r = r + x + y + 1;
            r += x * y;
        }
        //cout << i << " " << r << endl;
        res  = (res + r) % mod;
    }
    return res;
}
};
