class Solution {
public:
    int uniqueLetterString(string S) {
        long long ret = 0;
        string T = S;
        sort(T.begin(), T.end());
        int X = distance(T.begin(), unique(T.begin(), T.end()));
        for(int i = 0 ; i < S.size() ; ++i) {
            int v[26] = {0};
            int C = 0;
            int P = 0;
            for(int j = i ; j < S.size() ;++j ) {
                v[S[j]-'A']++;

                if(v[S[j]-'A']==1) {
                    C++;
                    ret += C;
                } else {
                    if(v[S[j]-'A']==2) {
                        C--;
                        P++;
                    }
                    ret += C;
                }
                if(P >= X) {
                    break;
                }        
            }
        }
        return (int)(ret % 1000000007LL);
    }
};
