class Solution {
public:
    int uniqueLetterString(string S) {
        int mod = 1000000007;
        long long ans = 0;
        
        for (int i = 0; i < S.size(); i++) {
            vector<int> tmp(26, 0);
            int acu = 0, tmpa = 0;
            for (int j = i; j < S.size(); j++) {
                if (!tmp[S[j] - 'A']) {
                    tmp[S[j] - 'A'] = 1;
                    acu++;
                } else {
                    if (tmp[S[j] - 'A'] == 1) {
                        acu--;
                        tmp[S[j] - 'A'] = 2;
                    }
                }
                ans += acu; 
                // cout << acu << " ";
            }
            // cout << endl;
            // cout << tmpa << endl;
            // tmpa %= mod;
            // ans = (ans + tmpa) % mod;
        }
        return ans % mod;
    }
};
