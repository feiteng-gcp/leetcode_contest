class Solution {
public:
    vector<vector<int>> largeGroupPositions(string S) {
        int len=S.size();
        int start=0;
        int end=0;
        vector<vector<int> > res;
        if(len<3)
            return res;
        for(int i=1;i<len;i++){
            if(S[i]==S[start]){
                continue;
            }else{
                if(i-start>=3){
                    vector<int> tmp;
                    tmp.push_back(start);
                    tmp.push_back(i-1);
                    res.push_back(tmp);
                }
                start=i;
            }
        }
        if(len-start>=3){
            vector<int> tmp;
                    tmp.push_back(start);
                    tmp.push_back(len-1);
                    res.push_back(tmp);
        }
        return res;
    }
};
