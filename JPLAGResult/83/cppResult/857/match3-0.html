class Solution {
public:
    vector<vector<int>> largeGroupPositions(string S) {
        int idx1 = 0;
        int sz = S.length();
        char curr = S[0];
        int cnt = 1;
        vector<vector<int> > res;
        for(int i = 1; i < sz; i++){
            if(S[i] == curr){
                cnt++;
            }else{
                if(cnt >= 3){
                    vector<int>temp;
                    temp.push_back(idx1);
                    temp.push_back(i-1);
                    res.push_back(temp);
                }
                cnt = 1;
                idx1 = i;
                curr = S[i];
            }
        }
        
        if(cnt >= 3){
                    vector<int>temp;
                    temp.push_back(idx1);
                    temp.push_back(sz-1);
                    res.push_back(temp);
                }
        
        return res;
    }
};
