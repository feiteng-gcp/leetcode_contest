class Solution {
public:
    vector<vector<int>> largeGroupPositions(string S) {
        vector<vector<int> > res;
        if(S.size() < 3) return res;
        int pre = 0;
        int ct = 0;
        for(int i = 0; i < S.size(); i++){
            if(ct == 0){
                pre = i;
            }
            else{
                if(S[i] != S[pre]){
                    if(ct >= 3){
                        vector<int> tmp;
                        tmp.push_back(pre);
                        tmp.push_back(i-1);
                        res.push_back(tmp);
                    }
                    ct = 0;
                    pre = i;
                }
            }
            ct ++;
        }
        if(S[S.size()-1] == S[pre]){
            if(ct >= 3){
                vector<int> tmp;
                tmp.push_back(pre);
                tmp.push_back(S.size()-1);
                res.push_back(tmp);
            }
        }
        return res;
    }
};
