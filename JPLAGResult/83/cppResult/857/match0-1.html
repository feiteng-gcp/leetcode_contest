class Solution {
public:
    vector<vector<int>> largeGroupPositions(string S) {
        vector<vector<int>> vec;
        vector<int> tmp;tmp.push_back(0),tmp.push_back(1);
        int slen = S.size();

        int cnt = 1,pre = 0;
        for (int i = 1;i < slen;i++) {
            if (S[i] == S[i-1]) cnt++;
            else {
                if (cnt > 2) {
                    tmp[0] = pre;
                    tmp[1] = i - 1;
                    vec.push_back(tmp);
                }
                cnt = 1;
                pre = i;
            }
        }
        if (cnt > 2) {
            tmp[0] = pre;
            tmp[1] = slen - 1;
            vec.push_back(tmp);
        }
        return vec;
    }
};
