class Solution {
public:
    vector<vector<int>> largeGroupPositions(string S) {
        vector<vector<int>> res;
        vector<int> rs(2,0);
        int n = S.size();
        if(!n) return res;
        //char x = S[0]
        int cnt = 1, start = 0;
        for(int i = 1; i < n; ++i) {
            if(S[i] == S[i-1]) cnt++;
            else {
                if(cnt >= 3) {
                    rs[0] = start;
                    rs[1] = start + cnt - 1;
                    res.push_back(rs);
                }
                start = i;
                cnt = 1;
            }
        }
        if(cnt >= 3) {
                    rs[0] = start;
                    rs[1] = start + cnt - 1;
                    res.push_back(rs);
        }
        return res;
    }
};
