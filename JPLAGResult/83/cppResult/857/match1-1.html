class Solution {
public:
    vector<vector<int>> largeGroupPositions(string S) {
        vector<vector<int>> res;
        if(S.size()<3) return res;
        
        int cnt = 1;
        int start = 0;
        for(int i=1; i<S.size(); ++i){
            if(S[i] == S[i-1]){
                cnt++;
            }
            else{
                if(cnt>=3)
                    res.push_back({start, i-1});
                cnt = 1;
                start = i;
            }
        }
        if(cnt>=3)
            res.push_back({start, S.size()-1});
        
        return res;
    }
};
