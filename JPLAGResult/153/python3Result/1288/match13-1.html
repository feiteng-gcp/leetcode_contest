class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        f0 = [-1e5]*len(arr)
        f1 = [-1e5]*len(arr)
        f0[0] = arr
[0]
        ma = arr[0]
        for i in range(1,len(arr)):
            f1[i] = max(f0[i-1], f1[i-1]+arr[i])
            f0[i] = max(f0[i-1]+arr[i], arr[i])
            ma = max(ma, f0[i], f1[i])
        
        return ma
