class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        if max(arr) <= 0:
            return max(arr)
        while arr[-1] <0:
            arr.pop()
        arr = arr[::-1]
        while arr[-1] < 0:
            arr.pop()
        if min(arr) >= 0:
            return sum(arr)
        def get(arr):
            s = [i for i in arr]
            for i in range(1,len(arr)):
                s[i] += s[i-1]
            m = [i for i in s]
            for i in range(1,len(arr)):
                m[i] = min(m[i-1],m[i])
            return s,m
        s1,m1 = get(arr)
        s2,m2 = get(arr[::-1])
        s2,m2 = s2[::-1],m2[::-1]
        res = 0
        for i in range(1,len(arr)-1):
            tmp = s1[i-1] - min(0,m1[i-1]) + s2[i+1] - min(0,m2[i+1])
            res = max(res,tmp)
        return res
