class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        f = [[-100000 for i in range(2)] for j in range(len(arr)+1)]
        ans = -100000
        for i in range(1, len(arr)+1):
            f[i][0] = max(f[i-1][0]+arr
[i-1], arr[i-1])
            f[i][1] = max(f[i-1][0], f[i-1][1]+arr[i-1])
            ans = max(ans, max(f[i][0], f[i][1]))
        return ans
