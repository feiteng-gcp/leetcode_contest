class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        n = len(arr)
        dp = [[-10000000000 for _ in range(2)] for _ in range(n)]
        best = arr[0]
        dp[0][0] = arr[0]
        for i in range(1, n):
            dp[i][0] = max(dp[i-1][0] + arr
[i], arr[i])
            dp[i][1] = max(dp[i-1][0], dp[i-1][1] + arr[i])
            best = max(best, max(dp[i][0], dp[i][1]))
        return best
