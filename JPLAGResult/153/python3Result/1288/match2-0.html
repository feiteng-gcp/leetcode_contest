class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        n = len(arr)
        
        forward = [0 for k in range(n)] 
        backward = [0 for k in range(n)] 

        cur_max, max_so_far = arr[0], arr[0] 

        forward[0] = arr[0]
        for i in range(1, n): 
            cur_max = max(arr[i], cur_max + arr[i]) 
            max_so_far = max(max_so_far, cur_max) 
            forward[i] = cur_max 

        backward[n-1] = arr[n-1] 
        cur_max = max_so_far = backward[n-1]
        
        i = n-2
        while i >= 0: 
            cur_max = max(arr[i], cur_max + arr[i]) 
            max_so_far = max(max_so_far, cur_max) 

            backward[i] = cur_max 
            i -= 1
        
        res = max_so_far 

        for i in range(1, n-1): 
            res = max(res, forward[i - 1] + backward[i + 1]) 

        return res
