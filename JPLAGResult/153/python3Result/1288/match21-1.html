class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        if len(arr) == 1:
            return arr[0]
        del_dp = [0] * len(arr)
        non_del_dp = [0] * len(arr)
        non_del_dp[0] = arr[0]
        for i in range(1, len
(arr)):
            del_dp[i] = max(del_dp[i-1]+arr[i], non_del_dp[i-1])
            non_del_dp[i] = max(arr[i], non_del_dp[i-1]+arr[i])
        del_max = max(del_dp[1:])
        non_del_max = max(non_del_dp)
        return max(del_max, non_del_max)
