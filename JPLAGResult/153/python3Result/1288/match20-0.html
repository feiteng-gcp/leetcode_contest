class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        dp0 = [arr[0]]
        dp1 = [max(0, arr[0])]
        
        if all(num <= 0 for num in arr):
            return max(arr)
        
        for i in range(1, len(arr)):
            dp0.append(max(dp0[-1] + arr[i], arr[i]))
            dp1.append(max([dp0[-1], dp1[-1] + arr[i], dp0[-2]]))
                
        return max(max(dp0), max(dp1))
