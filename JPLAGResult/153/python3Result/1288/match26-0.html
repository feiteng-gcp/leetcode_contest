class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        preSum = [0]
        for num in arr:
            preSum.append(preSum[-1]+num)
        res = float('-inf')
        MIN = float('inf')
        pre = []
        for s in preSum:
            
            res = max(res, s-MIN)
            pre.append(s-MIN)
            MIN = min(MIN, s)
            
        length = len(arr)
        afterSum = [0]
        for i in range(length-1, -1, -1):
            afterSum.append(afterSum[-1]+arr[i])
        
        after = []
        MIN = float('inf')
        t = float('-inf')
        for s in afterSum:
            
            t = max(t, s-MIN)
            after.append(s-MIN)
            MIN = min(MIN, s)
        
        t = float('-inf')
        for i in range(length):
            t = max(t, pre[i-1+1]+after[length-(i+1)])
        print(preSum, afterSum)
        print(res, t, pre, after)
        res = max(res, t)
        return res
