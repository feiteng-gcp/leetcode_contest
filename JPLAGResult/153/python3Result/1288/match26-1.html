class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        now = -int(1e5)
        max_sum = -int(1e5)
        
        a_end_max = []
        a_begin_max = []
        
        
for a in arr:
            now = max(a, now + a)
            a_end_max.append(now)
            max_sum = max(max_sum, now)
            
        now = -int(1e5)
        for a in reversed(arr):
            now = max(a, now + a)
            a_begin_max.append(now)
            
        a_begin_max.reverse()
        
        N = len(arr)
        if N == 1:
            return arr[0]
        
        for i in range(1, N-1):
            left = a_end_max[i-1]
            right = a_begin_max[i + 1]
            max_sum = max(max_sum, left + right)
        
        return max_sum
