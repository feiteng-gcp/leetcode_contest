class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        maxleft = [arr[0]] * len(arr)
        for i in range(1, len(arr)):
            maxleft[i] = arr[i] + max(maxleft[i-1], 0)
        maxright = [arr[-1]] * len(arr)
        for i in range(len(arr)-2, -1, -1):
            maxright[i] = arr[i] + max(maxright[i+1], 0)
        max_single = max
(maxleft)
        max_delete = -float("inf")
        for i in range(1, len(arr)-1):
            max_delete = max(max_delete, maxleft[i-1] + maxright[i+1])
        return max(max_single, max_delete)
