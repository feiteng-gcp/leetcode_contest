class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        first  = []
        second = []
        n = len(arr)
        for i in range(n):
            first.append(0)
            second.append(0)
        currentM= arr[0]
        myMax   = arr[0]
        first[0] = arr[0]
        for i in range(1,n):
            currentM = max(arr[i],currentM+arr[i])
            myMax    = max(myMax,currentM)
            first[i] = currentM
        currentM= arr[-1]
        myMax   = arr[-1]
        second[n-1] = arr[-1]
        for i in range(n-2,-1,-1):
            currentM  = max(arr[i],currentM+arr[i])
            myMax     = max(myMax,currentM)
            second[i] = currentM    
        print (first,second)
        output = myMax
        for i in range(1,n-1):
            output = max(output,first[i-1]+second[i+1])
        return output
