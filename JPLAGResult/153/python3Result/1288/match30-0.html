class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        left=[0]*len(arr)
        right=[0]*len(arr)
        left[0]=arr[0]
        right[-1]=arr[-1]
        for i in range(1,len(arr)):
            left[i]=max(arr[i],left[i-1]+arr[i])
        for j in range(len(arr)-2,-1,-1):
            right[j]=max(arr[j],right[j+1]+arr[j])
        

        res=-float("inf")
        flag=False
        for i in range(len(arr)):
            if arr[i]<0:
                flag=True
            tmp=0
            f=False
            if i>0:
                if left[i-1]>=0:
                    tmp+=left[i-1]
                    f=True
            if i<len(arr)-1:
                if right[i+1]>=0:
                    tmp+=right[i+1]
                    f=True
            if not f:
                res=max(res,arr[i])
            else:
                res=max(res,tmp)
        if not flag:
            return sum(arr)
        return res
