class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        left_deleted=-10000000
        left_notdeleted=0
        ret=arr[0]
        if max(arr)<0:
            return max(arr)
        for i in range(len(arr)):
            left_deleted1=max(left_deleted+arr
[i],left_notdeleted)
            left_notdeleted1=max(left_notdeleted+arr[i],arr[i])
            left_deleted=left_deleted1
            left_notdeleted=left_notdeleted1
            ret=max(ret,left_notdeleted,left_deleted)
        return ret
