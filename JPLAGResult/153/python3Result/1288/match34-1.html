class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        maxa=-10**5
        for i in range(len(arr)):
            maxa=max(arr[i],maxa)
        if maxa<=0:
            return maxa
        left=[0 for i in range(len(arr))]
        i=1
        while(i<len
(arr)):
            left[i]=max(left[i],left[i-1]+arr[i-1])
            left[i]=max(left[i],arr[i-1])
            i=i+1
        
        right=[0 for i in range(len(arr))]
        i=len(arr)-2
        while(i>=0):
            right[i]=max(right[i],right[i+1]+arr[i+1])
            right[i]=max(right[i],arr[i+1])
            i=i-1
        for i in range(len(arr)):
            maxa=max(maxa,left[i]+right[i])
        return maxa
