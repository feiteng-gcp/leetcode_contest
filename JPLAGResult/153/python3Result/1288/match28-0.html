class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        forward = self.maxSubSum(arr)
        backward = self.maxSubSum(arr[::-1])[::-1]
        l = [0]*len(arr)
        l[0] = backward[len(arr)-2]
        l[len(arr)-1] = forward[len(arr)-2]
        #print(forward)
        #print(backward)
        for i in range(1, len(arr)-1):
            l[i] = forward[i-1] + backward[i+1]
        #print(l)
        return max(max(l), max(forward))
        
    def maxSubSum(self, arr):
        sum = [0]*len(arr)
        sum[0] = arr[0]
        for i in range(1,len(arr)):
            if sum[i-1] < 0:
                sum[i] = arr[i]
            else:
                sum[i] = sum[i-1] + arr[i]
        return sum
