from typing import *

class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        if all(a <= 0 for a in arr):
            return max(arr)
        if all(a >= 0 for a in arr):
            return sum(arr)

        current_best = 0
        deleted_cumsum = 0
        nondeleted_cumsum = 0
        for a in arr:
            deleted_cumsum = max(deleted_cumsum + a, nondeleted_cumsum)
            nondeleted_cumsum = max(nondeleted_cumsum + a, 0)
            current_best = max(nondeleted_cumsum, deleted_cumsum, current_best)

        return current_best
