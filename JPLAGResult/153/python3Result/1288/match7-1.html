class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        
if all(v < 0 for v in arr):
            return max(arr)
        
        full_acc = 0
        drop_acc = 0
        result = 0
        for v in arr:
            drop_acc = max(full_acc, drop_acc + v, 0)
            full_acc = max(full_acc + v, 0)
            result = max(result, drop_acc, full_acc)
            
        return result
