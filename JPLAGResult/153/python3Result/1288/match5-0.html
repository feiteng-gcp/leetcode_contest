class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        n = len(arr)
        left, right = [0] * n, [0] * n
        curr = maxi = float("-inf")
        for i, val in enumerate(arr):
            curr = max(val, curr+val)
            maxi = max(maxi, curr)
            left[i] = curr
        curr = maxi = float("-inf")
        for i in range(n-1, -1, -1):
            curr = max(arr[i], curr+arr[i])
            maxi = max(maxi, curr)
            right[i] = curr
        ans = maxi
        for i in range(1, n-1):
            ans = max(ans, left[i-1] + right[i+1])
        return ans
