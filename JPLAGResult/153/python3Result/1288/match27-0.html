class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        right_max = []
        left_max = []
        for i in range(len(arr)):
            if i == 0:
                left_max.append(arr[0])
            else:
                left_max.append(arr[i] + max(0, left_max[-1]))
        for i in range(len(arr)-1, -1, -1):
            if i == len(arr)-1:
                right_max.append(arr[i])
            else:
                right_max.append(arr[i] + max(0, right_max[-1]))
        right_max = right_max[::-1]
        res = float('-inf')
        for i in range(1, len(arr) -1):
            res = max(res, left_max[i-1] + right_max[i+1])
        
        #print(res)
        #print(left_max)
        #print(right_max)
        return max([res, max(left_max), max(right_max)])
