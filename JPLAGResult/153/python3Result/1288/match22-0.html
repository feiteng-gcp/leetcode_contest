class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        max_ends_with_i, max_begins_with_i = [], []
        prev_sum = 0
        for i in range(len(arr)):
            if i == 0:
                prev_sum = arr[i]
            else:
                prev_sum = max(prev_sum, 0) + arr[i]
            max_ends_with_i.append(prev_sum)
        post_sum = 0
        for i in range(len(arr)-1, -1, -1):
            if i == len(arr)-1:
                post_sum = arr[i]
            else:
                post_sum = max(post_sum, 0) + arr[i]
            max_begins_with_i.append(post_sum)
        max_begins_with_i.reverse()
        res = max(arr)
        #print(max_begins_with_i, max_ends_with_i)
        for i in range(len(arr)-1):
            res = max(res, max_ends_with_i[i] + max_begins_with_i[i+1])
            if i < len(arr)-2:
                res = max(res, max_ends_with_i[i] + max_begins_with_i[i+2])
        return res
