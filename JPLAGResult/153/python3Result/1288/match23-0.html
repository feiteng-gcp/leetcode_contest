class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        max_ending_here = []
        
        total = 0
        for num in arr:
            total += num
            if num > total:
                total = num
            max_ending_here.append(total)
        
        total = 0
        max_starting_here = []
        for num in arr[::-1]:
            total += num
            if num > total:
                total = num
            max_starting_here.append(total)
        max_starting_here.reverse()
            
        ans = max(max_ending_here+max_starting_here)
        
        for i in range(1,len(arr)-1):
            ans = max(ans, max_ending_here[i-1]+max_starting_here[i+1])
        return ans
