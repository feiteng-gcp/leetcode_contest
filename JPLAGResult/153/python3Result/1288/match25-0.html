class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        
        dp = []
        dp.append([arr[0], 0, max(arr[0], 0)])
        for i in range(1, len(arr)):
            neverSkipped = max(dp[i-1][0] + arr[i], arr[i])
            skippedOnce = max(dp[i-1][1] + arr[i], dp[i-1][0])
            best = max(neverSkipped, skippedOnce)
            dp.append([neverSkipped, skippedOnce, best])
            
        # print(dp)
        if max(arr) < 0:
            return max(arr)
        else:
            return max([x[2] for x in dp])
