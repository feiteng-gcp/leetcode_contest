class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        dp1 = [None] * len(arr)
        dp2 = [None] * len(arr)
        dp1[0] = arr[0]
        for i in range(1, len(arr)):
            if dp1[i - 1] > 0:
                dp1[i] = dp1[i - 1] + arr[i]
            else:
                dp1[i] = arr[i]
        dp2[-1] = arr[-1]
        for i in range(len(arr) - 2, -1, -1):
            if dp2[i + 1] > 0:
                dp2[i] = dp2[i + 1] + arr[i]
            else:
                dp2[i] = arr[i]
        ans = max(max(dp1), max(dp2))
        for i in range(len(arr) - 2):
            ans = max(ans, dp1[i] + dp2[i + 2])
        return ans
