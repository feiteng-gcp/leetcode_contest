class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        current_max = arr[0]
        
        max_with_last_has_dropped = arr[0]
        max_with_last_no_drop = arr[0]
        
        for i in range(1, len(arr)):
            new_max_with_last_no_drop = max(arr[i], max_with_last_no_drop + arr[i])
            new_max_with_last_has_dropped = max(arr[i], max_with_last_no_drop, max_with_last_has_dropped + arr[i])
            
            max_with_last_no_drop = new_max_with_last_no_drop
            max_with_last_has_dropped = new_max_with_last_has_dropped
            
            current_max = max(max_with_last_has_dropped, max_with_last_no_drop, current_max)
            
        return current_max
