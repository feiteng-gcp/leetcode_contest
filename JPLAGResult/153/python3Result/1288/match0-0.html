class Solution:
    def maximumSum(self, arr: List[int]) -> int:
        if max(arr) <= 0:
            return max(arr)
        while arr[-1] < 0:
            arr.pop()
        arr = arr[::-1]
        while arr[-1] <0:
            arr.pop()
        def get(arr):
            sum_arr = [i for i in arr]
            for i in range(1,len(arr)):
                sum_arr[i] += sum_arr[i-1]
            min_arr = [i for i in sum_arr]
            for i in range(1,len(arr)):
                min_arr[i] = min(min_arr[i],min_arr[i-1])
            return sum_arr,min_arr
        print(arr)
        if min(arr) >= 0:
            return sum(arr)
        s1,m1 = get(arr)
        s2,m2 = get(arr[::-1])
        s2, m2 = s2[::-1], m2[::-1]
        #print(s1)
        #print(m1)
        #print(s2)
        #print(m2)
        res = -float('inf')
        for i in range(1,len(arr)-1):
            tmp = s1[i-1] - min(m1[i-1],0) + s2[i+1] - min(0,m2[i+1])
            res = max(res,tmp)
        return res
