class Solution:
    def makeArrayIncreasing(self, arr1: List[int], arr2: List[int]) -> int:
        arr2 = sorted(list(set
(arr2)))
        n = len(arr1)
        
        memo = {}
        def backtracking(i, prev):
            nonlocal res
            if i == n:
                return 0
            
            if (i, prev) in memo:
                return memo[(i, prev)]
            
            ans = float("Inf")
                
            if arr1[i] <= prev: # must replace
                replace = bisect.bisect_right(arr2, prev)
                if replace < len(arr2):
                    ans = 1 + backtracking(i+1, arr2[replace])
                
            else: # optional replace
                # no replace
                ans = min(ans, backtracking(i+1, arr1[i]))
                # replace
                replace = bisect.bisect_right(arr2, prev)
                if replace < len(arr2) and arr2[replace] < arr1[i]:
                    ans = min(ans, 1 + backtracking(i+1, arr2[replace]))
                    
            memo[(i, prev)] = ans
            return ans     
            
        res = backtracking(0, -1)
        
        return res if res < float("Inf") else -1
