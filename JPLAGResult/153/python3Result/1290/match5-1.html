def find(x, arr):
    l = 0
    r = len(arr)-1
    pos = len(arr)
    while l<r:
        mid = (r+l)//2
        if x>=arr[mid]:
            l = mid+1
        else:
            r = mid
            pos = r
    if arr[l]>x:
        pos = l
    if pos>=len(arr):
        return -1
    # print(x, pos)
    return arr[pos]
    
class Solution:    
    def makeArrayIncreasing(self, arr1: List[int], arr2: List[int]) -> int:
        arr2 = sorted(list(set(arr2)))
        # print(arr2)
        f = [[-1 for i in range(len(arr1)+2)] for j in range(len(arr1)+2)]
        f[0][0] = arr1[0]
        f[0][1] = arr2[0]
        for i in range(1, len(arr1)):
            for j in range(i+2):
                if j>0 and f[i-1][j-1]>=0:
                    f[i][j] = find(f[i-1][j-1], arr2)
                    # print(i, j, f[i-1][j-1], f[i][j])
                if f[i-1][j]>=0 and arr1[i]>f[i-1][j]:
                    if f[i][j] != -1:
                        f[i][j] = min(arr1
[i], f[i][j])
                    else:
                        f[i][j] = arr1[i]
        # for i in range(len(arr1)):
        #     print(arr1[i], f[i][0:len(arr1)+1])
        for i in range(len(arr1)+1):
            if f[len(arr1)-1][i]>0:
                return i
        return -1
