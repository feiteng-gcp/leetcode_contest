from bisect import bisect_right

class Solution:
    def makeArrayIncreasing(self, arr1, arr2) -> int:
        if len(arr1) == 1:
            return 0
        arr2 = sorted(set(arr2))
        f = [[-1]*(len(arr1)+1) for _ in range(len(arr1))]
        f[0][0] = arr1[0]
        f[0][1] = min(arr1[0], min(arr2))

        for i in range(len(arr1)-1):
            for j in range(len(arr1)+1):
                if f[i][j] == -1:
                    continue
                if arr1[i+1] > f[i][j]:
                    if f[i+1][j] == -1 or f[i+1][j] > arr1[i+1]:
                        f[i+1][j] = arr1[i+1]
                if j <= len(arr1):
                    kk = bisect_right(arr2, f[i][j])
                    if kk == len(arr2):
                        continue
                    if f[i+1][j+1] == -1 or f[i+1][j+1] > arr2[kk]:
                        f[i+1][j+1] = arr2[kk]
        for j in range(len(arr1)+1):
            if f[-1][j] != -1:
                return j
        return -1
