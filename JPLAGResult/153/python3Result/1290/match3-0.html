import bisect

class Solution:
    def makeArrayIncreasing(self, a: List[int], b: List[int]) -> int:
        b.sort()
        lb = len(b)
        n = len(a)
        # MAXN = 2010
        # d = [[None]*n for _ in range(MAXN)]
        d = {}

        def dp(k, i):
            if i == n:
                return 0
            if (k,i) in d:
                return d[(k,i)]
            
            low = bisect.bisect_left(b, k+1)
            # print(b,k+1,low)
            # high = bisect.bisect_right(b, a[i])
            maxx = float('inf')
            for idx in range(low, lb):
                # print(b[idx], i+1)
                r = dp(b[idx], i+1)
                if r != -1:
                    # print(k, b[idx], i+1, r+(b[idx] != a[i]))
                    maxx = min(maxx, r+(b[idx] != a[i]))
                break
            if a[i] > k:
                r = dp(a[i], i+1)
                if r!=-1:
                    maxx = min(maxx, dp(a[i], i+1))
            maxx = -1 if maxx == float('inf') else maxx
            d[(k,i)] = maxx
            return maxx
        
        return dp(-1, 0)
