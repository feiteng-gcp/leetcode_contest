class Solution:
    def dayOfTheWeek(self, day: int, month: int, year: int) -> str:
        week = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"]
        monthcnt = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31 ,30, 31]
        daycnt = 0
        daycnt += (year-1971)*365 + ((year-1971+2) // 4)
        for i in range(month-1):
            daycnt += monthcnt[i]
            if year!=2100 and year%4==0 and i==1:
                daycnt += 1
        daycnt += day-1
        return week[(daycnt+5) % 7]
