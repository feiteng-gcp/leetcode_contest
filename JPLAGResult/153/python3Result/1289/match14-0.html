import datetime

class Solution:
    def dayOfTheWeek(self, day: int, month: int, year: int) -> str:
        today = datetime.datetime(year, month, day)
        mp = {7: "Sunday", 1: "Monday", 2: "Tuesday", 3: "Wednesday", 4: "Thursday", 5: "Friday", 6: "Saturday"}
        return mp[today.isoweekday()]
