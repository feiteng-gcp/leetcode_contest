class Solution:
    def distanceBetweenBusStops(self, distance: List[int], start: int, destination: int) -> int:
        total=sum(distance)
        if start>destination:
            temp=start
            start=destination
            destination=temp
        ret1=sum(distance[start:destination])
        ret2=total-ret1
        return min(ret1,ret2)
