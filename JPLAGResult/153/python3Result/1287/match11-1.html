class Solution:
    def distanceBetweenBusStops(self, distance: List[int], start: int, destination: int) -> int:
        N=len(distance)
        s1=0
        a=start
        while(not a==destination):
            s1=s1+distance[a]
            a=(a+1)%N
        s2=0
        a=start
        while(not a==destination):
            s2=s2+distance[(a+N-1)%N]
            a=(a+N-1)%N
        return min(s1,s2)
