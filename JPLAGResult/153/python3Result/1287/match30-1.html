class Solution:
    def distanceBetweenBusStops(self, distance: List[int], start: int, des: int) -> int:
        if start==des:
            return 0
        
        tot 
= sum(distance)
        
        low, high = min(start, des), max(start, des)
        d = sum(distance[low:high])
        return min(d, tot-d)
