class Solution:
    def distanceBetweenBusStops(self, distance: List[int], start: int, end: int) -> int:
        if start == end:
            return 0
        
if start > end:
            start, end = end, start
        
        all_ = sum(distance)
        val = sum(distance[start:end])
        return min(all_-val, val)
