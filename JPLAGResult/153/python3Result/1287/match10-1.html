class Solution:
    def distanceBetweenBusStops(self, distance: List[int], start: int, destination: int) -> int:
        # Not inclusive of the latter one.
        lowest = min(start, destination)
        highest = max(start, destination)
        inclusive_sum = sum(distance[lowest:highest])
        exclusive_sum = sum(distance[:lowest] + distance[highest:])
        return min(inclusive_sum, exclusive_sum)
