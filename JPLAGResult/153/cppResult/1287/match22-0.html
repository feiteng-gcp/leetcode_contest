class Solution {
public:
    int distanceBetweenBusStops(vector<int>& d, int start, int dest) {
        int ret = INT_MAX; int n = d.size();
        int curr1 = 0;
        for(int i = start; i != dest; i = (i + 1) % n){
            curr1 += d[i % n];
        }
        int curr2 =0;
        for(int i = dest; i != start; i = (i + 1) % n){
            curr2 += d[i % n];
        }
        ret = min(curr1, curr2);
        return ret;
    }
};
