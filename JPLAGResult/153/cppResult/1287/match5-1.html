class Solution {
public:
    int distanceBetweenBusStops(vector<int>& dist, int st, int des) {
        int sum = accumulate(dist.begin(), dist.end(), 0);
        if(st > des) swap(st, des);
        // make st <= des
        int cw = accumulate(dist.begin()+st, dist.begin()+des, 0); // clockwise
        return min(cw, sum - cw);
    }
};
