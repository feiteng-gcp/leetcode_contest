class Solution {
public:
    int distanceBetweenBusStops(vector<int>& distance, int start, int destination) {
        int i = start;
        int size = distance.size();
        int answer = numeric_limits<int>::max();
        
        int sum = 0;
        while (i != destination) {
            sum += distance[i];
            i = (i + 1) % size;
        }
        
        answer = min(answer, sum);
        sum = 0;
        while (i != start) {
            sum += distance[i];
            i = (i + 1) % size;
        }
        
        answer = min(answer, sum);
        
        return answer;
    }
};
