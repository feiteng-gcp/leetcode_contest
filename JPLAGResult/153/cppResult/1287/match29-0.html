class Solution {
public:
    int distanceBetweenBusStops(vector<int>& distance, int start, int destination) {
        int res = 0;
        int l = min(start, destination);
        int r = max(start, destination);
        if (l == r) return 0;
        for (int i = l; i < r; ++i)
            res += distance[i];
        int res2 = 0;
        for (int i = 0; i < l; ++i)
            res2 += distance[i];
        for (int i = r; i < distance.size(); ++i)
            res2 += distance[i];
        return min(res, res2);
    }
};
