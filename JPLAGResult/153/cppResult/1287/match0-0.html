#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    int distanceBetweenBusStops(vector<int>& distance, int start, int destination) {
        int d1 = 0, d2 = 0;
        for (int i = start; i != destination; i = (i+1)%distance.size()) {
        	d1 += distance[i];
        }
         for (int i = destination; i != start; i = (i+1)%distance.size()) {
        	d2 += distance[i];
        }
        return min(d1, d2);
    }
};
