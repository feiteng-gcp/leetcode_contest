class Solution {
public:
    int distanceBetweenBusStops(vector<int>& a, int s, int d) {
        int dist1, dist2;
        dist1 = dist2 = 0;
        int n = a.size();
        for (int i = s; i != d; i = (i + 1) % n) {
            dist1 += a[i];
        }
        for (int i = s; i != d; i = (i - 1 + n) % n) {
            dist2 += a[(i - 1 + n) % n];            
        }
        // cout << dist1 << " " << dist2 << endl;
        return min(dist1, dist2);
        
    }
};
