class Solution {
public:
    int distanceBetweenBusStops(vector<int>& distance, int start, int destination) {
        int o1 = 0;
        int o2 = 0;
        
        int s = distance.size();
        vector<int> d(3*s);
        for(int i = 0; i < s; i++) {
            d[i] = distance[i];
        }
        for(int i = 0; i < s; i++) {
            d[i+s] = distance[i];
        }

        if(start >= destination) swap(start, destination);
        for(int i = start; i < destination; i++) {
            o1 += d[i];
        }
        
        for(int i = destination; i < start+s; i++) {
            o2 += d[i];
        }
      //  cout<<o1<<o2<<endl;
        return min(o1,o2);
    }
};
