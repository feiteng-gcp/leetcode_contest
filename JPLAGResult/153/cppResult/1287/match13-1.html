class Solution {
public:
    int distanceBetweenBusStops(vector<int>& distance, int start, int destination) {
        int sum = 0;
        for (int dis : distance)
            sum += dis;

        int st = min(start, destination), ed = max(start, destination);
        int res = 0;
        for (int i=st; i<ed; i++)
            res += distance[i];
        return min(res, sum-res);
    }
};
