class Solution {
public:
    int dis(vector<int>& dd, int s, int d, int n) {
        int res = 0;
        while (s != d) {
            res += dd[s];
            s = (s + 1) % n;
        }
        return res;
    }
    int distanceBetweenBusStops(vector<int>& distance, int start, int destination) {
        int n = distance.size();
        return min(dis(distance, start, destination, n), dis(distance, destination, start, n));
    }
};
