class Solution {
public:
    int distanceBetweenBusStops(vector<int>& distance, int start, int destination) {
        int s=0;
        int t=0;

        if(start>destination){
            int a= start;
            start = destination;
            destination=a;
        }
        for(int i=0;i<distance.size();++i){
            s+=distance[i];
        }
        for(int i=start;i<destination;++i){
            t+=distance[i];
        }
        return min(s-t,t);
    }
};
