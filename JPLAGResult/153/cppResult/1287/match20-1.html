class Solution {
public:
    int distanceBetweenBusStops(vector<int>& a, int s, int d) {
        if(s==d){
            return 0;
        }
        int d1=0, d2=0;
        if(s>d){
            swap(s,d);
        }

        for(int i = s ; i < d; i++){
            d1 += a[i];
        }
        for(int i= 0 ; i < a.size(); i++){
            d2 += a[i];
        }
        return min(d1, abs(d2-d1));
    }
};
