class Solution {
public:
    int distanceBetweenBusStops(vector<int>& distance, int start, int destination) {

        int sum = accumulate(distance.begin(), distance.end(), 0);

        int a = min(start, destination), b = max(start, destination);
        int x = 0;
        for(int i = a; i < b; i ++)
            x += distance[i];
        return min(x, sum - x);
    }
};
