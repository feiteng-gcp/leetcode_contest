class Solution {
public:
    int dis(vector<int>& d, int n, int s, int e) {
        int ans = 0;
        while (s != e) {
            ans += d[s];
            s = (s + 1) % n;
        }
        return ans;
    }
    
    int distanceBetweenBusStops(vector<int>& distance, int start, int destination) {
        return min(dis(distance, distance.size(), start, destination), dis(distance, distance.size(), destination, start));
    }
};
