class Solution {
public:
    int maximumSum(vector<int>& arr) {
        int n = arr.size();
        vector<vector<int>> dp(n, vector<int>(2, 0));
        dp[0][0] = arr[0];
        dp[0][1] = 0;
        int m = dp[0][0];
        for(int i = 1; i < n; i++) {
            dp[i][0] = max(dp[i-1][0] + arr[i], arr[i]);
            dp[i][1] = max(dp[i-1][1] + arr[i], dp[i-1][0]);
            m = max(dp[i][0], max(dp[i][1], m));
        }
        return m;
    }
};
