class Solution {
public:
    int maximumSum(vector<int>& arr) {
        int n = arr.size();

        vector< int > f(n, 0), b(n, 0);

        int cmax = arr[0], rmax = arr[0]; 

        f[0] = arr[0]; 
        for (int i=1; i<n; i++) { 
            cmax = max(arr[i], cmax + arr[i]); 
            rmax = max(rmax, cmax); 
            f[i] = cmax; 
        } 

        cmax = rmax = b[n-1] = arr[n-1]; 
        for (int i=n-2; i>=0; i--){ 
            cmax = max(arr[i], cmax + arr[i]); 
            rmax = max(rmax, cmax); 
            b[i] = cmax; 
        } 

        int res = rmax; 

        for (int i = 1; i < n - 1; i++) 
            res = max(res, f[i - 1] + b[i + 1]); 

        return res; 
    } 
};
