class Solution {
public:
    int maximumSum(vector<int>& a) {
        int n = a.size();
        vector<int> fw(n), bw(n);
        int mx = a[0], cur = a[0];
        fw[0] = a[0];
        for(int i = 1; i < n; i++) {
            cur = max(a[i], cur+a[i]);
            mx = max(mx, cur);
            fw[i] = cur;
        }
        cur = mx = bw[n-1] = a[n-1];
        for(int i = n-2; i >= 0; i--) {
            cur = max(a[i], cur+a[i]);
            mx = max(mx, cur);
            bw[i] = cur;
        }
        int res = mx;
        for(int i = 1; i < n-1; i++)
            res = max(res, fw[i-1] + bw[i+1]);
        return res;
    }
};
