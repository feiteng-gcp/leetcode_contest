class Solution {
public:
    int maximumSum(vector<int>& arr) {
        int n=arr.size();
        int fw[n],bw[n];
        int curr=arr[0],max1=arr[0];
        fw[0]=arr[0];
        for(int i=1;i<n;i++)
        {
            curr=max(arr[i],arr[i]+curr);
            max1=max(max1,curr);
            fw[i]=curr;
        }
        bw[n-1]=arr[n-1];
        curr=arr[n-1],max1=arr[n-1];
        for(int i=n-2;i>=0;i--)
        {
            curr=max(arr[i],curr+arr[i]);
            max1=max(max1,curr);
            bw[i]=curr;
        }
        int ans=max1;
        for(int i=1;i<n-1;i++)
        {
            ans=max(ans,fw[i-1]+bw[i+1]);
        }
        return ans;
        
        
    }
};
