class Solution {
public:
    int maximumSum(vector<int>& arr) {
        int n = arr.size();
        int fw[n], bw[n]; 
        int c = arr[0], m = arr[0]; 
        fw[0] = arr[0]; 
        for (int i = 1; i < n; i++) { 
            c = max(arr[i], c + arr[i]); 
            m = max(m, c); 
            fw[i] = c; 
        } 
        c = m = bw[n-1] = arr[n-1]; 
        for (int i = n-2; i >= 0; i--) { 
            c = max(arr[i], c + arr[i]); 
            m = max(m, c); 
            bw[i] = c; 
        } 
        int ans = m; 
        for (int i = 1; i < n - 1; i++) 
            ans = max(ans, fw[i - 1] + bw[i + 1]); 
        return ans; 
    } 
};
