class Solution {
public:
    int dp[100010][2];
    int maximumSum(vector<int>& arr) {
        int n=arr.size();
        int ans=-1e9;
        dp[1][1]=-1e9;
        for(int i=1;i<=n;i++){
            dp[i][0]=max(arr[i-1],arr[i-1]+dp[i-1][0]);
            if(i>1)dp[i][1]=arr[i-1]+max(dp[i-2][0],dp[i-1][1]);
            ans=max(ans,max(dp[i][0],dp[i][1]));
        }
        return ans;
    }
};
