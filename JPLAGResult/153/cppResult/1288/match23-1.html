class Solution {
public:
    int maximumSum(vector<int>& arr) {
        int n = arr.size();
        vector<int> forward(n);
        vector<int> backward(n);
        int curv = arr[0];

        int maxv = arr[0];
        forward[0] = arr[0];
        for (int i = 1; i < n; ++i) {
            curv = max(arr[i], curv + arr[i]);
            maxv = max(maxv, curv);
            forward[i] = curv;
        }
        backward[n - 1] = curv = maxv = arr[n - 1];
        for (int i = n - 2; i >= 0; --i) {
            curv = max(arr[i], curv + arr[i]);
            maxv = max(maxv, curv);
            backward[i] = curv;
        }
        int res = maxv;
        for (int i = 1; i < n - 1; i++) {
            res = max(res, forward[i - 1] + backward[i + 1]); 
        }
        return res;
    }
};
