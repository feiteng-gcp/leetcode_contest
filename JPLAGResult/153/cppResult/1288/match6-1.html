class Solution {
public:
    int maximumSum(vector<int>& arr) {

        int n = arr.size();
        if(n == 1) return arr[0];
        
        vector<int> dp1(n, 0);
        dp1[0] = arr[0];
        for(int i = 1; i < n; i ++)
            if(dp1[i - 1] <= 0) dp1[i] = arr[i];
            else dp1[i] = arr[i] + dp1[i - 1];
//        cout << "dp1 : "; for(int e: dp1) cout << e << " "; cout << endl;

        int res = *max_element(dp1.begin(), dp1.end());
        vector<int> dp2(n, 0);
        dp2[1] = arr[1];
        for(int i = 2; i < n; i ++)
            dp2[i] = max(arr[i] + dp2[i - 1], arr[i] + dp1[i - 2]);
//        cout << "dp2 : "; for(int e: dp2) cout << e << " "; cout << endl;

        res = max(res, *max_element(dp2.begin() + 1, dp2.end()));
        return max(res, accumulate(arr.begin(), arr.begin() + (n - 1), 0));
    }
};
