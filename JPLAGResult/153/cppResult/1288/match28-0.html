class Solution {
public:
    int maximumSum(vector<int>& arr) {
        int n = arr.size();
        vector<int> dp1(n, 0), dp2(n, 0);
        int ret = arr[0];
        dp1[0] = dp2[0] = arr[0];
        for (int i = 1; i < n; ++i) {
            dp1[i] = max(dp2[i-1], dp1[i-1] + arr[i]);// shan chu
            dp2[i] = max(dp2[i-1] + arr[i], arr[i]);
            ret = max(ret, max(dp1[i], dp2[i]));
        }
        return ret;
        
    }
};
