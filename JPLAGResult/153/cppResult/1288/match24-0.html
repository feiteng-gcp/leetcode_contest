class Solution {
public:
    int maximumSum(vector<int>& arr) {
        int n = arr.size();
        vector<int> dp1(n, 0);
        vector<int> dp2(n, 0);
        int curMax = arr[0], globalMax = arr[0];
        dp1[0] = arr[0];
        
        for(int i = 1; i < n; ++i)
        {
            curMax = max(arr[i], curMax + arr[i]);
            globalMax = max(curMax, globalMax);
            dp1[i] = curMax;
        }
        
        curMax = globalMax = dp2[n-1] = arr[n-1];
        for(int i = n - 2; i >= 0; --i)
        {
            curMax = max(arr[i], curMax + arr[i]);
            globalMax = max(globalMax, curMax);
            dp2[i] = curMax;
        }
        
        int res = globalMax;
        for(int i = 1; i < n - 1; ++i)
        {
            res = max(res, dp1[i-1] + dp2[i+1]);
        }
        return res;
        
    }
};
