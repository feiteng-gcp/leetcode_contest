class Solution {
public:
    int maximumSum(vector<int>& arr) {
        int n=arr.size();
        vector<int>tmp1(n),tmp2(n); 
        int cur = arr[0],mx=arr[0]; 
        tmp1[0] = arr[0]; 
        for (int i = 1; i < n; i++) { 
            cur = max(arr[i], cur + arr[i]); 
            mx = max(mx, cur); 
            tmp1[i] = cur; 
        } 
        cur = mx = tmp2[n-1] = arr[n-1]; 
        for (int i = n-2; i >= 0; i--) { 
            cur = max(arr[i], cur + arr[i]); 
            mx = max(mx, cur); 
            tmp2[i] = cur; 
        } 
        int ret = mx; 
        for (int i = 1; i < n - 1; i++) 
            ret = max(ret, tmp1[i - 1] + tmp2[i + 1]); 

        return ret;
    }
};
