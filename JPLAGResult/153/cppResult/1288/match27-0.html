class Solution {
public:
    int maximumSum(vector<int>& arr) {
        int n=arr.size();
        int forward[n], backward[n];
        int curMax=arr[0], res=arr[0];
        forward[0]=arr[0];
        for(int i=1; i<n;i++){
            curMax=max(arr[i], curMax+arr[i]);
            res=max(res, curMax);
            forward[i]=curMax;
        }
        
        curMax=res=backward[n-1]=arr[n-1];
        for(int i=n-2; i>=0;i--){
            curMax=max(arr[i],curMax+arr[i]);
            res=max(res,curMax);
            backward[i]=curMax;
        }
        
        int maxRemove=res;
        for(int i=1; i<n-1;i++){
            maxRemove=max(maxRemove, forward[i-1]+backward[i+1]);
        }
        
        return maxRemove;
    }
};
