class Solution {
public:
    int maximumSum(vector<int>& arr) {
        int size = arr.size();
        vector<int> forward(size, 0);
        vector<int> backward(size, 0);
        int cur = arr[0], overall = arr[0];
        forward[0] = arr[0];
        for (int i = 1; i < arr.size(); ++i) {
            cur = max(arr[i], cur + arr[i]);
            forward[i] = cur;
            
            overall = max(cur, overall);
        }
        
        backward[size - 1] = arr[size - 1];
        cur = arr[size - 1];
        overall = arr[size - 1];
        
        for (int i = size - 2; i >= 0; --i) {
            cur = max(arr[i], cur + arr[i]);
            backward[i] = cur;
            
            overall = max(overall, cur);
        }
        
        int ret = overall;
        for (int i = 1; i < size - 1; i++) {
            ret = max(ret, forward[i - 1] + backward[i + 1]);
        }
        
        return ret;
    }
};
