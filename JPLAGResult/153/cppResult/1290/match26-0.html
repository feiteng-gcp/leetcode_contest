class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        
        map<int, int> mp;
        
        reverse(arr1.begin(), arr1.end());
        arr1.push_back(INT_MIN);
        reverse(arr1.begin(), arr1.end());
        
        
        sort(arr2.begin(), arr2.end());
        
        int dp[2100][2100];
        
        for (int i = 0; i < 2100; i++)
            for (int j = 0; j < 2100; j++)
                dp[i][j] = INT_MAX - 2;
        
        dp[0][0] = 0;
        
        for (int i = 1; i < arr1.size(); i++)
            for (int j = 0; j <= arr2.size(); j++) {
                int cur = 0;
                if (j == 0)
                    cur = arr1[i - 1]; else 
                    cur = arr2[j - 1];
                
                if (cur < arr1[i]) {
                    dp[i][0] = min(dp[i][0], dp[i - 1][j]);
                }
                
                
                auto tutu = upper_bound(arr2.begin(), arr2.end(), cur);
                if (tutu != arr2.end()) {
                    dp[i][tutu - arr2.begin() + 1] =  min(dp[i][tutu - arr2.begin() + 1], dp[i - 1][j] + 1);
                }
            }
        int ans = INT_MAX - 2;
        
        for (int j = 0; j <= arr2.size(); j++)
            ans = min(ans, dp[arr1.size() - 1][j]);
        if (ans == INT_MAX - 2)
            return -1; else
            return ans;
        
    }
};
