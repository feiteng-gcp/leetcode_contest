class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2)
{
    set<int> tab;
    int len = (int)arr2.size();
    for(int i = 0; i < len; i++) tab.insert(arr2[i]);
    len = (int)arr1.size();
    vector<vector<int> > dp(len, vector<int>(len + 1, 0x3f3f3f3f));

    for(int i = 0; i < len; i++)
    {
        for(int j = 0; j <= i + 1; j++)
        {
            if(i == 0 || arr1[i] > dp[i - 1][j])
                dp[i][j] = min(dp[i][j], arr1[i]);
            if(j > 0)
            {
                auto p = tab.begin();
                if(i > 0) p = tab.upper_bound(dp[i - 1][j - 1]);
                if(p == tab.end()) continue;
                dp[i][j] = min(dp[i][j], *p);
            }
        }
    }
    for(int i = 0; i <= len; i++)
        if(dp[len - 1][i] < 0x3f3f3f3f) return i;
    return -1;
}
};
