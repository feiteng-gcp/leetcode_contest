const int maxN = 5010;
int f[maxN][maxN];

class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        map<int, int> values;
        for (int x : arr1) values[x] = 0;
        for (int x : arr2) values[x] = 0;
        int curr_id = 0;
        for (auto& p : values) p.second = curr_id++;
        for (int& x : arr1) x = values[x];
        for (int& x : arr2) x = values[x];
        unordered_set<int> opt(arr2.begin(), arr2.end());
        /*for (int x : arr1) cout << x << " ";
        cout << endl;
        for (int x : arr2) cout << x << " ";
        cout << endl;*/
        int n = arr1.size();
        for (int i = 0; i < n; ++i) for (int j = 0; j < curr_id; ++j)
            f[i][j] = n + 1;
        for (int i = 0; i < n; ++i)
            for (int j = i; j < curr_id; ++j) {
                if (j) f[i][j] = min(f[i][j], f[i][j - 1]);
                // No-switch
                if (j == arr1[i]) {
                    if (j) f[i][j] = min(f[i][j], i ? f[i - 1][j - 1] : 0);
                    else f[i][j] = 0;
                }
                // Switch
                if (opt.count(j))
                    f[i][j] = min(f[i][j], (i ? f[i - 1][j - 1] : 0) + 1);
            }
        int res = n + 1;
        for (int i = 0; i < curr_id; ++i)
            res = min(res, f[n - 1][i]);
        if (res > n) return -1;
        return res;
    }
};
