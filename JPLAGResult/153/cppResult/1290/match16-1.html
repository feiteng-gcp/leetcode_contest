#define INF 1000000000

class Solution {
public:
    int cost(int a, int b, const unordered_set<int>& arr2_values) {
        if (a == b) return 0;
        if (arr2_values.find(b) == arr2_values.end()) return INF;
        return 1;
    }
    
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        unordered_set<int> arr2_values(arr2.begin(), arr2.end());
        
        vector<int> values = arr1;
        values.insert(values.end(), arr2.begin(), arr2.end());
        
        sort(values.begin(), values.end());
        auto it = unique(values.begin(), values.end());
        values.resize(it - values.begin());
        
        /*r (int& x : values) cout << x << " ";
        cout << endl;
        cout << endl;*/
        
        int n = arr1.size(), m = values.size();
        vector<vector<int>> f(n, vector<int>(m, INF));
        

        for (int i = 0; i < n; ++i) {
            int min_ = INF;
                
            for (int j = 0; j < m; ++j) {
                if (i == 0) {
                    f[i][j] = cost(arr1[i], values[j], arr2_values);
                }
                else if (j == 0) {
                    f[i][j] = INF;

                    min_ = min(min_, f[i-1][j]);
                }
                else {
                    f[i][j] = min(f[i][j], min_ + cost(arr1[i], values[j], arr2_values));
                    min_ = min(min_, f[i-1][j]);
                }
            }
        }
        
        /*r (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (f[i][j] == INF) cout << "# ";
                else cout << f[i][j] << " ";
            }
            cout << endl;
        }*/
        
        int res = INF;
        for (int j = 0; j < m; ++j) {
            res = min(res, f[n-1][j]);
        }
        if (res == INF) res = -1;
        return res;
    }
};
