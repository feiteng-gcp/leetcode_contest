class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        std::sort(arr2.begin(), arr2.end());
        arr2.erase(unique(arr2.begin(), arr2.end()), arr2.end());
        int n = arr2.size();
        int dp[2020][2020];
        memset(dp, 0x3f3f3f3f, sizeof(dp));

        for (int j = 0; j < n; ++j)
            dp[0][j] = 1;
        dp[0][n] = 0;
        for (int i = 1; i < arr1.size(); ++i)
        {
            for (int j = 0; j < n; j++)
            {
                if (dp[i-1][j] != 0x3f3f3f3f && j != n - 1) 
                    dp[i][j+1] = std::min(dp[i-1][j] + 1, dp[i][j+1]);
                if (arr2[j] > arr1[i-1]) dp[i][j] = std::min(dp[i][j], dp[i-1][n] + 1);
            }
            for (int k = 0; k < n && arr2[k] < arr1[i]; ++k)
                dp[i][n] = std::min(dp[i][n], dp[i-1][k]);
            if (arr1[i] > arr1[i-1]) dp[i][n] = std::min(dp[i][n], dp[i-1][n]);
        }
        /*
        for (int i = 0; i < arr1.size(); ++i)
        {
            printf("i %d\n", i);
            for (int j = 0; j <= n; ++j)
            {
                printf("dp[%d][%d] %d\n", i, j, dp[i][j]);
            }
        }
        */
        int ans = 0x3f3f3f3f;
        for (int i = 0; i <= n; i++)
        {
            ans = std::min(dp[arr1.size()-1][i], ans);
        }
        if (ans == 0x3f3f3f3f) return -1;
        return ans;
    }
};
