#include <map>
#include <vector>
#include <algorithm>
using namespace std;
const int MAXN = 2100;

int cv[MAXN];
int dp[MAXN];
map <int, int> mm;
int nguy;

int gb (int nlo, int nhi)
{
    auto it = mm.lower_bound(nlo + 1);
    auto it2 = mm.lower_bound(nhi);
    return it2->second - it->second;
}

class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        mm.clear();
        int N = arr1.size();
        for (int x : arr2)
            mm[x];
        mm[2e9];
        nguy = 0;
        for (auto it = mm.begin(); it != mm.end(); it++)
            it->second = nguy++;
        
        cv[0] = -1e9;
        for (int i = 0; i < N; i++)
            cv[i+1] = arr1[i];
        cv[N+1] = 1.1e9;
        N += 2;
        
        for (int i = 0; i < MAXN; i++)
            dp[i] = -1e9;
        dp[0] = 1;
        
        for (int i = 0; i < N; i++)
        {
            for (int j = i + 1; j < N; j++)
            {
                if (cv[i] < cv[j] && gb (cv[i], cv[j]) >= j - i - 1)
                    dp[j] = max (dp[j], dp[i] + 1);
            }
        }
        
        if (dp[N-1] < -1e8)
            return -1;
        return N - dp[N-1];
    }
};
