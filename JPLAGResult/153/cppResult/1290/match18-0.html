class Solution {
    const static int MAX = 0x3f3f3f3f;
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        sort(arr2.begin(), arr2.end());
        arr2.erase(unique(arr2.begin(), arr2.end()), arr2.end());
        int n1 = arr1.size(), n2 = arr2.size();
        vector<vector<int>> dp(n1 + 1, vector<int>(n2 + 1, 0));
        dp[0][0] = 0;
        for(int j = 1; j <= n2; j++) dp[0][j] = 1;
        for(int i = 1; i < n1; i++) {
            int m = MAX;
            if(arr1[i] > arr1[i-1]) m = min(m, dp[i-1][0]);
            for(int j = 1; j <= n2; j++) {
                if(arr1[i] > arr2[j-1]) m = min(m, dp[i-1][j]);
            }
            dp[i][0] = m;
            //cout << dp[i][0] << ' ';
            for(int j = 1; j <= n2; j++) {
                m = MAX;
                if(arr2[j-1] > arr1[i-1]) m = min(m, dp[i-1][0] + 1);
                if(j > 1) {
                    m = min(m, dp[i-1][j-1] + 1);
                    m = min(m, dp[i][j-1]);
                }
                dp[i][j] = m;
           //// cout << dp[i][j] << ' ';
            }
           // cout << endl;
        }
        int ans = *min_element(dp[n1-1].begin(), dp[n1-1].end());
        if(ans >= MAX) return -1;
        return ans;
    }
};
