class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        int n = arr1.size(), m = arr2.size();
        int f[n + 1][m + 1], g[n + 1][m + 1], mf[n + 1][m + 1], mg[n + 1][m + 1];
        sort(arr2.begin(), arr2.end());
        f[0][0] = n + 1;
        g[0][0] = 0;

        for (int i = 1; i <= m; ++i) f[0][i] = g[0][i] = n + 1;
        for (int i = 0; i <= n; ++i)
            for (int j = 0; j <= m; ++j)
                mf[i][j] = n + 1;
        memset(mg, 0, sizeof(mg));

        for (int i = 1; i <= n; ++i) {
            for (int j = 0; j <= m; ++j) {
                if (j < 2 || arr2[j - 1] > arr2[j - 2])
                    f[i][j] = min(j != 0 ? mf[i - 1][j - 1] : n + 1,
                                  j != 0 && (i == 1 || arr2[j - 1] > arr1[i - 2]) ? mg[i - 1][j - 1] : n + 1) + 1; else
                    f[i][j] = n + 1;
                g[i][j] = min((j != 0 && arr1[i - 1] > arr2[j - 1] ? mf[i - 1][j] : n + 1),
                              (i == 1 || arr1[i - 1] > arr1[i - 2] ? mg[i - 1][j] : n + 1));
            }
            mf[i][0] = f[i][0];
            mg[i][0] = g[i][0];
            for (int j = 1; j <= m; ++j) {
                mf[i][j] = min(mf[i][j - 1], f[i][j]);
                mg[i][j] = min(mg[i][j - 1], g[i][j]);
            }
        }
        return min(mf[n][m], mg[n][m]) > n ? -1 : min(mf[n][m], mg[n][m]);
    }
};
