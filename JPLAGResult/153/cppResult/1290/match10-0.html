class Solution {
public:
    int makeArrayIncreasing(vector<int>& a1, vector<int>& a2) {
        int n1 = a1.size();
        set<int> s2(a2.begin(), a2.end());
        vector<int> a3(s2.begin(), s2.end());
        int n3 = a3.size();
        
        vector<vector<int>> dp(n1, vector<int>(n3, 1e9));
        vector<int> dp1(n1, 1e9);
        for(int i = 0; i < n3; i++)
            dp[0][i] = (a1[0] != a3[i]);
        dp1[0] = 0;
        for(int i = 1; i < n1; i++) {
            int cur = dp[i - 1][0];
            for(int j = 1; j < n3; j++) {
                dp[i][j] = min(dp[i][j], cur + (a1[i] != a3[j]));
                cur = min(cur, dp[i - 1][j]);
            }
            for(int j = 0; j < n3; j++) {
                if(a3[j] > a1[i - 1])
                    dp[i][j] = min(dp[i][j], dp1[i - 1] + (a1[i] != a3[j]));
            }
            for(int j = 0; j < n3; j++) {
                if(a3[j] < a1[i])
                    dp1[i] = min(dp1[i], dp[i - 1][j]);
            }
            if(a1[i - 1] < a1[i])
                dp1[i] = min(dp1[i], dp1[i - 1]);
        }
        int res = 1e9;
        for(int i = 0; i < n3; i++)
            if(dp[n1 - 1][i] >= 0)
                res = min(res, dp[n1 - 1][i]);
        if(dp1[n1 - 1] >= 0)
            res = min(res, dp1[n1 - 1]);
        return res >= 1e9 ? -1 : res;
        
    }
};
