class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        sort(arr2.begin(),arr2.end());
        auto end_unique = unique(arr2.begin(), arr2.end()); 
        arr2.erase(end_unique, arr2.end());
        //for (auto i:arr2) cout<<i<<' '; cout<<endl;
        
        vector<vector<int>> f(0);
        int self=arr2.size();
        for (int i=0;i<arr1.size();i++)
        {
            f.push_back(vector<int>(arr2.size()+1));
        }
        for (int j=0;j<arr2.size();j++)
        {
            f[0][j]=1;
        }

        f[0][self]=0;
        
        for (int i=1;i<arr1.size();i++)
        {
            for (int j=0;j<arr2.size();j++)
            {
                if (j==0)
                {
                    if (arr2[j]>arr1[i-1] && f[i-1][self]!=-1) f[i][j]=1;
                    else f[i][j]=-1;
                    continue;
                }
                f[i][j]=-1;
                if (arr2[j]>arr1[i-1] && f[i-1][self]!=-1) f[i][j]=f[i-1][self]+1;
                if (f[i-1][j-1]!=-1) 
                {
                    if (f[i][j]==-1) f[i][j]=f[i-1][j-1]+1;
                    else f[i][j]=min(f[i][j],f[i-1][j-1]+1);
                }
            }
            
            if (arr1[i]>arr1[i-1]) f[i][self]=f[i-1][self];
            else f[i][self]=-1;
            //cout<<i<<":"<<f[i][self]<<endl;
            for (int j=0;j<arr2.size();j++)
            {
                if (arr1[i]>arr2[j] && f[i-1][j]!=-1)
                {
                    if (f[i][self]==-1) {f[i][self]=f[i-1][j]; }

                    else f[i][self]=min(f[i][self],f[i-1][j]);
                }
            }
        }
        /*
        for (auto i:f)
        {
            for(auto j:i)
                cout<<j<<' ';
            cout<<endl;
        }
        */
        int ret=f[arr1.size()-1][self];
        for (int j=0;j<=self;j++)
        {
            if (f[arr1.size()-1][j]!=-1)
            {
                if (ret==-1) ret=f[arr1.size()-1][j];
                else ret=min(ret,f[arr1.size()-1][j]);
            }
        }
        return ret;
    }
};
