class Solution {
public:
    int A[4004],B[4004],cnt,INF=0x3f3f3f3f;
    int dp[2004][4004],minn[2004][4004];
    bool vis[4004];
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        int n=arr1.size();
        for(int i=0;i<arr1.size();i++){
            A[++cnt]=arr1[i];
        }
        for(int i=0;i<arr2.size();i++){
            A[++cnt]=arr2[i];
        }
        sort(A+1,A+1+cnt);
        int m=unique(A+1,A+1+cnt)-A-1;
        sort(arr2.begin(),arr2.end());
        for(int i=1;i<=arr2.size();i++){
            B[i]=lower_bound(A+1,A+1+m,arr2[i-1])-A;
            vis[B[i]]=1;
        }
        memset(dp,0x3f,sizeof(dp));
        memset(minn,0x3f,sizeof(minn));
        for(int i=0;i<=m;i++)dp[0][i]=minn[0][i]=0;
        for(int i=1;i<=n;i++){
            int p=lower_bound(A+1,A+1+m,arr1[i-1])-A;
            for(int j=1;j<=m;j++){
                if(vis[j]==1||p==j){
                    dp[i][j]=minn[i-1][j-1]+(p==j?0:1);
                    minn[i][j]=min(minn[i][j-1],dp[i][j]);
                }
                else{
                    dp[i][j]=dp[i][j-1];
                    minn[i][j]=minn[i][j-1];
                }
            }
        }
        int ans=INF;
        for(int i=1;i<=m;i++)ans=min(ans,dp[n][i]);
        if(ans==INF)return -1;
        return ans;
    }
};
