class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        set<int> s;
        for (int x : arr1)
            s.insert(x);
        for (int x : arr2)
            s.insert(x);
        int cnt = 0;
        unordered_map<int, int> m;
        for (auto now : s)
            m[now] = cnt++;
        for (int i = 0; i < arr1.size(); i++)
            arr1[i] = m[arr1[i]];
        for (int i = 0; i < arr2.size(); i++)
            arr2[i] = m[arr2[i]];
        
        s.clear();
        for (int x : arr2) 
            s.insert(x);
        vector<int> change(cnt, -1);
        for (int i = 0; i < cnt; i++) {
            auto it = s.upper_bound(i);
            if (it != s.end())
                change[i] = *it;
        }
        
        vector<int> ths(4001, 0x3f3f3f3f);
        ths[arr1[0]] = 0;
        for (int i : s) {
            ths[i] = min(ths[i],1);
        }
        for (int i = 1; i < arr1.size(); i++) {
            vector<int> nxt(4001, 0x3f3f3f3f);
            for (int j = 0; j < cnt; j++) {
                if (change[j] != -1) {
                    nxt[change[j]] = min(nxt[change[j]],ths[j]+1);
                }
            }
            int mn = 0x3f3f3f3f;
            for (int j = 0; j < arr1[i]; j++) {
                mn = min(mn,ths[j]);
            }
            nxt[arr1[i]] = min(nxt[arr1[i]], mn);
            ths = nxt;
        }
        
        int res = 0x3f3f3f3f;
        for (int i = 0; i < cnt; i++) {
            res = min(res, ths[i]);
        }
        return res == 0x3f3f3f3f ? -1: res;
    }
};
