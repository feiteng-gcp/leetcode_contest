class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) 
    {
        int n = arr1.size();
        auto dp = vector<vector<int>>(n,vector<int>(n+1,INT_MAX));
        
        sort(arr2.begin(),arr2.end());

        dp[0][0] = arr1[0];
        dp[0][1] = arr2[0];
        
        for (int i=1; i<n; i++)
        {
            for (int k=0; k<=i+1; k++)
            {
                if (k==0)
                {
                    if (dp[i-1][0] == INT_MAX)
                        dp[i][0] = INT_MAX;
                    else
                        dp[i][0] = (arr1[i]>arr1[i-1])?arr1[i]:INT_MAX;
                }
                else if (k==i+1)
                {
                    if (dp[i-1][i]==INT_MAX)
                        dp[i][k] = INT_MAX;
                    else 
                    {
                        int c2 = INT_MAX;
                        int num = dp[i-1][k-1];
                        auto iter = upper_bound(arr2.begin(),arr2.end(),num);
                        if (iter!=arr2.end())
                            c2 = *iter;
                        if (c2!=INT_MAX)
                            dp[i][k] = c2;
                    }
                }
                else
                {
                    int c1 = INT_MAX, c2 = INT_MAX;
                    
                    if (dp[i-1][k]<arr1[i])
                        c1 = arr1[i];
                    
                    if (dp[i-1][k-1]!=INT_MAX)
                    {
                        int num = dp[i-1][k-1];
                        auto iter = upper_bound(arr2.begin(),arr2.end(),num);
                        if (iter!=arr2.end())
                            c2 = *iter;
                    }                    
                    
                    if (c1!=INT_MAX || c2!=INT_MAX)
                        dp[i][k] = min(c1,c2);                    

                }
                
                //cout<<i<<" "<<k<<" "<<dp[i][k]<<endl;
            }
        }
        
        cout<<"OK"<<endl;
        
        for (int k=0; k<=n; k++)
        {
            if (dp[n-1][k]!=INT_MAX)
                return k;
        }
        
        return -1;
    }
};
