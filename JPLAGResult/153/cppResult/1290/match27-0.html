class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        int len1 = arr1.size();
        int len2 = arr2.size();
        vector<int> data(arr1);
        for (int i = 0; i < len2; i++)
            data.push_back(arr2[i]);
        sort(data.begin(), data.end());
        int len = unique(data.begin(), data.end()) - data.begin();
        if (len < len1) return -1;
        for (int i = 0; i < len1; i++)
            arr1[i] = lower_bound(data.begin(), data.begin() + len, arr1[i]) - data.begin() + 1;
        vector<bool> can_reach(len + 1, 0);
        for (int i = 0; i < len2; i++) {
            int idx = lower_bound(data.begin(), data.begin() + len, arr2[i]) - data.begin() + 1;
            can_reach[idx] = 1;
        }
        vector<int> row(len + 1, 100000000);
        vector<vector<int>> opt(len1 + 1, row);
        vector<vector<int>> best(len1 + 1, row);
        for (int j = 0; j <= len; j++) {
            opt[0][j] = 0;
            best[0][j] = 0;
        }
        for (int i = 1; i <= len1; i++)
            for (int j = i; j <= len; j++) {
                int change = 1;
                if (arr1[i - 1] == j) change = 0;
                if (!can_reach[j] && change != 0) change = 100000000;
                if (best[i - 1][j - 1] + change < opt[i][j])
                    opt[i][j] = best[i - 1][j - 1] + change;
                best[i][j] = best[i][j - 1];
                if (opt[i][j] < best[i][j])
                    best[i][j] = opt[i][j];
            }
        int ret = 100000000;
        for (int j = 1; j <= len; j++)
            if (opt[len1][j] < ret) ret = opt[len1][j];
        if (ret == 100000000) return -1;
        return ret;
    }
};
