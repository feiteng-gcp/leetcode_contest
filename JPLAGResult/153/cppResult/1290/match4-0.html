class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        const int n = arr1.size();
        const int m = arr2.size();
        vector<vector<int> > dp(n, vector<int>(m + 1, -1));
        sort(arr2.begin(), arr2.end());
        for (int i = 0; i < m; ++i) dp[0][i] = 1;
        dp[0][m] = 0;
        for (int i = 0; i < n - 1; ++i) {
            for (int j = 0; j <= m; ++j) {
                if (dp[i][j] == -1) continue;
                int x = j < m ? arr2[j] : arr1[i];
                for (int h = (j < m) ? j + 1 : 0; h <= m; ++h) {
                    int y = h < m ? arr2[h] : arr1[i + 1];
                    if (x < y) {
                        if (dp[i + 1][h] == -1 || dp[i + 1][h] > dp[i][j] + (h < m)) {
                            dp[i + 1][h] = dp[i][j] + (h < m);
                            h = m - 1;
                        }
                    }
                }
            }
        }
        int ans = -1;
        for (int i = 1; i <= m; ++i) {
            if (dp[n - 1][i] == -1) continue;
            if (ans == -1 || ans > dp[n - 1][i]) ans = dp[n - 1][i];
        }
        return ans;
    }
};
