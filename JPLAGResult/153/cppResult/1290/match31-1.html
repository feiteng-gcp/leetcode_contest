class Solution {
public:
    void update(int &x, int val) {
    if (x == -1)
        x = val;
    else
        if (val != -1)
            x = min(x, val);
    }
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        int ret = -1;
        int n = arr1.size(), m = arr2.size();
        sort(arr2.begin(), arr2.end());
        unique(arr2.begin(), arr2.end());
        for(int i = 1; i < m; ++i) {
            if (arr2[i] <= arr2[i - 1]) {
                m = i;
                break;
            }
        }
        vector< vector<int> > dp(n, vector<int>(m+1));
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j <= m; ++j) {
                dp[i][j] = -1;
            }
        }
        dp[0][m] = 0;

        for (int i = 0; i < m; ++i) dp[0][i] = 1;
        for (int i = 1; i < n; ++i) {

            for(int j = 0; j < m; ++j) {
                if (dp[i - 1][j] != -1) {

                    if (arr2[j] < arr1[i]) update(dp[i][m], dp[i - 1][j]);

                    if (j + 1 != m) {
                        update(dp[i][j + 1], dp[i - 1][j] + 1);
                    }
                }
            }

            if (dp[i - 1][m] != -1) {
                for(int j = 0; j < m; ++j) {
                    if (arr2[j] > arr1[i - 1]) {
                        update(dp[i][j], dp[i - 1][m] + 1);
                    }
                }
                if (arr1[i] > arr1[i - 1]) {
                    update(dp[i][m], dp[i - 1][m]);
                }
            }
        }
        for (int i = 0; i <= m; ++i) update(ret, dp[n - 1][i]);
        return ret;
    }
};
