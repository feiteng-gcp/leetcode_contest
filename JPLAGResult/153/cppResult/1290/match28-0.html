class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        sort(arr2.begin(), arr2.end());
        int t=1;
        for (int i=1; i<arr2.size(); i++)
            if (arr2[i]!=arr2[t-1]) arr2[t++]=arr2[i];
        arr2=vector<int>(arr2.begin(), arr2.begin()+t);
        vector<vector<int>> f(arr1.size(), vector<int>(arr2.size()+1, 10000));
        for (int i=0; i<arr2.size(); i++) f[0][i]=1;
        f[0].back()=0;
        for (int i=1; i<arr1.size(); i++) {
            for (int j=arr2.size()-1; ~j; j--) {
                if (j>0) f[i][j]=f[i-1][j-1]+1;
                if (arr2[j]>arr1[i-1]) f[i][j]=min(f[i][j], f[i-1].back()+1);
            }
            f[i].back()=arr1[i]>arr1[i-1]?f[i-1].back():10000;
            for (int j=0; j<arr2.size() && arr2[j]<arr1[i]; j++)
                f[i].back()=min(f[i].back(), f[i-1][j]);
        }
        /*for (int i=0; i<arr2.size(); i++) cout << arr2[i] << ' ';
        cout << endl;
        for (int i=0; i<arr1.size(); i++) {
            for (const auto& x: f[i]) cout << x << ' ';
             cout << endl;
        }*/
        int m=10000;
        for (const auto& x: f.back()) m=min(m, x);
        return m<=arr2.size()?m:-1;
    }
};
