class Solution {
#define maxn 2345
	int dp[maxn][maxn * 2];
	int A[maxn], B[maxn];
	int Hash[2 * maxn];
public:
	int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
		int n = arr1.size(), m = arr2.size();
		int cnt = 0;
		for (int i = 0; i < n; ++i) A[i + 1] = arr1[i], Hash[++cnt] = arr1[i];
		for (int i = 0; i < m; ++i) B[i + 1] = arr2[i], Hash[++cnt] = arr2[i];
		sort(Hash + 1, Hash + cnt + 1);
		cnt = unique(Hash + 1, Hash + cnt + 1) - Hash - 1;
		for (int i = 1; i <= n; ++i) A[i] = lower_bound(Hash + 1, Hash + cnt + 1, A[i]) - Hash;
		sort(B + 1, B + m + 1);
		set<int> s;
		for (int i = 1; i <= m; ++i) B[i] = lower_bound(Hash + 1, Hash + cnt + 1, B[i]) - Hash, s.insert(B[i]);
		for (int i = 1; i <= n; ++i)
			for (int j = 1; j <= cnt; ++j)
				dp[i][j] = 1e5;
		// for (int i = 1; i <= m; ++i)
		// 	cout << B[i] << " ";
		// cout << endl;
		int ans = 1e5;
		for (int i = 1; i <= n; ++i) {
			for (int j = i; j <= cnt; ++j) {
				if (j == A[i]) dp[i][j] = dp[i - 1][j - 1];
				else {
					if (s.find(j) != s.end()) dp[i][j] = dp[i - 1][j - 1] + 1;
				}
				if (i == n)
					ans = min(ans, dp[i][j]);
			}
			for (int j = 2; j <= cnt; ++j)
				dp[i][j] = min(dp[i][j], dp[i][j - 1]);

		}
		// cout << dp[2][1] << endl;
		if (ans == 1e5)
			return -1;
		return ans;
	}
};
