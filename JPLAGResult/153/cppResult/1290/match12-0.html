#define N 2011
long long dp[N][N];
class Solution {
public:
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        int n = arr1.size(), m = arr2.size();
        set<long long> tmp(arr2.begin(), arr2.end());
        for(int i=0;i<N;i++) for(int j=0;j<N;j++) dp[i][j] = 1e14;
        for(int i=0;i<n;i++) dp[0][i] = -1;
        for(int i=0;i<n;i++) {
            for(int j=0;j<=i+1;j++) {
                if(dp[i][j] < arr1[i]) dp[i+1][j] = arr1[i];
                if(j > 0 && dp[i][j-1] < 1e14) {
                    auto it = tmp.lower_bound(dp[i][j-1] + 1);
                    if(it != tmp.end()) {
                        dp[i+1][j] = min(dp[i+1][j], *it);
                    }
                }
            }
        }
        for(int i=0;i<=n;i++) {
            if(dp[n][i] < 1e14) return i;
        }
        return -1;
    }
};
