const int INF = 1000000000;

template<class T>
void print(const T &a)
{
	for (auto x : a) printf("%d ", x); puts("");
}

class Solution {
public:
    int makeArrayIncreasing(vector<int>& a, vector<int>& b) {
        vector<int> p;
        int n = a.size();
        int m = b.size();
        for (auto x : a) p.push_back(x);
        for (auto x : b) p.push_back(x);
        sort(p.begin(), p.end());
        p.erase(unique(p.begin(), p.end()), p.end());
        int k = p.size();
        for (auto &x : a) x = lower_bound(p.begin(), p.end(), x)-p.begin()+1;
        for (auto &x : b) x = lower_bound(p.begin(), p.end(), x)-p.begin()+1;
        
        vector<int> u(k+1);
        for (auto x : b) u[x] = 1;
        
        vector<vector<int>> F(n+1, vector<int>(k+1, INF));
        for (int j = 0; j <= k; ++ j)
            F[0][j] = 0;
        for (int i = 1; i <= n; ++ i)
        {
            for (int j = 1; j <= k; ++ j)
            {
                F[i][j] = F[i][j-1];
                if (a[i-1] == j)
                {
                    F[i][j] = min(F[i][j], F[i-1][j-1]);
                }
                if (u[j])
                    F[i][j] = min(F[i][j], F[i-1][j-1]+1);
            }
        }
        
        int ret = INF;
        for (int j = 1; j <= k; ++ j)
            ret = min(ret, F[n][j]);
        if (ret == INF) return -1;
        return ret;
    }
};
