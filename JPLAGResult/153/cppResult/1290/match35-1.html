int dp[2010][2010];
int dpx[2010][2010];
class Solution {
public:
   

int makeArrayIncreasing(vector<int>& a, vector<int>& b) {
	sort(b.begin(), b.end());
    b.erase(unique(b.begin(), b.end()), b.end());
    memset(dp, -1, sizeof dp);
    memset(dpx, -1, sizeof dpx);
    int n = a.size();
    dp[1][0] = a[0];
    if(b[0] < a[0]) {
    	dp[1][1] = b[0];
    	dpx[1][1] = 0;
    }

    int c = b.size();
   // cout << c << endl;
    for (int i = 2; i <= n; i++) {
    	for (int j = 0; j <= min(i, c); j++) {
    		//if(dp[i - 1][j] != -1) 
    		if(dp[i - 1][j] != -1 && a[i - 1] > dp[i - 1][j]) {
    			dp[i][j] = a[i - 1];
    			dpx[i][j] = dpx[i - 1][j]; 
    		} else if(dp[i - 1][j] != -1 && a[i - 1] <= dp[i - 1][j]) {
    			/*int t = dpx[i - 1][j];
    			while(t + 1 < c && b[t] <= dp[i - 1][j - 1]) {
    				t++;
    			}
    			if(t < c && b[t] > dp[i - 1][j - 1]) {
    				dp[i][j] = b[t];
    				dpx[i][j] = t;
    			} */
    		}
    		if(j - 1 >= 0 && dp[i - 1][j - 1] != -1 && dpx[i - 1][j - 1] + 1 < c) {
    			int t = dpx[i - 1][j - 1] + 1;
    			while(t < c && b[t] <= dp[i - 1][j - 1]) {
    				t++;
    			}
    			//cout << t << " " << b[t] << endl;
    			if(t < c && b[t] > dp[i - 1][j - 1]) {
    				if(dp[i][j] == -1 || dp[i][j] > b[t] || (
    					dp[i][j] == b[t] && t < dpx[i][j])) {
    					dp[i][j] = b[t];
    					dpx[i][j] = t;
    				}
    			}
    		}
    		//cout << i  << " " << j << " " << dp[i][j] << " " << dpx[i][j] << endl;
    	}
    }
    for (int i = 0; i <= c; i++) {
    	if(dp[n][i] != -1)
    		return i;
    }
    return -1;
}
};
