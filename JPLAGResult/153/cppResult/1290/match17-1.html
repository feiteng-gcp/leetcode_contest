class Solution {
public:
    vector<vector<int> > dp;
    
    int makeArrayIncreasing(vector<int>& arr1, vector<int>& arr2) {
        sort(arr2.begin(), arr2.end());
        vector<int> tarr2;
        
        for(int i = 0 ; i < arr2.size() ; i ++)
        {
            if(i == 0 || arr2[i] > arr2[i - 1])
            {
                tarr2.push_back(arr2[i]);
            }
        }
        
        arr2 = tarr2;
        
        int sz1 = arr1.size();
        int sz2 = arr2.size();
        
        for(int i = 0 ; i <= sz1 ; i ++)
        {
            vector<int> tmp(sz2 + 1, 30000);
            dp.push_back(tmp);
        }
        
        dp[0][0] = 0;
        
        for(int i = 1 ; i <= sz1 ; i ++)
        {
            // not pick
            if(i == 1 || arr1[i - 1] > arr1[i  - 2])
            {
                dp[i][0] = min(dp[i][0], dp[i - 1][0]);
            }
            

            for(int j = 1 ; j <= sz2 ; j ++)
            {
                if(arr1[i - 1] > arr2[j - 1])
                {
                    dp[i][0] = min(dp[i][0], dp[i - 1][j]);
                }
            }
            
            // pick
            int preMin = 30000;
            for(int j = 1 ; j <= sz2 ; j ++)
            {
                if(i == 1 || arr2[j - 1] > arr1[i - 2]) // pre-0
                {
                    dp[i][j] = min(dp[i][j], dp[i - 1][0] + 1);
                }
                
                dp[i][j] = min(dp[i][j], preMin + 1);
                preMin = min(preMin, dp[i - 1][j]);
            }

        }
        
        int res = 30000;
        for(int i = 0 ; i <= sz2 ; i ++)
        {
            res = min(res, dp[sz1][i]);
        }
        return res == 30000 ? -1 : res;
    }
};
