class Solution {
public:
    const string week[7] = {"Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"};
    
    string dayOfTheWeek(int d, int m, int y) {
         if(m <= 2) {
            m += 12;
            --y;
        }
        
        return week[(d+2*m+3*(m+1)/5+y+y/4-y/100+y/400)%7];
    }
};
