class Solution 
{
    vector<string> days = {"Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"};

public:
    string dayOfTheWeek(int d, int m, int y) 
    {
       if((m==1)||(m==2))
       {
            m+=12;
            y--;
       }
        int W= (d+2*m+3*(m+1)/5+y+y/4-y/100+y/400) % 7;
        return days[W];
    }
};
