class Solution {
public:
    string dayOfTheWeek(int day, int month, int year) {
        int days = 0;
        for (int y = 1971; y < year; ++y) {
            days += isRun(y) ? 366 : 365;

        }
        
        vector<int> dInMonth = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        if (isRun(year)) {
            dInMonth[1] = 29;
        }
        
        for (int m = 1; m < month; ++m) {
            days += dInMonth[m - 1];
        }
        
        days += day - 1;
        vector<string> table = {"Friday", "Saturday", "Sunday", "Monday", "Tuesday", "Wednesday", "Thursday"};
        return table[days%7];
    }
    
    bool isRun(int year) {
        if (year % 100 != 0)
            return year % 4 == 0;
        return year % 400 == 0;
    }
};
