class Solution {
public:
    string dayOfTheWeek(int day, int month, int year) {
        tm t = { 0, 0, 0, day, month-1, year-1900 };
        time_t t1 = mktime(&t);
        tm *t2 = localtime(&t1);
        string s[] = { "Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday" };
        return s[t2->tm_wday];
    }
};
