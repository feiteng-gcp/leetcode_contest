class Solution {
public:
    bool isLeap(int x)
    {
        return x % 4 == 0 && (x % 400 == 0 || x % 100 != 0);
    }
    

    string dayOfTheWeek(int day, int month, int year) {
        int d = 4;

        for (int i = 1971; i < year; ++i)
        {
            if (isLeap(i))
            {
                d += 366;
            }
            else
            {
                d += 365;
            }
        }
        
        for (int i = 1; i < month; ++i)
        {
            if (i == 4 || i == 6 || i == 9 || i == 11)
            {
                d += 30;
            }
            else if (i == 2)
            {

                if (isLeap(year))
                {
                    d += 29;
                }
                else
                {
                    d += 28;
                }
            }
            else
            {
                d += 31;
            }
        }
        
        d += day;
        
        const string ret[] = {"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"};
        
        return ret[d % 7];
    }
};
