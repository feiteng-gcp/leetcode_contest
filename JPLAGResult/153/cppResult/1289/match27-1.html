class Solution {
public:
    string dayOfTheWeek(int day, int month, int year) {
        int a=day-1;
        while(month>1){
            --month;
            if(month==1 or month==3 or  month==5 or month==7 or month==8 or
               month==10)a+=31;
            if(month==4 or month==6 or  month==9 or month==11)a+=30;
            if(month==2 and (year%4==0 and (year%100!=0 or year%400==0)))a+=29;
            if(month==2 and (year%4!=0 or (year %100==0 and year%400!=0)))a+=28;
        }
        while(year>1971){
            --year;
            if((year%4==0 and (year%100!=0 or year%400==0)))a+=366;
            if((year%4!=0 or (year %100==0 and year%400!=0)))a+=365;
        }

        a=a%7;
        switch(a){
            case 0:
                return "Friday";
            case 1:
                return "Saturday";
            case 2:
                return "Sunday";
            case 3:
                return "Monday";
            case 4:
                return "Tuesday";
            case 5:
                return "Wednesday";
            case 6:
                return "Thursday";
        }
        return " ";
    }
};
