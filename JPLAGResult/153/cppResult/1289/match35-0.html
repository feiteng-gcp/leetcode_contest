template <typename T1, typename T2>
vector<T2> accum(vector<T1>& a, T2 init) {
    int L = a.size();
    vector<T2> acc(L + 1);
    acc[0] = init;
    for (int i=0; i<L; ++i) acc[i+1] = acc[i] + a[i];
    // partial_sum(ALL(a), acc.begin()+1);
#ifdef DEBUG
    cerr << a << " -> accum -> " << acc << endl;
#endif
    return move(acc);
}

class Solution {
public:
    string dayOfTheWeek(int day, int month, int year) {
        vi _days {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        vi days = accum(_days, 0);
        vs yob {"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"};

        int ofs = 6 + 365*(year - 1972) + days[month-1] + (day-1);
        ofs += (year+3 - 1972)/4;
//        if (year == 2100) --ofs;
        if (year % 4 == 0 && year != 2100 && month >= 3) ++ofs;
        return yob[ofs % 7];
    }
};
