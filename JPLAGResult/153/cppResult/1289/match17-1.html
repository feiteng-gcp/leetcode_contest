class Solution {
public:
    std::unordered_map<int, string> week = {
        {0, "Sunday"},
        {1, "Monday"},
        {2, "Tuesday"},
        {3, "Wednesday"},
        {4, "Thursday"},
        {5, "Friday"},
        {6, "Saturday"},

    };
    string dayOfTheWeek(int day, int month, int year) {
        if (month == 1 || month == 2)
        {
            month += 12;
            --year;
        }
        return week[(day + 1 + 2 * month + 3 * (month + 1) / 5 + year + year / 4 - year / 100 + year / 400) % 7];
    }
};
