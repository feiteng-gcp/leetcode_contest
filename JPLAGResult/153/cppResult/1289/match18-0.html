const int monthDays[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
const string weekDay[] = {"Friday", "Saturday", "Sunday", "Monday", "Tuesday", "Wednesday", "Thursday"};

class Solution {
public:
    struct Date {
        int day, month, year;
        Date(int _day, int _month, int _year) {
            day = _day;
            month = _month;
            year = _year;
        }
        
        bool leapYear(int i) {
            return (i % 4 == 0 && i % 100 != 0 || i % 400 == 0);
        }
        
        int diffDays(Date other) {
            int sum = 0;
            for(int i = year; i < other.year; ++i) {
                if (leapYear(i)) {
                    sum += 366;
                } else {
                    sum += 365;
                }
            }
            for(int i = 1; i < other.month; ++i) {
                if(i == 2) {
                    if(leapYear(other.year)) {
                        sum += 29;
                    } else {
                        sum += 28;
                    }
                } else {
                    sum += monthDays[i];
                }
            }
            sum += other.day - 1;
            return sum;
        }
    };
    string dayOfTheWeek(int day, int month, int year) {
        Date firstDay = Date(1, 1, 1971); // friday
        Date searchDay = Date(day, month, year);
        return weekDay[firstDay.diffDays(searchDay) % 7];
    }
};
