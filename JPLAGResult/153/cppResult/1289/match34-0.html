class Solution {
public:
    string dayOfTheWeek(int d, int m, int y) {
         static int t[] = { 0, 3, 2, 5, 0, 3, 
                       5, 1, 4, 6, 2, 4 };  
    y -= m < 3;  
    int x= ( y + y / 4 - y / 100 + y / 400 + t[m - 1] + d) % 7; 
        if(x==1)
            return "Monday";
        if(x==2)
            return "Tuesday";
        if(x==3)
            return "Wednesday";
        if(x==4)
            return "Thursday";
        if(x==5)
            return "Friday";
        if(x==6)
            return "Saturday";
        if(x==0)
            return "Sunday";
        return "";
    }
};
