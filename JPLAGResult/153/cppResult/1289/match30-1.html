class Solution {
public:
	string dayOfTheWeek(int day, int month, int year) {
		if (1 == month || 2 == month){
			month += 12;
			year--;
		}
		int ret = (day + 2 * month + 3 * (month + 1) / 5 + year + year / 4 - year / 100 + year / 400) % 7;
		vector<string> week{  "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday","Sunday" };
		return week[ret];
	}
};
