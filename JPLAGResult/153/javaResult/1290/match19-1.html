    int f(int i, int last) {
        if (i == 0) {
            if (arr1[i] <= last) {
                return 0;
            } else if (arr2[0] <= last) {
                return 1;
            } else {
                return 1 << 29;
            }
        }
        long key = last * 10000L + i;
        if (f.containsKey(key)) {
            return f.get(key);
        
}

        int ans = 1 << 29;
        if (arr1[i] <= last) {
            ans = Math.min(ans, f(i - 1, arr1[i] - 1));
        }
        Integer hehe = tree.floor(last);
        if (hehe != null) {
            ans = Math.min(ans, 1 + f(i - 1, hehe - 1));
        }

        f.put(key, ans);
