import java.util.Map.Entry;
import java.util.TreeMap;
import java.util.TreeSet;

class Solution {
	public int makeArrayIncreasing(int[] arr1, int[] arr2) {
		TreeSet<Integer> s = new TreeSet<Integer>();
		for (int i : arr2)
			s.add(i);
		TreeMap<Integer, Integer> cur = new TreeMap<Integer, Integer>();
		cur.put(Integer.MIN_VALUE, 0);
		for (int i = 0; i < arr1.length; i++) {
			TreeMap<Integer, Integer> m2 = new TreeMap<Integer, Integer>();
			for (Entry<Integer, Integer> en : cur.entrySet()) {
				Integer next = s.ceiling(en.getKey() + 1);
				if (next != null) {
					m2.put(next, en.getValue() + 1);
				}
			}
			Entry<Integer, Integer> en = cur.floorEntry(arr1[i] - 1);
			if (en != null) {
				Entry<Integer, Integer> en2 = null;
				while ((en2 = m2.ceilingEntry(arr1[i])) != null) {
					if (en2.getValue() >= en.getValue())
						m2.remove(en2.getKey());
					else
						break;
				}
				if (!m2.containsKey(arr1[i]))
					m2.put(arr1[i], en.getValue());
			}
			cur = m2;
//			Helper.tr(cur);
		}
		if (cur.isEmpty())
			return -1;
		int ans = Integer.MAX_VALUE;
		for (int i : cur.values())
			ans = Math.min(ans, i);
		return ans;

	}
}
