import java.util.HashMap;
import java.util.Map;
import java.util.TreeSet;

class Solution {
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
        TreeSet<Integer> set = new TreeSet<>();
        for (int i : arr2) {
            set.add(i);
        }

        Map<Integer, Integer> map = new HashMap<>();
        map.put(set.first(), 1);
        map.put(arr1[0], 0);

        for (int i = 1; i < arr1.length; i++) {
            Map<Integer, Integer> next = new HashMap<>();
            int curr = arr1[i];
            for (Map.Entry<Integer, Integer> e : map.entrySet()) {
                int last = e.getKey(), times = e.getValue();
                Integer tar = set.higher(last);
                if (tar != null)
                    next.put(tar, Math.min(next.getOrDefault(tar, 2001), times + 1));
                if (curr > last)
                    next.put(curr, Math.min(next.getOrDefault(curr, 2001), times));
            }
            map = next;
        }

        if (map.size() == 0) return -1;
        int ans = 2009;
        for (Map.Entry<Integer, Integer> e : map.entrySet()) {
            ans = Math.min(e.getValue(), ans);
        }
        return ans;
    }
}
