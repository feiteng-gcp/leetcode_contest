class Solution {
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
        int[] distinct = Arrays.stream(arr2)
        .sorted()
        .distinct()
        .toArray();
        //if(distinct.length > arr1.length) return 0;
        
        int[][] ans = new int[2008][2008];
        for(int i=1;i<=distinct.length;i++) {
            ans[0][i] = 1;
        }
        
        for(int i=1;i<arr1.length;i++) {
            for(int j=1;j<distinct.length;j++) {
                ans[i][j+1] = ans[i-1][j] + 1;
                if(arr1[i-1] < distinct[j]) {
                    int temp = ans[i-1][0] + 1;
                    if(temp < ans[i][j+1]) ans[i][j+1] = temp;
                }
            }
            
            if(arr1[i-1] < distinct[0]) {
                int temp = ans[i-1][0];
                ans[i][1] = temp + 1;    
            } else {
                ans[i][1] = 99999999;
            }
            
            ans[i][0] = 99999999;
            for(int j=0;j<distinct.length;j++) {
                if(distinct[j] < arr1[i]) {
                    int temp = ans[i-1][j+1];
                    if(temp < ans[i][0]) {
                        ans[i][0] = temp;
                    }
                }
            }
            if(arr1[i-1] < arr1[i]) {
                int temp = ans[i-1][0];
                if(temp < ans[i][0]) ans[i][0] = temp;
            }
            
            
        }
        
        //System.out.println(Arrays.toString(distinctOrdered));
        int min = Integer.MAX_VALUE;
        
for(int i=0;i<=distinct.length;i++) {
           if(ans[arr1.length-1][i] < min) min = ans[arr1.length-1][i];
        }
        for(int i=0;i<arr1.length;i++) {
            //System.out.println("i = " + i + ": " + Arrays.toString(ans[i]));
        }
        
        if(min > arr1.length) return -1;
        return min;
    }
}
