class Solution {
    TreeSet<Integer> set;
    List<Map<Integer,Integer>> dp;
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
        set = new TreeSet<>();
        for(int i : arr2) set.add(i);
        dp = new ArrayList<>();
        for(int i=0;i<arr1.length;i++)
            dp.add(new HashMap<>());
        int res = solve(0,Integer.MIN_VALUE,arr1);
        if(res == Integer.MAX_VALUE) return -1;
        return res;
    }
    public int solve(int curIdx,int prevVal,int[] arr1){
        if(curIdx == arr1.length) return 0;
        if(dp.get(curIdx).containsKey(prevVal))
            return dp.get(curIdx).get(prevVal);
        int best = Integer.MAX_VALUE;
        
        // not replace if 
        if(arr1[curIdx] > prevVal){
            best = Math.min(best,solve(curIdx+1, arr1[curIdx], arr1));
        }
        // replace
        Integer nextHigher = set.higher(prevVal);
        if(nextHigher != null ){
            int cost = solve(curIdx+1, nextHigher, arr1);
            if(cost != Integer.MAX_VALUE){
                best = Math.min(best, cost+1);
            }
        }
        dp.get(curIdx).put(prevVal, best);
        return best;
    }
}
