class Solution {
int[][] dp;
    int[] arr1;
    int[] arr2;
    TreeMap<Integer, Integer> map;
    int n1, n2;
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
        n1 = arr1.length;
        n2 = arr2.length;
        this.arr1 = arr1;
        this.arr2 = arr2;
        map = new TreeMap<>();
        for (int i = 0; i < n2; i++) {
            map.put(arr2[i], i);
        }

        boolean ok = true;
        for (int i = 1; i < n1; i++) {
            if (arr1[i] <= arr1[i-1]) ok = false;
        }
        if (ok) return 0;

        dp = new int[n2+1][n1];
        for (int i = 0; i <=n2;i++) {
            Arrays.fill(dp[i], -1);
        }

        int res = n1+1;
        for (int i = 0; i < n2; i++) {
            res = Math.min(res, solve(i, 1) + 1);
        }
        res = Math.min(res, solve(n2, 1));

        if (res == n1+1) return -1;
        return res;
    }

    int solve(int pre, int idx) {
        if (idx == n1) return 0;
        if (dp[pre][idx] != -1) return dp[pre][idx];
        int last = 0;
        if (pre == n2) last = arr1[idx-1];
        else last = arr2[pre];

        int res = n1+1;
        if (arr1[idx] > last) {
            res = solve(n2, idx+1);
        }
        Map.Entry<Integer, Integer> nxt = map.higherEntry(last);
        if (nxt != null) {
            res = Math.min(res, solve(nxt.getValue(), idx+1) + 1);
        }
        dp[pre][idx] = res;
        return res;
    }
}
