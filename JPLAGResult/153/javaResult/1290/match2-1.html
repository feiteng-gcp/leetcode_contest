class Solution {
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
        Set<Integer> set = new HashSet<>();
        for (int a: arr2) {
            set.add(a);
        }
        int[] a2 = new int[set.size()];
        Iterator<Integer> ite = set.iterator();
        for (int i = 0; i < a2.length; i++) {
            a2[i] = ite.next();
        }
        Arrays.sort(a2);
        int n = arr1.length;
        int m = a2.length;
        int[][] dp = new int[n][m+1];
        for (int i = 0; i < n; i++) Arrays.fill(dp[i], Integer.MAX_VALUE);
        for (int i = 0; i < m; i++) {
            dp[0][i] = 1;
        }
        dp[0][m] = 0;
        
        for (int i = 1; i < n; i++) {
            int curMin = Integer.MAX_VALUE;
            for (int j = 0; j < m; j++) {
                if (j > 0) {
                    curMin = Math.min(curMin, dp[i-1][j-1]);
                    if (a2[j] > arr1[i-1]) {
                        if (dp[i-1][m] != Integer.MAX_VALUE) {
                            dp[i][j] = dp[i-1][m] + 1;
                        }
                    }
                    if (curMin != Integer.MAX_VALUE) {
                        dp[i][j] = Math.min(curMin + 1, dp[i][j]);
                    }
                }
            }
            // dp[i][m]
            if (arr1[i] > arr1[i-1]) {
                dp[i][m] = dp[i-1][m];
            }
            for (int j = 0; j < m; j++) {
                if (a2[j] < arr1[i]) {
                    dp[i][m] = Math.min(dp[i][m], dp[i-1][j]);
                }
            }
        }
        
        /*for (int i = 0; i < dp.length; i++) {
            for (int j = 0; j < dp[i].length; j++) {
                System.out.print(dp[i][j] + " ");
            }
            System.out.println();
        }*/
        
        int min = Integer.MAX_VALUE;
        for (int i = 0; i < m + 1; i++) {
            min = Math.min(min, dp[n-1][i]);
        }
        
        if (min == Integer.MAX_VALUE) return -1;
        return min;
    }
}
