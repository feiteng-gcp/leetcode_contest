class Solution {
    int dp[][] = new int[2001][2003];
    
    public int makeArrayIncreasing(int[] a, int[] b) {
        Arrays.sort(b);
        int s = go(a,b,0,0);
        if(s>=10000000){
            return -1;
        }
        return s;
    }
    
    public int bigger(int b[],int k){
        int lo = 0;
        int hi = b.length;
        while(lo<hi){
            int m = (lo+hi)>>1;
            if(b[m]>k){
                hi = m;
            }else{
                lo = m+1;
            }
        }
        return lo;
        
    }
    
    
    public int go(int a[],int b[],int i,int j){
        if(i==a.length){
            return 0;
        }
        if(dp[i][j]!=0){
            return dp[i][j];
        }
        
        
        if(i==0){
            
            int r  = go(a,b,i+1,2002);
            int t = a[i];
            a[i] = b[0];
            r = Math.min(r,1 + go(a,b,i+1,0));
            a[i] = t;
            return dp[i][j] = r;
            
        }
        
        
        
        
        if(a[i]>a[i-1]){
            int r  = go(a,b,i+1,2002);
            int idx = bigger(b,a[i-1]);
            if(idx<b.length){
                int t = a[i];
                a[i] = b[idx];
                r = Math.min(r,1 + go(a,b,i+1,idx));
                a[i] = t;
            }
            return dp[i][j] = r;
        }else{
         
            int idx = bigger(b,a[i-1]);
            if(idx<b.length){
                int t = a[i];
                a[i] = b[idx];
                int r =1 + go(a,b,i+1,idx);
                a[i] = t;
                return dp[i][j] = r;
            }else{
                return dp[i][j] = 10000000;
            }
           
            
            
        }
        
        
        
        
    }
}
