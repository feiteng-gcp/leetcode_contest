class Solution {
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
    	int len = arr1.length;
        TreeSet<Integer> set = new TreeSet<>();
        for(int v: arr2) {
        	set.add(v);
        }
        int[][] dp = new int[len+1][len];
        dp[0][0] = arr1[0];
        boolean x = false;
        for(int i = 1 ; i < len ; i++) {
        	if(arr1[i] <= arr1[i-1]) {
        		x = true;
        	}
        	if(x) {

        		dp[0][i] = Integer.MAX_VALUE;
        	}else {
        		dp[0][i] = arr1[i];
        	}
        }
        if(dp[0][len-1] != Integer.MAX_VALUE) {
        	return 0;
        }
        for(int r = 1; r <= len; r++) {
        	dp[r][0] = set.first();
        	
        	for(int i = 1 ; i < len ; i++) {
        		int best = Integer.MAX_VALUE;
        		int v = arr1[i];
        		if(v > dp[r][i-1]) {
        			best = Math.min(best,  v);
        		}

        		Integer c = set.higher(dp[r-1][i-1]);
        		if(c != null) {
        			best = Math.min(best, c);
        		}
        		
        		dp[r][i] = best;
        	}
        	if(dp[r][len-1] != Integer.MAX_VALUE) {
        		return r;
        	}
        	
        }
        return -1;
        
    }
}
