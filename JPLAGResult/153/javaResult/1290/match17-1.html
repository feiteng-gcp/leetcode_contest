class Solution {
    int input[];
    TreeSet<Integer> set;
    int n=0;
    int dp[][];
    HashMap<Integer,Integer> mapping;
    
    int dfs(int i,int prev){
        if(i==n){
            return 0;
        }
        if(dp[i][prev]!=-1){
            return dp[i][prev];
        }
        int ans=Integer.MAX_VALUE/3;
        
int cc=mapping.get(input[i]);
        if(cc<=prev&&i!=0){
            Integer aa=set.higher(prev);
            if(aa!=null){
                ans=1+dfs(i+1,aa);
            }
        }
        else{
            ans=dfs(i+1,cc);
            Integer aa=set.higher(prev);
            if(i==0){
                aa=set.higher(-1);
            }
            if(aa!=null)
                ans=Math.min(ans,1+dfs(i+1,aa));
        }
        dp[i][prev]=ans;
        return dp[i][prev];
    }
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
        input=arr1;
        n=input.length;
        TreeSet<Integer> st=new TreeSet<>();
        for(int u:arr2){
            st.add(u);
        }
        for(int u:arr1){
            st.add(u);
        }
        mapping=new HashMap<>();
        int cnt=0;
        for(int u:st){
            mapping.put(u,cnt);
            cnt++;
        }
        set=new TreeSet<>();
        for(int u:arr2){
            set.add(mapping.get(u));
        }
        dp=new int[n][4001];
        for(int ar[]:dp)
                Arrays.fill(ar,-1)
;
        int ans=dfs(0,0);
        if(ans>=Integer.MAX_VALUE/5)return -1;
        return ans;
    }
}
