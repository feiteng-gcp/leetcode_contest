class Solution {
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
        int len1 = arr1.length, len2 = arr2.length;
        Arrays.sort(arr2);
        TreeMap<Integer, Integer> map = new TreeMap<>();
        map.put(Integer.MIN_VALUE, 0);
        for (int i = 0; i < len1; ++i){
            if (map.isEmpty()){
                return -1;
            }
            TreeMap<Integer, Integer> next = new TreeMap<>();
            Integer floor = map.lowerKey(arr1[i]);
            boolean isContainArr1 = false;
            int cc = -1;
            if (floor != null){
                isContainArr1 = true;
                next.put(arr1[i], map.get(floor));
                cc = map.get(floor);
            }
            for (int j = 0; j < len2; ++j){
                if (isContainArr1 && arr2[j] >= arr1[i]){
                    floor = map.lowerKey(arr2[j]);
                    int t = map.get(floor) + 1;
                    if (t <= cc){
                        next.put(arr2[j], t);
                    }
                } else {
                    floor = map.lowerKey(arr2[j]);
                    if (floor != null){
                        next.put(arr2[j], map.get(floor) + 1);
                    }
                }
            }
            map = next;
        }
        if (map.isEmpty()){
            return -1;
        
}
        int min = Integer.MAX_VALUE;
        for (int i : map.keySet()){
            min = Math.min(map.get(i), min);
        }
        return min == Integer.MAX_VALUE ? -1 : min;
    }
}
