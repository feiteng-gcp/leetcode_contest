class Solution {
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
        
        TreeSet<Integer> set = new TreeSet<Integer>();
        for(int i = 0; i < arr2.length; i++)
            set.add(arr2[i]);
        int[] arr3 = new int[set.size()];
        int k = 0;
        for(Integer val : set)
            arr3[k++] = val;
        
        int[][] dp =  new int[arr1.length][arr3.length];
        int[] dp0 = new int[arr1.length];
        
        for(int j = 0; j < arr3.length; j++)
        {
            if(arr3[j] != arr1[0])
                dp[0][j] = 1;
        }
        
        
        for(int i = 1; i < arr1.length; i++)
        {
            dp0[i] = (arr1[i] > arr1[i-1] ? dp0[i-1] : -1); 
            for(int j = 0; j < arr3.length; j++)
            {
                dp[i][j] = -1;
                if(j > 0 && dp[i-1][j-1] != -1)
                {
                    int cnt = dp[i-1][j-1] + (arr1[i] == arr3[j] ? 0 : 1);
                    dp[i][j] = (dp[i][j] == -1) ? cnt : Math.min(dp[i][j], cnt);
                }
                
                if(arr3[j] > arr1[i-1] && dp0[i-1] != -1)
                {
                    int cnt = dp0[i-1] + (arr1[i] == arr3[j] ? 0 : 1);
                    dp[i][j] = (dp[i][j] == -1) ? cnt : Math.min(dp[i][j], cnt);
                }
                
                if(arr1[i] > arr3[j] && dp[i-1][j] != -1)
                {
                    dp0[i]= (dp0[i] == -1) ? dp[i-1][j] : Math.min(dp0[i], dp[i-1][j]);
                }
            }
        }
        
        int res = dp0[arr1.length-1];
        for(int j = 0; j < arr3.length; j++)
        {
            if(dp[arr1.length-1][j] != -1)
            {
                res = (res == -1 ? dp[arr1.length-1][j] : Math.min(res, dp[arr1.length-1][j]));
            }
        }
        
        return res;
    }
}
