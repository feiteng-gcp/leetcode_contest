class Solution {
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
        Arrays.sort(arr2);
        List<Integer> b = new ArrayList<>();
        for (int v : arr2) {
            if (b.size() == 0 || v != b.get(b.size() - 1)) 
                b.add(v);
        }
        
        int n = arr1.length, m = b.size();
        int[][] dp = new int[n][m];
        int[] f = new int[n];
        int[] his = new int[n];
        for (int  i = 0; i < n; i++) {
            his[i] = bsearch(b, arr1[i]);
        }
        final int INF = 1000000;
        f[0] = 0;
        for (int i = 0; i < m; i++) dp[0][i] = 1;
        
        for (int i = 1; i < n; i++) {
            f[i] = INF;
            if (his[i] >= 0) f[i] = dp[i-1][his[i]];
            if (arr1[i-1] < arr1[i]) f[i] = Math.min(f[i], f[i-1]);
            
            for (int j = 0; j < m; j++) {
                dp[i][j] = INF;
                
                if (arr1[i-1] < b.get(j)) dp[i][j] = 1 + f[i-1];
                
                if (j > 0) 
                    dp[i][j] = Math.min(dp[i][j], 1 + dp[i-1][j-1]);
            }
            
        }
        
        
        int ans = Math.min(f[n-1], dp[n-1][m-1]);
        if (ans < INF) return ans;
        return -1;
    }
    
    private int bsearch(List<Integer> b, int v) {
        int n = b.size();
        if (v <= b.get(0)) return -1;
        if (v > b.get(n-1)) return n-1;
        
        int lo = 0, hi = n - 1;
        while (lo <= hi) {
            int mid = (lo + hi) / 2;
            if (b.get(mid) < v) lo = mid + 1;
            else hi = mid - 1;
        }
        return hi;
    }
    
}
