class Solution {
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
        final int N = arr1.length;
        final int M = arr2.length;
        HashMap<Integer, Integer> ranks = makeRanks(arr1, arr2);
        final int K = ranks.size();
        int[] A = new int[N];
        for (int i = 0; i < N; ++i) {
            A[i] = ranks.get(arr1[i]);
        }
        int[] B = new int[M];
        for (int i = 0; i < M; ++i) {
            B[i] = ranks.get(arr2[i]);
        }
        // System.out.println(Arrays.toString(A));
        // System.out.println(Arrays.toString(B));
        
        int[][] dp = new int[N][K];
        for (int i = 0; i < N; ++i) {
            Arrays.fill(dp[i], N + 1);
        }
        for (int i = 0; i < N; ++i) {
            if (i == 0) {
                for (int v : B) {
                    dp[i][v] = 1;
                }
                dp[i][A[i]] = 0;
            } else {
                if (A[i] > 0) {
                    dp[i][A[i]] = Math.min(dp[i][A[i]], dp[i - 1][A[i] - 1]);
                }
                for (int v : B) {
                    if (v > 0) {
                        dp[i][v] = Math.min(dp[i][v], 1 + dp[i - 1][v - 1]);
                    }
                }
            }
            // System.out.format("dp[%d][%d] = %d\n", i, 0, dp[i][0]);
            for (int j = 1; j < K; ++j) {
                dp[i][j] = Math.min(dp[i][j], dp[i][j - 1]);
                // System.out.format("dp[%d][%d] = %d\n", i, j, dp[i][j]);
            }
        }
        int ans = dp[N - 1][K - 1];
        
if (ans > N) {
            return -1;
        }
        return ans;
    }
    
    private static HashMap<Integer, Integer> makeRanks(int[] a, int[] b) {
        HashSet<Integer> uniq = new HashSet<>();
        for (int v : a) {
            uniq.add(v);
        }
        for (int v : b) {
            uniq.add(v);
        }
        Integer[] arr = uniq.toArray(new Integer[0]);
        Arrays.sort(arr);
        HashMap<Integer, Integer> ranks = new HashMap<>();
        for (int i = 0; i < arr.length; ++i) {
            ranks.put(arr[i], i);
        }
        return ranks;
    }
}
