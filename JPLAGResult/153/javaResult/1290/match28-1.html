class Solution {
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
        Arrays.sort(arr2);
        Map<String, Integer> memo = new HashMap<>();
        return dfs(arr1, arr2, 0, memo);
    
}
    
    public int dfs(int[] arr1, int[] arr2, int index, Map<String, Integer> memo) {
        if (index == arr1.length) {
            return 0;
        }
        String key = index + " " + (index == 0 ? "zero" : arr1[index - 1]);
        if (memo.containsKey(key)) {
            return memo.get(key);
        }
        int count = Integer.MAX_VALUE;
        if (index == 0) {
            int rest = dfs(arr1, arr2, index + 1, memo);
            if (rest != -1) {
                count = Math.min(count, rest);
            }
            for (int i = 0; i < arr2.length; i++) {
                if (arr2[i] < arr1[index]) {
                    int old = arr1[index];
                    arr1[index] = arr2[i];
                    int right = dfs(arr1, arr2, index + 1, memo);
                    if (right != -1) {
                        count = Math.min(count, right + 1);
                    }
                    arr1[index] = old;
                    break;
                }
            }
        } else {
            if (arr1[index - 1] < arr1[index]) {
                int rest = dfs(arr1, arr2, index + 1, memo);
                if (rest != -1) {
                    count = Math.min(count, rest);
                }
            }
            for (int i = 0; i < arr2.length; i++) {
                if (arr2[i] > arr1[index - 1]) {
                    int old = arr1[index];
                    arr1[index] = arr2[i];
                    int right = dfs(arr1, arr2, index + 1, memo);
                    if (right != -1) {
                        count = Math.min(count, right + 1);
                    }
                    arr1[index] = old;
                    break;
                }
            }
        }
        int res = count == Integer.MAX_VALUE ? -1 : count;
        memo.put(key, res);
        return res;
    }
}
