class Solution {
	private void build(Map<Integer , Integer> dp) {
		list = new ArrayList<>(dp.keySet());
		Collections.sort(list);		
		for (int i = 0;i < list.size();i ++) {
			min[i] = dp.get(list.get(i));
			if (i > 0) {
				min[i] = Math.min(min[i] , min[i - 1]);	
			}
		}
	}
	
	private int[] min = new int[10000];
	private List<Integer> list;	
	
    public int makeArrayIncreasing(int[] arr1, int[] arr2) {
    	Arrays.sort(arr2);    	
    	Map<Integer , Integer> dp1 = new HashMap<>() , dp2 = new HashMap<>();
    	int i , j , n = arr1.length;    	
    	dp1.put(Integer.MIN_VALUE , 0);    	
    	for (i = 0;i < n;i ++) {
    		build(dp1);
    		// use current
    		int low = 0 , high = list.size();
    		while (low < high) {
    			int mid = (low + high) / 2;
    			if (list.get(mid) < arr1[i]) {
    				low = mid + 1;
    			} else {
    				high = mid;    				
    			}
    		}
    		if (high - 1 >= 0) {
    			int result = min[high - 1];    			
    			dp2.put(arr1[i] , result);
    		}    		
    		// use a new value
    		int k = 0;    		
    		for (j = 0;j < arr2.length;j ++) {
    			if (j == 0 || arr2[j] != arr2[j - 1]) {
    				while (k < list.size() && list.get(k) < arr2[j]) {
    					k ++;    					
    				}
    				if (k - 1 >= 0) {
    					int result = min[k - 1] + 1;
    					if (result < dp2.getOrDefault(arr2[j] , Integer.MAX_VALUE)) {
    						dp2.put(arr2[j] , result);    						
    					}
    				}
    			}
    		}
    		dp1 = dp2;
    		dp2 = new HashMap<>();   
    	}    	
    	int ans = - 1;
    	for (int value : dp1.values()) {
    		if (value < ans || ans < 0) {
    			ans = value;
    		}
    	}
    	return ans;
    }
}
