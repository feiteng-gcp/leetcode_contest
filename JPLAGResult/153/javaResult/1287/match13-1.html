class Solution {
    public int distanceBetweenBusStops(int[] distance, int start, int destination) {
        int des = 0;
        for(int i = start; i != destination; i = (i + 1)%distance.length){
            des += distance[i];
        }
        int sum = 0;
        
for(int i =0;i < distance.length; ++i){
            sum += distance[i];
        }
        return Math.min(des, sum - des);
    }
}
