class Solution {
    public int distanceBetweenBusStops(int[] distance, int start, int destination) {
        int len = distance.length;
        int dis1 = 0, sum = 0;
        for (int i = 0; i < len; ++i){
            if ((start + i) % len != destination){
                dis1 = dis1 + distance[(start + i) % len];
            } else {
                break;
            }
        }
        
for (int i = 0; i < len; ++i){
            sum += distance[i];
        }
        return Math.min(dis1, sum - dis1);
    }
}
