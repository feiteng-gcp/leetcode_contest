class Solution {
    public int distanceBetweenBusStops(int[] distance, int start, int destination) {
        int first = 0, seconde = 0;
        if (start == destination) {
            return 0;
        }
        //clockwise
        for (int i = start; i < distance.length * 2; i++) {
            int index = i % distance.length;
            if (index == destination) {
                break;
            }
            first += distance[index];
        }
        //counterclockwise
        for (int i = start - 1; i >= -distance.length * 2; i--) {
            int index = ((i % distance.length) + distance.length) % distance.length;
            seconde += distance[index];
            if (index == destination) {
                break;
            
}
        }
        return Math.min(first, seconde);
    }
}
