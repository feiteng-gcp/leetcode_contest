class Solution {
    public int distanceBetweenBusStops(int[] distance, int start, int destination) {
        int len = distance.length;
        long sum = 0;
        long res1 = 0;
        for (int i = 0; i < len; i++) {
            sum += distance[i];
        }
        if (start > destination) {
            int tmp = start;
            start = destination;
            destination = tmp;
        }
        for (int i = start; i < destination; i++) {
            res1 += distance[i];
        }
        return (int) Math.min(res1, sum - res1);
    }
}
