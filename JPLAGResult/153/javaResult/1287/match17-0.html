class Solution {
    public int distanceBetweenBusStops(int[] distance, int start, int destination) {
        int clock = 0, antiClock = 0;
        int small = start<destination ? start : destination;
        int big = start<destination ? destination : start;
        for(int i=small; i<big; i++){
            clock += distance[i];
        }
        int cur = big;
        while(cur!=small){
            antiClock += distance[cur];
            cur = (cur+1)%distance.length;
        }
        return Math.min(clock, antiClock);
    }
}
