class Solution {
    public int distanceBetweenBusStops(int[] distance, int start, int destination) {
        if (start > destination) {
            int tmp = start;
            start = destination;
            destination = tmp;
        }
        int d = 0;
        for (int i = start; i < destination; ++i) {
            d += distance[i];
        }

        int max = Arrays.stream(distance).sum();
        return Math.min(d, max - d);
    }
}
