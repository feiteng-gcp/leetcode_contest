class Solution {
    public int distanceBetweenBusStops(int[] distance, int s, int d) {
        if (d < s) {
            int temp = s;
            s = d;
            d = temp;
        }
         int n = distance.length;
        int t = 0;
        for (int i = s; i < d; ++i) {
            t += distance[i];
        }
        
        int t2 = 0;
        for (int i = d; i < s + n; ++i) {
            t2 += distance[i % n];
        }
        return Math.min(t, t2);
    }
}
