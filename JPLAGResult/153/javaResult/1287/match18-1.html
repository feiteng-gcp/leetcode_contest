class Solution {
    public int distanceBetweenBusStops(int[] distance, int start, int destination) {
        int ans=Integer.MAX_VALUE;
        int n=distance.length;
        int cur=start;
        int a1=0;
        while(cur!=destination){
            a1+=distance[cur];
            cur=(cur+1)%n;
        }
        cur=start;
        int a2=0;
        while(cur!=destination){
            int nx=(cur-1+n)%n;
            a2+=distance[nx];
            cur=nx;
        }
        return Math.min(a1,a2);
    }
}
