class Solution {
    public int distanceBetweenBusStops(int[] distance, int start, int destination) {
        int min = Integer.MAX_VALUE;
        int cur = 0;
        int id = start;
        int m = distance.length;
        while (id != destination) {
            cur += distance[id];
            id++;
            id = id % m;
        }
        min = Math.min(min, cur);
        id = destination;
        cur = 0;
        while (id != start) {
            cur += distance[id];
            id++;
            id = id % m;
        }
        min = Math.min(min, cur);
        return min;
    }
}
