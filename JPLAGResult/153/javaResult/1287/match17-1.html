class Solution {
    public int distanceBetweenBusStops(int[] distance, int start, int destination) {
        int clockwise = 0;
        int counterClock = 0;
        int n = distance.length;        
        int tmp = start;
        while (tmp != destination) {
            clockwise += distance[tmp];
            tmp = (tmp + 1) % n;
        }
        tmp = destination;
        
while (tmp != start) {
            counterClock += distance[tmp];
            tmp = (tmp + 1) % n;
        }
        return Math.min(clockwise, counterClock);
    }
}
