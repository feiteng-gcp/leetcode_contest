class Solution {
    public int distanceBetweenBusStops(int[] d, int s, int des) {
        int len = 0;
        int sum = 0;
        for (int i = 0; i < d.length; i++) {
            sum += d[i];
        }
        int min = Math.min(s, des);        int max = Math.max(s, des);

        for (int i = min; i < max; i++) {
            len += d[i];
        }


        return len < sum - len ? len : sum - len;
    }
}
