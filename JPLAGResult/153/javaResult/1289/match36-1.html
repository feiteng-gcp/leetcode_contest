class Solution {
    public String dayOfTheWeek(int day, int month, int year) {
        // Jan 1 1970 : Thursday
        String days[] = {"Thursday", "Friday", "Saturday","Sunday", "Monday", "Tuesday", "Wednesday"};
        int[] months = {31,28,31,30,31,30,31,31,30,31,30,31};
        int y = year - 1970;
        int m = month;
        int d = day-1;
        
        for (int i = 1970; i < year; i++) {
            if (isLeap(i)) {
                d++;
            }
            d += 365;
        }
        if (m > 2 && isLeap(year))
            d++;
        for (int i = 0; i < m-1; i++) {
            d += months[i];
        }
      //  System.out.println(d);
        return days[d%7];
    }
    
    public boolean isLeap(int y) {
        if (y%400 == 0)
            return true;
        if (y%100 == 0)
            return false;
        return (y%4 == 0);
    }
}
