class Solution {
    public String dayOfTheWeek(int d, int m, int y) {
        int t[] = { 0, 3, 2, 5, 0, 3, 5, 1, 4, 6, 2, 4 }; 
        String[] dict = new String[]{"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"};
        y -= (m < 3) ? 1 : 0; 
        int tmp = ( y + y/4 - y/100 + y/400 + t[m-1] + d) % 7;
        return dict[tmp];
    }
}
