class Solution {
    public String dayOfTheWeek(int day, int month, int year) {
        int[] m = new int[] {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        String[] days = new String[] {"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"};
                        
        // 2019-1-1 Tuesday
        int start = Math.min(year, 2019);
        int end = Math.max(year, 2019);

        int total = 0;
        for (int i=start; i<end; i++) total += isLeap(i) ? 366 : 365;

        int p = 0;
        for (int i=1; i<month; i++) p += m[i-1];
        if (isLeap(year) && month > 2) p++;
        p += day;

        //System.out.println(total + " " + p + " " + (total - p));
        
        if (year < 2019) {
            total -= p;
            return days[(8 - total % 7) % 7];
        }
        else {
            total += p;
            return days[(total + 1) % 7];
        }
    }
    
    private boolean isLeap(int year) {
        return year % 400 == 0 || year % 100 != 0 && year % 4 == 0;
    }
}
