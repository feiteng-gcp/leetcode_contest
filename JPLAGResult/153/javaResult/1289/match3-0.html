class Solution {
    public String dayOfTheWeek(int d, int m, int y) {
          // https://www.hackerearth.com/blog/developers/how-to-find-the-day-of-a-week/
        
        String[] names = new String[]{"Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"};
        int[] t = new int[]{0, 3, 2, 5, 0, 3, 5, 1, 4, 6, 2, 4};
        y = m < 3 ? y - 1 : y;
        int index =  (y + y/4 - y/100 + y/400 + t[m-1] + d) % 7;
        return names[index];
    }
}
