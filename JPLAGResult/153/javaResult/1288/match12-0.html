class Solution {
    public int maximumSum(int[] arr) {
        
        int n = arr.length;
        int dp1[] = new int[n];
        int dp2[] = new int[n];
        int curm = arr[0];
        int tm = arr[0];
        dp1[0] = arr[0];
        for (int i = 1; i < n; i++) {
            curm = Math.max(arr[i], curm + arr[i]);
            tm = Math.max(tm, curm);
            dp1[i] = curm;
        }
        curm = arr[n-1];
        tm = arr[n-1];
        dp2[n-1] = arr[n-1];
        for (int i = n - 2; i >= 0; i--) {
            curm = Math.max(arr[i], curm + arr[i]);
            tm = Math.max(tm, curm);
            dp2[i] = curm;
        }
        int ans = tm;
        for (int i = 1; i < n - 1; i++)
            ans = Math.max(ans, dp1[i - 1] + dp2[i + 1]);
        return ans;
    
    }
}
