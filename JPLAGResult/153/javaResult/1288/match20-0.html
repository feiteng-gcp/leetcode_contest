class Solution {
    public int maximumSum(int[] arr) {
        
        int n = arr.length;
        
        int pos_dir[] = new int[n]; 
        int neg_dir[] = new int[n]; 
  
        int curr = arr[0];
        int maximum = arr[0]; 
  
        pos_dir[0] = arr[0]; 
  
        for (int i = 1; i < n; i ++) { 
  
            curr = Math.max(arr[i], curr + arr[i]); 
            maximum = Math.max(maximum, curr); 
            pos_dir[i] = curr; 
        } 
  
        curr = maximum = neg_dir[n - 1] = arr[n - 1]; 
          
        for (int i = n - 2; i >= 0; i--) { 
  
            curr = Math.max(arr[i], curr + arr[i]); 
            maximum = Math.max(maximum, curr); 
            neg_dir[i] = curr; 
        } 
  
        int flag = maximum; 
  
        for (int i = 1; i < n - 1; i++) 
            flag = Math.max(flag, pos_dir[i - 1] + neg_dir[i + 1]); 
  
        return flag;
    }
}
