class Solution {
    public int maximumSum(int[] arr) {
        int n = arr.length;
        int left[] = new int[n];
        int right[] = new int[n];
        int c = arr[0], max = arr[0];
        left[0] = arr[0];
        for (int i = 1; i < n; i++) {
            c = Math.max(arr[i], c + arr[i]);
            max = Math.max(max, c);
            left[i] = c;
        }
        c = max = right[n - 1] = arr[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            c = Math.max(arr[i], c + arr[i]);
            max = Math.max(max, c);
            right[i] = c;
        }
        int ret = max;
        for (int i = 1; i < n - 1; i++) ret = Math.max(ret, left[i - 1] + right[i + 1]);
        return ret;
    }
}
