class Solution {
    public int maximumSum(int[] arr) {
        int n = arr.length;
        int[] fw = new int[n];
        int[] bw = new int[n];
        int curmax = arr[0];
        int maxnow = arr[0];
        fw[0] = arr[0];
        for (int i = 1; i < n; i++) {
            curmax = Math.max(arr[i], curmax + arr[i]);
            maxnow = Math.max(maxnow, curmax);
            fw[i] = curmax;
        }
        curmax = maxnow = bw[n - 1] = arr[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            curmax = Math.max(arr[i], curmax + arr[i]);
            maxnow = Math.max(maxnow, curmax);
            bw[i] = curmax;
        }
        int f = maxnow;
        for (int i = 1; i < n - 1; i++) {
            f = Math.max(f, fw[i - 1] + bw[i + 1]);
        }
        return f;
    }
}
