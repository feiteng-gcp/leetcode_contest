class Solution {
    public int[] processQueries(int[] queries, int m) {
        List<Integer> vals = new ArrayList<>();
        for (int i = 1; i <= m; i++) vals.add(i);
        int[] res = new int[queries.length];
        
        for (int i = 0; i < queries.length; i++) {
            int q = queries[i];
            
res[i] = vals.indexOf(q);
            vals.remove(res[i]);
            vals.add(0, q);
        }
        return res;
    }
}
