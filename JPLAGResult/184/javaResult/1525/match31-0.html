class Solution {
    public int[] processQueries(int[] queries, int m) {
        int[] res = new int[queries.length];
        List<Integer> ls = new ArrayList<>();
        for (int i = 0; i < m; i++) {
            ls.add(i + 1);
        }
        for (int i = 0; i < queries.length; i++) {
            res[i] = findNext(queries[i], ls);
        }
        return res;
    }

    int findNext(int target, List<Integer> ls) {
        int idx = -1;
        for (int i = 0; i < ls.size(); i++) {
            if (ls.get(i).equals(target)) {
                idx = i;
                break;
            }
        }
        ls.remove(idx);
        ls.add(0, target);
        return idx;
    }
}
