class Solution {
    public int[] processQueries(int[] queries, int m) {
int[] result = new int[queries.length];
		int[] arr = new int[m];
		for (int i = 0; i < m; ++i) {
			arr[i] = i+1;
		}
		for (int i = 0; i < queries.length; ++i) {
			int index = -1;
			for (int j = 0; j < m; ++j) {
				if (arr[j] == queries[i]) {
					index = j;
					break;
				}
			}
			result[i] = index;
			int tmp = arr[index];
			for (int j = index - 1; j >= 0; --j) {
				arr[j+1] = arr[j];
			}
			arr[0] = tmp;
		}
		return result;
    }
}
