class Solution {
    public int[] processQueries(int[] queries, int m) {
        int N = queries.length;
        
int[] ret = new int[N];
        int[] t = new int[m];
        for(int i = 0; i < m; i++){
            t[i] = i + 1;
        }
        for(int i = 0; i < N; i++){
            int j = 0;
            while(queries[i] != t[j]) j++;
            
ret[i] = j;
            int tmp = t[j];
            for(int k = j; k > 0; k--){
                t[k] = t[k-1];
            }
            t[0] = tmp;
        }
        return ret;
    }
}
