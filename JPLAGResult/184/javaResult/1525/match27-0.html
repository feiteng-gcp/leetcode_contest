class Solution {
    public int[] processQueries(int[] queries, int m) {
        List<Integer> l = new ArrayList<Integer>();
        for (int i=0; i<m; ++i) {
            l.add(i+1);
        }
        
        int[] res = new int[queries.length];
        for (int i=0; i<queries.length; ++i) {
            for(int j=0; j<m; ++j) {
                if (l.get(j) == queries[i]) {
                    l.remove(j);
                    l.add(0, queries[i]);
                    res[i] = j;
                    break;
                }
            }
        }
        return res;
        
    }
}
