class Solution {
    public int[] processQueries(int[] queries, int m) {
        int[] M = new int[m];
        int[] ans = new int[queries.length];
        for(int i = 1; i <= m; i++) {
        	M[i-1] = i;
        }
        
        for(int i = 0; i < queries.length; i++) {
        	
        	for(int
 j = 0; j < M.length; j++) {
        		if(M[j] == queries[i]) {
        			ans[i] = j;
        			break;
        		}
        	}
        	
        	int[] newM = new int[m];
        	int idx = 1;
        	newM[0] = M[ans[i]];
        	for(int j = 0; j < M.length; j++) {
        		if(j == ans[i]) continue;
        		newM[idx] = M[
j];
        		idx++;
        	}
        	M = newM;
        }
        //System.out.println(Arrays.toString(ans));
        return ans;
    }
}
