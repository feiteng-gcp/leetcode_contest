class Solution {
 public int[] processQueries(int[] queries, int m) {
        List<Integer> list = new ArrayList<>();
        for (int i = 1; i <= m; i++) {
            list.add(i);
        }

        int[] ans = new int[queries.length];
        int cur = 0;
        for (int q : queries) {
            int index = 0;
            for (int i = 0; i < m; i++) {
                if (list.get(i) == q) {
                    index = i;
                    break;
                }
            }

            
list.remove(index);
            list.add(0, q);
            ans[cur++] = index;
        }
        return ans;
    }
}
