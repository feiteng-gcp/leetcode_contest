class Solution {
    public int[] processQueries(int[] q, int m) {
        LinkedList<Integer> a = new LinkedList<>();
        for (int i = 0; i < m; ++i) {
            a.add(i + 1);
        }
        
        int n = q.length;
        int[] r = new int[n];
        for (int i = 0; i < n; ++i) {
            int qx = q[i];
            int p = -1;
            for (int j = 0; j < m; ++j) {
                if (qx == a.get(j)) {
                    p = j;
                    break;
                }
            }
            r[i] = p;
            a.remove(p);
            a.addFirst(qx);
        }
        return r;
    }
}
