class Solution {
    public int numOfWays(int n) {
        long mod = 1000000007;
        long[] d = new long[n + 1];
        long[] s = new long[n + 1];
        d[1] = 6;
        s[1] = 6;
        for(int i = 2; i <= n; i++){
            d[i] = (2 * d[i - 1] + 2 * s[i - 1]) % mod;
            s[i] = (2 * d[i - 1] + 3 * s[i - 1]) % mod;
        }
        
        return (int)((d[n] + s[n]) % mod);
        
    }
}
