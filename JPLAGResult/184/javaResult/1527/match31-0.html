class Solution {
    int mod = (int) (1e9 + 7);

    public int numOfWays(int n) {
        long dp[] = new long[12];
        for (int i = 0; i < 12; i++) {
            dp[i] = 1;
        }
        for (int i = 1; i < n; i++) {
            long tmp[] = new long[12];
            tmp[0] = (dp[1] + dp[2] + dp[4] + dp[5] + dp[10]) % mod;
            tmp[1] = (dp[0] + dp[3] + dp[6] + dp[8] + dp[11]) % mod;
            tmp[2] = (dp[0] + dp[3] + dp[6] + dp[7]) % mod;
            tmp[3] = (dp[1] + dp[2] + dp[7] + dp[10]) % mod;
            tmp[4] = (dp[0] + dp[6] + dp[8] + dp[9]) % mod;
            tmp[5] = (dp[0] + dp[6] + dp[7] + dp[9] + dp[10]) % mod;
            tmp[6] = (dp[1] + dp[2] + dp[4] + dp[5] + dp[11]) % mod;
            tmp[7] = (dp[2] + dp[3] + dp[5] + dp[11]) % mod;
            tmp[8] = (dp[1] + dp[4] + dp[9] + dp[10]) % mod;
            tmp[9] = (dp[4] + dp[5] + dp[8] + dp[11]) % mod;
            tmp[10] = (dp[0] + dp[3] + dp[5] + dp[8] + dp[11]) % mod;
            tmp[11] = (dp[1] + dp[6] + dp[7] + dp[9] + dp[10]) % mod;
            dp = tmp;
        }
        long res = 0;
        for (int i = 0; i < 12; i++) {
            res = (res + dp[i]) % mod;
        }
        return (int) res;
    }
}
