class Solution {
    public int numOfWays(int n) {
        int mod = 1000000007;
        long[] dp2 = new long[n + 1];
        long[] dp3 = new long[n + 1];
        dp2[1] = 6;
        dp3[1] = 6;
        for(int i = 2; i <= n; i++) {
            dp2[i] = (dp2[i - 1] * 3 + dp3[i - 1] * 2) % mod;
            dp3[i] = (dp2[i - 1] * 2 + dp3[i - 1] * 2) % mod;
        }
        return (int)((dp2[n] + dp3[n]) % mod);
    }
}
