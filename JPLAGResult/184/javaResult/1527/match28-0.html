class Solution {
    public int code(int a, int b, int c) {
        return a * 9 + b * 3 + c;
    }

    public int numOfWays(int n) {
        long mod = 1000000007;
        if (n == 1) {
            return 12;
        }
        long[][] dp = new long[n + 1][28];
        for (int a = 0; a < 3; a++) {
            for (int b = 0; b < 3; b++) {
                for (int c = 0; c < 3; c++) {
                    if (a != b && b != c) {
                        dp[1][code(a, b, c)] = 1;
                    }
                }
            }
        }
        for (int i = 2; i <= n; i++) {
            for (int a = 0; a < 3; a++) {
                for (int b = 0; b < 3; b++) {
                    for (int c = 0; c < 3; c++) {
                        if (a != b && b != c) {
                            for (int aa = 0; aa < 3; aa++) {
                                for (int bb = 0; bb < 3; bb++) {
                                    for (int cc = 0; cc < 3; cc++) {
                                        if (aa != bb && bb != cc && aa != a && bb != b && cc != c) {
                                            dp[i][code(a, b, c)] += dp[i - 1][code(aa, bb, cc)];
                                            dp[i][code(a, b, c)] %= mod;
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        int ans = 0;
        for (int i = 0; i <= 27; i++) {
            ans += dp[n][i];
            ans %= mod;
        }
        return ans;
    }
}
