class Solution {

	private boolean validate(int color1 , int color2 , int color3) {
		return color1 != color2 && color2 != color3;		
	}
	
	private boolean validate(int color11 , int color12 , int color13 , 
			int color21 , int color22 , int color23) {
		if (validate(color11 , color12 , color13) && validate(color21 , color22 , color23)) {
			return color11 != color21 && color12 != color22 && color13 != color23;			
		} else {
			return false;
		}
	}
	
	private long MOD = 1000_000_007;
	private long[][][][] dp;	
	
    public int numOfWays(int n) {
    	dp = new long[n][3][3][3];    	
    	for (int i = 0;i < n;i ++) {
    		if (i == 0) {
    			for (int color1 = 0;color1 < 3;color1 ++) {
    				for (int color2 = 0;color2 < 3;color2 ++) {
    					for (int color3 = 0;color3 < 3;color3 ++) {
    	    				if (validate(color1 , color2 , color3)) {
    	    					dp[0][color1][color2][color3] ++;    	    					
    	    				}
    	    			}	
        			}	
    			}
    		} else {
    			for (int color11 = 0;color11 < 3;color11 ++) {
    				for (int color12 = 0;color12 < 3;color12 ++) {
    					for (int color13 = 0;color13 < 3;color13 ++) {
	    					for (int color21 = 0;color21 < 3;color21 ++) {
	    	    				for (int color22 = 0;color22 < 3;color22 ++) {
	    	    					for (int color23 = 0;color23 < 3;color23 ++) {
	    	    	    				if (validate(color11 , color12 , color13 , 
	    	    	    						color21 , color22 , color23)) {
	    	    	    					dp[i][color21][color22][color23] = 
	    	    	    							(dp[i][color21][color22][color23] + 
	    	    	    									dp[i - 1][color11][color12][color13]) % MOD;
	    	    	    				}
	    	    	    			}	
	    	        			}	
	    	    			}				
    	    			}	
        			}	
    			}
    		}
    	}
    	long ans = 0;
    	for (int color1 = 0;color1 < 3;color1 ++) {
    		for (int color2 = 0;color2 < 3;color2 ++) {
    			for (int color3 = 0;color3 < 3;color3 ++) {
    	    		ans = (ans + dp[n - 1][color1][color2][color3]) % MOD;
    	    	}	
        	}	
    	}
    	return (int) ans;
    }
}
