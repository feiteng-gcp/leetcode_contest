class Solution {

   public int numOfWays(int n) {
       int[] dp = new int[n + 1];
       dp[0] = 3;
       dp[1] = 12;
       for (int i = 2; i <= n; i++) {
           dp[i] = (int) ((5L * dp[i - 1] - 2 * dp[i - 2] + 10000000070L) % 1000000007);
       }
       return dp[n];
   }
}
