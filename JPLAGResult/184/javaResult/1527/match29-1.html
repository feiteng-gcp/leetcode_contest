class Solution {
    public int numOfWays(int n) {
        long mod = 1000000000 + 7;
        long f2 = 6;
        long f3 = 6;
        
for (int i = 2; i <= n; i++) {
            f3 = (2 * f2 + 2 * f3) % mod;
            f2 = (f3 + f2) % mod;
            
        }
        return (int)((f2 + f3) % mod);
    }
}
