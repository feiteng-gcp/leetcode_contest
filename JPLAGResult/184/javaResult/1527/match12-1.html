class Solution {
    long modu = 1000000007;
    public int numOfWays(int n) {
        
long x = 6;
        long y = 6;
        for (int i=1; i<n; i++) {
            long xx = (x * 3 + y * 2) % modu;
            long yy = (x * 2 + y * 2) % modu;
            x = xx;
            y = yy;
        }
        return (int) ((x + y) % modu);
    }
}
