class Solution {
public:
    vector<int> processQueries(vector<int>& q, int m) {
        vector<int> res(q.size()), v(m);
        for (int i = 0; i < m; ++i)
            v[i] = i+1;
        
        for (int i = 0; i < q.size(); ++i) {
            int t = q[i];
            for (int j = 0; j < m; ++j)
                if (v[j] == t) {
                    res[i] = j;
                    v.erase(v.begin() + j);
                    break;
                }
            
            v.insert(v.begin(), t);
        }
        return res;
    }
};
