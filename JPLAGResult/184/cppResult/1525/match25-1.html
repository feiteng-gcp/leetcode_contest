class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> res;
        res.clear();
        for (int i = 0; i < m; ++i) {
            res.push_back(i + 1);
        }
        vector<int> res0;
        res0.clear();
        for (int i = 0; i < queries.size(); ++i) {
            int q = queries[i];
            int k = 0;
            for (k = 0; k < m; ++k) {
                if (res[k] == q) {

                    break;
                }
            }
            // cout << "k = " << k << endl;
            res0.push_back(k);
            for (int j = k; j > 0; --j) {
                res[j] = res[j - 1];
            }
            res[0] = q;
        }
        return res0;
    }
};
