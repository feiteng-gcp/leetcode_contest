class Solution {
public:
    vector<int> processQueries(vector<int>& qq, int m) {
        vector<int> vm;
        for(int i = 0; i < m; i++) vm.push_back(i+1);
        vector<int> ret;

        for(int q: qq) {
            int pos = -1;
            for(int i = 0; i < m; i++) {
                if(vm[i] == q) {pos = i; break;}
            }
            ret.push_back(pos);
            for(int i = pos; i >0; i--) vm[i] = vm[i-1];
            vm[0] = q;
        }
        return ret;       

    }
};
