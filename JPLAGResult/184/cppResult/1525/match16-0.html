class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> v(m);
        for (int i = 1; i <= m; ++i) {
            v[i - 1] = i;
        }
        vector<int> r;
        for (int q : queries) {
            int p = -1;
            for (int i = 0; p < 0 && i < m; ++i) {
                if (v[i] == q) p = i;
            }
            for (int i = p; i > 0; --i) {
                v[i] = v[i - 1];
            }
            v[0] = q;
            r.push_back(p);
            
        }
        return r;
        
    }
};
