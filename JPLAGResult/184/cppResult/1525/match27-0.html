class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> p(m);
        for(int i = 0; i < m; i++) {
            p[i] = i + 1;
        }
        vector<int> ret;
        for(auto x : queries) {
            ret.push_back(calc(p, x));
        }
        return ret;
    }
    int calc(vector<int>& p, int x) {
        int idx = 0;
        for(int i = 0; i < p.size(); i++) {
            if (p[i] == x) {
                idx = i;
                for(int j = idx; j > 0; j--) {
                    p[j] = p[j - 1];
                }
                p[0] = x;
                return idx;
            }
        }
        return idx;
    }
};
