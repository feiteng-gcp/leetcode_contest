class Solution {
public:

    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> res;
        vector<int> p(m+1),q(m);
        for(int i=1;i<=m;++i){
            q[i-1]=i;
            p[i]=i-1;
        }

        for(int z:queries){
            int t=p[z];
            res.push_back(t);
            for(int i=t;i>0;--i){
                q[i]=q[i-1];
                ++p[q[i-1]];
            }
            p[z]=0;
            q[0]=z;
        }
        return res;
    }
};
