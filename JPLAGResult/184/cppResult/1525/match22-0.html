class Solution {
public:
    vector<int> processQueries(vector<int>& q, int n) {
        vector<int> a(n);
        for (int i = 0; i < n; ++i) a[i] = i + 1;
        vector<int> ret;
        for (auto& x : q) {
            int k = -1;
            for (int i = 0; i < n; ++i) {
                if (a[i] == x) {
                    k = i;
                    break;
                }
            }
            ret.push_back(k);
            for (int i = k; i >= 1; --i) a[i] = a[i - 1];
            a[0] = x;
        }
        return ret;
    }
};
