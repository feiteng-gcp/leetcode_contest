class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> res;
        vector <int> a;
        for(int i = 1; i <= m; i++){
            a.push_back(i);
        }
        for(int i = 0; i < (int)queries.size(); i++){
            int idx = 0;
            for(int j = 0; j < m; j++){
                if(a[j] == queries[i]){
                    idx = j;
                    break;
                }
            }
            res.push_back(idx);
            a.erase(a.begin() + idx);
            a.insert(a.begin(), queries[i]);
        }
        return res;
    }
};
