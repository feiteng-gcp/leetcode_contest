class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> v, ans;
        for(int i = 1; i <= m; ++i){
            v.push_back(i);
        }
        for(auto x: queries){
            int j = 0;
            while(v[j] != x)
                ++j;
            ans.push_back(j);
            for(int i = j - 1; i >= 0; --i)
                v[i + 1] = v[i];
            v[0] = x;
        }
        return ans;
    }
};
