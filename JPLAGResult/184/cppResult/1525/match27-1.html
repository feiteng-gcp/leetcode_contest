class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        
        vector<int> a(m);
        for (int i = 0; i<m; ++i) {
            a[i]=i+1;
        }
        vector<int> ret{};

        for (auto x: queries) {
            ret.push_back(insert(a,x));
        }
        
        return ret;
    }
    
    
    int insert(vector<int>& a, int x) {
        //moves elt x to beginning of array a;
        int i = 0;

        while (a[i]!=x) ++i;
        
        for (int j = i; j>=1; --j) {
            a[j] = a[j-1];
        }
        a[0] = x;
        return i;
    }
};
