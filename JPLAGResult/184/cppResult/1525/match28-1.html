class Solution {
public:

	vector<int> processQueries(const vector<int>& queries, int m) {
		vector<int> permutation(m);
		vector<int> value_to_index(m + 1);
		for (int i = 0; i < m; i++) {
			permutation[i] = i + 1;
			value_to_index[i + 1] = i;
		}
		vector<int> res;
		for (int query : queries) {

			int index = value_to_index[query];
			res.emplace_back(index);
			for (int i = index - 1; i >= 0; i--) {
				permutation[i + 1] = permutation[i];
				value_to_index[permutation[i + 1]] = i + 1;
			}
			permutation[0] = query;
			value_to_index[query] = 0;
		}
		return res;
	}
};
