class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> a(m + 1);
        vector<int> ind(m + 1);
        for (int i = 1; i <= m; ++i) {
            a[i] = i;
            ind[i] = i;
        }
        vector<int> ans;
        for (auto x : queries) {
            int i = ind[x];
            ans.push_back(i - 1);
            for (int j = i - 1; j >= 1; --j) {
                a[j + 1] = a[j];
                ind[a[j]] = j + 1;
            }
            a[1] = x;
            ind[x] = 1;
        }
        return ans;
    }
};
