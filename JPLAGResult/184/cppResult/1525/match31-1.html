class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> p(m);
        for (int i=1; i<=m; ++i) {
            p[i-1]=i;
        }
        vector<int> ans;
        for (int i=0; i<queries.size(); ++i) {
            int j;

            for (j=0; j<m && p[j] != queries[i]; ++j) {}
            assert(j<m);
            ans.push_back(j);
            for (int k=j; k>0; --k) {
                p[k] = p[k-1];
            }
            p[0] = queries[i];
        }
        return ans;
    }
};
