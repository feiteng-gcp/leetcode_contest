class Solution {
 public:
  vector<int> processQueries(vector<int>& queries, int m) {
    vector<int> idx(m + 1);
    for (int i = 0; i <= m; i++) {
      idx[i] = i;
    }
    int n = queries.size();
    vector<int> res(n);
    for (int i = 0; i < n; i++) {
      int q = queries[i];
      int cur_pos = idx[q - 1];
      res[i] = cur_pos;
      for (int j = 0; j <= m; j++) {
        if (idx[j] < cur_pos) {
          idx[j]++;
        }
      }
      idx[q - 1] = 0;
    }
    return res;
  }
};
