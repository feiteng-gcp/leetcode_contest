class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> ans;
        vector<int> sb;
        for (int i = 1; i <= m; i++)
            ans.push_back(i);
        for (int i = 0; i < queries.size(); i++) {
            int k = queries[i];
            int p = 0;
            while (ans[p] != k) p++;
            for (int j = p; j > 0; j--)
                ans[j]=ans[j-1];
            ans[0]=k;
            sb.push_back(p);
        }
        return sb;
    }
};
