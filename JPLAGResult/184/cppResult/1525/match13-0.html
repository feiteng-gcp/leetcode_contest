class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> v, ret;
        for (int i = 1; i <= m; ++i) v.push_back(i);
        
        for (auto q: queries) {
            int i;
            for (i = 0; i < v.size(); ++i) {
                if (v[i] == q) {
                    ret.push_back(i);
                    break;
                }
            }
            reverse(v.begin(), v.begin() + i + 1);
            reverse(v.begin() + 1, v.begin() + i + 1);
        }
        
        return ret;
    }
};
