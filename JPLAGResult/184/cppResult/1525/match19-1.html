class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m) {
        vector<int> p(m);
        for(int i = 0; i < m; i ++) p[i] = i;
        vector<int> ans;
        for(int q: queries) {
            q --;

            int idx = -1;
            for(int i = 0; i < m; i ++) if(p[i] == q) {
                ans.push_back(i);
                idx = i;
                break;
            }
            p.erase(p.begin() + idx);
            p.insert(p.begin(), q);
        }
        return ans;
    }
};
