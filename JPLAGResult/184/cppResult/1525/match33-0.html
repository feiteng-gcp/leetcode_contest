class Solution {
public:
    vector<int> processQueries(vector<int>& queries, int m)
    {
        vector<int> P(m);
        for (int i = 0; i < m; ++i) {
            P[i] = i + 1;
        }
        vector<int> ans;
        for (int q : queries) {
            auto it = find(P.begin(), P.end(), q);
            int found = it - P.begin();
            ans.push_back(found);
            for (int i = found - 1; i >= 0; --i) {
                P[i + 1] = P[i];
            }
            P[0] = q;
        }
        return ans;
    }
};
