class Solution {
public:
    string entityParser(string s) {
        /*&quot;
        &apos;
        &amp;
        &gt;
        &lt;
        &frasl;
        
    */
        int n = s.length(), i = 0;
        string t;
        
        while (i < n) {
            if (s[i] != '&') {
                t += s[i];
                ++i;
            } else {
                if (s.substr(i, 6) == "&quot;") {
                    t += '"';
                    i += 6;
                } else if (s.substr(i, 6) == "&apos;") {
                    t += '\'';
                    i += 6;
                } else if (s.substr(i, 5) == "&amp;") {
                    t += '&';
                    i += 5;
                } else if (s.substr(i, 4) == "&gt;") {
                    t += '>';
                    i += 4;
                } else if (s.substr(i, 4) == "&lt;") {
                    t += '<';
                    i += 4;
                }else if (s.substr(i, 7) == "&frasl;") {
                    t += '/';
                    i += 7;
                }else {
                    t += s[i];
                    ++i;
                }
            }
        }
        return t;
    }
};
