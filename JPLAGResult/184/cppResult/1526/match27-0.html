class Solution {
public:
    string entityParser(string text) {
    	string ans="";
		for (int i=0;i<text.size();++i)	
		{
			if (text[i]!='&') {ans+=text[i];continue;}
			if (i+5<text.size()&&text[i+1]=='q'&&text[i+2]=='u'&&text[i+3]=='o'&&text[i+4]=='t'&&text[i+5]==';')
			{
				ans+="\"";
				i+=5;
				continue;
			}
			if (i+5<text.size()&&text[i+1]=='a'&&text[i+2]=='p'&&text[i+3]=='o'&&text[i+4]=='s'&&text[i+5]==';')
			{
				ans+="\'";
				i+=5;
				continue;
			}
			if (i+4<text.size()&&text[i+1]=='a'&&text[i+2]=='m'&&text[i+3]=='p'&&text[i+4]==';')
			{
				ans+="&";
				i+=4;
				continue;
			}
			if (i+3<text.size()&&text[i+1]=='g'&&text[i+2]=='t'&&text[i+3]==';')
			{
				ans+=">";
				i+=3;
				continue;
			}
			if (i+3<text.size()&&text[i+1]=='l'&&text[i+2]=='t'&&text[i+3]==';')
			{
				ans+="<";
				i+=3;
				continue;
			}
			if (i+6<text.size()&&text[i+1]=='f'&&text[i+2]=='r'&&text[i+3]=='a'&&text[i+4]=='s'&&text[i+5]=='l'&&text[i+6]==';')
			{
				ans+="/";
				i+=6;
				continue;
			}
			ans+=text[i];
		}
		return ans;	
    }
};
