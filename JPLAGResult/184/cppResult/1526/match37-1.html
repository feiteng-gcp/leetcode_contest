class Solution {
public:
    string entityParser(string text) {
        string ans = "";

        map<string, char> m;
        m["&quot;"] = '\"';
        m["&amp;"] = '&';
        m["&apos;"] = '\'';
        m["&gt;"] = '>';
        m["&lt;"] = '<';
        m["&frasl;"] = '/';
        for (int i = 0; i < text.size(); ++i) {
            bool match = 0;
            for (auto j : m) {

                if (text.substr(i, j.first.size()) == j.first) {
                    ans += j.second;
                    i += j.first.size() - 1;
                    match = true;
                    break;
                }
            }
            if (!match) {
                ans += text[i];
            }
        }
        return ans;
    }
};
