class Solution {
public:
    string entityParser(string a) {
        int n = a.size();
        string ans;
        for (int i = 0; i < n; i++) {
            if (i + 5 < n && a[i] == '&' && a[i + 1] == 'q' && a[i + 2] == 'u' && a[i + 3] == 'o' && a[i + 4] == 't') ans += "\"", i += 5;
            else if (i + 5 < n && a[i] == '&' && a[i + 1] == 'a' && a[i + 2] == 'p' && a[i + 3] == 'o' && a[i + 4] == 's') ans += "\'", i += 5;    
            else if (i + 4 < n && a[i] == '&' && a[i + 1] == 'a' && a[i + 2] == 'm' && a[i + 3] == 'p') ans += "&", i += 4;   
            else if (i + 3 < n && a[i] == '&' && a[i + 1] == 'g' && a[i + 2] == 't') ans += ">", i += 3;  
            else if (i + 3 < n && a[i] == '&' && a[i + 1] == 'l' && a[i + 2] == 't') ans += "<", i += 3; 
            else if (i + 6 < n && a[i] == '&' && a[i + 1] == 'f' && a[i + 2] == 'r' && a[i + 3] == 'a' && a[i + 4] == 's' && a[i + 5] == 'l') ans += "/", i += 6;
            else ans += a[i];
        }
        return ans;
    }
};

// Quotation Mark: the entity is &quot; and symbol character is ".
// Single Quote Mark: the entity is &apos; and symbol character is '.
// Ampersand: the entity is &amp; and symbol character is &.
// Greater Than Sign: the entity is &gt; and symbol character is >.
// Less Than Sign: the entity is &lt; and symbol character is <.
// Slash: the entity is &frasl; and symbol character is /.
