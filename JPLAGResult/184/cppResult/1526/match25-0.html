class Solution {
public:
    string entityParser(string text) {
        string result = "";
        for (int i = 0;i < text.length();++i){
            if (text[i] != '&'){
                result+=text[i];
            }
            else{
// Quotation Mark: the entity is &quot; and symbol character is ".
// Single Quote Mark: the entity is &apos; and symbol character is '.
// Ampersand: the entity is &amp; and symbol character is &.
// Greater Than Sign: the entity is &gt; and symbol character is >.
// Less Than Sign: the entity is &lt; and symbol character is <.
// Slash: the entity is &frasl; and symbol character is /.            
                if (text.substr(i, 6).compare("&quot;") == 0){
                    result += '"';
                    i += 5;
                }
                else if (text.substr(i, 6).compare("&apos;") == 0){
                    result += '\'';
                    i += 5;
                }
                else if (text.substr(i, 5).compare("&amp;") == 0){
                    result += '&';
                    i += 4;
                }
                else if (text.substr(i, 4).compare("&gt;") == 0){
                    result += '>';
                    i += 3;
                }
                else if (text.substr(i, 4).compare("&lt;") == 0){
                    result += '<';
                    i += 3;
                }
                else if (text.substr(i, 7).compare("&frasl;") == 0){
                    result += '/';
                    i += 6;
                }
                else {
                    result+='&';
                }
            }
        }
        return result;
    }
};
