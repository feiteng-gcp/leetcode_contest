class Solution {
public:
    const int N=5010;
    const int mod = 1e9+7;
    int numOfWays(int n) {

        long long dp[N][2];
        dp[1][0]=6;
        dp[1][1] = 6;
        for(int i=2;i<=n;++i){
            dp[i][0] = ((dp[i-1][0] * 3)%mod + (dp[i-1][1] *2)%mod)%mod;
            dp[i][1] = ((dp[i-1][0] * 2)%mod + (dp[i-1][1] *2)%mod)%mod;
        }
        return (dp[n][0]+dp[n][1])%mod;
    }
};
