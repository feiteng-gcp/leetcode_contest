class Solution {
public:
    typedef long long LL;
int numOfWays(int n)
{
    LL L = 6, R = 6;
    const int mod = 1e9 + 7;
    for(int i = 2; i <= n; i++)
    {
        LL nL = (3 * L + 2 * R) % mod;
        LL nR = (2 * L + 2 * R) % mod;
        L = nL; R = nR;
    }
    return (L + R) % mod;
}
};
