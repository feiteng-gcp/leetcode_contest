class Solution {
    const int MOD = 1e9 + 7;
public:
    int numOfWays(int n) {

        vector<vector<int>> color = {
            {0, 1, 0},
            {0, 1, 2},
            {0, 2, 0},
            {0, 2, 1},
            {1, 0, 1},
            {1, 0, 2},
            {1, 2, 0},
            {1, 2, 1},
            {2, 1, 0},
            {2, 1, 2},
            {2, 0, 1},
            {2, 0, 2},
        };
        vector<long long> dp(12, 1);
        for (int i = 2; i <= n; i++) {
            vector<long long> next(12, 0);

            for (int j = 0; j < 12; j++) {
                for (int k = 0; k < 12; k++) {
                    bool flag = true;
                    for (int m = 0; m < 3; m++) {
                        if (color[j][m] == color[k][m]) {
                            flag = false;
                            break;
                        }
                    }
                    if (flag) {
                        next[j] += dp[k];
                    }
                }
            }
            for (int j = 0; j < 12; j++) {
                dp[j] = next[j] % MOD;
            }
        }

        long long sum = 0;
        for (int i = 0; i < 12; i++) {
            sum += dp[i];
            sum %= MOD;
        }
        return sum % MOD;
    }
};
