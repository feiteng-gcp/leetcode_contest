class Solution {
public:
    int numOfWays(int n) 
    {
        long long dp[n+1][3][3][3];
        memset(dp,0,sizeof(dp));
        int mo=1e9+7;
        for(int a=0;a<3;a++)
            for(int b=0;b<3;b++)
                for(int c=0;c<3;c++)
                {
                    if(a!=b&&b!=c)
                        dp[1][a][b][c]=1;
                }
        for(int i=2;i<=n;i++)
        {
            for(int a=0;a<3;a++)
                for(int b=0;b<3;b++)
                {
                    if(a==b)
                        continue;
                    for(int c=0;c<3;c++)
                    {
                        if(b==c)
                            continue;
                        for(int aa=0;aa<3;aa++)
                            for(int bb=0;bb<3;bb++)
                            {
                                if(aa==bb)
                                    continue;
                                for(int cc=0;cc<3;cc++)
                                {
                                    if(bb==cc)
                                        continue;
                                    if(a!=aa&&b!=bb&c!=cc)
                                        dp[i][a][b][c]=(dp[i][a][b][c]+dp[i-1][aa][bb][cc])%mo;
                                }
                            }
                    }
                }
        }
        long long res=0;
        for(int a=0;a<3;a++)
            for(int b=0;b<3;b++)
                for(int c=0;c<3;c++)
                {
                    if(a!=b&&b!=c)
                        res=(res+dp[n][a][b][c])%mo;
                }
        return (int)res;
    }
};
