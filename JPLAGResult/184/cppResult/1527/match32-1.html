class Solution {
public:
    int numOfWays(int n) {
    	int id[5][5][5],nxt[30][30];
    	int num=0;
    	int dp[5005][30];
    	memset(dp,0,sizeof(dp));
    	memset(nxt,0,sizeof(nxt));
    	int mod=1e9+7;

		for (int i=1;i<=3;++i)
			for (int j=1;j<=3;++j)
				for (int k=1;k<=3;++k)
				{
					id[i][j][k]=-1;
					if (i==j) continue;
					if (j==k) continue;
					++num;id[i][j][k]=num;
				}

		for (int i=1;i<=3;++i)
			for (int j=1;j<=3;++j)
				for(int k=1;k<=3;++k)
					for (int p=1;p<=3;++p)
						for (int q=1;q<=3;++q)
							for (int r=1;r<=3;++r)
							{
								if (id[i][j][k]==-1) continue;
								if (id[p][q][r]==-1) continue;
								if (i==p) continue;
								if (j==q) continue;
								if (k==r) continue;
								nxt[id[i][j][k]][id[p][q][r]]=1;
							}

		for (int i=1;i<=num;++i) dp[1][i]=1;
		for (int i=2;i<=n;++i)
			for (int j=1;j<=num;++j)
				for (int k=1;k<=num;++k) dp[i][k]+=dp[i-1][j]*nxt[j][k],dp[i][k]%=mod;
		int ans=0;
		for (int i=1;i<=num;++i) ans+=dp[n][i],ans%=mod;
		return ans;	
    }
};
