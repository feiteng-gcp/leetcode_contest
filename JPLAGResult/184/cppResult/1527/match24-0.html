class Solution {
public:
    int numOfWays(int n) {
        long long x=6,y=6;
        int M = 1e9 + 7;
        for (int i=1;i<n;i++){
            long long xx = (3*x+2*y) % M;
            long long yy = (2*x + 2*y) % M;
            x =xx;
            y=yy;
        
        }
        return (x+y) % M;
    }
};
