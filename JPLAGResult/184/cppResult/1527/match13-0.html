int f[5005][3][3][3];
const int mod = 1000000007;
void pd(int &ret, int val){
    ret+=val;
    if(ret>=mod)ret-=mod;
}
class Solution {
public:
    int numOfWays(int n) {
        memset(f,0,sizeof(f));
        for(int i=0;i<3;i++)
            for(int j=0;j<3;j++)
                for(int k=0;k<3;k++){
                    if(!(i==j||j==k))f[1][i][j][k] = 1;
                }
        for(int i=1;i<n;i++)
            for(int j=0;j<3;j++)
                for(int k=0;k<3;k++)
                    for(int r=0;r<3;r++){
                        if(f[i][j][k][r]==0)continue;
                        for(int p=0;p<3;p++)
                            for(int q=0;q<3;q++)
                                for(int s=0;s<3;s++){
                                    if(p==q||q==s||p==j||q==k||s==r)continue;
                                    pd(f[i+1][p][q][s],f[i][j][k][r]);
                                }
                    }
        int ret = 0;
        for(int i=0;i<3;i++)
            for(int j=0;j<3;j++)
                for(int k=0;k<3;k++){
                    pd(ret, f[n][i][j][k]);
                }
        return ret;
    }
};
