class Solution {
public:
    int numOfWays(int n) {
        static const int MOD = 1e9+7;
        int dp[n+3][3][3][3];
        memset(dp, 0, sizeof dp);

        for (int i = 0; i < 3; ++i)
        for (int j = 0; j < 3; ++j) if (i != j)
        for (int k = 0; k < 3; ++k) if (j != k)
            dp[1][i][j][k] = 1;
        for (int t = 2; t <= n; ++t) {

            for (int i1 = 0; i1 < 3; ++i1)
            for (int j1 = 0; j1 < 3; ++j1) if (i1 != j1)
            for (int k1 = 0; k1 < 3; ++k1) if (j1 != k1)
            for (int i2 = 0; i2 < 3; ++i2) if (i1 != i2)
            for (int j2 = 0; j2 < 3; ++j2) if (i2 != j2 && j2 != j1)
            for (int k2 = 0; k2 < 3; ++k2) if (j2 != k2 && k2 != k1)
                (dp[t][i1][j1][k1] += dp[t-1][i2][j2][k2]) %= MOD;
        }
        int res = 0;
        for (int i = 0; i < 3; ++i)
        for (int j = 0; j < 3; ++j) if (i != j)
        for (int k = 0; k < 3; ++k) if (j != k)
            (res += dp[n][i][j][k]) %= MOD;
        return res;
    }
};
