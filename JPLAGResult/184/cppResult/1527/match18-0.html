class Solution {
public:
    long long dp[5005][4][4][4];
    long long mod=1e9+7;
    int numOfWays(int n) {
        for (int i=1; i<=3; i++) {
            for (int j=1; j<=3; j++) {
                if (i==j) continue;
                for (int k=1; k<=3; k++) {
                    if (k==j) continue;
                    dp[1][i][j][k]=1;
                }
            }
        }
        for (int len=2; len<=n; len++) {
            for (int i=1; i<=3; i++) {
                for (int j=1; j<=3; j++) {
                    if (i==j) continue;
                    for (int k=1; k<=3; k++) {
                        if (j==k) continue;
                        for (int i1=1; i1<=3; i1++) {
                            if (i==i1) continue;
                            for (int j1=1; j1<=3; j1++) {
                                if (j==j1) continue;
                                if (i1==j1) continue;
                                for (int k1=1; k1<=3; k1++) {
                                    if (k==k1) continue;
                                    if (j1==k1) continue;
                                    dp[len][i1][j1][k1]+=dp[len-1][i][j][k];
                                    dp[len][i1][j1][k1]%=mod;
                                }
                            }
                        }
                    }
                }
            }
        }
        long long res=0;
        for (int i=1; i<=3; i++) {
            for (int j=1; j<=3; j++) {
                if (i==j) continue;
                for (int k=1; k<=3; k++) {
                    if (k==j) continue;
                    res+=dp[n][i][j][k];
                    res%=mod;
                }
            }
        }
        return res;
        
    }
};
