class Solution {
public:
    
    int dp[5050][4][4][4];
    int mod = 1e9+7;
    

    int get(int ind, int a, int b, int c) {
        if (a == b || b == c) return 0;
        if (ind == 0) return 1;
        if (dp[ind][a][b][c] != -1) return dp[ind][a][b][c];
        
        

        int ans = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                for (int k = 0; k < 3; k++) {
                    if (i != a && j != b && k != c) {
                        ans += get(ind-1, i, j, k);
                        ans %= mod;
                    }
                }
            }
        }
        
        ans %= mod;
        dp[ind][a][b][c] = ans;
        return ans;
    }
    int numOfWays(int n) {
        memset(dp, -1, sizeof(dp));

        int ans = 0;
        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                for (int k = 0; k < 3; k++) {
                    ans += get(n-1, i, j, k);
                    ans %= mod;
                }
            }
        }
        ans %= mod;
        return ans;
    }
};
