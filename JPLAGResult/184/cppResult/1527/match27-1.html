class Solution {
    static constexpr long MOD = 1e9 + 7;
public:

    int numOfWays(int n) {
        long l2 = 6, l3 = 6;
        for(int i = 2; i <= n; ++i){
            long t2 = l2, t3 = l3;
            l2 = (t2 * 3 + t3 * 2) % MOD;
            l3 = (t2 * 2 + t3 * 2) % MOD;
        }
        return (l2 + l3) % MOD;
    }
};
