class Solution {
public:
    const long long mod = 1e9+7;
    long long dp[5010][3][3][3];
    int numOfWays(int n) {
        for (int i=1;i<=n;i++){

            if (i==1){
                for (int a1 = 0;a1<=2;a1++){
                    for (int a2 = 0;a2<=2;a2++){
                        if (a1==a2) continue;
                        for (int a3 = 0;a3<=2;a3++){
                            if (a3==a2) continue;
                            dp[i][a1][a2][a3]++;
                        }
                    }
                }
            }

            else{
                for (int a1 = 0;a1<=2;a1++){
                    for (int a2 = 0;a2<=2;a2++){
                        if (a1==a2) continue;
                        for (int a3 =0 ;a3<=2;a3++){
                            if (a3==a2) continue;
                            for (int b1=0;b1<=2;b1++){
                                if (b1==a1) continue;
                                for (int b2=0;b2<=2;b2++){
                                    if (b2==a2||b2==b1) continue;
                                    for (int b3=0;b3<=2;b3++){
                                        if (b3==b2||b3==a3) continue;
                                        dp[i][a1][a2][a3] = (dp[i][a1][a2][a3]+dp[i-1][b1][b2][b3])%mod;
                                    }

                                }
                            }
                        }
                    }
                }
            }

        }
        long long res = 0;
        for (int i=0;i<3;i++){
            for (int j=0;j<3;j++){
                for (int k=0;k<3;k++) res = (res+dp[n][i][j][k])%mod;
            }
        }
        return res;
    }
};
