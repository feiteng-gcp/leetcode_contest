class Solution {
public:
    int numOfWays(int n) {
        int MODULE = 1000000007;

        vector<int> abc(n+1);
        vector<int> aba(n+1);
        abc[1] = 6; aba[1] = 6;
        for (int i=2; i<=n; i++){
            abc[i] = (abc[i-1] + aba[i-1] ) % MODULE * 2 % MODULE;
            aba[i] = ((abc[i-1] + aba[i-1] ) % MODULE * 2 % MODULE + aba[i-1]) % MODULE;  
        }
        return (abc[n] + aba[n]) % MODULE;
    }
};
