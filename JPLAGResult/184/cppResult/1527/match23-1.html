class Solution {
public:
    int numOfWays(int n) {
        int M = 1e9 + 7;

        vector<long long> dp(n+1);
        dp[0] = 3;
        dp[1] = 12;
        for(int i = 2; i <= n; i++) {
            dp[i] = (5 * dp[i-1] - 2 * dp[i-2] + M * 2) % M;
        }
        return dp[n] % M; 
    }
};
