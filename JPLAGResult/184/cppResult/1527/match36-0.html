int f[5555][3][3][3];

class Solution {
public:
    int numOfWays(int n) {
        const int MOD = 1e9+7;
        static int flag = 0;
        if (flag == 0)
        {
            flag = 1;
            int N = 5000;
            for (int i = 0; i <= N; ++ i)
                for (int a = 0; a < 3; ++ a)
                    for (int b = 0; b < 3; ++ b)
                        for (int c = 0; c < 3; ++ c)
                            f[i][a][b][c] = 0;
            for (int a = 0; a < 3; ++ a)
                for (int b = 0; b < 3; ++ b) if (a != b)
                    for (int c = 0; c < 3; ++ c) if (b != c)
                    {
                        f[1][a][b][c] = 1;
                    }
            for (int i = 2; i <= N; ++ i)
                for (int a = 0; a < 3; ++ a)
                    for (int b = 0; b < 3; ++ b) if (a != b)
                        for (int c = 0; c < 3; ++ c) if (b != c)
                            for (int aa = 0; aa < 3; ++ aa) if (a != aa)
                                for (int bb = 0; bb < 3; ++ bb) if (b != bb && aa != bb)
                                    for (int cc = 0; cc < 3; ++ cc) if (c != cc && bb != cc)
                                        (f[i][a][b][c] += f[i-1][aa][bb][cc]) %= MOD;
        }
        int ret = 0;
        for (int a = 0; a < 3; ++ a)
            for (int b = 0; b < 3; ++ b)
                for (int c = 0; c < 3; ++ c)
                    (ret += f[n][a][b][c]) %= MOD;
        return ret;
    }
};
