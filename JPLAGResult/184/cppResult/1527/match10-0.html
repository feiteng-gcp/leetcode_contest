class Solution {
public:
    int numOfWays(int n) {
        const int mod = 1e9 + 7;
        vector<long long> dp2(n+1,0),dp3(n+1,0);
        dp2[1]=3*2;//两种颜色
        dp3[1]=6;//三种颜色全排列;
        for(int i=2;i<=n;i++){
            dp2[i]= (3 *dp2[i-1] + 2*dp3[i-1])%mod;
            dp3[i]= (2*dp2[i-1]+2*dp3[i-1])%mod;
        }
        return (dp2[n]+dp3[n]) % mod;
        
    }
};
