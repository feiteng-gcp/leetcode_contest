class Solution {
public:
    int numOfWays(int n) {
        const int MOD = 1000000007;
        using lint = long long int;

        lint n2 = 6, n3 = 6;
        for (int i = 1; i < n; i++)
        {
            lint n2n = n2 * 3 + n3 * 2;
            lint n3n = n2 * 2 + n3 * 2;
            n2 = n2n % MOD;
            n3 = n3n % MOD;
        }
        return (n2 + n3) % MOD;
    }
};
