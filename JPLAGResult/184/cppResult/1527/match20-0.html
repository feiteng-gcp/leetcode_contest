class Solution {
private:
    vector< vector<int> > stat={
        {0,1,0},
        {0,1,2},
        {0,2,0},
        {0,2,1},
        {1,0,1},
        {1,0,2},
        {1,2,0},
        {1,2,1},
        {2,0,1},
        {2,0,2},
        {2,1,0},
        {2,1,2}
    };
    static const long long MODULO=1000000007;
public:
    int numOfWays(int n) {
        vector< vector<int> > trans(12);
        bool flag;
        for(int i=0;i<12;i++){
            for(int j=0;j<12;j++){
                flag=false;
                for(int k=0;k<3;k++){
                    if(stat[i][k]==stat[j][k]){
                        flag=true;
                        break;
                    }
                }
                if(!flag){
                    trans[i].push_back(j);
                }
            }
        }
        vector<vector<unsigned long long> > dp(n,vector<unsigned long long>(12,0));
        for(int i=0;i<12;i++){
            dp[0][i]=1;
        }
        for(int i=1;i<n;i++){
            for(int j=0;j<12;j++){
                for(int k=0;k<trans[j].size();k++){
                    dp[i][j]+=dp[i-1][trans[j][k]];
                }
                dp[i][j]%=MODULO;
            }
        }
        int result=0;
        for(int i=0;i<12;i++){
            result+=dp[n-1][i];
            result%=MODULO;
        }
        return result;
    }
};
