class Solution {
public:
    int numOfWays(int n) {
        if (n == 1) return 12;
        int f[2][3][3][3];
        int cnt = 0 , base = 1000000007;

        memset(f, 0, sizeof(f));
        for (int i = 0;i < 3;i ++)
            for (int j = 0;j < 3;j ++)
                for (int k = 0;k < 3;k ++)
                    if (i != j && j != k)
                        f[cnt][i][j][k] = 1;
                    else
                        f[cnt][i][j][k] = 0;
        for (int p = 1;p < n;p ++) {

            cnt = 1-cnt;
            for (int i = 0;i < 3;i ++)
                for (int j = 0;j < 3;j ++)
                    for (int k = 0;k < 3;k ++)
                        if (i != j && j != k) {
                            f[cnt][i][j][k] = 0;

                            for (int i1 = 0;i1 < 3;i1 ++) 
                                for (int j1 = 0;j1 < 3;j1 ++) 
                                    for (int k1 = 0;k1 < 3;k1 ++)
                                        if (i1 != j1 && j1 != k1 && i != i1 && j != j1 && k != k1) {
                                            f[cnt][i][j][k] = (f[cnt][i][j][k] + f[1-cnt][i1][j1][k1]) % base;
                                        }
                        } else {
                            f[cnt][i][j][k] = 0;

                        }
        }
        int result = 0;
        for (int i = 0;i < 3;i ++)
            for (int j = 0;j < 3;j ++)
                for (int k = 0;k < 3;k ++)
                    result = (result + f[cnt][i][j][k]) % base;
        return result;
    }
};
