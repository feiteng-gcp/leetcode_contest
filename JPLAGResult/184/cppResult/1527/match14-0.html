class Solution {
public:
    int numOfWays(int n) {
        long long dp[5005][3][3][3] = {0};
        long long mod = 1000000007;
        for (int a = 0; a < 3; a++)
        for (int b = 0; b < 3; b++)
        for (int c = 0; c < 3; c++) {
            if (a == b || b == c) continue;
            dp[0][a][b][c] = 1;
        }
        for (int i = 1; i < n; i++) {
            for (int a = 0; a < 3; a++)
            for (int b = 0; b < 3; b++)
            for (int c = 0; c < 3; c++) {
                if (a == b || b == c) continue;
                for (int x = 0; x < 3; x++)
                for (int y = 0; y < 3; y++)
                for (int z = 0; z < 3; z++) {
                    if (x == y || y == z) continue;
                    if (a == x || b == y || c == z) continue;
                    dp[i][a][b][c] += dp[i - 1][x][y][z];
                    dp[i][a][b][c] %= mod;
                }
            }
        }
        long long ans = 0;
        for (int a = 0; a < 3; a++)
        for (int b = 0; b < 3; b++)
        for (int c = 0; c < 3; c++) {
            if (a == b || b == c) continue;
            ans += dp[n - 1][a][b][c];
            ans %= mod;
        }
        return ans;
        
    }
};
