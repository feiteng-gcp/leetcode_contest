import re

class Solution:
    def entityParser(self, text: str) -> str:
        text = re.sub(r"&quot;", "\"", text)
        text = re.sub(r"&apos;", r"'", text)
        text = re.sub(r"&amp;", r"&", text)
        text = re.sub(r"&gt;", r">", text)
        text = re.sub(r"&lt;", r"<", text)
        text = re.sub(r"&frasl;", r"/", text)
        return text
