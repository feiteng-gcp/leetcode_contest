class Solution:
    def entityParser(self, t: str) -> str:
        t = t.replace("&quot;","\"")
        t = t.replace("&apos;","\'")
        t = t.replace("&amp;","&")
        t = t.replace("&gt;",">")
        t = t.replace("&lt;","<")
        t = t.replace("&frasl;","/")
        return t
