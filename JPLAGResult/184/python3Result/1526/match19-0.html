class Solution:
    def entityParser(self, s: str) -> str:
        s = s.replace("&quot;", '"')
        s = s.replace("&apos;", "'")
        s = s.replace("&amp;", '&')
        s = s.replace("&gt;", '>')
        s = s.replace("&lt;", '<')
        s = s.replace("&frasl;", '/')
        return s
