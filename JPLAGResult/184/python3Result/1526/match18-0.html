class Solution:
    def entityParser(self, text: str) -> str:
        text = text.replace('&quot;', r'"')
        text = text.replace('&apos;', '\'')
        text = text.replace('&amp;', r'&')
        text = text.replace('&gt;', r'>')
        text = text.replace('&lt;', r'<')
        text = text.replace('&frasl;', r'/')
        return text
