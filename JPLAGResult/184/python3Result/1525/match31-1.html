class Solution:
    def processQueries(self, Q: List[int], m: int) -> List[int]:
        A = [i + 1 for i in range(m)]
        ret = []
        for q in Q :
            r = A.index(q)
            ret.append(r)
            A.remove(q)
            A.insert(0, q)
        return ret
