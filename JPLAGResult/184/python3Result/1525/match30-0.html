class Solution:
    def processQueries(self, queries: List[int], m: int) -> List[int]:
        P = [ x+1 for x in range(m)]
        res = []
        for query in queries:
            cur_loc = P.index(query)
            res.append(cur_loc)
            P.pop(cur_loc)
            P.insert(0,query)
        return res
