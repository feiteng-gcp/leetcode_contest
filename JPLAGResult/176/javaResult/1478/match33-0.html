class Solution {
    public int maxEvents(int[][] es) {
        
        Arrays.sort(es,(a,b)->{
            if(a[0]!=b[0]){
                return a[0] - b[0];
            }
            return a[1]-b[1];
        });
        int p = 0;
        int lst =-1;
        int r = 0;
        PriorityQueue<Integer> que = new PriorityQueue<>();
        for(int d=1;d<=100000;++d){
            while(p<es.length&&es[p][0]==d){
                que.offer(es[p][1]);
                p++;
            }
            while(que.size()>0&&que.peek()<d){
                que.poll();
            }
            if(que.size()>0&&que.peek()>=d){
                que.poll();
                r++;
            }
            
            
        }
        return r;
        
            
            
        
        
    }
}
