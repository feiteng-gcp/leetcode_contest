class Solution {
    public int maxEvents(int[][] events) {
        
int st = events[0][0];
        int en = events[0][1];
        for(int i = 0; i < events.length; ++i) {
            st = Math.min(st, events[i][0]);
            en = Math.max(en, events[i][1]);
        }
        Arrays.sort(events, new Comparator<int[]>() {
            public int compare(int[] a, int[] b) {
                return a[0] - b[0];
            }
        });
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>(new Comparator<int[]>() {
            public int compare(int[] a, int[] b) {
                if (a[1] == b[1]) {
                    return a[0] - b[0];
                } else {
                    return a[1] - b[1];
                }
            }
        });
        
int j = 0;
        int res = 0;
        for(int i = st; i <= en; ++i) {
            while(j < events.length && events[j][0] == i) {
                pq.add(events[j]);
                ++j;
            }
            while(pq.isEmpty() == false && pq.peek()[1] < i) {
                pq.poll();
            }
            if (pq.isEmpty() == false) {
                pq.poll();
                ++res;
            }
        }
        return res;
    }
    
}
