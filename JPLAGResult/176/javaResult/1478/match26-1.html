class Solution {
    public int maxEvents(int[][] events) {
        // PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[0] == b[0] ? a[1]-b[1] : a[0]-b[0]);
        int count = 0;
        boolean[] visited = new boolean[100001];
        Arrays.sort(events, (a, b) -> a[1]-b[1]);
        
for (int[] e : events) {
            for (int i = e[0]; i <= e[1]; i++) {
                if (!visited[i]) {
                    visited[i] = true;
                    count++;
                    break;
                }
            }
        }
        return count;
    }
}
