class Solution {
    public int maxEvents(int[][] es) {
        Arrays.sort(es, (a,b)->{return a[0] - b[0];});
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)->{return a[1]-b[1];});
        int cnt = 0, day = es[0][0];
        for (int i = 0; i < es.length || !pq.isEmpty(); ) {
            while (i < es.length && es[i][0] <= day) pq.offer(es[i++]);
            while (!pq.isEmpty() && pq.peek()[1] < day) pq.poll();
            if (pq.isEmpty()) {
                if (i < es.length)
                    day = es[i][0];
            } else {
                int[] e = pq.poll();
                System.out.println(Arrays.toString(e));
                
day = Math.max(day + 1, e[0] + 1);
                cnt++;
            }
            
        }
        return cnt;
    }
}
