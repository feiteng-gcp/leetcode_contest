class Solution {
    public int maxEvents(int[][] events) {
        // System.out.println("...");
        Queue<int[]> q = new PriorityQueue<>((a, b)->(a[1] == b[1] ? a[0] - b[0] : a[1] - b[1]));
        // Arrays.sort(events, (a, b)->(a[1] != b[1] ? a[0] - b[0] : a[1] - b[1]));
        int maxi = 1;
        Map<Integer, List<int[]>> map = new HashMap<>();

        for(int[] e : events) {
            maxi = Math.max(maxi, e[1]);
            int e0 = e[0];
            if(!map.containsKey(e0)) map.put(e0, new ArrayList<>());
            map.get(e0).add(e);
        }

        int re = 0;
        // for(int[] e : q)
        //     System.out.println(Arrays.toString(e));

        
        for(int i = 1; i <= maxi; i++)
        {
            if(map.containsKey(i))
            {
                for(int[] next : map.get(i)) q.offer(next);
            }
            while(!q.isEmpty())
            {
                int[] p = q.poll();
                // System.out.printf("%d %s", i, Arrays.toString(p));
                int from = p[0], to = p[1];
                if(from <= i && i <= to) 
                {
                    // System.out.printf("*\n");
                    re++;    
                    break;
                }
                // System.out.printf("\n");
            }
            
        }
        return re;
    }
}
