class Solution {
    public int maxEvents(int[][] events) {
        Arrays.sort(events, (a, b) -> {
            if (a[1] == b[1]) return a[0] - b[0];
            return a[1] - b[1];
        });
        int res = 0;
        Set<Integer> used = new HashSet<>();
        for (int[] e : events) {
            for (int t = e[0]; t <= e[1]; t++) {
                if (!used.contains(t)) {
                    used.add(t);
                    res++;
                    break;
                }
            }
        }
        return res;
    }
}
