class Solution {
    public int maxEvents(int[][] events) {
        Arrays.sort(events, (a,b)->(a[1]-b[1]==0 ? a[0]-b[0] : a[1]-b[1]));
        int res = 0;
        boolean[] book = new boolean[500000];
        for(int[] event : events){
            for(int i=event[0]; i<=event[1]; i++){
                if(!book[i]){
                    res++;
                    book[i] = true;
                    break;
                }
            }
        }
        return res;
    }
}
