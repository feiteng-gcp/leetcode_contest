class Solution {
    public int maxEvents(int[][] events) {
        Arrays.sort(events, (a, b) -> a[0] - b[0] == 0 ? a[1] - b[1] : a[0] - b[0]);
        // stores all events with start time <= next available date
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[1] - b[1]);
        int result = 0;
        int idx = 0;
        int nextAvail = events[0][0];
        pq.add(events[idx++]);
        while (!pq.isEmpty()) {
            while (idx < events.length && events[idx][0] <= nextAvail) pq.add(events[idx++]);
            
            while (!pq.isEmpty() && pq.peek()[1] < nextAvail) pq.poll();
            
            if (!pq.isEmpty()) {
                nextAvail = Math.max(nextAvail, pq.poll()[0]) + 1;
                result++;
            }
            if (pq.isEmpty() && idx < events.length) pq.add(events[idx++]);
        }
        return result;
    }
}
