class Solution {
    public int maxEvents(int[][] events) {
        // sort by end time
        Arrays.sort(events, new Comparator<int[]>(){
            public int compare(int[] a, int[] b){
                // if(a[0] == b[0]) return (a[1] - a[0]) - (b[1] - b[0]);
                // else return a[0] - b[0];
                return a[0] - b[0];
            }
        });
        int res = 0, cur = 1, counter = 0;
        PriorityQueue<int[]> canDo = 
new PriorityQueue<>(new Comparator<>(){
            public int compare(int[] a, int[] b){
                return a[1] - b[1];
            }
        });
        
        
        for(int i = 1; counter < events.length || !
canDo.isEmpty(); i++){
            // if(counter < events.length && i < events[counter][0]) i = events[counter][0];
            
            while(counter < events.length && i >= events[counter][0]){
                canDo.add(events[counter]);
                counter++;
            }
            // System.out.println(canDo.size());
            // get rid of expired event
            while(!canDo.isEmpty() && i > canDo.peek()[1]) canDo.poll();
            
            // System.out.println(Arrays.toString(canDo.poll()));
            if(!canDo.isEmpty()){
                canDo.poll();
                res++;
            }
        }
        
        
        return res;
    }
