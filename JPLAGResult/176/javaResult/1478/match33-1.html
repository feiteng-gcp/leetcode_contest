import java.util.Arrays;
import java.util.PriorityQueue;

class Solution {
  public static void main(String args[]) {
    int[][] input = {
        {1, 4},
        {1, 4},
        {1, 3},
        {1, 2},
        {1, 1}
    };
    Solution sol = new Solution();
    System.out.println(sol.maxEvents(input));
  }
  
  public int maxEvents(int[][] events) {
    PriorityQueue<int[]> endingEvents = new PriorityQueue<>(events.length, (e1, e2) -> e1[1] - e2[1]);
    Arrays.sort(events, (e1, e2)-> {
      int cmp1 = e1[0] - e2[0];
      int cmp2 = e1[1] - e2[1];
      
if (cmp1 != 0) {
        return cmp1;
      }
      return cmp2;
    });
    int i = 0;
    int count = 0;
    
for (int curr = 1; curr <= 100000; curr++) {
      while (i < events.length && events[i][0] <= curr) {
        endingEvents.add(events[i]);
        i++;
      }
      while (!endingEvents.isEmpty() && endingEvents.peek()[1] < curr) {
        endingEvents.poll();
      }
      if (!endingEvents.isEmpty()) 
{
        endingEvents.poll();
        count++;
      }
    }
    return count;
  }
}
