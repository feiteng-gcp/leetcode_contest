class Solution {
    public int maxEvents(int[][] events) {
        
        Arrays.sort(events, (a,b)-> {
            return a[0]-b[0];
        });
        
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> {
            return a[1]-b[1];
        });
        
        int res = 0;
        int eIdx = 0;
        for(int i = 1; i<=100000; i++) {
            while(eIdx<events.length && events[eIdx][0]==i) {
                pq.add(events[eIdx]);
                eIdx++;
            }
            
            while(pq.size()>0 && i>pq.peek()[1]) pq.poll();
            if(pq.size() == 0) continue;
            
            int[] event = pq.peek();
            if(i>=event[0] && i<=event[1]) {
                res++;
                pq.poll();
            }
        }
        return res;  
    }
}
