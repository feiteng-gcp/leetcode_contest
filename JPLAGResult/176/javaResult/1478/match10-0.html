class Solution {
    public int maxEvents(int[][] events) {
        if (events == null || events.length == 0) {
            return 0;
        }
        int mind = 100000;
        int maxd = 1;
        Map<Integer, List<Event>> map = new HashMap<>();
        for(int[] event : events) {
            map.putIfAbsent(event[0], new ArrayList<>());
            map.get(event[0]).add(new Event(event[0], event[1]));
            if (mind > event[0]) {
                mind = event[0];
            }
            if (maxd < event[1]) {
                maxd = event[1];
            }
        }
        PriorityQueue<Event> queue = new PriorityQueue<>((a, b) -> a.r - b.r);
        int tot = 0;
        for(int d = mind; d <= maxd; ++d) {
            List<Event> list = map.get(d);
            if (list == null || list.isEmpty()) {
                if (queue.isEmpty()) {
                    continue;
                }
            } else {
                for(Event event : list) {
                    queue.offer(event);
                }
            }
            while(!queue.isEmpty()) {
                Event event = queue.poll();
                if (event.l <= d && event.r >= d) {
                    ++tot;
                    break;
                }
            }
        }
        return tot;
    }
    
    private static class Event {
        public int l;
        public int r;
        public Event(int l, int r) {
            this.l = l;
            this.r = r;
        }
    }
}
