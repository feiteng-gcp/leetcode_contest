class Solution {
    public int maxEvents(int[][] events) {
        Arrays.sort(events, (e1, e2) -> e1[0] == e2[0] ? e1[1] - e2[1] : e1[0] - e2[0]);
        
PriorityQueue<int[]> queue = new PriorityQueue<>((e1, e2) -> e1[1] - e2[1]);
        int cnt = 0;
        int id = 0;
        for (int time = 0; time <= 100000; time++) {
            while (id < events.length && events[id][0] <= time && events[id][1] >= time) queue.offer(events[id++]);
            
while (!queue.isEmpty() && queue.peek()[1] < time) queue.poll();
            if (!queue.isEmpty()) {
                queue.poll();
                cnt++;
            }
        }

        return cnt;
    }
}
