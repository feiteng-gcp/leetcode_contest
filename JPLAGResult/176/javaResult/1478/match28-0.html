class Solution {
    public int maxEvents(int[][] events) {
        PriorityQueue<int[]> heap = new PriorityQueue<>((ev1, ev2) -> ev1[1] - ev2[1]);
        int start = Integer.MAX_VALUE;
        int end = Integer.MIN_VALUE;
        for (int i = 0; i < events.length; i++) {
            start = Math.min(start, events[i][0]);
            end = Math.max(end, events[i][1]);
        }
        
        Arrays.sort(events, (e1, e2) -> e1[0] - e2[0]);
        int res = 0;
        int idx = 0;
        
        for (int i = start; i <= end; i++) {
            while (idx < events.length && events[idx][0] <= i) {
                heap.offer(events[idx++]);
            }
            
            while (!heap.isEmpty() && heap.peek()[1] < i) {
                heap.poll();
            }
            if (!heap.isEmpty()) {
                res++;
                heap.poll();
            }
        }
        
        
        return res;
        
    }
}
