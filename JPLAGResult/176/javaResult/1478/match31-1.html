class Solution {
    public int maxEvents(int[][] events) {
        PriorityQueue<int[]> queue = new PriorityQueue<int[]>((a,b)->a[0]==b[0]?a[1]-b[1]: a[0]-b[0]);
        
for(int[] e: events) {
            queue.add(e);
        }
        int cnt = 0;
        int time = 0;
        while(queue.size()>0) {
            int[] head = queue.poll();
            if (time < head[0]) {
                time = head[0];
                cnt++;
            } else if (head[1]>time) {
                head[0] = time +1;
                
queue.add(head);
            }
        }
        return cnt;
    }
}
