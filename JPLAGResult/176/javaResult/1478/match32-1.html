class Solution {
    
    public int maxEvents(int[][] events) {
    
        Arrays.sort(events, (e1, e2)->{
            return e1[1]-e2[1];
        });
        
        TreeSet<Integer> set=new TreeSet<>();
        for(int i=1; i<=100_000; i++)
            set.add(i);
        
        int cnt=0;
        
        for(int a[]: events) {
            
Integer ceil=set.ceiling(a[0]);
            if(ceil!=0 && ceil<=a[1]) {
                cnt+=1;
                set.remove(ceil);
            }
        }
        
        return cnt;
    }
}
