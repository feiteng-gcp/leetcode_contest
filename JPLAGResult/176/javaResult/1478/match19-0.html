class Solution {
    public int maxEvents(int[][] events) {
        PriorityQueue<int[]> heap = new PriorityQueue<>((e1, e2) -> e1[0] == e2[0] ? Integer.compare(e1[1], e2[1]) : Integer.compare(e1[0], e2[0]));
        for (int[] event : events) {
            heap.add(event);
        }
        int day = 1, count = 0;
        while (!heap.isEmpty()) {
            int[] event = heap.remove();
            if (day <= event[0]) {
                day = event[0] + 1;
                count++;
            } else if (day > event[0] && day <= event[1]) {
                heap.add(new int[]{day, event[1]});
            }
        }
        return count;
    }
}
