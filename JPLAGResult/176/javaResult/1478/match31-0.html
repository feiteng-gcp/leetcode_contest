class Solution {
    public int maxEvents(int[][] events) {
        PriorityQueue<int[]> q = new PriorityQueue<int[]>(new Comparator<int[]>(){
            @Override
            public int compare(int [] a, int [] b){
                if(a[0] == b[0]){
                    return a[1] - b[1];
                }
                return a[0] - b[0];
            }
        });
        for(int [] cur: events){
            q.offer(cur);
        }
        int s = -1;
        int count = 0;
        while(q.size() > 0){
            int [] cur = q.poll();
            if(cur[0] > s){
                s = cur[0];
                count++;
            }else if(s >= cur[1]){
                continue;
            }else{
                cur[0] = Math.min(cur[1], s + 1);
                q.offer(cur);
            }
        }
        
        return count;
    }
}
