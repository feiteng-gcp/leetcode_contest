class Solution {
    public int maxEvents(int[][] events) {
    PriorityQueue<int[]> queue = new PriorityQueue<>(new Comparator<int[]>(){
      @Override
      public int compare(int[] a, int[] b){
        if(a[0] != b[0]){
          return a[0] - b[0];
        } else {
          return a[1] - b[1];
        }
      
}
    });
    for(int[] event: events){
      queue.add(event);
    }
    int res = 0;
    int day = 1;
    while(!queue.isEmpty()){
      int[] event = queue.poll();
      if(day > event[1]){
        continue;
      } else if(event[0] < day){
        queue.add(new int[]{day, event[1]});
      } else {
        day = 
Math.max(day + 1, event[0] + 1);
        res++;
      }
    }
    return res;
  }
}
