class Solution {
    public int maxEvents(int[][] events) {
        Arrays.sort(events, (e1, e2)-> e1[0]-e2[0]);
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>((e1, e2)->e1[1] - e2[1]);
        int idx = 0, res=0;
        // for(int[] e: events) {
        //     System.out.println(Arrays.toString(e));
        // }
        for(int i=0; i<=100000; i++) {
            if(idx >= events.length && pq.isEmpty()) break;
            for(; idx<events.length; idx++) {
                int[] e = events[idx];
                if(i < e[0]) break;
                pq.add(e);
            }
            //System.out.println(i+" "+pq.size());
            while(!pq.isEmpty() && pq.peek()[1] < i) pq.poll();
            if(!pq.isEmpty()) {
                res++;
                pq.poll();
            }
        }
        return res;
    }
}
