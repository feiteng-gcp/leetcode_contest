class Solution {
    public int maxEvents(int[][] events) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((n1, n2) -> n1[0] == n2[0] ? n1[1] - n2[1] : n1[0] - n2[0]);
        for (int[] e : events) {
            pq.offer(e);
        }
        int count = 0;
        int currTime = -1;
        while (!pq.isEmpty()) {
            int[] e = pq.poll();
            if (currTime <= e[0]) {
                count++;
                currTime = e[0] + 1;
            } else if (currTime <= e[1]) {
                count++;
                currTime = currTime + 1;
            }
            while (!pq.isEmpty() && pq.peek()[0] < currTime) {
                int[] ev = pq.poll();
                ev[0] = currTime;
                if (ev[0] <= ev[1]) {
                    pq.offer(ev);
                }
            }
            
        }
        return count;
    }
}
