class Solution {
    public int maxEvents(int[][] events) {
        
boolean []states = new boolean[100001];
        Arrays.sort(events, (o1,o2) -> o1[1] - o2[1]);
        for(int i=0;i<events.length;i++){
            for(int j=events[i][0];j<=events[i][1];j++){
                if(!states[j]){
                    states[j] = true;
                    break;
                }
            }
        }
        int ans = 0;
        for(int i=1;i<100001;i++){
            if(states[i]) ans++;
        }
        return ans;
    }
}
