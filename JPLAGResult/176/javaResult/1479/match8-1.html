class Solution {
    public boolean isPossible(int[] target) {
        PriorityQueue<Integer> qu = new PriorityQueue<>(Collections.reverseOrder());
        long sum = 0;
        for(int num : target){
            qu.offer(num);
            sum += num;
        }
        while(true)
{
            if(qu.peek() == 1)
                return true;
            int cur = qu.poll();
            long pre = 2 * cur - sum;
            if(pre < 1)
                return false;
            sum = cur;
            qu.offer((int)pre);
        }
    }
}
