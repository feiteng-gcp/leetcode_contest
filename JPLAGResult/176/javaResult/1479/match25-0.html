class Solution {
    public boolean isPossible(int[] target) {
        long sum = 0;
        PriorityQueue<Long> pq = new PriorityQueue<>(Collections.reverseOrder());
        for (int num : target) {
            pq.offer((long)num);
            sum += num;
        }
        
        while (pq.peek() != 1) {
            long cur = pq.poll();
            if (cur - (sum - cur) <= 0) {
                return false;
            }
            pq.offer(cur * 2 - sum);
            sum -= sum - cur;
        }
        return true;
    }
}
