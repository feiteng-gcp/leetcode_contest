class Solution {
    public boolean isPossible(int[] t) {
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
        long sum = 0;
        for (int i : t) {
            sum += i;
            pq.offer(i);
        }
        while (!pq.isEmpty()) {
            if (pq.peek() == 1) return true;
            int curMax = pq.poll();
            sum -= curMax;
            long before = curMax - sum;
            
if (before < 1) return false;
            pq.offer((int) before);
            sum += before;
        }
        return true;
    }
}
