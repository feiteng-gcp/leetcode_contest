class Solution {
    public boolean isPossible(int[] target) {
        int maxIdx = 0;
        int max = -1;
        int sum = 0;
        for  (int i = 0; i < target.length; i++) {
            sum += target[i];
            if (target[i] > max) {
                max = target[i];
                maxIdx = i;
            }
        }
        if (max == 1) return true;
        
        int other = sum - max;
        if (max - other < 1) return false;
        int origin = max % other;
   
        target[maxIdx] = origin;
        return isPossible(target);
    }
}
