class Solution {
    public boolean isPossible(int[] target) {
        PriorityQueue<Integer> pq = new PriorityQueue(Collections.reverseOrder());
        
        int sum = 0;
        for (int n : target) {
            pq.offer(n);
            sum += n;
        }
        
        while (!pq.isEmpty()) {
            int cur = pq.poll();
            if (cur == 1) continue;
            
            
            
int sumOfRem = sum - cur;
            cur -= sumOfRem;
            sum -= sumOfRem;
            
            if (cur < 1 || sumOfRem == 0) return false;
            pq.offer(cur);
        }
        return true;
    }
}
