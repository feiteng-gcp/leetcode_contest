class Solution {
    public boolean isPossible(int[] target) {
         PriorityQueue<Long> heap = new PriorityQueue<Long>(Collections.reverseOrder());
         long sum = 0;
         for(int elem : target){
             heap.add((long)elem);
             sum += elem;
         }   
          while(true)
{
            long largest = heap.poll();
            if(sum == target.length)
                return true;
            long temp = sum - largest;
            largest = largest - temp; 
            sum -= temp;  
            if(largest < 1)
                return false;
            
            heap.add(largest);  
           
        }
