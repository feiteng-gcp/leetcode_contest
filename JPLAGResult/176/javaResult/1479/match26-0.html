class Solution {
    public boolean isPossible(int[] target) {
        long sum = 0;
        
        PriorityQueue<Long> q = new PriorityQueue<Long>((x, y) -> y.compareTo(x));
        for (int num : target)
        {
            q.offer((long)num);
            sum += num;
        }
        
        while(q.peek() != 1)
        {
            long cur = q.poll();
            sum -= cur;
            cur -= sum;
            if (cur <= 0)
            {
                return false;
            }
            
            sum += cur;
            q.offer(cur);
        }
        
        return true;
    }
}
