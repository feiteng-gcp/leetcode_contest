class Solution {
    public boolean isPossible(int[] target) {
        PriorityQueue<Long> queue = new PriorityQueue<>((a, b) -> b - a > 0 ? 1 : -1);
        long sum = 0l;
        for(int num : target) {
            queue.offer((long) num);
            sum += num;
        }
        while(!queue.isEmpty()) {
            long max = queue.poll();
            if (max == 1l) {
                return true;
            }
            //之前的值
            long pre = 2l * max - sum;
            if (pre <= 0l) {
                return false;
            }
            queue.offer(pre);
            sum = sum - max + pre;
        }
        return false;
    }
}
