class Solution {
    public boolean isPossible(int[] target) {
        Queue<Integer> q = new PriorityQueue<>(Collections.reverseOrder());
        int sum = 0;

        for(int t : target)
        {
            sum += t;
            q.offer(t);
        }
        while(!q.isEmpty() && q.peek() > 1)
        {
            int next = q.poll();
            if(sum <= next) return false;
            int diff = sum - next;
            int nnext = next - diff;
            if(nnext < 1) return false;
            sum = next;
            q.offer(nnext);
        }
        int n = target.length;
        if(q.size() != n) {
            return false;
        
}
        
        
        while(!q.isEmpty())
        {
            if(q.poll() != 1) 
                return false;
        }
        return true;
    }
}
