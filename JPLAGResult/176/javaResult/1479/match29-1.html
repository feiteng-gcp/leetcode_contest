class Solution {
    public boolean isPossible(int[] target) {
        long sum = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>((a, b) -> (b - a));
        for (int i : target) {
            sum += i;
            pq.add(i);
        }
        while (!pq.isEmpty()) {
            int maxOne = pq.poll();
            if (maxOne == 1) {
                return true;
            }
            if (maxOne < 0) {
                return false;
            }
            long other = sum - maxOne;
            sum -= other;
            if (maxOne < other) {
                return false;
            }
            maxOne -= other;
            if (maxOne < 0) {
                return false;
            }
            if (maxOne > 1) {
                pq.add(maxOne);
            }
        }
        return true;
    }
}
