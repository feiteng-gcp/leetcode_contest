import java.util.Comparator;
import java.util.PriorityQueue;


class Solution {
	public boolean isPossible(int[] target) {
		PriorityQueue<Long> pq = new PriorityQueue<Long>(Comparator.reverseOrder()); 
		long sum = 0;
		for (int i : target) {
			pq.add((long) i);
			sum += i;
		}
		while (!pq.isEmpty()) {
			Long c = pq.poll();
			if (c == 1)
				return true;
			if (c < 1)
				return false;
			long x = 2 * c - sum;

//			Helper.tr(sum, c, x);
			if (x < 1)
				return false;
			pq.add(x);
			sum = sum - c + x;
		}
		return false;
	}
}
