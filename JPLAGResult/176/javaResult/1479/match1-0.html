class Solution {
    
  public boolean isPossible(int[] target) {
    long sum = 0;
    PriorityQueue<Integer> queue = new PriorityQueue<>(Collections.reverseOrder());
    for (int num : target) {
      sum += num;
      queue.add(num);
    }
    while (true) {
      int max = queue.poll();
      if (max == 1) {
        return true;
      }
      int pre = max;
      if(max <= sum - pre){
        return false;
      }
      max %= sum - pre;
      sum += max - pre;
      queue.add(max);
    }
  }
}
