class Solution {
    
    // sum: len, 2*len-1, 3*len-1
    // subsitute max, + len - 1
    // others 
    
    // 最大的是由其他的和组成的（正数性质）
    // 所以恢复原来的数即可
    public boolean isPossible(int[] target) {
        PriorityQueue<Long> heap = new PriorityQueue<>(Comparator.reverseOrder());
        long sum = 0;
        for(int n : target) {
            sum += n;
            heap.add((long)n);
        }
        
        // max 是之前的和
        // 其他数的和是 sum - max
        // 所以原数是 max - (sum - max);
        while(sum != target.length){
            long max = heap.poll();
            long origin = 2 * max - sum;
            
            // System.out.println(max + " " + origin);
            if(origin < 1) return false;
            heap.add(origin);
            sum -= max - origin;
        }        
        return true;
    }
}
