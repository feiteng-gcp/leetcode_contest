class Solution {
    public boolean isPossible(int[] target) {
        return compute(target);
    }
    
    
private boolean compute(int[] target) {
        long max = Integer.MIN_VALUE;
        int index = -1;
        long sum = 0L;
        for (int i = 0; i < target.length; i++) {
            sum += target[i];
            if (target[i] > max) {
                max = target[i];
                index = i;
            }
        }
        if (sum == target.length) return true;
        long origin = max - (sum - max);
        if (origin <= 0 || origin > Integer.MAX_VALUE) return false;
        target[index] = (int)origin;
        return compute(target);
    }
}
