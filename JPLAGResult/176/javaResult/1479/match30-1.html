class Solution {
	public static Comparator<Integer> cmp = new Comparator<Integer>() {
		@Override
		public int compare(Integer o1, Integer o2) {
			// TODO Auto-generated method stub
			return o2 - o1;
		}
	};
    public boolean isPossible(int[] target) {
        Queue<Integer> pq = new PriorityQueue<Integer>(cmp);
        
int count_sum=0;
        for (int i=0;i<target.length;i++) {
        	pq.add(target[i]);
        	count_sum += target[i];
        }
        while(! pq.isEmpty()) {
        	int num = pq.poll();
        	if (num == 1){
        		break;
        	}

        	int tar = num - (count_sum - num);
        	if ((tar <=0) || (count_sum == num)){
        		return false;
        	}
        	pq.add(tar);
        	count_sum = count_sum - num + tar;
        }
        return true;
    }
}
