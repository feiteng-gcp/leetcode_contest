class Solution {
    public boolean isPossible(int[] target) {
        Queue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
        long sum = 0;
        for (int n : target) {
            pq.offer(n);
            sum += n;
        }
        // int i = 0;
        while (pq.size() > 0) {
            int n = pq.poll();
            if (n == 1) return true;
            long pre = 2 * n - sum;
            sum = n;
            if (pre == 1) continue;
            
if (pre < 0) return false;
            pq.offer((int) pre);
            // i++;
            // if (i % 1000 == 0) System.out.println(i);
        }
        return true;
    }
}
