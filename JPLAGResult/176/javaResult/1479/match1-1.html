class Solution {
    public boolean isPossible(int[] target) {
        PriorityQueue<Long> queue = new PriorityQueue<>(Comparator.reverseOrder());
        long sum = 0;
        
for (long integer : target) {
            sum += integer;
            queue.add(integer);
        }
        while (true) {
            long last = queue.poll();
            if (last == 1) {
                return true;
            }
            long newNum = last * 2 - sum;
            if (newNum <= 0) {
                return false;
            }
            sum -= (last - newNum);
            queue.add(newNum);
        }
    }
}
