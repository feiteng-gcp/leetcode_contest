import java.util.PriorityQueue;


class Solution {
    public boolean isPossible(int[] target) {
        long sum = 0;
        PriorityQueue<Integer> queue = new PriorityQueue<>((o1, o2) -> o2 - o1);
        
for (int i : target) {
            sum += i;
            queue.add(i);
        }
        while (!queue.isEmpty()) {
            int current = queue.poll();
            if (current == 1) {
                return true;
            }
            sum -= current;
            if (current <= sum) {
                return false;
            }
            int next = (int) (current - sum);
            queue.add(next);
            sum+=next;
        }
        return false;
    }
}
