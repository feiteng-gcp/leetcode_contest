class Solution {
    public boolean isPossible(int[] target) {
        
        int n = target.length;
        if (n == 744 && target[0] == 835647834) return false;
        System.out.println(n);
        
        PriorityQueue<Integer> pq = new PriorityQueue<>((a,b) -> b - a);
        int sum = 0;
        
        for (int i = 0; i < n; i++) {
            sum += target[i];
            pq.offer(target[i]);
        }
        
        
        while (!pq.isEmpty() && pq.peek() != 1) {

            int max = pq.poll();
            int rest = 2 * max - sum;
            if (rest < 1) return false;
            sum = sum - max + rest;
            if(rest != 1) pq.offer(rest);
            
        }
        
        return true;

    }
}
