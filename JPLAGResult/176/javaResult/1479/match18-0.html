class Solution {
    public boolean isPossible(int[] target) {
        PriorityQueue<Long> pq = new PriorityQueue<Long>(Collections.reverseOrder());
        
        long sum = 0;
        for(int v : target) {
            sum += v;
            pq.add((long)v);
        }
        
        
        while(pq.peek() != 1) {
            long max = pq.peek();
            pq.poll();
            long remain = sum - max;
            long prevVal = max - remain;
            if(prevVal < 1) return false;
            pq.add(prevVal);
            sum = remain + prevVal;
        }
        
        return true;
    }
}
