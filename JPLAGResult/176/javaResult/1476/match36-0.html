class Solution {
    public int countNegatives(int[][] grid) {
        int res = 0;
        for(int r=0;r<grid.length;r++){
            for(int c=0;c<grid[0].length;c++){
                if(grid[r][c]<0){
                    res += grid[r].length - c;
                    break;
                } 
            }
        }
        return res;
    }
}
