class ProductOfNumbers {
    
    private List<Integer> nums = new ArrayList<>();

    public ProductOfNumbers() {
        
    }
    
    public void add(int num) {
        nums.add(num);
    }
    
    public int getProduct(int k) {
        int len = nums.size();
        int prod = 1;
        for (int i = 1; i <= k; i++) {
            prod *= nums.get(len - i);
        }
        return prod;
    }
}
