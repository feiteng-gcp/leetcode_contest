class ProductOfNumbers {

    List<Integer> nums;
    
    TreeMap<Integer, Integer> map;
    
    public ProductOfNumbers() {
        nums = new ArrayList<>();
        map = new TreeMap<>();
    }
    
    public void add(int num) {
        nums.add(num);
        map.clear();
    }
    
    public int getProduct(int k) {
        int idx = nums.size() - 1;
        int res = 1;
        
        int original = k;
        
        Integer pre = map.floorKey(k);
        
        
        if (pre == null) {

        } else {

            res = map.get(pre);
            if (res == 0) {
                map.put(k, 0);
                return 0;
            }
            k -= pre;
            idx -= pre;
        }
        
        while (k-- > 0) {
            res *= nums.get(idx--);
        }  
        
        map.put(original, res);

        return res;
    }
}
