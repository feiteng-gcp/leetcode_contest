class ProductOfNumbers {
    ArrayList<Integer> al;

    public ProductOfNumbers() {
        al = new ArrayList<Integer>();
    }
    
    public void add(int num) {
        al.add(num);
    }
    
    public int getProduct(int k) {
        int answer = 1;
        for (int i=al.size()-k; i<al.size(); i++) {
            answer *= al.get(i);
        }
        return answer;
    }
}
