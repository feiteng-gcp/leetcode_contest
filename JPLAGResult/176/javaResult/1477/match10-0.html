class ProductOfNumbers {

    ArrayList<Integer> nums;
    
    public ProductOfNumbers() {
        nums = new ArrayList<Integer>();
    }
    
    public void add(int num) {
        nums.add(num);
    }
    
    public int getProduct(int k) {
        int res = 1;
        for(int i = nums.size() - k; i < nums.size(); i++){
            res *= nums.get(i);
        }
        return res;
    }
}
