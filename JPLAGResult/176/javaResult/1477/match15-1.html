class ProductOfNumbers {
    
    List<Integer> list = null;

    public ProductOfNumbers() {
        list = new ArrayList<>();
    }
    
    public void add(int num) {
        list.add(num);
    }
    
    public int getProduct(int k) {
        int ans = 1;
        for(int i= list.size()-k;i<list.size();i++){
            ans *= list.get(i);
        }
        return ans;
    }
}

 * obj.add(num);
