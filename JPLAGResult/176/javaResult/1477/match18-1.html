class ProductOfNumbers {

    List<Integer> l;
    int p = 1, zeroIdx = -1;
    public ProductOfNumbers() {
        l = new ArrayList<>();
    }
    
    public void add(int num) {
        p *= num;
        if (num == 0) {
            zeroIdx = l.size();
            p = 1;
        }
        l.add(p);
    }
    
    public int getProduct(int k) {
        int pre = l.size() - 1 - k;
        if (pre < zeroIdx) return 0;
        return l.get(l.size() - 1) / (pre < 0 ? 1 : l.get(pre));
    }
}
