class ProductOfNumbers {
    
    List<Integer> list;
    
    public ProductOfNumbers() {
        list = new ArrayList<Integer>();
    }
    
    public void add(int num) {
        list.add(num);
    }
    
    public int getProduct(int k) {
        int ans = 1;
        for(int i=list.size()-1; i>=list.size()-k; i--) {
            ans *= list.get(i);
        }
        return ans;
    }
}
