class ProductOfNumbers {

    List<Integer> pro;
    List<Integer> zeros;
    public ProductOfNumbers() {
        pro = new ArrayList<>();
        zeros = new ArrayList<>();
        pro.add(1);
    }
    
    public void add(int num) {
        if (num == 0) {
            pro.add(1);
            zeros.add(pro.size());
            return;
        }
        pro.add(pro.get(pro.size() - 1) * num);
    }
    
    public int getProduct(int k) {
        //whether there is zero in the last k numbers
        if (zeros.size() > 0 && zeros.get(zeros.size() - 1) > pro.size() - k) {
            return 0;
        }
        return pro.get(pro.size() - 1) / pro.get(pro.size() - k - 1);
    }
}
