class ProductOfNumbers {
    
    List<Integer> list;
    public ProductOfNumbers() {
        list=new ArrayList<>();
    }
    
    public void add(int num) {
        list.add(num);
    }
    
    public int getProduct(int k) {
        int res=1;
        int n=list.size();
        
for(int i=1;i<=k;i++){
            res*=list.get(n-i);
        }
        return res;
    }
}
