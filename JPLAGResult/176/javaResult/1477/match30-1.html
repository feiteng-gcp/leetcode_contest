class ProductOfNumbers {

  List<int[]> l;

  public ProductOfNumbers() {
    l = new ArrayList<>();
  }

  public void add(int num) {
    int[] cres;
    if(l.size() == 0) {
      cres = new int[101];
    } else {
      cres = l.get(l.size()-1).clone();
    }
    cres[num]++;
      // System.out.println(Arrays.toString(cres));
    
l.add(cres);
  }

  public int getProduct(int k) {
    int si = l.size() - k - 1;
    int res = 1;
    int[] o = (si >= 0) ? l.get(si) : new int[101];
    int[] n = l.get(l.size()-1);
    if(n[0] > o[0]) return 0;
    
for(int i=1; i<101; i++) {
      res *= (int) Math.pow(i, n[i] - o[i]);
    }
    return res;
  }
}
