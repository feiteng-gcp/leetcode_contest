class ProductOfNumbers {
        private List<Integer> list;
    public ProductOfNumbers() {
        this.list = new ArrayList<>();
    }
    
    public void add(int num) {
        list.add(num);
    }
    
    public int getProduct(int k) {
        int idx = list.size() - 1;
        int prod = 1;
        for (int i=0; i<k; i++) {
            prod *= list.get(idx-i);
        }
        return prod;
    }
}
