class Solution:
    def isPossible(self, target: List[int]) -> bool:
        q = [-x for x in target]
        s = sum(target)
        heapq.heapify(q)
        while q:
            if s == len(target):
                return True
            x = -heapq.heappop(q)
            y = 2*x-s
            if y < 1:
                return False
            s -= x-y
            heapq.heappush(q, -y)
        return False
