import heapq as pq


class Solution:
    def isPossible(self, target):
        total = 0
        q = []
        for num in target:
            total += num
            pq.heappush(q, -num)
        while q:
            if q[0] == -1:
                return True
            maxv = -pq.heappop(q)
            res = total-maxv
            nv = maxv-res
            if nv < 1:
                return False
            total = total-maxv+nv
            pq.heappush(q, -nv)
        return False
