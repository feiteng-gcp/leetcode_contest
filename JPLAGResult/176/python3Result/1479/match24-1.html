import heapq

class Solution:
    def isPossible(self, target: List[int]) -> bool:
        heap = [-x for x in target]
        summed = sum(heap)
        heapq.heapify(heap)
        
        while heap[0] < -1:
            current 
= heapq.heappop(heap)
            new = current - (summed - current)
            if new > -1:
                return False
            summed += (new - current)
            heapq.heappush(heap, new)
        return True
