class Solution:
    def isPossible(self, target: List[int]) -> bool:
        heap = [-x for x in target]
        heapq.heapify(heap)
        n = len(target)
        while heap:
            curr = heapq.heappop(heap)
            if curr == -1:
                return True
            total = sum(heap)
            left = curr - total
            if left > -1:
                return False
            heapq.heappush(heap, left)
        return True
