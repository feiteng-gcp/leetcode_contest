class Solution:
    def isPossible(self, target: List[int]) -> bool:
        target = [-i for i in target]
        heapq.heapify(target)
        cursum = sum(target)
        while(cursum<-len(target)):
            m = heapq.heappop(target)
            if 2*m - cursum >=0:
                return False
            heapq.heappush(target,2*m - cursum)
            cursum = m
        return cursum == -len(target)
