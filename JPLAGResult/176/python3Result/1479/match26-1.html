import heapq


class Solution:
    def isPossible(self, target: List[int]) -> bool:
        nums = []
        for num in target:
            heapq.heappush(nums, -num)
        total = sum
(target)

        while True:
            elem = -heapq.heappop(nums)
            if elem == 1:
                return True
            left = total - elem
            prev = elem - left
            if prev < 0:
                return False
            heapq.heappush(nums, -prev)
            total -= elem - prev
