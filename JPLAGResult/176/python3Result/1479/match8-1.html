class Solution:
    def isPossible(self, target: List[int]) -> bool:
        hq 
= []
        n = len(target)
        
        for i in range(n):
            heapq.heappush(hq, (-target[i],target[i]))
        
        res = sum(target)
        while True:
            if res == n:
                return True
            idx, cur = heapq.heappop(hq)
            k = res - cur
            cur -= k
            res -= k 
            if cur < 1:
                return False
            else:
                heapq.heappush(hq, (-cur, cur))
