class Solution:
    def isPossible(self, target: List[int]) -> bool:
        count = 0
        for num in target:
            if num == 1:
                count += 1
        L = len(target)
        A = [1] * L

        total = sum(target)
        for num in target:
            heapq.heappush(heap, -num)
        while heap:
            curr = -heapq.heappop(heap)
            nxt = curr * 2 - total
            # print(total, curr, nxt)
            if nxt > curr or nxt < 0: return False
            total = total - curr + nxt
            if nxt == 1:
                count += 1
            else:
                heapq.heappush(heap, -nxt)
            if count == L: return True
