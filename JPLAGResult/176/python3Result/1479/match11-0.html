class Solution:
    def isPossible(self, target: List[int]) -> bool:
        s = sum(target)
        h = []
        for x in target:
            heapq.heappush(h, -x)
        while h[0] < -1:
            #print(h)
            y = heapq.heappop(h)
            y = -y
            temp = 2*y - s
            if temp < 1:
                return False
            heapq.heappush(h, -temp)
            s = y
        #print(h)
        if set(h) == set([-1]):
            return True
        return False
