import heapq

class Solution:
    def isPossible(self, target: List[int]) -> bool:
        if len(target)==1:
            return target==[1]
        s,t=sum(target),[-x for x in target]
        heapq.heapify(t)
        while True:
            x=-heapq.heappop(t)
            if x==1:
                return True
            n=(x-1)//(s-x)
            x,s=x-(s-x)*n,s-(s-x)*n
            if x<1 or n==0:
                return False
            heapq.heappush(t,-x)
