import heapq 
class Solution:
    def pop(self):
        return -heapq.heappop(self.heap)

    def push(self, x):
        heapq.heappush(self.heap, -x)

    def max(self):
        return -self.heap[0]

    def build(self, xs):
        self.heap = [-x for x in xs]
        heapq.heapify(self.heap)

    def isPossible(self, target: List[int]) -> bool:
        self.build(target)
        total = sum(target)
        while self.max() != 1:
            top = self.pop()
            rest = total - top
            new = top - rest 
            if new < 1:
                return False
            self.push(new)
            total = rest + new
        
        return True
