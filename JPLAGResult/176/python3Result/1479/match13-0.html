class Solution:
    def isPossible(self, target: List[int]) -> bool:
        
        N = len(target)
        count = N
        for num in target:
            if num == 1:
                count -= 1
        heap = []
        total = sum(target)
        
        for num in target:
            heapq.heappush(heap, -num)
            
        while heap:
            curr = -heapq.heappop(heap)
            nxt = curr * 2 - total
            if nxt > curr or nxt < 0: 
                return False
            total = total - curr + nxt
            if nxt == 1:
                count -= 1
            else:
                heapq.heappush(heap, -nxt)
            if count == 0: return True
