class Solution:
    def isPossible(self, target: List[int]) -> bool:
        target.sort()
        ss = sum(target)
        while target[-1] > 1:
            largest 
= target.pop()
            ss -= largest
            largest -= ss
            if largest < 1:
                return False
            bisect.insort(target, largest)
            ss += largest
        return True
