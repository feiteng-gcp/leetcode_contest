class Solution:
    def isPossible(self, target: List[int]) -> bool:
        heap = [-x for x in target]
        heapq.heapify(heap)
        s = sum(target)
        while s > len(target):
            x = -heapq.heappop(heap)
            d = s - x
            x -= d
            if x < 1:
                return False
            s -= d
            heapq.heappush(heap, -x)
        return True
