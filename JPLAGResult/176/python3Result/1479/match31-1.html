class Solution:
    def isPossible(self, target: List[int]) -> bool:
        sums = sum(target)
        
        target = [-x for x in target]
        heapq.heapify(target)
        
        
while target[0] != -1:
            maxi = -heapq.heappop(target)
            
            others = sums - maxi
            prev_value = maxi - others
            
            if prev_value < 1:
                return False
            
            heapq.heappush(target, -prev_value)
            sums = others + prev_value
        
        return True
