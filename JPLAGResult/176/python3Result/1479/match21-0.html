class Solution:
    def isPossible(self, target: List[int]) -> bool:
        heap = []
        s = sum(target)
        for t in target:
            heapq.heappush(heap, -t)
        while heap and heap[0] != -1:
            t = -heapq.heappop(heap)
            n = 2* t -s
            s = t
            if n < 1:
                return False
            heapq.heappush(heap, -n)
        return True
