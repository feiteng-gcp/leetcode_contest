class Solution:
    def isPossible(self, target: List[int]) -> bool:
        cursum = 0
        maxheap = []
        for num in target:
            cursum += num
            heapq.heappush(maxheap, -num)
        
        while cursum>len(target):
            curmax = -heapq.heappop(maxheap)
            cursum -= curmax
            othersum = curmax - cursum
            if othersum <= 0:
                return False
            heapq.heappush(maxheap, -othersum)
            cursum += othersum
        return maxheap == [-1]*len(target)
