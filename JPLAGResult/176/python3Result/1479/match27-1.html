class Solution:
    def isPossible(self, t: List[int]) -> bool:
        sm = sum(t)
        arr = [-i for i in t]
        sm = -sm
        heapq
.heapify(arr)
        while 1:
            t = heapq.heappop(arr)
            if t == -1:
                return True
            t1 = t
            t = 2 * t - sm
            if t > -1:
                return False
            sm = sm - t1 + t
            heapq.heappush(arr, t)
