class Solution:
    def isPossible(self, target: List[int]) -> bool:
        s = sum(target)
        L = [-x for x in target]
        heapq.heapify(L)
        while True:
            x = heapq.heappop(L)
            if x==-1: return True
            s += x
            y = x+s
            s -= y
            if y>=0: return False
            heapq.heappush(L,y)
