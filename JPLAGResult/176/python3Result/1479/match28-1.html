class Solution:
    def isPossible(self, target: List[int]) -> bool:
        a = [-x for x in target]
        heapq.heapify(a)
        s = sum(target)
        while a[0] != -1:
            x = -a[0] * 2 - s
            if x < 1:
                return False
            s += a[0] + x
            heapq.heapreplace(a, -x)
        return True
