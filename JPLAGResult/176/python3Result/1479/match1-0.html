class Solution:
    def isPossible(self, target: List[int]) -> bool:
        import heapq
        heap = []
        size = len(target)
        for i in range(size):
            heapq.heappush(heap, (-target[i],target[i]))
        ans = sum(target)
        while True:
            if ans == size:
                return True
            tmp, cur = heapq.heappop(heap)
            k = ans - cur
            cur -= k
            ans -= k 
            if cur < 1:
                return False
            else:
                heapq.heappush(heap, (-cur, cur))
