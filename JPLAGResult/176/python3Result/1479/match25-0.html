import heapq

class Solution:
    def isPossible(self, target: List[int]) -> bool:
        target_sum = sum(target)
        for i, v in enumerate(target):
            target[i] = -v
        heapq.heapify(target)
        while target_sum > len(target):
            v = -heapq.heappop(target)
            new_v = 2 * v - (target_sum)
            if new_v < 1:
                return False
            target_sum -= (v - new_v)
            heapq.heappush(target, -new_v)
        return True
