import heapq

class Solution:
    def isPossible(self, target: List[int]) -> bool:
        curr_sum = sum(target)
        heap = [-n for n in target]
        heapq
.heapify(heap)
        while True:
            prev_sum = -heapq.heappop(heap)
            if prev_sum == 1:
                return True
            diff = curr_sum - prev_sum
            old = prev_sum - diff
            # print(old, prev_sum, curr_sum)
            if old < 1:
                return False
            heapq.heappush(heap, -old)
            curr_sum = prev_sum
