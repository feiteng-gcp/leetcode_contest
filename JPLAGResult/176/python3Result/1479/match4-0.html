class Solution:
    def isPossible(self, target: List[int]) -> bool:
        import heapq
        
        heap = []
        n = len(target)
        
        for i in range(n):
            heapq.heappush(heap, (-target[i],target[i]))
        
        
        ans = sum(target)
        
        while True:
            if ans == n:
                return True
            idx, cur = heapq.heappop(heap)
            k = ans - cur
            cur -= k
            ans -= k 
            if cur < 1:
                return False
            else:
                heapq.heappush(heap, (-cur, cur))
