import heapq


class Solution:
    def isPossible(self, target: List[int]) -> bool:
        target = [-t for t in target]
        heapq.heapify(target)
        total = -sum(target)
        while True:
            # print(target)
            top = -heapq.heappop(target)
            if top == 1:
                return True
            rest = total - top
            if top <= rest:
                return False
            total -= (top // rest) * rest
            top %= rest
            heapq.heappush(target, -top)
        return True
