class ProductOfNumbers:

    def __init__(self):
        self.ps = [1]
        

    def add(self, num: int) -> None:
        if num == 0:
            self.ps = [1]
        else:
            self.ps.append(self.ps[-1] * num)

    def getProduct(self, k: int) -> int:
        if k >= len(self.ps):
            return 0
        return int(self.ps[-1] / self.ps[-k-1])
