class ProductOfNumbers:

    def __init__(self):
        self.lst = [1]

    def add(self, num: int) -> None:
        if num == 0:
            self.lst = [1]
        else:
            self.lst.append(self.lst[-1] * num)

    def getProduct(self, k: int) -> int:
        if len(self.lst) > k:
            return self.lst[-1] // self.lst[-k-1]
        return 0
