class ProductOfNumbers:

    def __init__(self):
        self.nums = [1]
        self.zero = [0]

    def add(self, num: int) -> None:           
        self.nums.append(self.nums[-1] * num if num != 0 else self.nums[-1])
        self.zero.append(self.zero[-1] + (1 if num == 0 else 0))

    def getProduct(self, k: int) -> int:
        i = len(self.nums) - k - 1
        
if self.zero[-1] - self.zero[i] > 0:
            return 0
        return int(self.nums[-1] // self.nums[i])
