class ProductOfNumbers:

    def __init__(self):
        self.pro = [1]
        self.zerocount = [0]

    def add(self, num: int) -> None:
        if num==0:
            self.zerocount.append(self.zerocount[-1]+1)
            self.pro.append(self.pro[-1])
        else:
            self.zerocount.append(self.zerocount[-1])
            self.pro.append(self.pro[-1]*num)

    def getProduct(self, k: int) -> int:
        if self.zerocount[-1]>self.zerocount[-k-1]:
            return 0
        else:
            return int(self.pro[-1]/self.pro[-k-1])
