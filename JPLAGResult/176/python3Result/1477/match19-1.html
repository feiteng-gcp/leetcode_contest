class ProductOfNumbers:
    def __init__(self):
        self.cummul = [1]
        self.zero_cnt = [0]

    def add(self, num: int) -> None:
        if num == 0:
            self.cummul.append(self.cummul[-1] * 1)
            self.zero_cnt.append(self.zero_cnt[-1] + 1)
        else: 
            self.cummul.append(self.cummul[-1] * num)
            self.zero_cnt.append(self.zero_cnt[-1])

    def getProduct(self, k: int) -> int:
        if self.zero_cnt[-1] - self.zero_cnt[-k-1] > 0:
            return 0 
        else:
            return self.cummul[-1] // self.cummul[-k-1]
