class ProductOfNumbers:

    def __init__(self):
        self.a = [1]

    def add(self, num: int) -> None:
        if num != 0:
            self.a.append(self.a[-1] * num)
        else:
            self.a = [1]

    def getProduct(self, k: int) -> int:
        if len(self.a) > k:
            return self.a[-1] // self.a[-1-k]
        else:
            return 0
