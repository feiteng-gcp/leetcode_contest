class ProductOfNumbers:

    def __init__(self):
        self.nums = [1]
        self.memo = []

    def add(self, num: int) -> None:
        if num:
            self.nums.append(self.nums[-1]*num)
        else:
            self.memo.append(len(self.nums))
            self.nums.append(self.nums[-1])

    def getProduct(self, k: int) -> int:
        if self.memo and  len(self.nums)-self.memo[-1] <= k:
            return 0
        return self.nums[-1]//self.nums[-(k+1)]
