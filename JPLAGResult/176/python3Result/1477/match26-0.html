class ProductOfNumbers:

    def __init__(self):
        self.s=[None]
        self.z=[0]
        self.p=[1]

    def add(self, num: int) -> None:
        self.s.append(num)
        if num==0:
            self.z.append(self.z[-1]+1)
            self.p.append(1)
        else:
            self.z.append(self.z[-1])
            self.p.append(self.p[-1]*num)

    def getProduct(self, k: int) -> int:
        if self.z[-1]>self.z[-k-1]:
            return 0
        else:
            return self.p[-1]//self.p[-k-1]
