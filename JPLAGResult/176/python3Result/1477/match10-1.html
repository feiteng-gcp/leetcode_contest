class ProductOfNumbers:

    def __init__(self):
        self.list = []
        self.cache = {}

    def add(self, num: int) -> None:
        self.list.append(num)
        self.cache = {}
    
    
    def getProduct(self, k: int) -> int:
        if k in self.cache:
            return self.cache[k]
        
        pro = 1
        for e in self.list[-k:]:
            pro *= e
        
        self.cache[k] = pro
        return pro
