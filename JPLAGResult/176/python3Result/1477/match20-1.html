class ProductOfNumbers:

        self.cnt0 = [0]
        self.pro = [1]


    def add(self, num: int) -> None:
        self.cnt0.append(self.cnt0[-1] + (1 if num == 0 else 0))
        self
.pro.append(self.pro[-1] * (1 if num == 0 else num))

    def getProduct(self, k: int) -> int:
        if self.cnt0[-1] - self.cnt0[-k - 1]:
            return 0
        return self.pro[-1] // self.pro[-k - 1]
