from operator import mul
from functools import reduce
class ProductOfNumbers:

    def __init__(self):
        self.prod=[1]
        self.zero=[0]
        

    def add(self, num: int) -> None:
        
if num==0:
            self.zero.append(self.zero[-1]+1)
            self.prod.append(1)
        else:
            self.zero.append(self.zero[-1])
            self.prod.append(self.prod[-1]*num)
        

    def getProduct(self, k: int) -> int:  
        end=-1
        start=-k-1
        
if self.zero[end]!=self.zero[start]:
            return 0
        else:
            return self.prod[end]//self.prod[start]
