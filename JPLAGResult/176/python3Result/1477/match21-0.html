class ProductOfNumbers:

    def __init__(self):
        self.cumprod = [1]
        self.cumzero = [0]

    def add(self, num: int) -> None:
        self.cumzero.append(self.cumzero[-1] + int(num == 0))
        self.cumprod.append(self.cumprod[-1] * (num if num!=0 else 1))

    def getProduct(self, k: int) -> int:
        if self.cumzero[-1] - self.cumzero[-k-1] > 0:
            return 0
        return int(self.cumprod[-1]/self.cumprod[-k-1])
