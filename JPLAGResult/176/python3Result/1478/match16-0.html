class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        a = math.inf
        b = -math.inf
        events.sort()
        for x, y in events:
            a = min(a, x)
            b = max(b, y)
        
        h =[]
        ans = 0

        for s in range(a, b + 1):
            while events and events[0][0] == s:
                x, y = events.pop(0)     
                heapq.heappush(h, y)
            while h and h[0] < s:
                heapq.heappop(h)
            if h:
                heapq.heappop(h)
                ans += 1
        return ans
