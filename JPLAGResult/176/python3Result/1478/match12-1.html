class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events.sort()
        
        # print(events)
        
        res = 0
        last = -sys.maxsize
        pq = []
        
        i = 0
        while i < len(events):
            if not pq:
                last = events[i][0]
                res += 1
                i += 1
                
            while i < len(events) and events
[i][0] <= last + 1:
                heapq.heappush(pq, events[i][1])
                i += 1
            while pq and pq[0] <= last:
                heapq.heappop(pq)
            if pq:
                heapq.heappop(pq)
                res += 1
                last += 1
        
        while pq:
            while pq and pq[0] <= last:
                heapq.heappop(pq)
            if pq:
                heapq.heappop(pq)
                res += 1
                last += 1
        
        return res
