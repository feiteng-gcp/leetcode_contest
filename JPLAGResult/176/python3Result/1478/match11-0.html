class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events.sort(key=lambda x: x[0])
        q = []
        n = len(events)
        sol = 0
        i = 0
        T = max(e[1] for e in events)
        for t in range(T+1):
            while i<n and events[i][0] == t:
                heapq.heappush(q, events[i][1])
                i += 1

            while q and q[0] < t:
                heapq.heappop(q)
            if q:
                heapq.heappop(q)
                sol += 1
        return sol
