class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events.sort(key=lambda x: (x[1], x[0]))
        max_day = max(day[1] for day in events)
        days_busy = [0] * (max_day + 1)
        days_sum = [0] * (max_day + 1)
        for start, end in events:
            if days_sum[end] - days_sum[start-1] == end - start + 1:
                continue
            for d in range(start, end+1):
                days_sum[d] = days_sum[d-1]
                if not days_busy[d]:
                    days_busy[d] = 1
                    days_sum[d] += 1
                    break
        # print(days_sum)
        return sum(days_busy)
