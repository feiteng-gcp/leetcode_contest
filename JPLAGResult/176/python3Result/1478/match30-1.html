class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        result = 0
        events.sort(key=lambda x: x[0])
        n, i, day, h = len(events), 0, 0, []
        while i < n:
            if events[i][0] <= day:
                heapq.heappush(h, events[i][1])
                i += 1
            else:
                while h and day < events[i][0]:
                    x = heapq.heappop(h)
                    if x >= day:
                        result += 1
                        day += 1
                day = events[i][0]
                heapq.heappush(h, events[i][1])
                i += 1
        while h:
            x = heapq.heappop(h)
            if x >= day:
                result += 1
                day += 1
        return result
