import heapq
class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events.sort()
        current=[]
        now=1
        i=0
        cnt=0
        while i<len(events) or current:
            while i<len(events):
                start,end=events[i]
                if start<=now:
                    heapq.heappush(current,(end,start))
                else:
                    break
                i+=1
            #print(now,current,cnt)
            if current:
                heapq.heappop(current)
                cnt+=1
            now+=1
            while current and current[0][0]<now:
                heapq.heappop(current)
        return cnt
