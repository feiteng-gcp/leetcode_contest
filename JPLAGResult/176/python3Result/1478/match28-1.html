class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        d = collections.defaultdict(list)
        for s, e in events:
            d[s].append(e)
        d = sorted(d.items(), key=lambda kv: kv[0])
        s, i, h, ret = d[0][0], 1, d[0][1], 0
        heapq.heapify(h)
        while h:
            ret += 1
            heapq.heappop(h)
            s += 1
            while h and h[0] < s:
                heapq.heappop(h)
            if i < len(d) and (d[i][0] == s or not h):
                for e in d[i][1]:
                    heapq.heappush(h, e)
                s = d[i][0]
                i += 1
        return ret
