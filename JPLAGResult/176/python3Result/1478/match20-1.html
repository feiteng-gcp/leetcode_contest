class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        res = 0
        sd = 1
        index = 0
        events.sort()
        heap = []
        heapq.heapify(heap)
        while index < len(events):
            while index < len(events) and events
[index][0] == sd:
                heapq.heappush(heap, [events[index][1],events[index][0]])
                index += 1
            while heap != []:
                end, start = heapq.heappop(heap)
                if end >= sd:
                    res += 1
                    break
            sd += 1
        while heap != []:
            end, start = heapq.heappop(heap)
            if end >= sd:
                res += 1
                sd += 1
        return res
