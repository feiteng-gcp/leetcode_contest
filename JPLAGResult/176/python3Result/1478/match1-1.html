class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events.sort(key=lambda x: x
[1])
        d=set()
        for s,e in events:
            for i in range(s,e+1):
                if i not in d:
                    d.add(i)
                    break
        return len(d)
