from heapq import heappush, heappop

class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events.sort(key = lambda x: x[0])
        total_days = -float('inf')
        for event in events:
            total_days = max(total_days, event[1])
        
        heap = []
        d = 1
        ans = 0
        eid = 0
        while d <= total_days:
            if eid < len(events) and not heap:
                d = max(d, events[eid][0])
            while eid < len(events) and events[eid][0] <= d:
                heappush(heap, events[eid][1])
                eid += 1
            while heap and heap[0] < d:
                heappop(heap)
            if heap:
                heappop(heap)
                ans += 1
            d += 1
        return ans
