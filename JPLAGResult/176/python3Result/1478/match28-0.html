class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        ranges = collections.defaultdict(list)
        for a, b in events:
            ranges[a].append(b)
        newdays = sorted(ranges.keys())
        heap = []
        curd = newdays[0]
        res = 0
        newdays.append((10 ** 5) * 2 + 1)
        for i in range(len(newdays) - 1):
            nd = newdays[i]
            for t in ranges[nd]:
                heapq.heappush(heap, t)
            curd = nd
            while curd < newdays[i + 1]:
                while heap and heap[0] < curd:
                    heapq.heappop(heap)
                if not heap:
                    break
                heapq.heappop(heap)
                res += 1
                curd += 1
                
        return res
