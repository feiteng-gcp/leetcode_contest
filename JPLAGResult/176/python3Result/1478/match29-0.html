from heapq import heappush, heappop

class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events.sort(reverse=True)
        ans = 0
        q = []
        cur = 0
        
        while events or q:
            while events and events[-1][0] <= cur:
                s, e = events.pop()
                heappush(q, e)
            if not q:
                cur = events[-1][0]
            else:
                e = heappop(q)
                if cur <= e:
                    ans += 1
                    cur = cur + 1
        return ans
