class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events = sorted(events)
        num = 0
        i = 0
        heap = []
        for date in range(events[0][0], max(e[1] for e in events)+1):
            
while i < len(events) and events[i][0] == date:
                heapq.heappush(heap, events[i][1])
                i += 1
            while heap and heap[0] < date:
                heapq.heappop(heap)
            if heap:
                heapq.heappop(heap)
                num += 1
        return num
