class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events.sort(key = lambda x:(x[1], x[0]))
        
        occupied = set()
        for start, end in events:

            for time in range(start, end + 1):
                if time not in occupied:
                    occupied.add(time)
                    break
        print(occupied)
        return len(occupied)
