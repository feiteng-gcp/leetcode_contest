class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events.sort(key=lambda e:e[0])
        start = events[0][0]
        res = 0
        heap = []
        i = 0
        n = len(events)
        
        while i < n or heap:
            while i < n and events[i][0] <= start:
                heapq.heappush(heap, (events[i][1], events[i][0]))
                i += 1
            while heap:
                end, _ = heapq.heappop(heap)
                if end >= start:
                    res += 1
                    break
            start += 1
        
        return res
