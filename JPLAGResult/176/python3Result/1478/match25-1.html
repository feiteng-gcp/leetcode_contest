class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        import heapq
        events.sort()
        N = len(events)
        e_ptr = 0
        ends = []
        maxE = 0
        
        for t in range(1, 100001):
            # print(t)
            while e_ptr < N and events
[e_ptr][0] <= t:
                heapq.heappush(ends, events[e_ptr][1])
                e_ptr += 1
            # print(ends)
            while len(ends) > 0 and ends[0] < t:
                heapq.heappop(ends)
            # print(ends)
            if len(ends) > 0:
                maxE += 1
                heapq.heappop(ends)
            # print(ends)
        
        return maxE
