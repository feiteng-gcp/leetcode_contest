class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        slots = set()
        res = 0
        events = sorted(events, key=lambda x: x[1])
        
for s, e in events:
            for t in range(s, e+1):
                if t not in slots:
                    slots.add(t)
                    res += 1
                    break
        return res
