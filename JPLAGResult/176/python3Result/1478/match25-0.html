import heapq
class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        if len(events) == 0:
            return 0

        heap = []
        events = sorted(events, key=lambda e: ((e[0], e[1])))
        cur = 0
        result = 0
        for i in range(1, 100000 + 1):
            while cur < len(events) and events[cur][0] <= i:
                heapq.heappush(heap, events[cur][1])
                cur += 1
            
            while len(heap) > 0 and heap[0] < i:
                heapq.heappop(heap)
            
            if len(heap) > 0 and heap[0] >= i:
                heapq.heappop(heap)
                result += 1
        
        return result
