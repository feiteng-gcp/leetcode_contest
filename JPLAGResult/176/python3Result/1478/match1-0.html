class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events.sort(key=lambda x: [x[1], x[0]])
        d = set()
        for start, end in events:
            for i in range(start, end + 1):
                if i not in d:
                    d.add(i)
                    break
        return len(d)
