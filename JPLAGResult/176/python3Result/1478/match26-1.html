class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        
        events = sorted(events)
        prev = 1
        i = 0
        q = []
        res = 0
        
        while i < len(events) or q:
            # print(prev)
            while i < len(events) and events[i][0] <= prev:
                heapq.heappush(q, (events[i][1], events[i][0]))
                i += 1
            # print(q)
            while q and q[0]
[0] < prev:
                heapq.heappop(q)
            # print(q)
            if q:
                heapq.heappop(q)
                res += 1
            # print(q)
            prev += 1
        return res
