class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        events.sort(reverse=True)
        count = 0
        d = 1
        q = []
        while events or q:
            while events and events[-1][0] <= d:
                s, e = events.pop()
                heapq.heappush(q, e)
            while q and q[0] < d:
                heapq.heappop(q)
            if q:
                count += 1
                heapq.heappop(q)
            d += 1
            #print(d, q, events)
        return count
