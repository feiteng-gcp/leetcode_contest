class Solution:
    def maxEvents(self, events: List[List[int]]) -> int:
        res = 0

        events.sort(key=lambda x: x[1])

        used = set()
        for start, end in events:
            for i in range(start, end + 1):
                if i not in used:
                    used.add(i)
                    res += 1
                    break

        return res
