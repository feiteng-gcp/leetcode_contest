class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        unordered_map<int, unordered_multiset<int>> um;
        for (auto e : events) {
            um[e[0]].insert(e[1]);
        }
        int res = 0;
        priority_queue<int, vector<int>, greater<int>> pq;
        for (int i = 1; i <= 100000; i++) {
            for (auto d : um[i])
                pq.push(d);
            while (pq.size() > 0 && pq.top() < i)
                pq.pop();
            if (pq.size() > 0) {
                pq.pop();
                res++;
            }
        }
        return res;
    }
};
