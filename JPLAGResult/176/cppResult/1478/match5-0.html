class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        int max_day = 0;
        sort(events.begin(), events.end(), [](auto& a, auto& b)->bool{
            return a[0] < b[0];
        });
        for (auto& e: events) {
            max_day = max(max_day, e[1]);
        }
        priority_queue<int, std::vector<int>, std::greater<int>> c;
        int day = 1;
        int ptr = 0;
        int ret = 0;
        while (day <= max_day) {
            while (ptr < events.size() && events[ptr][0] <= day) {
                c.push(events[ptr][1]);
                ptr++;
            }
            while (c.size() > 0 && c.top() < day) {
                c.pop();
            }
            if (c.size() > 0 && c.top() >= day) {
                // cout << c.top() << " " << day << endl;
                ret++;
                c.pop();
            }
            day++;
        }
        return ret;
    }
};
