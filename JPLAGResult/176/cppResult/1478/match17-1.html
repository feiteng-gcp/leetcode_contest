class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        sort(events.begin(), events.end());
        int start = 100005;
        int end = 1;
        for(auto it:events)
        {
            for(auto i:it)
            {
                start = min(start, i);
                end = max(end, i);
            }

        }
        int id = 0;
        priority_queue<int,vector<int>, greater<int>> pq;
        int res = 0;
        for(int i=start;i<=end+1;i++)
        {
            while(id<events.size()&&events[id][0]==i)
            {
                pq.push(events[id][1]);
                id++;
            }
            while(!pq.empty()&&pq.top()<i)
            {
                pq.pop();
            }
            if(!pq.empty()){
                //cout<<pq.top()<<endl;
                pq.pop();
                res ++;
            }
        }
        return res;
    }
};
