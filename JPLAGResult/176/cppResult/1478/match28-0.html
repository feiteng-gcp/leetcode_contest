class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        vector<int> days(100001,0);
        sort(events.begin(),events.end(),[](const vector<int> &a,const vector<int> &b){return a[1]<b[1];});
        int ans=0;
        for(auto event:events){
            int i=event[0];
            int j=event[1];
            for(int k=i;k<=j;++k){
                if(days[k]==0){
                    ++ans;
                    days[k]=1;
                    break;
                }
            }
        }
        return ans;
    }
};
