class Solution {
public:
    struct Meet{
        int st, ed;
        // bool operator <()
    };
    
    static bool cmp(Meet a, Meet b){
        if(a.ed != b.ed){
            return a.ed < b.ed;
        }
        else{
            return a.st < b.st;
        }
    }
    
    int maxEvents(vector<vector<int>>& events) {
        vector<Meet> m;
        for(int i = 0; i < events.size(); i++){
            Meet tmp;
            tmp.st = events[i][0];
            tmp.ed = events[i][1];
            m.push_back(tmp);
        }
        
        sort(m.begin(), m.end(), cmp);
        vector<int> flag(100005, 1);
        

        int ans = 0, ed = 1;
        for(int i = 0; i < m.size(); i++){
            for(int j = m[i].st; j <= m[i].ed; j++){
                if(flag[j]){
                    flag[j] = 0;
                    ans++;
                    break;
                }
            }
        }
        
        return ans;
    }
};
