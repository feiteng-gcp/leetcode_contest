class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        sort(events.begin(), events.end());
        priority_queue<int, vector<int>, greater<int>> pq;

        int pos = 0;
        int cnt = 0;
        for (int i = 0; i < events.size(); i++) {
            int b = events[i][0];
            int e = events[i][1];

            // [[1,2],[1,2],[1,5],[1,5],[3,3]]
            while (pos < b) {
                if (!pq.empty()) {
                    if (pq.top() >= pos) {
                        cnt++;
                    }
                    pq.pop();
                }
                pos++;
            }

            pq.push(e);

            if (pos > b) {
                continue;
            }
            
            while (!pq.empty() && pq.top() < pos) {
                pq.pop();
            }
            
            if (!pq.empty()) {
                pq.pop();
                cnt++;
                pos++;
            }
        }
        while (!pq.empty()) {
            if (pq.top() >= pos) {
                cnt++;
                pos++;
            }
            pq.pop();
        }
        return cnt;
    }
};
