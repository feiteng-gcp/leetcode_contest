class Solution {
private: 
	static int cmp(vector<int>& v1, vector<int>& v2){
		return v1[0]<v2[0];	
	}
	priority_queue<int,vector<int>,greater<int> >pq;
public:
    int maxEvents(vector<vector<int>>& vc) {
        sort(vc.begin(),vc.end(),cmp);
        int pos=0,ans=0,M=0;

        for(int i=0;i<vc.size();++i) M=max(M,vc[i][1]);
        for(int i=1;i<=M;++i)
        {
        	while(pos<vc.size() && vc[pos][0]<=i) {
				pq.push(vc[pos][1]);
				++pos;	
			}
			while(!pq.empty() && pq.top()<i) pq.pop();
			if(!pq.empty())
			{
				++ans;
				pq.pop();
			}
        }
        return ans;
    }
};
