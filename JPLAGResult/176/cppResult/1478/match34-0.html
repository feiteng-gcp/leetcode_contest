class Solution {
public:
    struct node {
        int s,t;
        node(int s,int t):s(s),t(t) {}
        bool operator<(const node &x)const {
            return s<x.s;
        }
    };
    
    int maxEvents(vector<vector<int>>& events) {
        int n=events.size();
        vector<node> e;
        for (auto i:events) {
            e.push_back(node(i[0],i[1]));
        }
        sort(e.begin(), e.end());
        priority_queue<int, vector<int>, greater<int>> pq;
        int ans=0, l=0;
  //      for (auto i:e) cout << i.s << endl;
        for (int i=1;i<=1e5;i++) {
            while (l<n&&e[l].s==i) {
                pq.push(e[l].t);
                l++;
            }
            while (!pq.empty()&&pq.top()<i) 
                pq.pop();
            if (!pq.empty()) {
                ans++;
                pq.pop();
            }
        }
        return ans;
    }
};
