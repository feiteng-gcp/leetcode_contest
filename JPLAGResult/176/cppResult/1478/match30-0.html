class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        priority_queue<int> pq;
        sort(events.begin(), events.end());
        int res=0, i=0;
        for(int cur=0;cur<=100000;++cur){
            while(i<events.size() && events[i][0]==cur){
                pq.push(-events[i][1]);
                ++i;
            }
            while(!pq.empty() && -pq.top()<cur) pq.pop();
            if(!pq.empty()){
                res++;
                pq.pop();
            }
        }
        return res;
    }
};
