class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        vector<pair<int, int>> p;
        for (auto it : events) {
            p.push_back(make_pair(it[1], it[0]));
        }
        sort(p.begin(), p.end());
        int n = p.size();
        int ans = 0;
        int vis[100010] = {0};

        for (int i = 0; i < n; i++) {
            int x = p[i].second;
            int y = p[i].first;
            for (int j = x; j <= y; j++) {
                if (vis[j] == 0) {
                    vis[j] = 1;
                    ans++;
                    break;
                }
            }
        }
        return ans;
    }
};
