class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        sort(events.begin(), events.end(), [](vector<int>& a, vector<int>& b) {
            return a[1] < b[1];
        });
        unordered_set<int> used;
        int res = 0;

        for (int i = 0; i < events.size(); i++) {
            for (int j = events[i][0]; j <= events[i][1]; j++) {
                if (used.find(j) == used.end()) {
                    res++;
                    used.insert(j);
                    break;
                }
            }
        }
        return res;
    }
};
