class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        sort(events.begin(), events.end());
        priority_queue<int, vector<int>, greater<int>> q;
        int ei = 0;
        int res = 0;
        for (int d = 1; d <= 100000; d++) {
            while (ei < events.size() && events[ei][0] == d) {
                q.push(events[ei][1]);
                ei++;
            }
            while (!q.empty() && q.top() < d) {
                q.pop();
            }
            if (!q.empty() && q.top() >= d) {
                q.pop();
                res++;
            }
        }
        return res;
    }
};
