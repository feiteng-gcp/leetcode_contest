class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        sort(events.begin(),events.end());
        priority_queue<int,vector<int>,greater<int>>  pq;
        int sd=0;
        int ans=0;
        for(int i=1;i<=100000;i++){
           while(sd<events.size() && events[sd][0]<=i){
               pq.push(events[sd][1]);
               sd++;
           }
           while(!pq.empty() && pq.top()<i) pq.pop();
            if(!pq.empty()){
               ans++;
               pq.pop();
            }
        }
        return ans;
    }
};
