class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {

        priority_queue<int, vector<int>, greater<int>> q;
        
        sort(events.begin(), events.end());
        int ret = 0;
        int last = 1;
        int i = 0;
        while (i < events.size() || q.size() > 0) {
            while (i < events.size() && events[i][0] == last) {
                q.push(events[i][1]);
                i++;
            }
            while (q.size() > 0 && q.top() < last) {
                q.pop();
            }
            if (q.size() != 0) {
                q.pop();
                ret++;
            }
            last++;
        }
        return ret;
    }
};
