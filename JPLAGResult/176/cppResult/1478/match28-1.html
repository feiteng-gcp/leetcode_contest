class Solution {
public:
    struct cmp
    {
        bool operator()(vector<int> eventA, vector<int> eventB)
        {
            if (eventA[1] < eventB[1]) return true;
            else return false;
        }
    } myobject;


    int maxEvents(vector<vector<int>>& events) {
        sort(events.begin(), events.end(), myobject);
        
        int maxDays = 0;
        for (auto event: events) {
            int end = event[1];
            maxDays = max(maxDays, end);
        }
        
        int count = 0;
        vector<int> mp(maxDays + 1, 0);
        

        for (auto event: events) {
            int start = event[0];
            int end = event[1];
            for (int i = start; i <= end; i++) {
                if (mp[i] == 0) {
                    mp[i] = 1;
                    count++;
                    break;
                }
            }
        }
        
        return count;
    }
};
