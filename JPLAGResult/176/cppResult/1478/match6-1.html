class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        int cnt = 0, n = events.size();
        sort(events.begin(), events.end());
        priority_queue<int, vector<int>, std::greater<int>> pq;
        int time = events[0][0], end = time;
        for (const auto& event : events) {
            end = max(end, event[1]);
        }

        int idx = 0;
        while (time <= end) {
            while (idx < n && events[idx][0] <= time) {
                pq.push(events[idx][1]);
                ++idx;
            }
            //events that cannot be done
            while (!pq.empty() && pq.top() < time) {
                pq.pop();
            }
            if (!pq.empty()) {
                ++cnt;
                pq.pop();
            }
            ++time;
        }
        return cnt;
    }
};
