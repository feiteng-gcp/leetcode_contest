class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        priority_queue<int, vector<int>, greater<int>> sime;
        sort(events.begin(), events.end());
        int i = 0;
        int now = 0;
        int ret = 0;
        while (true) {
            if (i == events.size() and sime.empty()) break;
            while (i < events.size() and events[i][0] <= now) {
                sime.push(events[i][1]);
                i++;
            }
            while (sime.size() and sime.top() < now) sime.pop();
            if (sime.size() and sime.top() >= now) {
                ret++;
                sime.pop();
            }
            now++;
        }
        return ret;
    }
};
