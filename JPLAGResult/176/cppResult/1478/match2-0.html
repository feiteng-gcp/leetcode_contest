class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        int max_val = 0;
        for (auto event: events)
            max_val = max(max_val, event[1]);
        
        vector<vector<int>> gevents(max_val+1);
        for (auto event: events)
            gevents[event[0]].push_back(event[1]);
        
        multiset<int> mset;
        
        int ans = 0, curr_end = 0;
        for (int i=0; i<=max_val; i++) {
            for (auto end: gevents[i])
                mset.insert(end);
            while (!mset.empty() && *(mset.begin()) < i)
                mset.erase(mset.begin());
            if (!mset.empty()) {
                ans++;
                // curr_end = max(curr_end + 1, *(mset.begin()) + 1);
                mset.erase(mset.begin());
            }
        }
        return ans;
    }
};
