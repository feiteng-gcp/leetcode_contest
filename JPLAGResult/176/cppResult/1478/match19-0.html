bool cmp(vector<int> a,vector<int> b){
    if(a[1] != b[1])return a[1] < b[1];
    return a[0] < b[0];
    
}
class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        sort(events.begin(), events.end(), cmp);
        map<int,bool> m;
        for (int i=0;i<events.size();i++){
            for(int j=events[i][0];j<=events[i][1];j++){
                if(m.find(j) == m.end()){
                m[j] = 1;
                break;
                }
            }
        }
        return m.size();
    }
};
