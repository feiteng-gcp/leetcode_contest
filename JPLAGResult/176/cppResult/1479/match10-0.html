class Solution {
public:
  bool isPossible(vector<int>& target) {
    priority_queue<long long> q;
    int n = target.size();
    long long sum = 0;
    for (auto x : target) {
      sum += x;
      q.emplace(x);
    }
    while (q.top() != 1) {
      long long x = q.top();
      q.pop();
      long long osum = sum - x;
      x = x - osum;
      if (x < 1) return false;
      sum = osum + x;
      q.emplace(x);
    }
    return true;
  }
};
