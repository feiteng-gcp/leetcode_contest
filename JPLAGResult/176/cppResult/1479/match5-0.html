class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<long long> Q;
		long long sum = 0;
		
		for(int i = 0; i < target.size(); i++){
			sum += target[i];
		 Q.push(target[i]);
		}
		
		while(Q.size()){
			long long x = Q.top(); Q.pop();
			long long y = x - (sum - x);
			sum = sum - (x - y);
			if(x == 1) break;
			if(y <= 0) return false;
			Q.push(y);
			
		}
		
		return true;
    }
};
