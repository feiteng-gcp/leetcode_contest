class Solution {
public:
	bool isPossible(vector<int> target) {
		priority_queue<long long>q;
		long long sum = 0;
		for (auto t : target)
		{
			q.push(t);
			sum += t;
		}

		while (!q.empty())
		{
			auto top = q.top();
			if (top == 1)
				return true;
			q.pop();
			auto other_sum = sum - top;
			auto rest = top - other_sum;
			if (rest < 0)
				return false;
			q.push(rest);
			sum = other_sum + rest;
		}
		return true;
	}
};
