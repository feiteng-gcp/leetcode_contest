class Solution {
public:
    bool isPossible(vector<int> &target) {
        long long sum = accumulate(target.begin(), target.end(), 0ll);
        priority_queue<long long> pq(target.begin(), target.end());
        while (!pq.empty()) {
            int n = pq.top();
            pq.pop();
            if (n == 1) return true;
            long long prev = n - (sum - n);
            if (prev < 0) return false;
            sum = sum - n + prev;
            pq.push(prev);
        }
        return true;
    }
};
