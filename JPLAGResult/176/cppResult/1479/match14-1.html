class Solution {
public:
    bool isPossible(vector<int>& target) {
        multiset<long long> ms;
        long long sum = 0;
        for (auto i: target) {
            ms.insert(i);
            sum += i;
        }
        while (true) {
            long long mx = *ms.rbegin();
            ms.erase(prev(ms.end()));
            long long nn = mx - (sum-mx);
            if (nn <= 0) return false;
            ms.insert(nn);
            sum = sum - mx + nn;
            if (sum == target.size()) return true;
        }
    }
};
