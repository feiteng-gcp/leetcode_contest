typedef long long ll;
class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<ll> pq;
        ll s = 0;
        for (int x : target) {
            s += x;
            pq.push(x);
        }
        while (pq.top() > 1) {
            ll x = pq.top(); 
            pq.pop();
            ll y = x - (s - x);     
            if (y < 1)
                return 0;
            s -= (x - y);
            pq.push(y);
        }
        return 1;
    }
};
