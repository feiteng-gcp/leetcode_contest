class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<long long> q;
        long long total = 0;
        
        for (int i = 0; i < target.size(); ++i) {
            q.push(target[i]);
            total += target[i];
        }
        while (q.top() > 1) {
            long long tmp = q.top();
            q.pop();
            total -= tmp;
            tmp -= total;
            if (tmp < 1) {
                return false;
            }
            total += tmp;
            q.push(tmp);
        }
        return true;
    }
};
