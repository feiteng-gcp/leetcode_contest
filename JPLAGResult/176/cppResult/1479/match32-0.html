#include <set>
typedef long long LL;

class Solution {
public:
    bool isPossible(vector<int>& a) {
        LL sum = 0;
        multiset<LL> b;
        for (auto &x: a) {
            sum += x;
            b.insert((LL) x);
        }
        while (!(*b.rbegin() == 1)) {
            LL d = *b.rbegin();
            b.erase(d);
            if (2 * d - sum < 1) return 0;
            b.insert(2 * d - sum);
            sum -= sum - d;
        }
        return 1;
    }
};
