typedef long long LL;

class Solution {
public:
    bool isPossible(vector<int>& target) {
        multiset<LL> S;
        LL sum = 0;
        for (auto v : target) {
            S.insert(-v);
            sum += v;
        }
        while (S.size() > 0) {
            auto it = S.begin();
            if (*it >= 0) return false;
            if (*it == -1) {
                for (auto x : S)
                    if (x != -1) return false;
                return true;
            }
            auto v = -(*it);
            S.erase(it);
            LL d = sum - v;
            if (d == 0) return false;
            //printf("v = %lld, d = %lld, insert %lld\n", v, d, v-d);
            S.insert(-(v - d));
            sum -= d;
        }
        //printf("hi?");
        return false;
    }
};
