class Solution {
public:
    bool isPossible(vector<int>& target) {
        int n = target.size();
        if(n == 1) return target[0] == 1;
        priority_queue<long> q;
        long sum = 0;
        for(int num : target){
            sum += num;
            q.push(num);
        }
        while(!q.empty() && q.top() > 1){
            long num = q.top(); q.pop();
            if(sum <= num) return false;
            long d = sum - num;
            long t = num - d;
            if(t <= 0) return false;
            sum = num;
            q.push(t);
        }
        if(q.size() != n) return false;
        while(!q.empty()) {
            if(q.top() != 1) return false;
            q.pop();
        }
        return true;
    }
};
