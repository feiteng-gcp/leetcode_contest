class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<int> q;
        long long sum = 0;
        for(int i = 0; i < target.size(); ++i) {
            q.push(target[i]);
            sum += target[i];
        }
        while(q.top() > 1) {
            long long tmp = q.top();
            q.pop();
            //cout << tmp << endl;
            if(tmp - (sum - tmp) < 1) return false;
            q.push(tmp - (sum - tmp));
            sum = tmp;
        }
        return true;
    }
};
