class Solution {
public:
    long long sum=0;
    bool isPossible(vector<int>& target) {
        int n=target.size();
        priority_queue<int>q;
        for(int i=0;i<n;i++)q.push(target[i]),sum+=target[i];

        while(!q.empty()){
            int x=q.top(); q.pop();
            if(x==1)return true;
            long long sub=sum-x;
            if(x-sub<1)return false;
            sum=sum-sub;
            q.push(x-sub);
        }
        return false;
    }
};
