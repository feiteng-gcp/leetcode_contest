class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<long long> pq;
        long long s(0);
        for (int i: target) {
            pq.push((long long)i);

            s += i;
        }
        
        while (pq.top() != 1) {
            long long t = pq.top();
            pq.pop();
            long long nt = t - (s - t);
            if (nt <= 0) return false;
            s = s - t + nt;
            pq.push(nt);
        }
        
        return true;
    }
};
