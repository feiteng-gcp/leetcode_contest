class Solution {
public:
    
    bool isPossible(vector<int>& target) {
        priority_queue<int> q;
        long long sum=0;
        for(auto tp: target)
        {
            sum+=tp;
            q.push(tp);
        }
        
        while(!q.empty())
        {
            long long cur = q.top();
            if(cur==1)
                return true;
            q.pop();
            long long nxt = (cur-(sum-cur));
            if(nxt<=0)
                return false;
            q.push(nxt);
            sum = (sum-cur)+nxt;
        }
        return true;
    }
};
