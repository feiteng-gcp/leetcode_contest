class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<int> q;
        long long s = 0;
        for (int i = 0;i < target.size();i++) {
            q.push(target[i]);
            s += target[i];
        }
        while (1) {
            if (q.top() == 1) {
                return true;
            }
            long long p = q.top();
            q.pop();
            long long x = p - (s - p);
            s = p;
            if (x <= 0) {
                return false;
            }
            q.push(x);
        }
        return false;
    }
};
