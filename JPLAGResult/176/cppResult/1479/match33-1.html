class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<int> p=priority_queue<int>();
        int n=target.size();
        long long sum=0;
        for(int i=0;i<n;i++){
            p.push(target[i]);
            sum+=target[i];
        }

        while(1){
            int v=p.top();
            if(v==1) return true;
            long long remain=sum-v;
            if(v-remain<1) return false;
            sum=sum-v+(v-remain);
            p.pop();
            p.push(v-remain);
        }
        return false;
    }
};
