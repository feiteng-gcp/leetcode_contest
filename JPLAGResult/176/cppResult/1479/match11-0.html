class Solution {
public:
    bool isPossible(vector<int>& A) {
        int n=A.size();
        long long sum=0;
        priority_queue<int> Q;
        for (int i=0;i<n;++i) sum+=A[i],Q.push(A[i]);
        while (!Q.empty())
        {
            int x=Q.top();Q.pop();
            if (x==1) return true;
            long long a=x*2-sum;
            if (a<=0) return false;
            Q.push(a);
            sum=sum-x+a;
        }
        return true;
    }
};
