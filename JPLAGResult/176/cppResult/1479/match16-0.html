class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<int> q;
        long long sum = 0;
        for (int i: target) {
            sum += i;
            q.push(i);
        }
        while (q.top() > 1) {
            long long ma = q.top();
            q.pop();
            long long bf = 2 * ma - sum;
            if (bf < 1) return false;
            sum -= ma - bf;
            q.push(bf);
        }
        return true;
    }
};
