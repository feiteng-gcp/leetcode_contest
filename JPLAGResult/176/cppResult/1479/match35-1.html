class Solution {
public:
    bool isPossible(vector<int>& target) {
        int size = target.size();

        long long sum = accumulate(target.begin(), target.end(), static_cast<long long>(0));
        auto m = max_element(target.begin(), target.end());
        while(*m != 1) {
            long long left = sum - *m;
            sum -= left;
            if(*m - left < 1)
                return false;
            *m -= left;
            m = max_element(target.begin(), target.end());
        }
        return true;
    }
};
