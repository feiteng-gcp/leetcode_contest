class Solution {
public:
    bool isPossible(vector<int>& target) 
    {
        long long sum = 0;
        priority_queue<int>pq;
        for (int i=0; i<target.size(); i++)
        {
            pq.push(target[i]);
            sum+=target[i];
        }
        
        while (pq.top()!=1)
        {
            long long x = pq.top();
            pq.pop();
            long long y = x-(sum-x);
            sum -= (x-y);
            if (y<=0) return false;
            pq.push(y);
        }
        return true;
    }
};
