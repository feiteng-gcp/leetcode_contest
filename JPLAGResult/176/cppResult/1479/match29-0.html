class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<int> q;
        int n = target.size();
        long long sum = 0;
        for(int i = 0; i < n; i++)
        {
            q.push(target[i]);
            sum += target[i];
        }
        while(1)
        {
            int t = q.top();
            q.pop();
            if(t == 1) return true;
            if(t<1) return false;
            long long dif = sum - t;
            if(t-dif<1) return false;
            q.push(t-dif);
            sum -= dif;
        }
        return false;
    }
};
