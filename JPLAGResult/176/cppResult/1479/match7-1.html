class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<long long> q;
        long long  sum = 0;

        for(int i = 0; i < target.size(); i++)
        {
            sum += target[i];
            q.push(target[i]);
        }
        while(q.top() > 1)
        {
            long long  t = q.top();
            q.pop();
            long long now = t - (sum - t);
            sum = sum - t + now;
            if(now < 1) return false;
            q.push(now);
        }
        while(!q.empty())
        {
            if(q.top() != 1)return false;
            q.pop();
        }
        return true;
    }
};
