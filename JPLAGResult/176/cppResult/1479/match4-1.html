using ll = long long;
class Solution {
public:
    bool isPossible(vector<int>& target) {
        ll tot = 0;
        priority_queue<ll> q;

        for(auto i:target){
            tot += i;
            q.push(i);
        }
        while(q.top() > 1){
            ll x = q.top(); q.pop();
            ll y = x*2 - tot;
            if(y <= 0) return false;
            q.push(y);
            tot = tot - x + y;
        }
        return true;
    }
};
