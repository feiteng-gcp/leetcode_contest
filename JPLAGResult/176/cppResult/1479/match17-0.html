class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<long long> pq;
        long long totalSum = 0;
        for (int v: target) {
            totalSum += v;
            pq.push(v);
        }
        while (!pq.empty() && pq.top() > 1) {
            long long now = pq.top();
            pq.pop();
            long long restSum = totalSum - now;
            if (now <= restSum) return false;
            long long repeats = now / restSum;
            pq.push(now % restSum);
            totalSum -= restSum * repeats;
        }
        return true;
    }
};
