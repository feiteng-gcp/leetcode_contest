class Solution {
public:
	bool isPossible(vector<int>& target) {
		priority_queue<int, vector<int>, less<int>> p;

		long long sum = 0;
		for (auto t : target) {
			sum += t;
			p.push(t);
		}
		while (p.top()!=1) {
			long long temp = p.top(); p.pop();
			long long ls = sum - temp;
			sum = temp;
			if (temp < ls) return false;
			temp -= ls;
			p.push(temp);
		}
		return true;
	}
};
