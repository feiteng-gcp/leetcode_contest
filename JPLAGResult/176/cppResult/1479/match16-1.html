class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<int> q;
        long long sum = 0;
        for (int t : target) {
            q.push(t);
            sum += t;
        }
        while (true) {
            long long t = q.top();
            q.pop();
            if (t == 1)
                break;
            long long others = sum - t;
            t = t - others;
            if (t < 1)
                return false;
            sum = others + t;
            q.push(t);
        }
        return true;
    }
};
