class Solution {
public:
    bool isPossible(vector<int>& tget) {
        vector<long long>target(tget.begin(),tget.end());
        auto it=max_element(target.begin(),target.end());
        while(*it>1){
            long long c=accumulate(target.begin(),target.end(),-*it);
            *it=*it-c;
            if(*it<1)return 0;
            it=max_element(target.begin(),target.end());
        }
        return 1;
    }
};
