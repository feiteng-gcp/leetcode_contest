class Solution {
public:
    bool isPossible(vector<int>& target) {
        long long sum=0, cnt=target.size();
        priority_queue<long long> q;

        for(int i=0;i<cnt;i++){
            q.push(target[i]);
            sum+=target[i];
        }
        
        while(sum!=cnt){
            long long maxn=q.top(); q.pop();
            long long newn=2*maxn-sum;
            if(newn<1) return false;
            q.push(newn);
            sum=sum-maxn+newn;
        }
        
        return true;
    }
};
