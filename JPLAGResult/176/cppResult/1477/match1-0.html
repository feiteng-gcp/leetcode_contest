class ProductOfNumbers {
public:
    vector<int> nums;
    ProductOfNumbers() {

    }

    void add( int num ) {
        nums.push_back( num );
    }

    int getProduct( int k ) {
        int ret = 1;

        for( int i = 1; i <= k; i++ ) {
            ret *= nums[nums.size() - i];
        }

        return ret;
    }
};
