class ProductOfNumbers {
public:
    vector<int> vec;
    ProductOfNumbers() {
        vec.clear();
    }
    
    void add(int num) {
        vec.push_back(num);
    }
    
    int getProduct(int k) {
        int n = vec.size();
        int ret = 1;
        for(int i = 0; i < k; i++) {
            ret *= vec[n-1-i];
        }
        return ret;
    }
};

 * ProductOfNumbers* obj = new ProductOfNumbers();
 * obj->add(num);
 * int param_2 = obj->getProduct(k);
