class ProductOfNumbers {
public:
    vector<int> hist;
    ProductOfNumbers() {
    }
    
    void add(int num) {
        hist.push_back(num);
    }
    
    int getProduct(int k) {
        int s = hist.size();
        int prod = 1;
        for (int i = s - k; i < s; i++) {
            prod = prod * hist[i];
        }
        
        return prod;
    }
};

 * ProductOfNumbers* obj = new ProductOfNumbers();
 * obj->add(num);
