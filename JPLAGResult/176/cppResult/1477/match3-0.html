class ProductOfNumbers {
public:
    vector<int> nums;
    //3, 0, 2, 10, 40
    set<int> zeroSet;
    ProductOfNumbers() {
        
    }
    
    void add(int num) {
        if(num == 0) {
            zeroSet.insert(nums.size());
        }
        if(nums.empty() || nums.back() == 0){
            nums.emplace_back(num);
        }else{
            nums.emplace_back(num * nums.back());
        }
    }
    
    int getProduct(int k) {
        int n = nums.size();
        auto itr = zeroSet.lower_bound(n - k);
        if(itr != zeroSet.end()) return 0;
        if(n == k || nums[n - k - 1] == 0) return nums.back();
        return nums.back() / nums[n - k - 1];
    }
};

/**
 * Your ProductOfNumbers object will be instantiated and called as such:
 * ProductOfNumbers* obj = new ProductOfNumbers();
 * obj->add(num);
 * int param_2 = obj->getProduct(k);
