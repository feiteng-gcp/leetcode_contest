class Solution {
    public int[] closestDivisors(int num) {
        int[] p1 = helper(num+1);
        int[] p2 = helper(num+2);
        if (Math.abs(p1[0] - p1[1]) <= Math.abs(p2[0] - p2[1])) {
            return p1;
        }
        return p2;
    }
    
    private int[] helper(int num) {
        int a = 1, b = num;
        for (int i = 1; i * i <= num; i++) {
            if (num % i == 0) {
                a = i;
                b = num / i;
            }
        }
        return new int[]{a, b};
    }
}
