import java.util.ArrayList;
import java.util.Arrays;

class Solution {
  static ArrayList<Integer> primes = new ArrayList<>();
 
  public static void main(String args[]) {
    System.out.println(Arrays.toString(new Solution().closestDivisors(1000000000)));
  }
  
  
public int[] closestDivisors(int num) {
    int[] best1 = getClosest(num + 1);
    int[] best2 = getClosest(num + 2);
    
    int score1 = Math.abs(best1[0] - best1[1]);
    int score2 = Math.abs(best2[0] - best2[1]);
    return score1 <= score2 ? best1 : best2;
  
}
  
  private static int[] getClosest(int num) {
    for (int i = (int)Math.sqrt(num); i >= 1; i--) {
      if (num % i == 0) {
        return new int[] {i, num / i};
      }
    }
    return null;
  }
}
