class Solution {

	public int[] closestDivisors(int num) {
		for (int i = (int) Math.ceil(Math.sqrt(num)); i <= num; i++) {
			if ((num + 1) / i * i == num + 1) {
				return new int[] { i, (num + 1) / i };
			} else if ((num + 2) / i * i == num + 2) {
				return new int[] { i, (num + 2) / i };
			}
		}
		return null;
	}
}
