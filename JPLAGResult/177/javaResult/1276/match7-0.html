class Solution {
    public int[] closestDivisors(int num) {
        int[] p1 = findPair(num + 1);
        int[] p2 = findPair(num + 2);
        if (Math.abs(p1[0] - p1[1]) > Math.abs(p2[0] - p2[1])) return p2;
        return p1;
    }
    private int[] findPair(int n) {
        int sqrt = (int) Math.sqrt(n);
        for (int i = sqrt; i > 1; i--) {
            if (n % i == 0) {
                return new int[]{i, n / i};
            }
        }
        return new int[]{1, n};
    }
}
