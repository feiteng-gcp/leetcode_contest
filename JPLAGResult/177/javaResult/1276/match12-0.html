class Solution {
    public int[] closestDivisors(int num) {
        int ceil = (int)Math.sqrt(num+2);
        int[] res = new int[2];
        res[1] = Integer.MAX_VALUE;
        for(int i=ceil; i>0; i--){
            if((num+1)%i==0){
                int j = (num+1)/i;
                if(Math.abs(i-j)<=Math.abs(res[0]-res[1])){
                    res[0] = i;
                    res[1] = j;
                }
            }
            if((num+2)%i==0){
                int j = (num+2)/i;
                if(Math.abs(i-j)<=Math.abs(res[0]-res[1])){
                    res[0] = i;
                    res[1] = j;
                }
            }
        }
        return res;
    }
}
