class Solution {
    public int[] closestDivisors(int num) {
        int n1 = num + 1, n2 = num + 2;
        int[] a = calc(n1), b = calc(n2);
        int d1 = Math.abs(a[0] - a[1]), d2 = Math.abs(b[1] - b[0]);
        if(d1 < d2) return a;
        return b;
    }
    
    int[] calc(int k)
    {
        int[] re = new int[2];
        int targ = 1 << 30;
        for(int i = 1; i <= Math.sqrt(k) + 1; i++)
        {
            if(k % i != 0) continue;
            int a = i, b = k / i;
            if(Math.abs(a - b) < targ)
            {
                targ = Math.abs(a - b);
                re = new int[]{a, b};
            }
        }
        return re;
    }
}
