class Solution {
    public int[] closestDivisors(int num) {
        int[] res = new int[2];
        int n = (int)Math.ceil(Math.sqrt(num + 2));
        for(; n >= 1; n--) {
            if((num + 1) % n == 0) {
                return new int[]{n, (num + 1) / n};
            }
            if((num + 2) % n == 0) {
                return new int[]{n, (num + 2) / n};
            }
            
        }
        return res;
    }
}
