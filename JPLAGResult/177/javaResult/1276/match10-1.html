class Solution {
    public int[] closestDivisors(int num) {
        int[] r1 = findDivisors(num + 1);
        int[] r2 = findDivisors(num + 2);
        if(Math.abs(r1[0] - r1[1]) < Math.abs(r2[0] - r2[1])) return r1;
        else return r2;
    }
    public int[] findDivisors(int num){
        int sqrt = (int)Math.sqrt(num);
        for(int i = sqrt; i > 0; i--)
{
            if(num % i == 0){
                return new int[]{i, num / i};
            }
        }
        return new int[]{1,num};
    }
}
