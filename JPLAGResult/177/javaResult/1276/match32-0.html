class Solution {
    public int[] closestDivisors(int num) {
        // 这个没啥难度啊。这个肯定是有答案的。
        int[] res1 = getAns(num+1);
        int[] res2 = getAns(num+2);
        if(res1[1] - res1[0]  < res2[1] - res2[0]) return res1;
        return res2;
    }
    
    int[] getAns(int num) {
        int[] res = new int[]{0, 0};
        int min = Integer.MAX_VALUE;
        for(int i = 1;i<=Math.sqrt(num);i++) {
            if(num%i == 0) {
                int b = num/i;
                res = new int[]{i, b};
            }
        }
        return res;
    }
}
