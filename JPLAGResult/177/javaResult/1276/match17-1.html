class Solution {
    public int[] closestDivisors(int num) {
        int tmp[] = solve(num + 1);
        int tmp2[] = solve(num + 2);
        int diff1 = tmp[1] != -1 ? tmp[1] - tmp[0] : Integer.MAX_VALUE;
        int diff2 = tmp2[1] != -1 ? tmp2[1] - tmp2[0] : Integer.MAX_VALUE;
        if(diff1 <= diff2) 
{
            return tmp;
        }
        return tmp2;
    }
    public int[] solve(int num) {
        int low = -1, high = -1;
        for(int i = 1; i <= (int)Math.sqrt(num); i++) 
{
            if(num%i == 0) {
                low = i;
                high = num/i;
            }
        }
        return new int[]{low, high};
    }
}
