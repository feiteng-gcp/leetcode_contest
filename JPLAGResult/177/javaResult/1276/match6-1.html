class Solution {
    public int[] closestDivisors(int num) {
        int[] d1 = findClosestDivisor(num + 1);
        int[] d2 = findClosestDivisor(num + 2);
        if (d1[0] + d1[1] < d2[0] + d2[1]) 
return d1;
        return d2;
    }
    
    private int[] findClosestDivisor(int num) {
        int sqrt = (int) Math.sqrt(num);
        for (int i = sqrt; i >= 1; i--) {
            if (num % i == 0) {
                return new int[]{i, num / i};
            }
        }
        return new int[]{-1, -1};
    }
}
