class Solution {
    public int[] closestDivisors(int num) {
        int[] a = get(num+1);
        int[] b = get(num+2);
        if (diff(a) < diff(b)) {
            return a;
        } else {
            return b;
        }
    }
    
    public int[] get(int num) {
        int[] best = null;
        for (int i = 1; i*i <= num; i++) {
            if (num % i == 0) {
                int[] x = new int[]{i, num/i};
                if (x[0] > x[1]) {
                    int tmp = x[0];
                    x[1] = x[0];
                    x[0] = tmp;
                }
                
                if (best == null || diff(best) > diff(x)) {
                    best = x;
                }
            }
        }
        return best;
    }
    
    public int diff(int[]a) {
        return a[1] - a[0];
    }
}
