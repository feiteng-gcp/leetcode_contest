class Solution {
    public int[] closestDivisors(int num) {
        int[] pair = new int[2];
        
        int sq = (int) Math.sqrt(num) + 1;
        int diff = Integer.MAX_VALUE;
        for (int i = 1; i <= sq; i++) {
            if ((num + 1) % i == 0) {
                int j = (num + 1) / i;
                if (Math.abs(i - j) < diff) {
                    diff = Math.abs(i - j);
                    pair[0] = i;
                    pair[1] = j;
                }
            }
            
            if ((num + 2) % i == 0) {
                int j = (num + 2) / i;
                if (Math.abs(i - j) < diff) {
                    diff = Math.abs(i - j);
                    pair[0] = i;
                    pair[1] = j;
                }
            }
        }
        
        return pair;
    }
}
