class Solution {
    public int[] closestDivisors(int num) {
        int[] ans = help(num + 1), ans2 = help(num + 2);
        int tmp = (int)Math.abs(ans[0] - ans[1]), tmp2 = (int)Math.abs(ans2[0] - ans2[1]);
        if (tmp < tmp2) {
            return ans;
        } else {
            return ans2;
        
}
    }
    public int[] help(int num) {
        int tmp = 1, tmp2 = num;
        int test = num - 1;
        for (int i = 1; i * i <= num; i++) {
            if (num % i != 0) {
                continue;
            }
            
int left = num / i;
            int ans = (int)Math.abs(left - i);
            if (ans < test) {
                test = ans;
                tmp = i;
                tmp2 = left;
            }
        }
        return new int[]{tmp, tmp2};
    }
}
