class Solution {
    public int[] closestDivisors(int num) {
        int[] c1 = c(num + 1);
        int[] c2 = c(num + 2);
        if (Math.abs(c1[0] - c1[1]) < Math.abs(c2[0] - c2[1])) {
            return c1;
        } else {
            return c2;
        }
    
}

    private int[] c(int n) {
        int a = 0;
        int b = n;
        for (int i = 1; i * i <= n; i++) {
            if (n % i == 0) {
                if (n / i - i < (b - a)) {
                    a = i;
                    b = n / i;
                }
            
}
        }
        return new int[]{a, b};
    }
}
