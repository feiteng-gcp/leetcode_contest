class Solution {
    public int[] closestDivisors(int num) {
        int dis = Integer.MAX_VALUE;
        int [] result = new int[2];
        int val1 = num + 1;
        for(int i = 1; i * i <= val1; i++){
            if(val1 % i == 0){
                int k = val1 / i;
                int curdis = k - i;
                if(curdis < dis){
                    dis = curdis;
                    result[0] = i;
                    result[1] = k;
                }
            }
        }
        val1++;
        for(int i = 1; i * i <= val1; i++){
            if(val1 % i == 0){
                int k = val1 / i;
                int curdis = k - i;
                if(curdis < dis){
                    dis = curdis;
                    result[0] = i;
                    result[1] = k;
                }
            }
        }
        return result;
    }
}
