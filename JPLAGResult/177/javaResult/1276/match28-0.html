class Solution {
    public int[] closestDivisors(int num) {
        int[] res_a = PrintAllPairProductIsN(num + 1);
        int[] res_b = PrintAllPairProductIsN(num + 2);
        int a, b;
        if (res_a[2] < res_b[2]) {
            return new int[]{res_a[0], res_a[1]};
        } else {
            return new int[]{res_b[0], res_b[1]};
        }
    }

    private int[] PrintAllPairProductIsN(long n) {
        long min_diff = (int)1e9+500;
        long r1 = 0, r2 = 0;

        for (long cur = 1; cur * cur <= n; cur ++) {
            if (n % cur == 0) {
                long a = cur;
                long b = n / cur;

                long t = Math.abs(a - b);
                if (t < min_diff) {
                    min_diff = t;
                    r1 = a;
                    r2 = b;
                }
                // System.out.println(a + " " + b);
            }
        }

        return new int[]{(int)r1, (int)r2, (int)min_diff};
    }

}
