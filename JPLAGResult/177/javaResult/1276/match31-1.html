class Solution {
    public int[] closestDivisors(int num) {
        int num1 = num + 1;
        int num2 = num + 2;
        int snum1 = (int) Math.sqrt(num1);
        int snum2 = (int) Math.sqrt(num2);
        if (snum1 * snum1 == num1) {
            return new int[] {snum1, snum1};
        }
        if (snum2 * snum2 == num2) {
            return new int[] {snum2, snum2};
        }
        int min = num2;
        int[] res = new int[2];
        for(int i = 1; i <= snum2; ++i) {
            if (num1 % i == 0) {
                
if (min > Math.abs(i - num1 / i)) {
                    min = Math.abs(i - num1 / i);
                    res[0] = i;
                    res[1] = num1 / i;
                }
            }
            if (num2 % i == 0) {
                
if (min > Math.abs(i - num2 / i)) {
                    min = Math.abs(i - num2 / i);
                    res[0] = i;
                    res[1] = num2 / i;
                }
            }
        }
        return res;
    }
}
