class Solution {
    public int[] closestDivisors(int num) {
        int[] res = new int[]{0, Integer.MAX_VALUE - 2};
        int num1 = num + 1;
        int num2 = num + 2;
        for (int i=1;i * i <=num1;i++) 
        if (num1%i==0) {
            int next = num1 / i;
            if ((next - i) < (res[1] - res[0])) {
                res[0] = i;
                res[1] = next;
            }
        }
         for (int i=1;i * i <=num2;i++) 
        if (num2%i==0) {
            int next = num2 / i;
            if ((next - i) < (res[1] - res[0])) {
                res[0] = i;
                res[1] = next;
            }
        }
         return res;   
    }
}
