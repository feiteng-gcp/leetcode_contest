class Solution {
    public int[] closestDivisors(int num) {
        int temp = num + 1;
        int[] res = new int[2];
        int min = Integer.MAX_VALUE;
        for (int i = 1; i <= Math.sqrt(temp); i++) {
            if (temp % i == 0) {
                
if (temp / i - i < min) {
                    min = temp / i - i;
                    res[0] = i;
                    res[1] = temp / i;
                }
            }
        }
        temp = num + 2;
        for (int i = 1; i <= Math.sqrt(temp); i++) {
            if (temp % i == 0) {
                
if (temp / i - i < min) {
                    min = temp / i - i;
                    res[0] = i;
                    res[1] = temp / i;
                }
            }
        }
        return res;
    }
}
