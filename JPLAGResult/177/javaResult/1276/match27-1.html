class Solution {
    public int[] closestDivisors(int num) {
        long[] n1 = findBest(num + 1);
        long[] n2 = findBest(num + 2);
        if (Math.abs(n1[0] - n1[1]) < Math.abs(n2[0] - n2[1])) {
            return new int[] { (int)n1[0], (int)n1[1] };
        } else {
            return new int[] { (int)n2[0], (int)n2[1] };
        
}
    }
    
    private static long[] findBest(long n) {
        long[] best = null;
        for (long i = 1; i * i <= n; ++i) {
            if (n % i == 0) {
                long j = n / i;
                if (best == null || Math.abs(j - i) < Math.abs(best[0] - best[1])) {
                    best = new long[] {i, j};
                
}
            }
        }
        return best;
    }
}
