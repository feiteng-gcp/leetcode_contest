class Solution {
    public int[] closestDivisors(int num) {
        int target = (int)(Math.sqrt(num)+1);
        while (target > 0) 
{
            if ((num+1) % target == 0) {
                return new int[] {(num+1)/target, target};
            }
            if ((num+2) % target == 0) {
                return new int[] {(num+2)/target, target};
            }
                
            target--;
        }
        throw new IllegalStateException("Error in closestDivisors: no solution found");
    }
}
