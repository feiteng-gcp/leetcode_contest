class Solution {
    public int[] closestDivisors(int num) {
        int[] first = findSmallest(num+1);
        int[] second = findSmallest(num+2);
        return Math.abs(first[0]-first[1])<=Math.abs(second[0]-second[1])?first:second;
    }
    
    public int[] findSmallest(int k){
        int sqr = (int)Math.sqrt(k);
        
        for(int i=sqr;i>0;i--){
            if(k%i == 0){
                return new int[]{i,k/i};
            }
        }
        return null;
    }
}
