class Solution {
   public int[] closestDivisors(int num) {
        int[] answer1 = divisorDiff(num + 1);
        int[] answer2 = divisorDiff(num + 2);
        int diff1 = Math.abs(answer1[0] - answer1[1]);
        int diff2 = Math.abs(answer2[0] - answer2[1]);
        
if (diff1 <= diff2) {
            return answer1;
        }
        return answer2;
    }
    
    private int[] divisorDiff(int num) {
        for (int i = (int)Math.sqrt(num); i >= 1; i--) {
            if (num % i == 0) {
                return new int[] {i, num / i};
            
}
        }
        return new int[] {1, num};
    }
}
