class Solution {
    public int[] closestDivisors(int num) {
        int diff = num + 3;
        int[] ans = new int[2];
        for (int i = (int)(Math.floor(Math.sqrt(num + 3))); i >= 1; i--) 
{
            if ((num + 1) % i == 0) {
                if (Math.abs(i - (num + 1) / i) < diff) {
                    diff = Math.abs(i - (num + 1) / i);
                    ans[0] = (num + 1) / i;
                    ans[1] = i;
                }
            }
            if ((num + 2) % i == 0) {
                if (Math.abs(i - (num + 2) / i) < diff) {
                    diff = Math.abs(i - (num + 2) / i);
                    ans[0] = (num + 2) / i;
                    ans[1] = i;
                }
            }
        }
        return ans;
    }
    
}
