import java.text.SimpleDateFormat;  
import java.util.Date;  
class Solution {
    public int daysBetweenDates(String date1s, String date2s) {        
        try{
        Date date1=new SimpleDateFormat("yyyy-MM-dd").parse(date1s);  
        Date date2=new SimpleDateFormat("yyyy-MM-dd").parse(date2s);  
        long dif = Math.abs(date1.getTime()-date2.getTime());            
        return (int)(dif/86400000);
        
} catch (Exception e){
            return -1;
        }
    }
}
