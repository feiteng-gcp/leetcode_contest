class Solution {
    class Date {
        int d,m,y;
        public Date(int d, int m, int y) {
            this.d = d; 
            this.m = m; 
            this.y = y;
        }
    }
    int monthDays[] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31}; 
    public int daysBetweenDates(String date1, String date2) {
        String[] d1 = date1.split("-");
        String[] d2 = date2.split("-");
        Date dt1 = new Date(Integer.valueOf(d1[2]), Integer.valueOf(d1[1]),Integer.valueOf(d1[0]));
        Date dt2 = new Date(Integer.valueOf(d2[2]), Integer.valueOf(d2[1]),Integer.valueOf(d2[0]));
        return Math.abs(getDifference(dt1, dt2));
    }
    
    private int countLeapYears(Date d) {
        int years = d.y; 

        if (d.m <= 2)  
        { 
            years--; 
        } 
        return years / 4 - years / 100 + years / 400; 
    }
    
    public int getDifference(Date dt1, Date dt2)  {
        int n1 = dt1.y * 365 + dt1.d; 
        for (int i = 0; i < dt1.m - 1; i++)  { 
            n1 += monthDays[i]; 
        } 

        n1 += countLeapYears(dt1); 

        int n2 = dt2.y * 365 + dt2.d; 
        for (int i = 0; i < dt2.m - 1; i++) 
        { 
            n2 += monthDays[i]; 
        } 
        n2 += countLeapYears(dt2); 

        return (n2 - n1); 
    }
}
