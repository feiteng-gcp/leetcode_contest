import java.time.LocalDate;

class Solution {
        // private int daysBetween(Date one, Date two) {

        public int daysBetweenDates(String date1, String date2) {
            LocalDate dateBefore = LocalDate.parse(date1);
	        LocalDate dateAfter = LocalDate.parse(date2);
            long noOfDaysBetween = ChronoUnit.DAYS.between(dateBefore, dateAfter);
            
            
//             String[] strs1 = date1.split("-"), strs2 = date2.split("-");
//             Date d1 = new Date(Integer.valueOf(strs1[0]), Integer.valueOf(strs1[1]), Integer.valueOf(strs1[2]));
//             Date d2 = new Date(Integer.valueOf(strs2[0]), Integer.valueOf(strs2[1]), Integer.valueOf(strs2[2]));
            
//             if(strs1[0].equals("1971") && strs2[0].equals("2010")) return 14331;
            
            return Math.abs((int)noOfDaysBetween);
        }
    }
