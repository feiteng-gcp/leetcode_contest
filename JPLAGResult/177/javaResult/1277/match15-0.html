class Solution {
    public String largestMultipleOfThree(int[] digits) {
        if(digits.length==0){
            return "";
        }
        List<List<Integer>> list = new ArrayList<>();
        for(int i=0; i<3; i++){
            list.add(new ArrayList<>());
        }
        int sum = 0;
        Arrays.sort(digits);
        if(digits[digits.length-1]==0){
            return "0";
        }
        for(int digit: digits){
            list.get(digit%3).add(digit);
            sum+=digit;
        }

        List<Integer> rm = new ArrayList<>();
        
        if(sum%3==1){
            if(!list.get(1).isEmpty()){
                rm.add(list.get(1).get(0));
            }else if(list.get(2).size()>=2){
                rm.add(list.get(2).get(0));
                rm.add(list.get(2).get(1));
            }
        }else if(sum%3==2){
            if(!list.get(2).isEmpty()){
                rm.add(list.get(2).get(0));
            }else if(list.get(1).size()>=2){
                rm.add(list.get(1).get(0));
                rm.add(list.get(1).get(1));
            }
        }
        
        StringBuilder res = new StringBuilder();
        for(int i=digits.length-1; i>=0; i--){
            if(!rm.remove(Integer.valueOf(digits[i]))){
                res.append(digits[i]);
            }
        }
        return res.toString();
    }
}
