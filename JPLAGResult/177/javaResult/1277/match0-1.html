class Solution {
    public String largestMultipleOfThree(int[] digits) {
        int sum = 0;
        for(int i = 0; i < digits.length; ++i) {
            sum += digits[i];
        }
        Arrays.sort(digits);
        int y = sum % 3;
        if (y == 0) {
            if (digits[digits.length - 1] != 0) {
                StringBuilder res = new StringBuilder();
                for(int i = 0; i < digits.length; ++i) {
                    res.append(digits[digits.length - i - 1]);
                }
                return res.toString();
            }
            return "0";
        } else {
            // System.out.println(y);
            int del = -1;
            for(int i = 0; i < digits.length; ++i) {
                // System.out.println(digits[i]);
                if (digits[i] % 3 == y) {
                    del = i;
                    break;
                }
            }
            if (del != -1) {
                StringBuilder res = new StringBuilder();
                for(int i = 0; i < digits.length; ++i) {
                    if (digits.length - i - 1 == del) { 
                        continue; 
                    };
                    if (res.length() == 0 && digits[digits.length - i - 1] == 0) {
                        return "0";
                    }
                    res.append(digits[digits.length - i - 1]);
                }
                return res.toString();
            } else {
                int del1 = -1;
                int del2 = -1;
                //两个余数是2或者两个为1
                for(int i = 0; i < digits.length; ++i) {
                    if ((y == 1 && digits[i] % 3 == 2) || (y == 2 && digits[i] % 3 == 1)) {
                        if (del1 == -1) {
                            del1 = i;
                        } else {
                            del2 = i;
                            break;
                        }
                    }
                }
                StringBuilder res = new StringBuilder();
                for(int i = 0; i < digits.length; ++i) {
                    if (digits.length - i - 1 == del1 || digits.length - i - 1 == del2) { 
                        continue; 
                    };
                    if (res.length() == 0 && digits[digits.length - i - 1] == 0) {
                        return "0";
                    }
                    res.append(digits[digits.length - i - 1]);
                }
                return res.toString();
            }
        }
    }
}
