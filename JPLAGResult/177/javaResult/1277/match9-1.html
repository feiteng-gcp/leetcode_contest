class Solution {
    public String largestMultipleOfThree(int[] digits) {
        List<Integer> m0 = new ArrayList<>(), m1 = new ArrayList<>(), m2 = new ArrayList<>();
        Arrays.sort(digits);
        int n = digits.length;
        String str = "";
        for(int k : digits) str += "" + k;
        str = new StringBuilder(str).reverse().toString();
        int p = 0;
        while(p < n && str.charAt(p) == '0') p++;
        if(p >= n) str = "0";
        else str = str.substring(p);
        
        int sum = 0;
        
        for(int k : digits)
        {
            sum += k;
            if(k % 3 == 0) m0.add(k);
            if(k % 3 == 1) m1.add(k);
            if(k % 3 == 2) m2.add(k);
        }
        Collections.sort(m1);
        Collections.sort(m2);
        
        if(sum % 3 == 1)
        {
            // take out 1
            if(m1.size() >= 1) return takeOut(str, m1.get(0));
            else if(m2.size() >= 2)
            {
                str = takeOut(str, m2.get(0));
                str = takeOut(str, m2.get(1));
                return str;
            }
            else return "";
        }
        
        if(sum % 3 == 2)
        {
            // take out 1
            if(m2.size() >= 1) return takeOut(str, m2.get(0));
            else if(m1.size() >= 2)
            {
                str = takeOut(str, m1.get(0));
                str = takeOut(str, m1.get(1));
                return str;
            }
            else return "";
        }
        
        return str;
        
    }
    
    String takeOut(String str, int k)
    {
        int n = str.length();
        for(int i = 0; i < n; i++)
        {
            if(str.charAt(i) - '0' == k) return str.substring(0, i) + str.substring(i + 1);
        }
        return "";
    }
}
