class Solution {
    public String largestMultipleOfThree(int[] digits) {
        if(digits.length==0){
            return "";
        }
        List<List<Integer>> list = new ArrayList<>();
        list.add(new ArrayList<>());
        list.add(new ArrayList<>());
        list.add(new ArrayList<>());
        
int sum = 0;
        Arrays.sort(digits);
        if(digits[digits.length-1]==0){
            return "0";
        }
        for(int digit: digits){
            sum+=digit;
            list.get(digit%3).add(digit);
        }

        List<Integer> remove = new ArrayList<>();
        
        if(sum%3==1){
            if(!list.get(1).isEmpty()){
                remove.add(list.get(1).get(0));
            }else if(list.get(2).size()>=2){
                remove.add(list.get(2).get(0));
                remove.add(list.get(2).get(1));
            }
        }else if(sum%3==2){
            if(!list.get(2).isEmpty()){
                remove.add(list.get(2).get(0));
            }else if(list.get(1).size()>=2){
                remove.add(list.get(1).get(0));
                remove.add(list.get(1).get(1));
            }
        }
        
        StringBuilder sb = new StringBuilder();
        for(int i=digits.length-1; i>=0; i--){
            if(!remove.remove(Integer.valueOf(digits[i]))){
                sb.append(digits[i]);
            }
        }
        return sb.toString();
    }
}
