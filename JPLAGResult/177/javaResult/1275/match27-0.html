class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        Set<Integer> used = new HashSet<>();
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(0);
        used.add(0);
        while (!queue.isEmpty()) {
            int temp = queue.poll();
            int left = leftChild[temp];
            int right = rightChild[temp];
            if (left != -1) {
                if (used.contains(left)) {
                    return false;
                } else {
                    used.add(left);
                    queue.offer(left);
                }
            }
            if (right != -1) {
                if (used.contains(right)) {
                    return false;
                } else {
                    used.add(right);
                    queue.offer(right);
                }
            }
        }
        return used.size() == n;
    }
}
