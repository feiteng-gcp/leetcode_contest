import java.util.Deque;
import java.util.LinkedList;


class Solution {
    public static boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int vis[] = new int[n];
        int degree[] = new int[n];

        for (int i=0; i<n; i++) {
            if (leftChild[i] != -1) {
                degree[leftChild[i]]++;
            }
            if (rightChild[i] != -1) {
                degree[rightChild[i]]++;
            }
        }

        int zeroCnt = 0;
        int rootIdx = -1;
        for (int i=0; i<n; i++) {
            if (degree[i] == 0) {
                rootIdx = i;
                zeroCnt++;
                if (zeroCnt > 1) {
                    return false;
                }
            } else {
                if (degree[i] > 2) {
                    return false;
                }
            }
        }

        if (rootIdx == -1) {
            return false;
        }

        Deque<Integer> queue = new LinkedList<>();
        vis[rootIdx] = 1;
        queue.add(rootIdx);
        while (!queue.isEmpty()) {
            int idx = queue.poll();
            if (leftChild[idx] != -1) {
                if (vis[leftChild[idx]] == 0) {
                    vis[leftChild[idx]] = 1;
                    queue.add(leftChild[idx]);
                } else {
                    return false;
                }
            }
            if (rightChild[idx] != -1) {
                if (vis[rightChild[idx]] == 0) {
                    vis[rightChild[idx]] = 1;
                    queue.add(rightChild[idx]);
                } else {
                    return false;
                }
            }
        }

        
for (int i=0; i<n; i++) {
            if (vis[i] == 0) {
                return false;
            }
        }

        return true;
    }
}
