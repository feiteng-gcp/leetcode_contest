import java.util.HashSet;

class Solution {
  public static void main(String args[]) {
    int n = 1;
    int[] left = {-1};
    int[] right = {-1};
    Solution sol = new Solution();
    System.out.println(sol.validateBinaryTreeNodes(n, left, right));
  }
  
public boolean validateBinaryTreeNodes(int n, int[] leftChildren, int[] rightChildren) {
    Node[] nodes = new Node[n];
    for (int i = 0; i < nodes.length; i++) {
      nodes[i] = new Node();
    }

    for (int i = 0; i < leftChildren.length; i++) {
      Node node = nodes[i];
      int parent = i;
      int leftChild = leftChildren[i];
      int rightChild = rightChildren[i];
      if (leftChild >= 0) {
        if (nodes[leftChild].parent != null) {
          return false;
        }
        nodes[leftChild].parent = parent;
        node.left = leftChild;
      }
      if (rightChild >= 0) {
        if (nodes[rightChild].parent != null) {
          return false;
        }
        nodes[rightChild].parent = parent;
        node.right = rightChild;
      
}
    }
    int root = -1;
    for (int i = 0; i < nodes.length; i++) {
      if (nodes[i].parent == null) {
        if (root >= 0) {
          return false;
        }
        root = i;
      
}
    }
    if (root < 0) {
      return false;
    }
    return !hasCycle(nodes, root, new HashSet<>());
  }
  
  public boolean hasCycle(Node[] graph, int curr, HashSet<Integer> path) {
    if (path.contains(curr)) {
      return true;
    }
    path.add(curr);
    Integer leftChild = graph[curr].left;
    Integer rightChild = graph[curr].right;
    if (leftChild != null && hasCycle(graph, leftChild, path)) {
      return true;
    }
    if (rightChild != null && hasCycle(graph, rightChild, path)) {
      return true;
    }
    
    path.remove(curr);
    return false;
  }
  
  static class Node {
    Integer parent;
    Integer left;
    Integer right;
  }
}
