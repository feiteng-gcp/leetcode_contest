class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] ind = new int[n];
        for (int i = 0; i < leftChild.length; i++) {
            int l = leftChild[i], r = rightChild[i];
            
if (l != -1) ind[l]++;
            if (r != -1) ind[r]++;
        }
        int zero = 0;
        for (int i = 0; i < n; i++) {
            if (ind[i] > 1) return false;
            if (ind[i] == 0) zero++;
        }
        return zero == 1;
    }
}
