class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] f = new int[n];
        Map<Integer, List<Integer>> map = new HashMap<>();
        
        for(int i = 0; i < n; i++) map.put(i, new ArrayList<>());
        
        for(int i = 0; i < n; i++)
        {
            int l = leftChild[i],
                r = rightChild[i];
            // i->l, i-> r
            if(l != -1) map.get(i).add(l);
            if(r != -1) map.get(i).add(r);
        }
        
        Queue<Integer> q = new LinkedList<>();
        q.offer(0);
        while(!q.isEmpty())
        {
            int p = q.poll();
            f[p]++;
            if(f[p] > 1) break;
            for(int next : map.get(p)) q.offer(next);
        }
        for(int i = 0; i < n; i++) if(f[i] != 1) return false;
        return true;
    }
}
