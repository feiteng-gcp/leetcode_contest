class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] incoming = new int[n];
        for (int i = 0; i < n; i++) {
            if (leftChild[i] >= 0) {
                incoming[leftChild[i]]++;
            }
            if (rightChild[i] >= 0) {
                incoming[rightChild[i]]++;
            }
        }
        int countOfZero = 0;
        for (int i = 0; i < n; i++) {
            if (incoming[i] >= 2) {
                return false;
            }
            if (incoming[i] == 0) {
                countOfZero++;
            }
        }
        return countOfZero == 1;
    }
}
