class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        Map<Integer, Integer> parents = new HashMap<>();
        Queue<Integer> q = new LinkedList<>();
        Set<Integer> set = new HashSet<>();
        int count = 0;
        Set<Integer> visited = new HashSet<>();
        for(int i=0; i<n; i++){
            
if(leftChild[i]!=-1){
                if(parents.containsKey(leftChild[i])){
                    return false;
                }
                parents.put(leftChild[i], i);
            }
            if(rightChild[i]!=-1){
                if(parents.containsKey(rightChild[i])){
                    return false;
                }
                parents.put(rightChild[i], i);
            }
            if(rightChild[i]==-1 && leftChild[i]==-1){
                q.offer(i);
                visited.add(i);
            }
        }
        
        
while(!q.isEmpty()){
            int cur = q.poll();
            if(parents.get(cur)==null){
                set.add(cur);
            }else if(visited.add(parents.get(cur))){
                q.offer(parents.get(cur));
            }
            count++;
        }
        return count==n&&set.size()==1;
    }
}
