class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] degree = new int[n];
        for(int i: leftChild) {
            if(i > -1) {
                degree[i] += 1;
                if(degree[i] > 1) {
                    return false;
                }
            }
            
        }
        for(int i: rightChild) {
            if(i > -1) {
                degree[i] += 1;
                if(degree[i] > 1) {
                    return false;
                }
            
}
            
        }
        int count = 0;
        for(int i: degree) {
            if(i == 0) {
                count++;
            }
        }
        return count == 1;
    }
}
