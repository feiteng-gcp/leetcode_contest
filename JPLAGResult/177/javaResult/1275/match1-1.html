class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        // every node has exact one parent except the root
        
        int[] parent = new int[n];
        for (int i = 0; i < n; i++)
            parent[i] = -1;
        
        
        
for (int i = 0; i < n; i++) {
            if (leftChild[i] != -1) {
                if (parent[leftChild[i]] != -1)
                    return false;
                parent[leftChild[i]] = i;
            }
            
            if (rightChild[i] != -1) {
                if (parent[rightChild[i]] != -1) 
                    return false;
                parent[rightChild[i]] = i;
            }
        }
        
        int countRoot = 0;
        for (int i = 0; i < n; i++) {
            if (parent[i] == -1)
                countRoot++;
        }
        
        return countRoot == 1;
    }
}
