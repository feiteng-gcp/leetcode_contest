class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        Set<Integer> set = new HashSet<Integer>();
        int sum = 0;
        for(int i: leftChild){
            if(i>=n){
                return false;
            }
            if(i==-1){
                continue;
            }
            if(set.contains(i)){
                return false;
            }
            sum+=i;
            set.add(i);
        }
        for(int i: rightChild){
            if(i>=n){
                return false;
            }
            if(i==-1){
                continue;
            }
            if(set.contains(i)){
                return false;
            }
            sum+=i;
            set.add(i);
        }
        return set.size()==n-1;
    }
}
