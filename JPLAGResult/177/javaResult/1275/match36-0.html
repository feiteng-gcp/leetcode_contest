class Solution {

	public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
		ArrayList<Integer>[] graph = new ArrayList[n];
		for (int i = 0; i < n; i++) {
			graph[i] = new ArrayList<>();
		}
		for (int i = 0; i < n; i++) {
			if (leftChild[i] != -1) {
				graph[i].add(leftChild[i]);
			}
			if (rightChild[i] != -1) {
				graph[i].add(rightChild[i]);
			}
		}
		boolean[] visited = new boolean[n];
		if (!validateBinaryTreeNodes(0, graph, visited)) {
			return false;
		}
		for (boolean b : visited) {
			if (!b) {
				return false;
			}
		}
		return true;
	}

	private boolean validateBinaryTreeNodes(int i, ArrayList<Integer>[] graph, boolean[] visited) {
		if (!visited[i]) {
			visited[i] = true;
			for (int j : graph[i]) {
				if (!validateBinaryTreeNodes(j, graph, visited)) {
					return false;
				}
			}
			return true;
		} else {
			return false;
		}
	}
}
