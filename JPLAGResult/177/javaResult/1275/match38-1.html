class Solution {
 		
	private int[] inDegree;	
	
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
    	inDegree = new
 int[n];
    	for (int i = 0;i < n;i ++) {
    		if (leftChild[i] >= 0) {
    			inDegree[leftChild[i]] ++;    			
    		}
    		if (rightChild[i] >= 0) {
    			inDegree[rightChild[i]] ++;
    		}
    	}
    	int root = - 1;
    	for (int i = 0;i < n;i ++) {
    		if (inDegree[i] == 0) {
    			if (root < 0) {
    				root = i;
    			} else {
    				return false;
    			}
    		}
    	}
    	if (root >= 0) {
    		Queue<Integer> queue = new LinkedList<>();
    		boolean[] vis = new boolean[n];
    		queue.add(root);
    		vis[root] = true;
    		while (!queue.isEmpty()) {
    			int current = queue.poll();
    			if (leftChild[current] >= 0) {
    				if (vis[leftChild[current]]) {
    					return false;    					
    				}
    				vis[leftChild[current]] = true;
    				queue.add(leftChild[current]);
    			}
    			if (rightChild[current] >= 0) {
    				if (vis[rightChild[current]]) {
    					return false;    					
    				}
    				vis[rightChild[current]] = true;
    				queue.add(rightChild[current]);
    			}
    		}
    		for (int i = 0;i < n;i ++) {
    			if (!vis[i]) {

    				return false;
    			}
    		}
    		return true;
    	} else {
    		return false;
    	}
    }
}
