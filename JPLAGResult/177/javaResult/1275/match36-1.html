class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        return !hasTwoParents(n, leftChild, rightChild) && !hasLoop(n, leftChild, rightChild) && allConnected(n, leftChild, rightChild);
    }
    private boolean hasTwoParents(int n, int[] leftChild, int[] rightChild) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            int left = leftChild[i];
            int right = rightChild[i];
            if (map.containsKey(left) || map.containsKey(right)) return true;
            if (left != -1) {
                map.put(left, i);
            }
            if (right != -1) {
                map.put(right, i);
            }
        }
        return false;
    }
    private boolean hasLoop(int n, int[] leftChild, int[] rightChild) {
        int[] state = new int[n];
        
for (int i = 0; i < n; i++) {
            if (state[i] == 0 && dfs(n, leftChild, rightChild, state, i)) {
                return true;
            }
        }    
        return false;
    }
    private boolean dfs(int n, int[] leftChild, int[] rightChild, int[] state, int curr) {
        if (state[curr] == 2) return false;
        if (state[curr] == 1) return true;
        state[curr] = 1;
        if (leftChild[curr] != -1) {
            if (dfs(n, leftChild, rightChild, state, leftChild[curr])) {
                return true;
            }
        }
        if (rightChild[curr] != -1) {
            if (dfs(n, leftChild, rightChild, state, rightChild[curr])) {
                return true;
            }
        }
        state[curr] = 2;
        return false;
    }
    
private boolean allConnected(int n, int[] leftChild, int[] rightChild) {
        int[] parent = new int[n];
        for (int i = 0;i < n; i++) {
            parent[i] = i;
        
}
        for (int i = 0; i < n; i++) {
            if (leftChild[i] != -1) {
                merge(leftChild[i], i, parent);
            }
            if (rightChild[i] != -1) {
                merge(rightChild[i], i, parent);
            }
        }   
        int p = find(0, parent);
        for (int i =1; i < n; i++) {
            if (find(i, parent) != p) 
{
                return false;
            }            
        }
        return true;
    }
    private void merge(int i, int j, int[] parent) {
        int p1 = find(i, parent);
        int p2 = find(j, parent);
        parent[p1] = p2;
    }
    private int find(int i, int[] parent) {
        while (i != parent[i]) {
            i = parent[i];
        }
        return i;
    }
}
