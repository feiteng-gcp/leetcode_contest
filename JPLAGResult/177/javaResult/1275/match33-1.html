class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int [] father = new int [n];
        Arrays.fill(father, -1);
        int count = n;
        for (int i = 0; i < leftChild.length; i++) {
            if (leftChild[i] != -1) {
                if (leftChild[i] == i) {
                    return false;
                }
                int result = connect(father, i, leftChild[i]);
                if (result == 1) 
{
                    return false;
                }
                count --;
            }
            
            if (rightChild[i] != -1) {
                if (rightChild[i] == i) {
                    return false;
                }
                int result = connect(father, i, rightChild[i]);
                if (result == 1) 
{
                    return false;
                }
                count --;
            }
        }
        return count == 1;
    }
    
    private int connect(int [] father, int a, int b) {
        int fa = find(father, a);
        int fb = find(father, b);
        if (fa == fb) {
            return 1;
        }
        father[Math.min(fa, fb)] = father[Math.max(fa, fb)];
        return -1;
    }
    
    private int find(int [] father, int a) {
        if (father[a] == -1) {
            return a;
        }
        father[a] = find(father, father[a]);
        return father[a];
    } 
}
