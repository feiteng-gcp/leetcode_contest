class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] parents = new int[n];
        Arrays.fill(parents, -1);
        for(int i = 0; i < n; i++) {
            int p = i;
            int lc = leftChild[i];
            int rc = rightChild[i];
            if(lc == -1 && rc == -1) continue;
            if((lc != -1 && parents[lc] != -1) || (rc != -1 && parents[rc] != -1)) {
                return false;
            }
            if((lc != -1 && (leftChild[lc] == p || rightChild[lc] == p)) || 
               (rc != -1 && (leftChild[rc] == p || rightChild[rc] == p))) {
                return false;
            }
            if(lc != -1) {
                parents[lc] = p;
            }
            if(rc != -1) {
                parents[rc] = p;
            
}
            
            
            
        }
        boolean root = false;
        for(int i = 0; i < n; i++) {
            if(parents[i] == -1) {
                if(!root) {
                    root = true;
                } else {
                    return false;
                
}
            }
        }
        return root;
    }
}
