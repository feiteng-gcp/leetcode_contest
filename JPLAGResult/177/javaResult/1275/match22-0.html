class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        boolean[] noParent = new boolean[n];
        Arrays.fill(noParent, true);
        for (int i = 0; i < n; i++) {
            int left = leftChild[i], right = rightChild[i];
            if (left != -1 && !noParent[left]) return false;
            if (left != -1) noParent[left] = false;
            if (right != -1 && !noParent[right]) return false;
            if (right != -1) noParent[right] = false;
        }
        
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            if (noParent[i]) {
                if (cnt > 1) return false;
                cnt++;
            }
        }
        
        return cnt == 1;
    }
}
