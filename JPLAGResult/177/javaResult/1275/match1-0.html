class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] fa = new int[n];
        Arrays.fill(fa, -1);
        // 如果一个节点有两个fa就是问题。或者循环的fa，或者是没有root。
        for(int i = 0;i<n;i++) {
            if(leftChild[i] != -1) {
                if(fa[leftChild[i]] != -1) return false;
                fa[leftChild[i]] = i;
            }
            if(rightChild[i] != -1) {
                if(fa[rightChild[i]] != -1) return false;
                fa[rightChild[i]] = i;
            }
        }

        int count = 0;
        for(int i = 0;i<n;i++) {
            if(fa[i] == -1) count++;
        }
        return count == 1;
    }
}
