class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] indegree = new int[n];
        for (int i = 0; i < n; i++) {
            if (leftChild[i] != -1) {
                indegree[leftChild[i]]++;
            }
            if (rightChild[i] != -1) {
                indegree[rightChild[i]]++;
            }
        }
        int root = 0;
        for (int i = 0;i < n; i++) {
            if (indegree[i] == 0) {
                root++;
            }
            if (indegree[i] > 1) return false;
        }
        return root == 1;
    }
}
