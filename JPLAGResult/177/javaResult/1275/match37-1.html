class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] parent = new int[n];
        if (n==0) return true;
        for(int i=0; i<n; i++){
            parent[i] = -1;
        }
        
for(int i=0; i<n; i++){
            if(leftChild[i] >= 0){
                if(parent[leftChild[i]] == -1){
                    parent[leftChild[i]] = i;
                }
                else{
                    return false;
                }
            }
            if(rightChild[i] >= 0){
                if(parent[rightChild[i]] == -1){
                    parent[rightChild[i]] = i;
                }
                else{
                    return false;
                }
            }
            
        }
        
        // check if only one people don't have parent
        int count = 0;
        for(int i=0; i<n; i++){
            if(parent[i] == -1){
                count = count + 1;
            }
        }
        if(count == 1){
            return true;
        }
        else{
            return false;
        }
        
    }
}
