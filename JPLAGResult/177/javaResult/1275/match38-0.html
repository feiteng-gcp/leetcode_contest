class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] inDegree = new int[n];
        for (int i = 0; i < n; i++) {
            if (leftChild[i] != -1) {
                inDegree[leftChild[i]]++;
            }
            if (rightChild[i] != -1) {
                inDegree[rightChild[i]]++;
            }
        }
        int count = 0;
        for (int i = 0; i < n; i++) {
            if (inDegree[i] == 0) {
                count++;
                if (count > 1) {
                    return false;
                }
            }
            if (inDegree[i] > 1) {
                return false;
            }
        }
        return count == 1;
    }
}
