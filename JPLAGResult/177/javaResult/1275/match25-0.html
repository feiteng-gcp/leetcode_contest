class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        boolean[] visited = new boolean[n];
        for (int i = 0; i < leftChild.length; i++) {
            if (leftChild[i] != -1) {
                int c = leftChild[i];
                if (visited[c]) return false;
                visited[c] = true;
            }
            if (rightChild[i] != -1) {
                int c = rightChild[i];
                if (visited[c]) return false;
                visited[c] = true;
            }
        }
        int root = 0;
        for (boolean v : visited) {
            if (!v) root++;
        }
        return root == 1;
    }
}
