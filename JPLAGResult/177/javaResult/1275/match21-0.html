class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] parent = new int[n];
        Arrays.fill(parent, -1);
        for(int i = 0; i < n; i++) {
            // i --> left, i --> right
            int left = leftChild[i];
            int right = rightChild[i];
            if(left != -1 && parent[left] != -1) {
                return false;
            }
            if(right != -1 && parent[right] != -1) {
                return false;
            }
            if(left != -1) {
                parent[left] = i;
            }
            if(right != -1) {
                parent[right] = i;
            }
        }
        // find the root
        int root = -1;
        for(int i = 0; i < n; i++) {
            if(parent[i] == -1) {
                if(root != -1) {
                    return false;
                }
                root = i;
            }
        }
        return root != -1;
    }
}
