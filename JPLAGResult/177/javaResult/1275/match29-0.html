class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        Set<Integer> vis = new HashSet<>();

        int sz_l = leftChild.length, sz_r = rightChild.length;
        if (n != sz_l || n != sz_r) return false;

        int cnt = 0;
        
        for (int u : leftChild) {
            if (u != -1) {
                if (vis.contains(u)) return false;
                vis.add(u);
                cnt ++;
            }
        }

        for (int u : rightChild) {
            if (u != -1) {
                if (vis.contains(u)) return false;
                vis.add(u);
                cnt ++;
            }
        }

        if (cnt != n - 1) return false;

        return true;
    }
}
