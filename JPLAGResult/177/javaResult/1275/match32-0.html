class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        boolean[] vis = new boolean[n];
        int[] in = new int[n];
        for(int i = 0; i < n; i++) {
            if(leftChild[i] != -1) {
            in[leftChild[i]] += 1;
            }
            if(rightChild[i] != -1) {
            in[rightChild[i]] += 1;
            }
        }
        if(in[0] != 0) {
            return false;
        }
        for(int i = 1; i < n; i++) {
            if(in[i] == 0 || in[i] > 1) {
                return false;
            }
        }
        return true;
    }
}
