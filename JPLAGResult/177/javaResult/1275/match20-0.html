class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] in = new int[n];
        int[] out = new int[n];
        
        for (int i = 0; i < n; i++) {
            int l = leftChild[i];
            int r = rightChild[i];
            if (l != -1) {
                in[l]++;
                out[i]++;
            }
            if (r != -1) {
                in[r]++;
                out[i]++;
            }
        }
        
        int root = 0;
        for (int e : in) {
            if (e == 0) root++;
            if (e > 1) return false;
        }
        if (root != 1) return false;
        for (int e : out) if (e > 2) return false;
        return true;
    }
}
