class Solution {
    public boolean validateBinaryTreeNodes(int n, int[] leftChild, int[] rightChild) {
        int[] in = new int[n];
        int[] out = new int[n];
        for (int i = 0; i < n; i++) {
            if (leftChild[i] >= 0)
                in[leftChild[i]]++;
            if (rightChild[i] >= 0)
                in[rightChild[i]]++;
            
        }
        
        int count = 0;
        for (int i = 0; i < n; i++) {
            if (in[i] > 1)
                return false;
            if (in[i] == 0)
                count++;
        }
        
        return count == 1;
    }
}
