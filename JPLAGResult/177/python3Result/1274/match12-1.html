from datetime import date


class Solution:
  def daysBetweenDates(self, date1: str, date2: str) -> int:
    y1, m1, d1 = date1.split('-')
    y2, m2, d2 = date2.split('-')

    do1 = date(int(y1), int(m1), int(d1))
    do2 = date(int(y2), int(m2), int(d2))
    return abs((do2 - do1).days)
