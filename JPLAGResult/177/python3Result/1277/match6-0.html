class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        s = sum(digits)
        digits.sort(reverse = True)
        if s % 3 == 0:
            return str(int(''.join(str(d) for d in digits)))
        n = len(digits)
        cdd = []
        for i in range(n - 1, -1, -1):
            v = digits[i]
            if v % 3 == 0: continue
            if v % 3 == s % 3:
                digits.pop(i)
                return str(int(''.join(str(d) for d in digits))) if digits else ""
            else:
                if len(cdd) < 2:
                    cdd.append(i)
            
        if len(cdd) == 2:
            digits = [digits[i] for i in range(n) if i not in cdd]
            return str(int(''.join(str(d) for d in digits))) if digits else ""
        
        return ""
