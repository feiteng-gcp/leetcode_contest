class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        s=sum(digits)
        if s%3==0:
            digits.sort(reverse=True)
            digits=map(str,digits)
            ans=''.join(digits)
            return str(int(ans))
        #res=set(digits)
        t1
=[]
        t2=[]
        t3=[]
        for i in digits:
            if i%3==0:
                t3.append(i)
            elif i%3==1:
                t1.append(i)
            elif i%3==2:
                t2.append(i)
        t1.sort(reverse=True)
        t2.sort(reverse=True)
        if s%3==1:
            if len(t1):
                t1.pop()
                digits=t1+t2+t3
                digits.sort(reverse=True)
                digits=map(str,digits)
                digits=''.join(digits)
                return str(int(digits)) if digits else ''
            t2.pop()
            t2.pop()
            digits=t1+t2+t3
            digits.sort(reverse=True)
            digits=map(str,digits)
            digits=''.join(digits)
            return str(int(digits)) if digits else ''
        if len(t2):
            t2.pop()
            digits=t1+t2+t3
            digits.sort(reverse=True)
            digits=map(str,digits)
            digits=''.join(digits)
            return str(int(digits)) if digits else ''
        t1.pop()
        digits=t1+t2+t3
        digits.sort(reverse=True)
        digits=map(str,digits)
        digits=''.join(digits)
        return str(int(digits)) if digits else ''
