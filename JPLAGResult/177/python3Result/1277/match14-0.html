class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        zeros = []
        ones = []
        twos = []
        for d in digits:
            if d % 3 == 0:
                zeros.append(d)
            elif d % 3 == 1:
                ones.append(d)
            else:
                twos.append(d)
        zeros.sort()
        ones.sort(reverse=True)
        twos.sort(reverse=True)
        
        mod = (1 * len(ones) + 2 * len(twos)) % 3
        # print('0:',len(zeros), '1:',len(ones),'2:',len(twos),'mod',mod)
        if mod == 1:
            if len(ones) > 0:
                ones.pop()
            elif len(twos) > 1:
                twos.pop()
                twos.pop()
            else:
                return ''
            
        elif mod == 2:
            if len(twos) > 0:
                twos.pop()
            elif len(ones) > 1:
                ones.pop()
                ones.pop()
            else:
                return ''
        if not zeros and not ones and not twos:
            return ''
        res = []
        res.extend(zeros)
        res.extend(ones)
        res.extend(twos)
        res.sort(reverse=True)
        if res[0] == 0:
            return '0'
        return ''.join([str(x) for x in res])
