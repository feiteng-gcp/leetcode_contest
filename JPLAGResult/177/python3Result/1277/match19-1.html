class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        r = []
        r0 = [x for x in digits if x % 3 == 0]
        r1 = sorted([x for x in digits if x % 3 == 1])
        r2 = sorted([x for x in digits if x % 3 == 2])
        
        r += r0
        diff = len(r1)%3 - len(r2)%3
        if diff == 1:
            r += r1[1:]
            r += r2
        elif diff == -1:
            r += r1
            r += r2[1:]
        elif diff == -2:
            if len(r1):
                r += r1[1:]
                r += r2
            else:
                r += r1
                r += r2[2:]
        elif diff == 2:
            print('here')
            if len(r2):
                r += r1
                r += r2[1:]
            else:
                r += r1[2:]
                r += r2
        else:
            r += r1
            r += r2
        r.sort(reverse=True)
        if len(r) == 0 and len
(digits) > 0:
            return ""
        if sum(r) == 0: return '0'
        return ''.join([str(ele) for ele in r])
