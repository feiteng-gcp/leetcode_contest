class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        count = collections.Counter(digits)
        dic= {1: [1,4,7], 2: [2,5,8]}
        s = sum(digits)
        s %= 3
        if s == 1:
            good 
= False
            for x in dic[1]:
                if count[x] > 0:
                    count[x] -= 1
                    good = True
                    break
            if not good:
                temp = []
                for x in (2, 5, 8):
                    temp += [x]*min(2, count[x])
                #temp.sort()
                for t in temp[:2]:
                    count[t] -= 1
        elif s == 2:
            good 
= False
            for x in dic[2]:
                if count[x] > 0:
                    count[x] -= 1
                    good = True
                    break
            if not good:
                temp = []
                for x in (1, 4, 7):
                    temp += [x]*min(2, count[x])
                #temp.sort()
                for t in temp[:2]:
                    count[t] -= 1
        ans = ''
        for x in sorted(count.keys(), reverse = True):
            ans += str(x)*(count[x])
        if set(ans) == set(['0']):
            return '0'
        return ans
