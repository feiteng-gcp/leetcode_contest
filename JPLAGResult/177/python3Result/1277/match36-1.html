class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        p = [[] for i in range(3)]
        for d in digits:
            p[d % 3].append(d)
        for i in range(3):
            p[i].sort()
        l1, l2 = (len(p[1]), len(p[2]))
        l = (l1 + 2 * l2) % 3
        r = p[0]
        if l == 1:
            if l1 > 0:
                r += p[1][1:] + p[2]
            else:
                r += p[2][2:] + p[1]
        elif l == 2:
            if l2 > 0:
                r += p[2][1:] + p[1]
            else:
                r += p[1][2:] + p[2]
        else:
            r += p[1] + p[2]
            
        ret = ''.join(map(str, sorted(r)[::-1]))
        if ret.count('0') == len(ret) > 0:
            return '0'
        return ret.lstrip('0')
