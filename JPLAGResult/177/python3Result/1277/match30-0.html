class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        cnt=[0]*10
        for x in digits:
            cnt[x]+=1
        r=sum(digits)%3
        if r!=0:
            for i in ((1,4,7,2,2,5,5,8,8) if r==1 else (2,5,8,1,1,4,4,7,7)):
                if cnt[i]:
                    cnt[i]-=1
                    r=(9+r-i)%3
                if r==0:
                    break
        if cnt[0]==sum(cnt):
            return "0" if cnt[0] else ""
        return "".join(str(i)*cnt[i] for i in range(9,-1,-1))
