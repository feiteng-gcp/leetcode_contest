class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        digits.sort(reverse = True)
        
        zero, one, two = [], [], []
        for n in digits:
            [zero, one, two][n % 3].append(n)
        
        summ = sum(digits)
        if summ % 3 == 1:
            if one:
                one.pop()
            else:
                two.pop()
                two.pop()
        elif summ % 3 == 2:
            if two:
                two.pop()
            else:
                one.pop()
                one.pop()
        
        nums = zero + one + two
        nums.sort(reverse = True)
        
        if not nums:
            return ''
        
        if nums[0] == 0:
            return '0'
        
        return ''.join([str(n) for n in nums])
