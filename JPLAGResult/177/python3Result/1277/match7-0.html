class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        sum_ = sum(digits)
        min_left_1, min_left_2, sec_left_1, sec_left_2 = float('inf'), float('inf'), float('inf'), float('inf')
        for num in digits:
            if num % 3 == 1:
                if num < min_left_1:
                    min_left_1, sec_left_1 = num, min_left_1
                elif num < sec_left_1:
                    sec_left_1 = num
            if num % 3 == 2:
                if num < min_left_2:
                    min_left_2, sec_left_2 = num, min_left_2
                elif num < sec_left_2:
                    sec_left_2 = num
        if sum_ % 3 == 1:
            if min_left_1 != float('inf'):
                digits.remove(min_left_1)
            elif min_left_2 != float('inf') and sec_left_2 != float('inf'):
                digits.remove(min_left_2)
                digits.remove(sec_left_2)
            else:
                return ""
        if sum_ % 3 == 2:
            if min_left_2 != float('inf'):
                digits.remove(min_left_2)
            elif min_left_1 != float('inf') and sec_left_1 != float('inf'):
                digits.remove(min_left_1)
                digits.remove(sec_left_1)
            else:
                return ""
        if not digits:
            return ""
        res = "".join(map(str, sorted(digits, reverse = True)))
        res = res.lstrip('0')
        return res if res else '0'
