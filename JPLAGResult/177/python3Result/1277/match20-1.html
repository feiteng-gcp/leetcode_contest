from typing import List


class Solution:
  def largestMultipleOfThree(self, digits: List[int]) -> str:
    s = sum(digits)

    if s % 3 == 0:
      if all(c == 0 for c in digits):
        return '0'
      return ''.join(list(sorted([str(d) for d in digits], reverse=True)))

    counts = [0] * 10
    for d in digits:
      counts
[d] += 1

    idx = 0
    for i in range(10):
      while counts[i] > 0:
        digits[idx] = i
        idx += 1
        counts[i] -= 1

    rem = s % 3
    if rem == 1:
      rem_2 = [0] * 2
      rem_2[0] = -1
      rem_2[1] = -1

      for i, d in enumerate(digits):
        if d % 3 == 1:
          return ''.join(reversed([str(d) for idx, d in enumerate(digits) if idx != i]))

        if d % 3 == 2:
          
if rem_2[0] == -1:
            rem_2[0] = i
          elif rem_2[1] == -1:
            rem_2[1] = i

      if rem_2[0] != -1 and rem_2[1] != -1:
        return ''.join(reversed([str(d) for idx, d in enumerate(digits) if idx not in rem_2]))

    elif rem == 2:
      rem_1 = [0] * 2
      rem_1[0] = -1
      rem_1[1] = -1

      for i,d in enumerate(digits):
        if d % 3 == 2:
          return ''.join(reversed([str(d) for idx, d in enumerate(digits) if idx != i]))
        if d % 3 == 1:
          
if rem_1[0] == -1:
            rem_1[0] = i
          elif rem_1[1] == -1:
            rem_1[1] = i

      if rem_1[0] != -1 and rem_1[1] != -1:
        return ''.join(reversed([str(d) for idx, d in enumerate(digits) if idx not in rem_1]))

    return ''
