class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        
        sum_ = sum(digits)
        min11, min21, min12, min22 = float('inf'), float('inf'), float('inf'), float('inf')
        for num in digits:
            
            if num % 3 == 1:
                if num < min11:
                    min11, min12 = num, min11
                elif num < min12:
                    min12 = num
                    
            if num % 3 == 2:
                if num < min21:
                    min21, min22 = num, min21
                elif num < min22:
                    min22 = num
        
        if sum_ % 3 == 1:
            if min11 != float('inf'):
                digits.remove(min11)
            elif min21 != float('inf') and min22 != float('inf'):
                digits.remove(min21)
                digits.remove(min22)
            else:
                return ''
            
        if sum_ % 3 == 2:
            if min21 != float('inf'):
                digits.remove(min21)
            elif min11 != float('inf') and min12 != float('inf'):
                digits.remove(min11)
                digits.remove(min12)
            else:
                return ''
            
        if not digits:
            return ''
            
        res = ''.join(map(str, sorted(digits, reverse = True)))
        res = res.lstrip('0')
        return res if res else '0'
