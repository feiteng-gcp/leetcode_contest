class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        s = sum(digits)
        if s%3==0:
            res_list = digits[:]
        else:
            r_0 = []
            r_1 = []
            r_2 = []
            for x in digits:
                if x%3 == 0:
                    r_0.append(x)
                elif x%3 == 1:
                    r_1.append(x)
                else:
                    r_2.append(x)
            
            if s%3 == 1:
                if len(r_1) >= 1:
                    r_1.sort()
                    res_list = r_0 + r_2 + r_1[1:]
                elif len(r_2) >= 2:
                    r_2.sort()
                    res_list = r_0 + r_2[2:] + r_1
                else:
                    res_list = r_0
            else:
                if len(r_2) >= 1:
                    r_2.sort()
                    res_list = r_0 + r_1 + r_2[1:]
                elif len(r_1) >= 2:
                    r_1.sort()
                    res_list = r_0 + r_1[2:] + r_2
                else:
                    res_list = r_0
  
        if len(res_list) == 0:
            return ""
        else:
            res_list.sort(key=lambda x: -x)
            res = 0
            for i in res_list:
                res = res*10 + i
            return str(int(res))
