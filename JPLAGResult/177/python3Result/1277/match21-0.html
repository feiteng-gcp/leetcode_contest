class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        digs = [0 for _ in range(10)]
        sm = 0
        for d in digits:
            digs[d] += 1
            sm = (sm + d) % 3
        if sm == 0:
            cdigs = [x for x in digs]
        elif sm == 1:
            sat = False
            cdigs = [x for x in digs]
            for i in [1,4,7]:
                if cdigs[i] > 0:
                    cdigs[i] -= 1
                    sat = True
                    break
            if not sat:
                ct = 2
                cdigs = [x for x in digs]
                for i in [2,5,8]:
                    if cdigs[i] > 0:
                        amt = min(cdigs[i], ct)
                        cdigs[i] -= amt
                        ct -= amt
                sat = (ct == 0)
            if not sat:
                return ""
        else:
            sat = False
            cdigs = [x for x in digs]
            for i in [2,5,8]:
                if cdigs[i] > 0:
                    cdigs[i] -= 1
                    sat = True
                    break
            if not sat:
                ct = 2
                cdigs = [x for x in digs]
                for i in [1,4,7]:
                    if cdigs[i] > 0:
                        amt = min(cdigs[i], ct)
                        cdigs[i] -= amt
                        ct -= amt
                sat = (ct == 0)
            if not sat:
                return ""
        ret = ""
        for i in range(9, -1, -1):
            x = str(i) * cdigs[i]
            ret += x
        if ret == "":
            return ""
        if ret[0] == '0':
            return '0'
        return ret
