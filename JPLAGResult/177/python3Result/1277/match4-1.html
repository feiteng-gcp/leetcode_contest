import functools

class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        
        sum_v = sum(digits)
        
        if sum_v % 3==0:
            
            if sum_v==0: return "0"
             
            digits.sort(reverse=True)
            
            return ''.join([str(x) for x in digits])
        
        digits.sort(reverse=True)
        
        v_del = -1
        
        for x in digits[::-1]:
            
            if x%3 == sum_v%3:
                
                v_del = x
                
                break
                
        if v_del == -1: 
            digits = [x for x in digits if x % 3 == 0]
        else:
            digits.remove(v_del)
            
        digits.sort(reverse=True)
        
        return ''.join([str(x) for x in digits])
