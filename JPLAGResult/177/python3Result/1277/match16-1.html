class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        ans = []
        mod1 = []
        mod2 = []
        for d in digits:
            if d % 3 == 0:
                ans.append(d)
            elif d % 3 == 1:
                mod1.append(d)
            else:
                mod2.append(d)
        ans.sort()
        mod1.sort(reverse=True)
        mod2.sort(reverse=True)
        n1 = len(mod1)
        n2 = len(mod2)
        
        if n1 == 0 or n2 == 0:
            n1 -= n1 % 3
            n2 -= n2 % 3
        else:
            if n1 % 3 == 2 and n2 % 3 == 0:
                n2 -= 1
            elif n1 % 3 == 0 and n2 % 3 == 2:
                n1 -= 1
            elif n1 % 3 > n2 % 3:
                n1 -= 1
            elif n1 % 3 < n2 % 3:
                n2 -= 1
        ans.extend(mod1[:n1])
        ans.extend(mod2[:n2])
        ans.sort(reverse=True)
        if not ans:
            return ''
        elif ans[0] == 0:
            return '0'
        else:
            return ''.join(map(str, ans))
