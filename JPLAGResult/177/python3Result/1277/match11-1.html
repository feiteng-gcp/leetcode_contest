class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        _sum = sum(digits)
        q0 
= []
        q1 = []
        q2 = []
        digits.sort(reverse=True)
        for d in digits:
            if d % 3 == 0:
                q0.append(d)
            elif d % 3 == 1:
                q1.append(d)
            else:
                q2.append(d)
        if _sum % 3 == 1:
            if q1:
                q1.pop()
            else:
                if len(q2) >= 2:
                    q2.pop()
                    q2.pop()
                else:
                    return ""
        elif _sum % 3 == 2:
            if q2:
                q2.pop()
            else:
                if len(q1) >= 2:
                    q1.pop()
                    q1.pop()
                else:
                    return ""
        _digits = []
        _digits.extend(q0)
        _digits.extend(q1)
        _digits.extend(q2)
        if not _digits:
            return ""
        _digits.sort(reverse=True)
        idx = 0
        while idx < len(_digits) and _digits[idx] == 0:
            idx += 1
        ans = 0
        for i in range(idx, len(_digits)):
            ans *= 10
            ans += _digits[i]
        return str(ans)
