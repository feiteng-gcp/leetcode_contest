class Solution:
    def largestMultipleOfThree(self, dd: List[int]) -> str:
        s = sum(dd) % 3
        a = [[], [], []]
        for x in dd:
            a[x % 3].append(x)
        for i in range(3):
            a[i].sort(reverse=True)
        def cc(d):
            if d == []:
                return ''
            d.sort(reverse=True)
            if d[0] == 0:
                return '0'
            r = ""
            for x in d:
                r += str(x)
            return r
        if s == 0:
            return cc(dd)
        if a[s] != []:
            a[s].pop()
            return cc(a[0]+a[1]+a[2])
        else:
            if len(a[3-s]) < 2:
                return ''
            else:
                a[3-s].pop()
                a[3-s].pop()
                return cc(a[0]+a[1]+a[2])
