class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        a = []
        b = []
        c = []
        for i in digits:
            if i % 3 == 0:
                a.append(i)
            elif i % 3 == 1:
                b.append(i)
            else:
                c.append(i)
        a.sort(reverse = True)
        b.sort(reverse = True)
        c.sort(reverse = True)
        r = a
        B = len(b)
        C = len(c)
        if B == 0 or C == 0:
            r += b[0:(B // 3 * 3)] + c[0:(C // 3 * 3)]
        elif (B % 3 == C % 3):
            r += b + c
        elif B % 3 == 0 and C % 3 == 1 or B % 3 == 1 and C % 3 == 2 or B % 3 == 2 and C % 3 == 0:
            r += b + c[0:(C - 1)]
        else:
            r += c + b[0:(B - 1)]
        r.sort(reverse = True)
        if len(r) == 0:
            return ''
        if max(r) == 0:
            return '0'
        return ''.join([str(i) for i in r])
