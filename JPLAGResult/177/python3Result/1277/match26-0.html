class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:        
        ten = [0] * 10
        l = len(digits)        
        for i in range(l):
            ten[int(digits[i])] += 1
        sumResidue = (ten[1] + ten[4] + ten[7] + 2*ten[2] + 2*ten[5] + 2*ten[8])%3
        if sumResidue == 1:
            l -= 2
            if ten[1] > 0:
                ten[1] -= 1
                l += 1
            elif ten[4] > 0:
                ten[4] -= 1
                l += 1
            elif ten[7] > 0:
                ten[7] -=1
                l += 1
            elif ten[2] > 1:
                ten[2] -= 2
            elif ten[2] == 1 and ten[5] > 0:
                ten[2] -= 1
                ten[5] -= 1
            elif ten[5] > 1:
                ten[5] -= 2
            elif ten[2] == 1 and ten[8] > 0:
                ten[2] -= 1
                ten[8] -= 1
            elif ten[5] == 1 and ten[8] > 0:
                ten[5] -= 1
                ten[8] -= 1
            else:
                ten[8] -= 2
        elif sumResidue == 2:
            l -= 2
            if ten[2] > 0:
                ten[2] -= 1
                l += 1
            elif ten[5] > 0:
                ten[5] -= 1
                l += 1
            elif ten[8] > 0:
                ten[8] -=1
                l += 1
            elif ten[1] > 1:
                ten[1] -= 2
            elif ten[1] == 1 and ten[4] > 0:
                ten[1] -= 1
                ten[4] -= 1
            elif ten[4] > 1:
                ten[4] -= 2
            elif ten[1] == 1 and ten[7] > 0:
                ten[1] -= 1
                ten[7] -= 1
            elif ten[4] == 1 and ten[7] > 0:
                ten[4] -= 1
                ten[7] -= 1
            else:
                ten[7] -= 2
        if l == 0:
            return ""
        if l == ten[0]:
            return "0"                
        res = "9"*ten[9]+"8"*ten[8]+"7"*ten[7]+"6"*ten[6]+"5"*ten[5]+"4"*ten[4]+"3"*ten[3]+"2"*ten[2]+"1"*ten[1]+"0"*ten[0]
        return res
