class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        cands0 = [n for n in digits if n % 3 == 0]
        cands1 = [n for n in digits if n % 3 == 1]
        cands2 = [n for n in digits if n % 3 == 2]
        cands1.sort(reverse=True)
        cands2.sort(reverse=True)
        diff = sum(digits) % 3
        if diff == 1:
            if cands1:
                cands1.pop()
            elif len(cands2) < 2:
                return ""
            else:
                cands2.pop()
                cands2.pop()
        elif diff == 2:
            if cands2:
                cands2.pop()
            elif len(cands1) < 2:
                return ""
            else:
                cands1.pop()
                cands1.pop()
        ans = cands0 + cands1 + cands2
        ans.sort(reverse=True)
        if not ans:
            return ""
        if all(it == 0 for it in ans):
            return "0"
        return "".join(str(it) for it in ans)
