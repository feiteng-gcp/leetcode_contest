from collections import Counter
class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        sumi= sum(digits)
        if sumi %3 == 0:
            cnt = Counter(digits)
            if len(cnt.keys()) == 1 and list(cnt.keys())[0] == 0:
                return "0"
            digits.sort(reverse=True)
            st = ""
            for dig in digits:
                st += str(dig)
            return st
        else:
            rem = sumi % 3
            cnt = Counter(digits)
            flag = True
            if rem == 2:
                for key in sorted(cnt.keys()):
                    if key%3 == 2:
                        cnt[key] -= 1
                        flag = False
                        break
                if flag: 
                    for key in sorted(cnt.keys()):
                        if key%3 == 1 and cnt[key]>1:
                            cnt[key] -= 2
                            flag = False
                            break
                if flag:
                    for key in sorted(cnt.keys()):
                        if key%3 == 1:
                            cnt[key] -= 1
                            break
                    for key in sorted(cnt.keys()):
                        if key%3 == 1 and cnt[key]>0:
                            cnt[key] -= 1
                            break

            else:
                for key in sorted(cnt.keys()):
                    if key%3 == 1:
                        cnt[key] -= 1
                        flag = False
                        break
                if flag:
                    for key in sorted(cnt.keys()):
                        if key%3 == 2 and cnt[key]>1:
                            cnt[key] -= 2
                            flag = False
                            break
                if flag:
                    for key in sorted(cnt.keys()):
                        if key%3 == 2:
                            cnt[key] -= 1
                            break
                            
                    for key in sorted(cnt.keys()):
                        if key%3 == 2 and cnt[key]>0:
                            cnt[key] -= 1
                            break
            st = ""
            lst = list(cnt.keys())
            for key in lst:
                if cnt[key] == 0:
                    del cnt[key]
            
            if len(cnt.keys()) == 1 and list(cnt.keys())[0] == 0:
                return "0"
            for key in sorted(cnt.keys(), reverse=True):
                for i in range(cnt[key]):
                    st += str(key)
            return st
