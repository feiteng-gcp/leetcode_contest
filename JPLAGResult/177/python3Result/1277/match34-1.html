class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        digits = self.filter
(digits)
        if not digits:
            return ''

        if all(i == 0 for i in digits):
            return '0'

        digits.sort(reverse=True)
        return ''.join(str(i) for i in digits)

    @staticmethod
    def filter(digits):
        reminder = sum(digits) % 3
        if reminder == 0:
            return digits

        bads = tuple(i for i in digits if i % 3 == reminder)
        if bads:
            bad = min(bads)
            digits.remove(bad)
            return digits

        new = 3 - reminder
        bads = [i for i in digits if i % 3 == new]
        if len(bads) >= 2:
            bad = min(bads)
            digits.remove(bad)

            bads.remove(bad)
            bad = min(bads)
            digits.remove(bad)
            return digits
        else:
            return []
