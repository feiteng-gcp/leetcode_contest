class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        d = collections.defaultdict(list)
        for n in digits:
            d[n%3].append(n)
            
        d[1].sort(reverse=True)
        d[2].sort(reverse=True)
        n1 = len(d[1])
        n2 = len(d[2])
        s1, s2 = 0, 0
        picked = []
        picked1, picked2 = [], []
        #print(d)
        if n1<=n2:
            r = (n2-n1)%3
            if r==0:
                s2 = sum(d[1])+sum(d[2])
                picked = d[1] + d[2]
            elif r==1:
                if n2>=1:
                    s2 = sum(d[1][:n1])+sum(d[2][:n2-1])
                    picked1 = d[1][:n1]+ d[2][:n2-1]
                if n1>=2:
                    if sum(d[1][:n1-2])+sum(d[2][:n2]) > s2:
                        s2 = max(s2, sum(d[1][:n1-2])+sum(d[2][:n2]))
                    picked2 = d[1][:n1-2] + d[2][:n2]
                if len(picked1)>len(picked2):
                    picked = picked1
                else:
                    picked = picked2
            else:
                if n1>=1:
                    s2 = sum(d[1][:n1-1])+sum(d[2][:n2])
                    picked1 = d[1][:n1-1] + d[2][:n2]
                if n2>=2:
                    if sum(d[1][:n1])+sum(d[2][:n2-2]) > s2:
                        s2 = max(s2, sum(d[1][:n1])+sum(d[2][:n2-2]))
                    picked2 = d[1][:n1] + d[2][:n2-2]
                if len(picked1)>len(picked2):
                    picked = picked1
                else:
                    picked = picked2
        else:
            r = (n1-n2)%3
            if r==0:
                s2 = sum(d[1])+sum(d[2])
                picked = d[1]+d[2]
            elif r==1:
                if n1>=1:
                    s2 = sum(d[1][:n1-1])+sum(d[2][:n2])
                    picked1 = d[1][:n1-1]+d[2][:n2]
                if n2>=2:
                    if sum(d[1][:n1])+sum(d[2][:n2-2]) > s2:
                        s2 = max(s2, sum(d[1][:n1])+sum(d[2][:n2-2]))
                    picked2 = d[1][:n1]+d[2][:n2-2]
                if len(picked1)>len(picked2):
                    picked = picked1
                else:
                    picked = picked2
                
            else:
                if n1>=2:
                    s2 = sum(d[1][:n1-2])+sum(d[2][:n2])
                    picked1 = d[1][:n1-2]+d[2][:n2]
                if n2>=1:
                    if sum(d[1][:n1])+sum(d[2][:n2-1]) > s2:
                        s2 = max(s2, sum(d[1][:n1])+sum(d[2][:n2-1]))
                    picked2 = d[1][:n1]+d[2][:n2-1]
                if len(picked1)>len(picked2):
                    picked = picked1
                else:
                    picked = picked2
        #print(picked)
        if len(d[0])>0:
            s1 = sum(d[0])
            picked+=d[0]
        #print(picked)
        picked.sort(reverse=True)
        ans = "".join([str(i) for i in picked])
        if len(ans)>0 and int(ans)==0:
            return "0"
        return ans
