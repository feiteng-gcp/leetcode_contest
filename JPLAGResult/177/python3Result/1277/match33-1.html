from collections import defaultdict
class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        dic = defaultdict(list)
        for x in digits:
            dic[x%3].append(x)
        for key in dic:
            dic[key].sort(reverse=True)
        zero, one, two = len(dic.get(0, [])), len(dic.get(1, [])), len(dic.get(2, []))
        sums = sum(digits) % 3
        if sums == 0:
            final = sorted(dic.get(0, []) + dic.get(1, []) + dic.get(2, []), reverse=True)
        elif sums == 1:
            if one >= 1:
                final = sorted(dic.get(0, []) + dic.get(1, [])[:one-1] + dic.get(2, []), reverse=True)
            else:
                final = sorted(dic.get(0, []) + dic.get(1, []) + dic.get(2, [])[:two-2], reverse=True)
        else:
            if two >= 1:
                final = sorted(dic.get(0, []) + dic.get(1, []) + dic.get(2, [])[:two-1], reverse=True)
            else:
                final = sorted(dic.get(0, []) + dic.get(1, [])[:one-2] + dic.get(2, []), reverse=True)
        
if not final:
            return ""
        if final[0] == 0:
            return "0"
        return ''.join(map(str,final))
