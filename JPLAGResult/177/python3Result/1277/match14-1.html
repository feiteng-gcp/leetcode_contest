class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        return solve(digits)


def solve(digits):
    digits.sort(reverse=True)
    s = sum(digits)
    d1 = [d for d in digits if d % 3 == 1]
    d2 = [d for d in digits if d % 3 == 2]
    d0 = [d for d in digits if d % 3 == 0]

    if not digits:
        return ''

    if s % 3 == 0:
        for i, d in enumerate(digits):
            if d != 0:
                break
        digits = digits[i:]
        if not digits:
            return '0'
        return ''.join(map(str, digits))
    elif s % 3 == 1:
        
if len(d1) >= 1:
            d1.pop()
        elif len(d2) >= 2:
            d2.pop()
            d2.pop()
        else:
            return ''
    else:
        if len(d2) >= 1:
            d2.pop()
        elif len(d1) >= 2:
            d1.pop()
            d1.pop()
        else:
            return ''
    digits = d1 + d2 + d0
    if len(digits) == 0:
        return ''
    digits.sort(reverse=True)

    for i, d in enumerate(digits):
        if d != 0:
            break
    digits = digits[i:]
    if not digits:
        return '0'
    return ''.join(map(str, digits))
