class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        sum_ = sum(digits)
        if sum_ == 0:
            return '0'
        mod0 = []
        mod1 = []
        mod2 = []
        digits.sort(reverse=True)
        for d in digits:
            if d%3==0:
                mod0.append(d)
            elif d%3==1:
                mod1.append(d)
            else:
                mod2.append(d)
        if sum_%3==1:
            if mod1:
                mod1.pop()
            else:
                if mod2 and len(mod2)>=2:
                    mod2.pop()
                    mod2.pop()
                else:
                    return '';
        elif sum_%3==2:
            if mod2:
                mod2.pop()
            else:
                if mod1 and len(mod1)>=2:
                    mod1.pop()
                    mod1.pop()
                else:
                    return '';
        res = mod0+mod1+mod2
        res.sort(reverse=True)
        return ''.join([str(a) for a in res])
