class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        s = sum(digits) % 3
        
        groups = [[], [], []]
        for d in digits:
            groups[d % 3].append(d)
        for i in range(3):
            groups[i] = sorted(groups[i])[::-1]
        if s == 1:
            if groups[1]:
                groups[1].pop()
            else:
                groups[2].pop()
                groups[2].pop()
        elif s == 2:
            if groups[2]:
                groups[2].pop()
            else:
                groups[1].pop()
                groups[1].pop()
        all_nums = sorted(groups[0] + groups[1] + groups[2])[::-1]
        
if not all_nums:
            return ""
        if all_nums[0] == 0:
            return "0"
        return "".join(map(str, all_nums))
