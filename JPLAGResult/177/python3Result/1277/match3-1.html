class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        sumv = sum(digits)
        if sumv % 3==0:
            if sumv==0: return "0"
            digits.sort(reverse=True)
            return ''.join([str(x) for x in digits])
        
        digits.sort(reverse=True)
        delv = -1
        
        for x in digits[::-1]:
            if x%3 == sumv%3:
                delv = x
                break
        if delv == -1: 
            digits = [x for x in digits if x%3==0]
        else:
            digits.remove(delv)
        digits.sort(reverse=True)
        
        return ''.join([str(x) for x in digits])
