class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        def convert(ds):
            if not ds:
                return ""
            if ds[0]==0:
                return "0"
            return "".join(str(x) for x in ds)
        ds = digits
        ds
.sort(reverse=True)
        sm=sum(ds)
        if sm%3==0:
            return convert(ds)
        a=sm%3
        A=[]
        for i in range(len(ds)-1, -1, -1):
            if ds[i]%3==a:
                ds=ds[:i]+ds[i+1:]
                return convert(ds)
            elif ds[i]%3 != 0 and len(A)<2:
                A.append(i)
        if len(A)==2:
            ds = [ds[i] for i in range(len(ds)) if i not in A]
            return convert(ds)
        return ""
