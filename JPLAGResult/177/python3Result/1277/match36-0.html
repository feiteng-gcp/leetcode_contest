class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        arr = [[] for _ in range(3)]
        for d in digits:
            arr[d % 3].append(d)
        for i in range(1, 3):
            arr[i].sort()
        l1 = len(arr[1]) % 3
        if len(arr[1]) > 0 and l1 == 0: l1 = 3
        l2 = len(arr[2]) % 3
        if len(arr[2]) > 0 and l2 == 0: l2 = 3
        for i in range(l1, len(arr[1])):
            arr[0].append(arr[1][i])
        for i in range(l2, len(arr[2])):
            arr[0].append(arr[2][i])

        newl = arr[1][:l1] + arr[2][:l2]
        newl.sort(reverse=True)
        best = 0

        def dfs(i, cur):
            nonlocal best
            if i >= len(newl):
                if cur % 3 == 0 and cur > best:
                    best = cur
                return
            dfs(i + 1, cur)
            dfs(i + 1, cur * 10 + newl[i])
        dfs(0, 0)

        while best > 0:
            arr[0].append(best % 10)
            best //= 10

        arr[0].sort(reverse=True)
        if len(arr[0]) == 0: return ''
        if arr[0][0] == 0: return '0'
        return ''.join(list(map(str, arr[0])))
