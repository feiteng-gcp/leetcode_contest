class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        cnt = [0] * 10
        for d in digits:
            cnt[d] += 1
        
        s = sum(digits)
        if s % 3 == 1:
            for i in range(1, 10, 3):
                if cnt[i] > 0:
                    cnt[i] -= 1
                    s -= i
                    break
                    
        if s % 3 == 1:
            for i in range(2, 10, 3):
                if cnt[i] > 0:
                    cnt[i] -= 1
                    s -= i
                    break
        if s % 3 == 2:
            for i in range(2, 10, 3):
                if cnt[i] > 0:
                    cnt[i] -= 1
                    s -= i
                    break
        if s % 3 == 2:
            for i in range(1, 10, 3):
                if cnt[i] > 0:
                    cnt[i] -= 1
                    s -= i
                    break
        if s % 3 == 1:
            for i in range(1, 10, 3):
                if cnt[i] > 0:
                    cnt[i] -= 1
                    s -= i
                    break
        res = sum([[i] * cnt[i] for i in range(9, -1, -1)], [])
        if s % 3 != 0 or not res:
            return ''
        
        if res[0] == 0:
            return '0'
        return ''.join(map(str, res))
