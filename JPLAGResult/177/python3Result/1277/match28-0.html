import heapq

class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        
        arr0 = []
        arr1 = []
        arr2 = []
        
        sumv = sum(digits)
        
        for i in digits:
            if i % 3 == 0:
                heapq.heappush(arr0, i)
            elif i % 3 == 1:
                heapq.heappush(arr1, i)
            else:
                heapq.heappush(arr2, i)
                
        if sumv % 3 == 1:
            if arr1:
                heapq.heappop(arr1)
            else:
                if len(arr2) < 2:
                    return ""
                
                heapq.heappop(arr2)
                heapq.heappop(arr2)
                
        elif sumv % 3 == 2:
            if arr2:
                heapq.heappop(arr2)
            else:
                if len(arr1) < 2:
                    return ""
                
                heapq.heappop(arr1)
                heapq.heappop(arr1)
                
        
        arr = arr0 + arr1 + arr2
        arr = [-x for x in arr]
        heapq.heapify(arr)
        
        st = ""
        while arr:
            dg = -(heapq.heappop(arr))
            st += str(dg)
            
        while len(st) > 1 and st[0] == '0':
            st = st[1:]
            
        return st
