class Solution:
    def largestMultipleOfThree(self, digits: List[int]) -> str:
        def convert(digits):
            if len(digits) == 0:
                return ""
            if digits[0] == 0:
                return "0"
            else:
                return "".join([str(d) for d in digits])
        digits.sort(reverse=True)
        
        sum_digit = sum(digits)
        if sum_digit % 3 == 0:
            return convert(digits)
        
        remove = None
        remove_modulo = sum_digit % 3 
        for i in range(len(digits) -1 ,-1, -1):
            if digits[i] % 3 == remove_modulo:
                remove = digits[i]
                break
        
        if remove != None:
            digits.remove(remove)
            return convert(digits)
        remove_modulo = 3 - remove_modulo
        remove1 = None
        remove2 = None
                
        for i in range(len(digits) -1 ,-1, -1):
            if digits[i] % 3 == remove_modulo:
                if remove1 == None:
                    remove1 = digits[i]
                elif remove2 == None:
                    remove2 = digits[i]
                else:
                    break
        
        digits.remove(remove1)
        digits.remove(remove2)
        return convert(digits)
