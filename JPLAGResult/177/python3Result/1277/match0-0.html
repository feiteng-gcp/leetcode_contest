class Solution:
    def largestMultipleOfThree(self, A: List[int]) -> str:
        sumv = sum(A)
        if sumv % 3==0:
            if sumv==0: 
                return "0"
            A.sort(reverse=True)
            return ''.join([str(x) for x in A])
        A.sort(reverse=True)
        delv = -1
        for x in A[::-1]:
            if x%3 == sumv%3:
                delv = x
                break
        if delv == -1: 
            A = [x for x in A if x%3==0]
        else:
            A.remove(delv)
        A.sort(reverse=True)
        return ''.join([str(x) for x in A])
