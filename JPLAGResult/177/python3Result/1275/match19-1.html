class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        cnts = [0] * n
        for child in leftChild:
            if child >= 0: cnts[child] += 1
        for child in rightChild:
            if child >= 0: cnts[child] += 1
        cnts_cnst = collections.Counter(cnts)
        return len(cnts_cnst) == 2 and cnts_cnst.get(0, 0) == 1 and cnts_cnst.get(1, 0) == n - 1
