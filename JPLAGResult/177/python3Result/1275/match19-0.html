class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        count = [0] * n
        for child in leftChild:
            if child >= 0:
                count[child] += 1
        for child in rightChild:
            if child >= 0:
                count[child] += 1
                
        total = 0
        
        for num in count:
            if num == 1:
                total += 1
            elif num > 1:
                return False
        
        return total == n - 1
