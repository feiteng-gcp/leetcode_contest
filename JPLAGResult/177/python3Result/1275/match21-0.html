class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        parent = [-1] * n
        for u, v in enumerate(leftChild):
            if v != -1:
                if parent[v] != -1:
                    return False
                parent[v] = u
        for u, v in enumerate(rightChild):
            if v != -1:
                if parent[v] != -1:
                    return False
                parent[v] = u
        return collections.Counter(parent)[-1] == 1
