class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        vis = [False] * n
        
        for a,b in zip(leftChild, rightChild):
            if a!=-1: vis[a] = True
            if b!=-1: vis[b] = True
                
        root = -1
        
        for i in range(n):
            if not vis[i]:
                if root!=-1: return False
                root = i
        
        if root==-1: return False
        
        self.num = 0
        vis = [False] * n
        self.flag = True
        
        def solver(r):
            
            if r==-1: return True
            self.num += 1
            if vis[r]: return False
            vis[r] = True
            
            return solver(leftChild[r]) and solver(rightChild[r])
        
        return solver(root) and self.num == n
