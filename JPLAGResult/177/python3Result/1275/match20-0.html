class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        
        d = [0] * n
        
        for i in range(n):
            
            if leftChild[i] != -1:
                d[ leftChild[i] ] += 1
                
            if rightChild[i] != -1:
                d[ rightChild[i] ] += 1
                
        c0 = 0
        
        for i in d:
            
            if i == 0:
                c0 += 1
                
            if i > 1:
                return False
            
        return c0 == 1
