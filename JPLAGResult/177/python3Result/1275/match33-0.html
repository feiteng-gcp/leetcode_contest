class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        vis = [0]*n
        for i in leftChild:
            if i>0:
                vis[i]=1
        for i in rightChild:
            if i>0:
                vis[i]=1
        root = []
        for i in range(n):
            if vis[i]==0:
                root.append(i)
        if len(root)!=1:
            return False
        vis = [0]*n
        def dfs(x):
            if x==-1:
                return True
            vis[x]=1
            if leftChild[x]!=-1 and vis[leftChild[x]]:
                return False
            if rightChild[x]!=-1 and vis[rightChild[x]]:
                return False
            return dfs(leftChild[x]) and dfs(rightChild[x])
        return dfs(root[0])
