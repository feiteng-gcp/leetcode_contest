class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        queue = [0]
        visited = [0]*len(leftChild)
        while queue:
            nque = []
            for n in queue:
                if visited[n]:
                    return False
                visited[n]=1
                if leftChild[n]!=-1:
                    nque.append(leftChild[n])
                if rightChild[n]!=-1:
                    nque.append(rightChild[n])
            queue = nque
        print(visited)
        return sum(visited)==len(leftChild)
