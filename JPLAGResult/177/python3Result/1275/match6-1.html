class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        inde = collections.defaultdict(int)
        oude = collections.defaultdict(list)
        m = len(leftChild)
        for i in range(n):
            a = leftChild[i]
            b = rightChild[i]
            if a != -1 and a == b: return False
            if a != -1:
                oude[i].append(a)
                inde[a] += 1
            if b != -1:
                oude[i].append(b)
                inde[b] += 1
        
        q = collections.deque()
        for i in range(n):
            if inde[i] == 0:
                q.append(i)
                
        seen = set([0])
        if len(q) != 1: return False
        while q:
            nxt = collections.deque()
            while q:
                cur = q.popleft()
                for child in oude[cur]:
                    if child in seen: return False
                    seen.add(child)
                    nxt.append(child)
            q = nxt
        
        # print(n, leftChild)
        return len(seen) == n
