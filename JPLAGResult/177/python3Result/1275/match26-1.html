class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        b = collections.deque()
        v = set()
        a = set(range(n))
        for i in range(n):
            a.discard(leftChild[i])
            a.discard(rightChild[i])
        if len(a) != 1:
            return False
        b.append(min(a))
        v.add(b[0])
        
while b:
            p = b.popleft()
            for i in [leftChild[p], rightChild[p]]:
                if i == -1:
                    continue
                if i in v:
                    return False
                v.add(i)
                b.append(i)
        return len(v) == n
