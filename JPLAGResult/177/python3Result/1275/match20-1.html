class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        parents_count = [0] * n
        for i in range(n):
            if leftChild[i] >= 0:
                parents_count[leftChild[i]] += 1
            if rightChild[i] >= 0:
                parents_count[rightChild[i]] += 1
        zero, one = 0, 0
        for i in range(n):
            if parents_count[i] == 0:
                zero += 1
            elif parents_count[i] == 1:
                one += 1

        if zero == 1 and one == n - 1:
            return True
        return False
