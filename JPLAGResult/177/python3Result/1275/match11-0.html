class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        indegree = collections.Counter()
        outdegree = collections.Counter()
        for i in range(n):
            if leftChild[i] != -1:
                outdegree[i] += 1
                indegree[leftChild[i]] += 1
            if rightChild[i] != -1:
                outdegree[i] += 1
                indegree[rightChild[i]] += 1
        root = 0
        for i in range(n):
            if indegree[i] == 0:
                root += 1
            if indegree[i] > 1:
                return False
            if outdegree[i] > 2:
                return False
        return root == 1
