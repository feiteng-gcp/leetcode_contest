class Solution:
    def validateBinaryTreeNodes(self, n: int, lC: List[int], rC: List[int]) -> bool:
        root = -1
        memo = [1] * len(lC)
        for i, j in zip(lC, rC):
            if i != -1:
                memo[i] = 0
            if j != -1:
                memo[j] = 0
        for i in range(n):
            if memo[i]:
                root = i
                break
        if root == -1:
            return False
        memo = [-1] * len(lC)
        que = collections.deque([root])
        while que:
            x = que.popleft()
            if memo[x] != -1: return False
            memo[x] = 1
            if lC[x] != -1: que.append(lC[x])
            if rC[x] != -1: que.append(rC[x])
        
        for i in memo:
            if i == -1:
                return False
        return True
