class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        fa = {i: -1 for i in range
(n)}
        for i, ch in enumerate(leftChild):
            if ch != -1:
                if fa[ch] != -1:
                    return False
                fa[ch] = i
        for i, ch in enumerate(rightChild):
            if ch != -1:
                if fa[ch] != -1:
                    return False
                fa[ch] = i
        
        rcnt = sum(1 for k, v in fa.items() if v == -1)
        if rcnt != 1:
            return False
        
        return True
