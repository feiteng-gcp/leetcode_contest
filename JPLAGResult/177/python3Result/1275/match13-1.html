class Solution:
    def validateBinaryTreeNodes(self, n: int, l: List[int], r: List[int]) -> bool:
        ind = collections.defaultdict(int)
        outd = {}
        for i in range(n):
            outd[i] = 2
            
if l[i] == -1:
                outd[i] -= 1
            else:
                ind[l[i]] += 1
            if r[i] == -1:
                outd[i] -= 1
            else:
                ind[r[i]] += 1
        root = 1
        for i in range(n):
            if ind[i] == 0:
                root -= 1
            elif ind[i] != 1:
                return False
        if root != 0:
            return False
        for i in range(n):
            if outd[i] > 2:
                return False
        return True
