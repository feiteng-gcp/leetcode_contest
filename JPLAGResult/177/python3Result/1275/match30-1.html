class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        if n == 0:
            return True
        in_degree 
= [0] * n
        for i in range(n):
            if leftChild[i] > -1:
                in_degree[leftChild[i]] += 1
            if rightChild[i] > -1:
                in_degree[rightChild[i]] += 1
        if sum(1 for i in range(n) if in_degree[i] == 0) != 1:
            return False
        
        seen= [0] * n
        def dfs(cur):
            seen[cur] = 1
            nxt_nodes = [leftChild[cur], rightChild[cur]]
            for nxt in nxt_nodes:
                if nxt == -1:
                    continue
                if seen[nxt] == 1 or seen[nxt] == 2 or not dfs(nxt):
                    return False
            seen[cur] = 2
            return True
        
        for i in range(n):
            if in_degree[i] == 0:
                return dfs(i)
        return True
