class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        par = [None] * n
        cc = [x for x in range(n)]
        for i, e in enumerate(leftChild):
            if e != -1:
                if par[e] is not None:
                    return False
                par[e] = i
        for i, e in enumerate(rightChild):
            if e != -1:
                if par[e] is not None:
                    return False
                par[e] = i
        ct = 0
        for i in range(n):
            if par[i] is None:
                ct += 1
        # print(par, ct)
        return ct == 1
