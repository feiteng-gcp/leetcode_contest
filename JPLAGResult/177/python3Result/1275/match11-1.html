class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        has_parent = [0] * n
        children = [0] * n
        
for i in range(n):
            if leftChild[i] != -1:
                children[i] += 1
                has_parent[leftChild[i]] += 1
            if rightChild[i] != -1:
                children[i] += 1
                has_parent[rightChild[i]] += 1
        
        zero_parent = False
        for i in range(n):
            if has_parent[i] == 0:
                if zero_parent:
                    return False
                zero_parent 
= True
            if has_parent[i] == 2:
                return False
            if children[i] > 2:
                return False
        
        return zero_parent
