class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        p = collections.defaultdict(int)
        for l, r in zip(leftChild, rightChild):
            if l != -1:
                p[l] += 1
            if r != -1:
                p[r] += 1
        n1, n2 = 0, 0
        for k, v in p.items():
            if v == 1:
                n2 += 1
            else:
                n1 += 1
        return n2 == n-1 and len(p) == n-1 and n1 == 0
