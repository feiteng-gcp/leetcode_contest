class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        visited = set([0])
        q = collections.deque([0])
        while q:
            node = q.popleft()
            for nxt in [leftChild[node], rightChild[node]]:
                if nxt == -1:
                    continue
                if nxt in visited:
                    return False
                q.append(nxt)
                visited.add(nxt)
        
        return len(visited) == n
