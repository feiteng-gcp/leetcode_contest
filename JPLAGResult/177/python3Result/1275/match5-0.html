class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        l = [i for i in leftChild if i!= -1]
        r = [i for i in rightChild if i != -1]
        if len(set(l) & set(r)) > 0 or len(set(l)) + len(set(r)) != n-1:
            return False
        return True
