from collections import deque

class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        s = set()
        for i in range(n):
            s.add(i)
        q = deque([0])
        while len(q):
            cur = q.popleft()
            if cur not in s:
                return False
            s.remove(cur)
            if leftChild[cur]!=-1:
                q.append(leftChild[cur])
            if rightChild[cur]!=-1:
                q.append(rightChild[cur])
        return len(s)==0
