class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        
        root = 0
        stack = [root]
        
        considered 
= set()
        
        while len(stack) > 0:
            pop = stack.pop()
            
            if pop in considered:
                return False
            
            considered.add(pop)
            
            if leftChild[pop] > -1:
                stack.append(leftChild[pop])
            if rightChild[pop] > -1:
                stack.append(rightChild[pop])
                
        for l in leftChild:
            if l > -1 and l not in considered:
                return False
            
        for r in rightChild:
            if r > -1 and r not in considered:
                return False
            
        return True
