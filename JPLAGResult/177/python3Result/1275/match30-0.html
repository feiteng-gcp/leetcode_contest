class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        visited = [0 for _ in leftChild]
        for i in range(n):
            if leftChild[i] != -1:
                visited[leftChild[i]] += 1
            if rightChild[i] != -1:
                visited[rightChild[i]] += 1
        
        c = collections.Counter(visited)
        #print(visited, c)
        return c[0] == 1 and c[1] == n-1
