class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        #n,leftChild,rightChild = 5, [2,2,-1,1,3], [2,2,1,2,1]
        #n,leftChild,rightChild = 5,[0,-1,3,1,3],[4,3,0,1,-1]
        vis = [False] * n
        for a,b in zip(leftChild, rightChild):
            if a!=-1: vis[a] = True
            if b!=-1: vis[b] = True
        root = -1
        for i in range(n):
            if not vis[i]:
                if root!=-1: return False
                root = i
        print(root)
        if root==-1: return False
        self.num = 0
        vis = [False] * n
        self.flag = True
        def solver(r):
            if r==-1: return True
            self.num += 1
            if vis[r]: return False
            vis[r] = True
            return solver(leftChild[r]) and solver(rightChild[r])
        return solver(root) and self.num == n
