class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        parents = [None]*n
        for i in range(n):
            left = leftChild[i]
            right = rightChild[i]
            
            
if left != -1:
                if parents[left] != None:
                    return False
                else:
                    parents[left] = i
            if right != -1:
                if parents[right] != None:
                    return False
                else:
                    parents[right] = i
        
        root_count = 0
        for p in parents:
            if p is None:
                root_count += 1
        return root_count == 1
