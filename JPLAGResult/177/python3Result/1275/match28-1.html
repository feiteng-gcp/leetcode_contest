class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        
        def bfs(src, left, right, done):
            queue = [src]
            while queue:
                cur = queue.pop(0)
                
if done[cur]:
                    return False
                done[cur] = True
                if left[cur] != -1:
                    queue.append(left[cur])
                if right[cur] != -1:
                    queue.append(right[cur])
            return all(done)
        
        done = [False] * n
        return bfs(0, leftChild, rightChild, done)
