class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        is_child = set()
        ct = 0
        for i in leftChild:
            is_child.add(i)
            if i == -1:
                ct += 1
        for i in rightChild:
            is_child.add(i)
            if i == -1:
                ct 
+= 1
        root = []
        for i in range(n):
            if i not in is_child:
                root.append(i)
        if len(root)!=1:
            return False
        if ct != n+1:
            return False
        return True
