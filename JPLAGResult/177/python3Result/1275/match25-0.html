class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: List[int], rightChild: List[int]) -> bool:
        visited = [False] * n
        for l, r in zip(leftChild, rightChild):
            if l != -1:
                if visited[l]:
                    return False
                else:
                    visited[l] = True
            if r != -1:
                if visited[r]:
                    return False
                else:
                    visited[r] = True
        res = sum([1 if not x else 0 for x in visited])
        
        return res == 1
