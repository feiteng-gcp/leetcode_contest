class Solution:
    def validateBinaryTreeNodes(self, n: int, leftChild: 'List[int]', rightChild: 'List[int]') -> bool:
        pre={}
        l=len(leftChild)
        edges=0
        for i in range(l):
            l1,r1=leftChild[i],rightChild[i]
            if l1!=-1:
                if l1 in pre:
                    return False
                pre[l1]=i
                edges+=1
            if r1!=-1:
                if r1 in pre:
                    return False
                pre[r1]=i
                edges+=1
        nopa=[]
        for i in range(n):
            if i not in pre:
                nopa.append(i)
        if len(nopa)!=1:
            return False
        if edges!=n-1:
            return False
        return True
