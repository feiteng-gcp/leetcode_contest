class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        res = [1, num+1]
        end = int(math.sqrt
(num+2))+1
        for i in range(1, end):
            if (num+1)%i == 0:
                if abs((num+1)//i-i) < abs(res[0]-res[1]):
                    res = [(num+1)//i, i]
            if (num+2)%i == 0:
                if abs((num+2)//i-i) < abs(res[0]-res[1]):
                    res = [(num+2)//i, i]
        return res
