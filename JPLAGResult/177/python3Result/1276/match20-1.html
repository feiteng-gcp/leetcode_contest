class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        from math import sqrt
        def cal(num):
            for i in range(int(sqrt(num)),0,-1):
                
if num%i==0:
                    return i,num//i
        a,b=cal(num+1)
        c,d=cal(num+2)
        return [a,b] if abs(a-b)<=abs(c-d) else [c,d]
