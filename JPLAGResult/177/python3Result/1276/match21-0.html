import math


class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        def closest_divisors_of_num(n):
            d1 = math.floor(math.sqrt(n))
            while n % d1 != 0:
                d1 -= 1
            return [d1, n // d1]
        
        ans1 = closest_divisors_of_num(num + 1)
        ans2 = closest_divisors_of_num(num + 2)
        diff1 = abs(ans1[1] - ans1[0])
        diff2 = abs(ans2[1] - ans2[0])
        if diff1 <= diff2:
            return ans1
        return ans2
