import math
class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        n1 = num + 1
        n2 = num + 2
        r = int(math.sqrt(n2))
        for i in range(r, 0, -1):
            if n1 % i == 0:
                return [i, n1 // i]
            if n2 % i == 0:
                return [i, n2 // i]
