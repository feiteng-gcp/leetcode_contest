class Solution:
    def divide(self, num):
        import math
        m = int(math
.floor(math.sqrt(num)))
        for i in range(m, 0, -1):
            if num % i == 0:
                return (i, num // i)
    def closestDivisors(self, num):
        d1 = self.divide(num + 1)
        d2 = self.divide(num + 2)
        if abs(d1[0] - d1[1]) <= abs(d2[0] - d2[1]):
            return d1
        else:
            return d2
