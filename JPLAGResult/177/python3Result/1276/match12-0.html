class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        def getDiv(x):
            for i in range(int(math.sqrt(x)),0,-1):
                div,mod=divmod(x,i)
                if not mod:
                    return [div,i]
        a1,b1=getDiv(num+1)
        a2,b2=getDiv(num+2)
        if abs(a1-b1)>abs(a2-b2):
            return [a2,b2]
        return [a1,b1]
