from math import sqrt

class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        ans1 = self.findDivisors(num + 1)
        ans2 = self.findDivisors(num + 2)
        if abs(ans1[1] - ans1[0]) > abs(ans2[1] - ans2[0]):
            return ans2
        else:
            return ans1
        
    def findDivisors(self, num):
        min_ = 1 if (num % 2) == 0 else 2
        # print(num, int(sqrt(num)), min_)
        for i in range(int(sqrt(num)), min_, -1):
            if num % i == 0:
                return [num // i, i]
        return [1, num]
