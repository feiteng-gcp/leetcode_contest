def closeDiv(n):
    s=math.floor(math.sqrt(n));
    for i in range(s,0,-1):
        if(n%i==0):
            return [i,n//i];

class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        ans1=closeDiv(num+1);
        ans2=closeDiv(num+2);
        if(ans1[1]-ans1[0]<ans2[1]-ans2[0]):
            return ans1;
        else:
            return ans2;
