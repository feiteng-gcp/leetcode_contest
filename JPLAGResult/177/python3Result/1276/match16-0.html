class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        def find2(num):
            for i in range(int(num**0.5), 0, -1):
                if num % i == 0:
                    return i, num // i
        d11, d12 = find2(num+1)
        d21, d22 = find2(num+2)
        # print([d11, d12], [d21, d22])
        if abs(d11-d12) <= abs(d21-d22):
            return [d11, d12]
        else:
            return [d21, d22]
