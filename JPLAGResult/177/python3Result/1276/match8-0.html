class Solution:
    def closestDivisors(self, num: int) -> 'List[int]':
        def findk(k):
            ss=int(k**0.5)
            for i in range(ss,0,-1):
                if k%i==0:
                    return [i,k//i]
        a=findk(num+1)
        b=findk(num+2)
        if abs(a[0]-a[1])<=abs(b[0]-b[1]):
            return a
        return b
