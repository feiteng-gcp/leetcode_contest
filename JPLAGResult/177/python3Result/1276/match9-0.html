class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        from math import sqrt
        def helper(num):
            
            mid = int(sqrt(num))
            while num % mid != 0:
                mid -= 1
            return [mid, int(num/mid)]
        solve1 = helper(num+1)
        solve2 = helper(num+2)
        
        if abs(solve1[0] - solve1[1]) > abs(solve2[0] - solve2[1]):
            return solve2
        return solve1
