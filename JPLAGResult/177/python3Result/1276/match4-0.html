from functools import reduce
class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        def factors(n):    
            return set(reduce(list.__add__, 
                        ([i, n//i] for i in range(1, int(n**0.5) + 1) if n % i == 0)))
        res = (1,num+1)
        cur = num
        for i in factors(num+1):
            if abs((num+1)//i - i) < cur:
                res = (i,(num+1)//i)
                cur = abs(i-(num+1)//i)
                print(cur)
        for i in factors(num+2):
            if abs((num+2)//i - i) < cur:
                res = (i,(num+2)//i)
                cur = abs(i-(num+2)//i)
        return res
