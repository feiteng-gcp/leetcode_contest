class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        n1, n2 = num + 1, num + 2
        def close(n):
            for i in range(int(n ** 0.5), 0, -1):
                j = n // i
                if i * j == n:
                    return j, i
        
        
        v1, v2 = close(n1)
        v3, v4 = close(n2)
        return [v1, v2] if abs(v1 - v2) <= abs(v3 - v4) else [v3, v4]
