from typing import List

import math


class Solution:
  def closestDivisors(self, num: int) -> List[int]:
    def calc(num):
      s = math.floor(math.sqrt(num))

      while num % s != 0:
        s -= 1

      return [s, num // s]

    a1, a2 = calc(num + 1)
    b1, b2 = calc(num + 2)

    if abs(a2 - a1) < abs(b2 - b1):
      return [a1, a2]
    return [b1, b2]
