class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        res = [0, 10**9]
        k = int((num+2)**0.5)
        for i in range(1, k+1):
            
            if (num+1) % i == 0:
                if abs(i-(num+1) // i) < abs(res[0]-res[1]):
                    res = [i, (num+1)//i]
            if (num+2) % i == 0:
                if abs(i-(num+2) // i) < abs(res[0]-res[1]):
                    
                    res = [i, (num+2)//i]
        
        return res
