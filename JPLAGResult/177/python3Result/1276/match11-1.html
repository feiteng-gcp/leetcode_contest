class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        def find(num):
            for i in range(int(num**0.5)+1, 0, -1):
                if num % i == 0:
                    return i, num // i
        x1, y1 = find(num + 1)
        x2, y2 = find(num + 2)
        return [x1,y1] if abs(x1-y1) <= abs(x2-y2) else [x2,y2]
