class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        attempt1 = self.closest(num + 1)
        attempt2 = self.closest(num + 2)
        
        if attempt1[1] - attempt1[0] < attempt2[1] - attempt2[0]:
            return attempt1
        else:
            return attempt2
    
    def closest(self, num: int) -> List[int]:
        d = int(math.sqrt(num))
        while num % d != 0:
            d -= 1
        return [d, num // d]
