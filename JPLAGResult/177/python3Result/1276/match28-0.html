class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        
        def close(n):
            minv = float('inf')
            a, b = 0, 0
            
            for i in range(1, int(n**0.5) + 1):
                if n % i == 0:
                    
                    if abs(n//i - i) < minv:
                        minv = abs(n//i - i)
                        a = n//i
                        b = i
                        
            return [a, b]
        
        a1, b1 = close(num + 1)
        a2, b2 = close(num + 2)
        
        if abs(a1 - b1) < abs(a2 - b2):
            return [a1, b1]
        return [a2, b2]
