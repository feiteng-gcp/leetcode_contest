class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        def close_exact_divisor(x):
            for i in range(int(x**0.5)+1, 0, -1):
                if x%i == 0:
                    return i, x//i

        x1, y1 = close_exact_divisor(num+1)
        x2, y2 = close_exact_divisor(num+2)
        if abs(x1-y1) < abs(x2 - y2):
            return [x1, y1]
        return [x2, y2]
