class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        l, r = 1, num + 2
        while l < r:
            mid = l + (r - l + 1) // 2
            if mid ** 2 > num + 2:
                r = mid - 1
            else:
                l = mid

        for x in range(l, 0, -1):
            if (num + 1) % x == 0:
                return [x, (num + 1) // x]
            if (num + 2) % x == 0:
                return [x, (num + 2) // x]
