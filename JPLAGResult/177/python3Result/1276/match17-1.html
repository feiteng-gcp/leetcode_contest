class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        for i in range(int((num+1)**0.5), 0, -1):
            if (num+1) % i == 0:
                ans = [i, (num+1) // i]
                break
        for i in range(int((num+2)**0.5), 0, -1):
            if (num+2) % i == 0:
                ans_ = [i, (num+2) // i]
                break
        # print(ans, ans_)
        return ans if abs(ans[0]-ans[1]) < abs(ans_[0]-ans_[1]) else ans_
