class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        def divisors(num):
            sq = int(num**0.5)
            for i in range(sq, 0, -1):
                if num % i == 0:
                    return [i, num//i]
        ans1 = divisors(num+1)
        ans2 = divisors(num+2)
        if abs(ans1[0]-ans1[1]) < abs(ans2[0]-ans2[1]):
            return ans1
        return ans2
