class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        a = self.get_r(num+1)
        b = self.get_r(num+2)
        if a[1] - a[0] <= b[1] - a[0]:
            return a
        else:
            return b
    
    def get_r(self, n):
        import math
        x = int(math.sqrt(n)) +1
        for i in list(range(1, x+1))[::-1]:
            if n%i == 0:
                return [i, int(n/i)]
