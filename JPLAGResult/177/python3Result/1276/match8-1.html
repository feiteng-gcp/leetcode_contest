class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        ans = float("inf")

        def f(num1):
            for c in range(int
(num1**0.5), 0, -1):
                if num1 % c == 0:
                    return [c, num1//c]

        ans = f(num+1)
        c = f(num + 2)
        if abs(c[0] - c[1]) < abs(ans[0] - ans[1]):
            ans = c[:]
        return ans
