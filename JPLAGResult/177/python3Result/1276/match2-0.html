class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        a1 = self.get_divisor(num+1)
        #print(a1)
        diff1 = a1[1]-a1[0]
        a2 = self.get_divisor(num+2)
        #print(a2)
        diff2 = a2[1]-a2[0]
        if diff1<diff2:
            return a1
        return a2
        
    
    def get_divisor(self, n):
        k = int(n**0.5)
        while k>1:
            if n%k==0:
                return k, n//k
            k-=1
        return k, n//k
