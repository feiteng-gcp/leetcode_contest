from math import sqrt
from math import ceil

class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        def maxProd(n):
            sq = int( sqrt(num))
            if sq * sq == n:
                return sq
            for i in range(int(sqrt(num) + 1), 0, -1):
                if n % i == 0:
                    return i
        def cand(n):
            x = maxProd(n)
            return [x, n//x]
        c1 = cand(num+1)
        c2 = cand(num+2)
        if abs(c1[0]-c1[1]) <= abs(c2[0]-c2[1]):
            return c1
        return c2
