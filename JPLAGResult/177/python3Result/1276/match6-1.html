class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        
        def find_close(n):
            s = int(math.sqrt(n))
            if s * s == n:
                return s
            
            while n % s != 0:
                s -= 1
            
            return s
        
        a = find_close(num+1)
        b = (num+1) //a 
        
        c = find_close(num+2)
        d = (num+2) // c
        
        if abs(a-b) < abs(c-d):
            return [a, b]
        else:
            return [c, d]
