class Solution:
    def closestDivisors(self, num: int) -> List[int]:
            
        res = None
        for n in [num + 1, num + 2]:
            
            for i in range(1, int((n + 1) ** 0.5 + 1)):
                if n % i == 0:
                    if not res:
                        res = (i, n // i)
                    else:
                        if n // i - i < (res[1] - res[0]):
                            res = (i, n // i)
                            
        return res
