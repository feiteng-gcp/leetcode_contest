class Solution:
    def closestDivisors(self, num: int) -> List[int]:
        
        
        for i in range(int((num+1)**0.5), 0, -1):
            if (num+1) % i == 0:
                ret1 = [i,(num+1)//i]
                break
        for i in range(int((num+2)**0.5), 0, -1):
            if (num+2) % i == 0:
                ret2 = [i,(num+2)//i]
                break
        
        if ret1[1] - ret1[0] >= ret2[1] - ret2[0]:
            return ret2
        else:
            return ret1
