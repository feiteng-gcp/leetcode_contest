class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        // find the root
        vector<int> incoming(n);
        for (int i=0; i<n; i++) {
            if (leftChild[i] != -1)
                incoming[leftChild[i]]++;
            if (rightChild[i] != -1)
                incoming[rightChild[i]]++;
        }

        int root = -1;
        for (int i=0; i<n; i++) {
            if (incoming[i] == 0) {
                if (root != -1)
                    return false;
                root = i;
            }
            else if (incoming[i] > 1)
                return false;
        }
        return root != -1;
    }
};
