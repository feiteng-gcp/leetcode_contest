class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& l, vector<int>& r) {

        vector<int>d=vector<int>(n, 0);
        for(int i=0;i<n;i++){
            if(l[i]!=-1)d[l[i]]++;
            if(r[i]!=-1)d[r[i]]++;
        }
        int cnt = 0;
        for(int i=0;i<n;i++){
            if(d[i]>=2)return false;
            if(d[i]==0)cnt++;
        }
        return cnt==1;
    }
};
