class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> ind(n, 0), outd(n, 0);
        
        for(int i = 0; i < n; ++i) {
            int l = leftChild[i], r = rightChild[i];
            if(l != -1) outd[i]++, ind[l]++;
            if(r != -1) outd[i]++, ind[r]++;
        }
        
        int cnt = 0;
        for(int i = 0; i < n; ++i) {
            if(ind[i] == 0) ++cnt;
            if(ind[i] > 1) return false;
            if(outd[i] > 2) return false;
        }
        if(cnt != 1) return false;
        return true;
    }
};
