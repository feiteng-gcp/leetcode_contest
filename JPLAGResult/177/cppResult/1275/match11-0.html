class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        // get parents
        vector<int> parent(n, -1);
        for(int i=0;i<n;++i) {
            if(leftChild[i] != -1) {
                if(parent[leftChild[i]] != -1) {
                    cout << "left node " << leftChild[i] << " has multiple parents" << endl;
                    return false;
                }
                parent[leftChild[i]] = i;
            }
            if(rightChild[i] != -1) {
                if(parent[rightChild[i]] != -1) {
                    cout << "right node " << rightChild[i] << " has multiple parents" << endl;
                    return false;
                }
                parent[rightChild[i]] = i;
            }
        }
        
        // find root
        int root = -1;
        for(int i=0;i<n;++i) {
            if(parent[i] == -1) {
                if(root != -1) {
                    cout << "multiple root" << endl;
                    return false;
                }
                root = i;
            }
        }
        if(root == -1)
            return false;
        
        return true;
    }
};
