class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> p(n,-1);
        for(int i=0;i<n;i++){
            int c=leftChild[i];
            if(c>=0){
                if(p[c]!=-1) return false;
                p[c]=i;
            }
        }
        
        for(int i=0;i<n;i++){
            int c=rightChild[i];
            if(c>=0){
                if(p[c]!=-1) return false;
                p[c]=i;
            }
        }
        
        int root=-1;
        for(int i=0;i<n;i++){
            if(p[i]==-1){
                if(root==-1){
                    root=i;
                }else return false;
            }
        }
        if(root==-1) return false;
        vector<bool> seen(n,false);
        queue<int> qq;
        qq.push(root);
        seen[root]=true;
        while(!qq.empty()){
            int c=qq.front();
            qq.pop();
            int l=leftChild[c];
            int r=rightChild[c];
            if(l>=0&&!seen[l]){
                qq.push(l);
                seen[l]=true;
            }
            if(r>=0&&!seen[r]){
                qq.push(r);
                seen[r]=true;
            }
        }
        for(int i=0;i<n;i++){
            if(!seen[i]) return false;
        }
        return true;
    }
};
