class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> parent(n, -1);

        for (int i = 0; i < n; i++) {
            if (leftChild[i] == i || rightChild[i] == i)
                return false;


            if (leftChild[i] >= 0) {
                if (parent[leftChild[i]] >= 0)
                    return false;

                parent[leftChild[i]] = i;
            }

            if (rightChild[i] >= 0) {
                if (parent[rightChild[i]] >= 0)
                    return false;

                parent[rightChild[i]] = i;
            }
        }

        int root = -1;

        for (int i = 0; i < n; i++)
            if (parent[i] < 0) {
                if (root >= 0)
                    return false;

                root = i;
            }

        if (root < 0)
            return false;

        vector<bool> visited(n, false);
        queue<int> q;
        visited[root] = true;
        q.push(root);

        while (!q.empty()) {
            int front = q.front(); q.pop();

            if (leftChild[front] >= 0) {
                if (visited[leftChild[front]])
                    return false;

                visited[leftChild[front]] = true;
                q.push(leftChild[front]);
            }

            if (rightChild[front] >= 0) {
                if (visited[rightChild[front]])
                    return false;

                visited[rightChild[front]] = true;
                q.push(rightChild[front]);
            }
        }

        for (int i = 0; i < n; i++)
            if (!visited[i])
                return false;

        return true;
    }
};
