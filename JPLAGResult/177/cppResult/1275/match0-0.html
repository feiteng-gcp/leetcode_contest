class Solution {
public:
    //An undirected graph is tree if it has following properties.
    //1) There is no cycle.
    //2) The graph is connected.
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> indegree(n, 0);;
        for (int i = 0; i < n; i++) { 
            if (leftChild[i] >= 0) { 
                indegree[leftChild[i]]++;
            }
            if (rightChild[i] >= 0) { 
                indegree[rightChild[i]]++;
            }
        }
        int source = -1;
        for (int i = 0; i < n; i++) { 
            if (source >= 0 && indegree[i] == 0) { 
                return false;
            }
            else if (indegree[i] == 0) { 
                source = i;
            }
        }
        if (source == -1) { 
            return false;
        }
        vector<bool> visited(n, false);
        queue<int> q;
        q.push(source);
        visited[source] = true;
        while (!q.empty()) { 
            int node = q.front();
            q.pop();
            int l = leftChild[node];
            int r = rightChild[node];
            if (l >= 0) { 
                if (visited[l]) { 
                    return false;
                }
                else { 
                    visited[l] = true;
                    q.push(l);
                }
            }
            if (r >= 0) { 
                if (visited[r]) { 
                    return false;
                }
                else { 
                    visited[r] = true;
                    q.push(r);
                }
            }
        }
        for (int i = 0; i < n; i++) { 
            if (visited[i] == false) { 
                return false;
            }
        }
        return true;
    }
};
