class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> in_deg(n, 0);
        for(auto n : leftChild){
            if(n != -1){
                in_deg[n] += 1;
            }
        }
        for(auto n : rightChild){
            if(n != -1){
                in_deg[n] += 1;
            }
        }
        int cnt0 = 0;
        for(auto i : in_deg){
            if(i == 0) cnt0+=1;
            else if(i != 1) return false;
        }
        return cnt0 == 1;
    }
};
