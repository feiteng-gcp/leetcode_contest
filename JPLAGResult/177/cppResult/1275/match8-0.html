class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> in(n, 0);
        for (int& n : leftChild) {
            if (n != -1) {
                ++in[n];
            }
        }
        
        for (int& n : rightChild) {
            if (n != -1) {
                ++in[n];
            }
        }
        
        if (in[0] > 0) return false;
        int all = 0;
        for (int n : in) {
            if (n > 1) return false;
            all += n;
        }
        return all == n - 1;
    }
};
