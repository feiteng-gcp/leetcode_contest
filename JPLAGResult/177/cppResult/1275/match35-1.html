class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> num(n, 0);
        for(int i = 0; i < n; ++i) {
            if(leftChild[i] >= 0) ++num[leftChild[i]];
            if(rightChild[i] >= 0) ++num[rightChild[i]];
        }
        int zero = 0;
        for(int i = 0; i < n; ++i) {
            if(0 == num[i]) {
                if(++zero > 1) return false;
            } else if(num[i] > 1) {
                return false;
            }
        }
        return zero == 1;
        
    }
};
