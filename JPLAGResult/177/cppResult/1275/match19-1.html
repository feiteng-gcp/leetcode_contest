class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> ct(n);
        for(auto i:leftChild){
            if(i!=-1){
                ct[i]++;
            }
        }
        for(auto i:rightChild){
            if(i!=-1){
                ct[i]++;
            }
        }
        int zero = 0;
        for(auto i:ct){
            if(i==0)zero++;
            if(i>1)return false;
        }
        return zero == 1;
    }
};
