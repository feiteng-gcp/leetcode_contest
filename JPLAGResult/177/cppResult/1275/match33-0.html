class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int> &leftChild, vector<int> &rightChild) {
        vector<int> parents(n, -1);
        for (int i = 0; i < n; i++) {
            int l = leftChild[i];
            int r = rightChild[i];
            if (l != -1) {
                if (parents[l] != -1) return false;
                parents[l] = i;
            }
            if (r != -1) {
                if (parents[r] != -1) return false;
                parents[r] = i;
            }
        }
        int roots = 0;
        for (int i : parents) roots += (i == -1);
        return roots == 1;
    }
};
