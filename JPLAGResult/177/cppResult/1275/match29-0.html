class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        
        vector<int> in_degrees(n, 0);
        for(int i=0;i<n;i++) {
            if (leftChild[i]>=0) {
                in_degrees[leftChild[i]]++;
            }
            if (rightChild[i]>=0) {
                in_degrees[rightChild[i]]++;                
            }
        }
        
        int rt=-1;
        for(int i=0;i<n;i++) {
            if (in_degrees[i]>1)
                return false;
            if (in_degrees[i]==0) {
                if (rt != -1)
                    return false;
                rt = i;
            }
        }
        if (rt == -1)
            return false;
        return true;
    }
};
