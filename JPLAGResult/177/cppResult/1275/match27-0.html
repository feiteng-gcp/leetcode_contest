class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        unordered_set<int> st;
        for (int i = 0; i < n; ++i) {
            if (leftChild[i] != -1) {
                if (st.count(leftChild[i]) || leftChild[i] == i) return false;
                st.insert(leftChild[i]);
            }
            if (rightChild[i] != -1) {
                if (st.count(rightChild[i]) || rightChild[i] == i) return false;
                st.insert(rightChild[i]);
            }         
        }
        return st.size() == n - 1;
    }
};
