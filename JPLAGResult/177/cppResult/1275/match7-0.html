class Solution {
   public:
    bool validateBinaryTreeNodes(const int N, vector<int> &L, vector<int> &R) {
        if (N < 1) return true;
        if (N == 1) {
            return L[0] == -1 && R[0] == -1;
        }
        // input count is 0
        int in[N];
        memset(in, 0, sizeof(in));

        for (int i = 0; i < N; i++) {
            const int par = i;
            const int l = L[i];
            const int r = R[i];
            if (l != -1) {
                in[l]++;
            }
            if (r != -1) {
                in[r]++;
            }
        }

        int root = -1;
        int root_cnt = 0;
        for (int i = 0; i < N; i++) {
            if (in[i] == 0) {
                root = i;
                root_cnt++;
            }
        }

        if (root_cnt != 1) return false;

        vector<int> cur{root}, tmp;
        bool vis[N];
        memset(vis, 0, sizeof(vis));

        while (!cur.empty()) {
            tmp.clear();

            for (auto &c: cur) {
                int l = L[c];
                int r = R[c];

                if (l != -1 && vis[l]) return false;
                if (r != -1 && vis[r]) return false;

                if (l != -1) {
                    tmp.push_back(l);
                    vis[l] = true;
                }
                if (r != -1) {
                    tmp.push_back(r);
                    vis[r] = true;
                }
            }

            cur.swap(tmp);
        }

        return true;
    }

};
