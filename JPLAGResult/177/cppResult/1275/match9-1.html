class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> visited(n, 0);
        vector<int> pointed(n, 0);
        vector<int> parent(n, -1);
        for(int i = 0; i < n; ++i){
            if(leftChild[i] != -1){
                int l = leftChild[i];
                if(pointed[l]) return false;
                pointed[l] = 1;
                //visited[l] = 1;
                if(!UNION(l, i, parent)) return false;
            }
            if(rightChild[i] != -1){
                int r = rightChild[i];
                if(pointed[r]) return false;
                pointed[r] = 1;
                //visited[r] = 1;
                if(!UNION(r, i, parent)) return false;
            }
            visited[i] = 1;
        }
        int p = -1;
        for(int i = 0; i < n; ++i){
            int pi = getParent(i, parent);
            if(p == -1){
                p = pi;
            }
            //cout<<pi<<endl;
            if(p != pi) return false;
        }
        //return accumulate(visited.begin(), visited.end()) == n;
        return true;
    }
    
    bool UNION(int x, int y, vector<int>& parent){
        int px = getParent(x, parent);
        int py = getParent(y, parent);
        if(px == py) return false;
        parent[x] = py;
        return true;
    }
    
    int getParent(int x, vector<int>& parent){
        int t = x;
        if(parent[x] == -1){
            parent[x] = x;
            return x;
        }
        while(parent[t] != t){
            t = parent[t];
        }
         parent[x] = t;
        return t;
     }
};
