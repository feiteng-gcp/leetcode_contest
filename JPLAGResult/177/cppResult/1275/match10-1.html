class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> visit(0);
        int c=0;
        for (int i=0;i<n;i++) visit.push_back(0);
        for (auto a:leftChild)
        {
            if (a==-1) continue;
            visit[a]++;
            c++;
        }
        for (auto b:rightChild)
        {
            if (b==-1) continue;
            visit[b]++;
            c++;
        }
        if (c!=n-1) return false;
        for (auto a:visit)
        {
            if (a>1) return false;
        }
        return true;
    }
};
