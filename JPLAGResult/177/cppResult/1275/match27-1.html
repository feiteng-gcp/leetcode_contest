class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& left, vector<int>& right) {
        set<int> st;
        for (int i=0; i<n; i++)
            st.insert(i);
        for (auto l : left)
            st.erase(l);
        for (auto r : right)
            st.erase(r);
        if(st.size()!=1)
            return false;
        int root=*st.begin();
        queue<int> q;
        st.clear();
        q.push(root); st.insert(root);
        while(!q.empty()) {
            int cur=q.front(); q.pop();
            int l=left[cur], r=right[cur];

            if(l!=-1) {
                if(st.count(l))
                    return false;
                st.insert(l);
                q.push(l);
            }
            if(r!=-1) {
                if(st.count(r))
                    return false;
                st.insert(r);
                q.push(r);
            }
        }
        return true;
    }
};
