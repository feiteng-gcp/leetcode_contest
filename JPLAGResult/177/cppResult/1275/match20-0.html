class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& l, vector<int>& r) {
        vector<int> ind(n, 0);
        for(int i = 0; i < n; ++i) {
            if(l[i] != -1) {
                if(ind[l[i]]++ != 0)
                    return false;
            }
            if(r[i] != -1) {
                if(ind[r[i]]++ != 0)
                    return false;
            }
        }
        bool hasRoot = false;
        for(int i = 0; i < n; ++i) {
            if(ind[i] == 0) {
                if(hasRoot)
                    return false;
                hasRoot = true;
            }else if(ind[i] != 1)
                return false;
        }
        return hasRoot;
    }
};
