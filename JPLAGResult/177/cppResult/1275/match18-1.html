class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> parent(n, -1);
        for (int i = 0; i < n; ++i) {
            int cur = leftChild[i];
            if (cur == -1) continue;
            if (parent[cur] != -1 && parent[cur] != i) return false;
            parent[cur] = i;
        }
        for (int i = 0; i < n; ++i) {
            int cur = rightChild[i];
            if (cur == -1) continue;
            if (parent[cur] != -1 && parent[cur] != i) return false;
            parent[cur] = i;
        }
        int cnt = 0;
        for (int i = 0; i < n; ++i) {
            if (parent[i] == -1) ++cnt;
        }
        return cnt == 1;
    }
};
