class Solution {
public:
    bool validateBinaryTreeNodes(int n, vector<int>& leftChild, vector<int>& rightChild) {
        vector<int> id(n, 0);
        
        for (int i = 0; i < n; ++i) {
            if (leftChild[i] != -1)
                id[leftChild[i]]++;
            if (rightChild[i] != -1) 
                id[rightChild[i]]++;
        }
        
        int count = 0;
        for (int i = 0; i < n; ++i) {
            if (id[i] == 2) return false;
            if (id[i] == 0) ++count;
            
        }
        
        return count == 1;
    }
};
