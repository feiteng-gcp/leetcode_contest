class Solution {
public:
   int days[12] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

bool isRun(int y) {
    return (y % 4 == 0) && (y % 100 != 0) || (y % 400 == 0);
}

int order(int y, int m, int d) {
    int total = 0;
    for (int i = 1970; i < y; i++) {
        total += isRun(i) ? 366 : 365;
    }
    for (int i = 1; i < m; i++) {
        if (isRun(y) && i == 2) total += 29;
        else total += days[i - 1];
    }
    total += d;
    return total;
}

int daysBetweenDates(string date1, string date2) {
    int y1 = stoi(date1.substr(0, 4));
    int y2 = stoi(date2.substr(0, 4));
    int m1 = stoi(date1.substr(5, 2));
    int m2 = stoi(date2.substr(5, 2));
    int d1 = stoi(date1.substr(8, 2));
    int d2 = stoi(date2.substr(8, 2));
    return abs(order(y1, m1, d1) - order(y2, m2, d2));
}
};
