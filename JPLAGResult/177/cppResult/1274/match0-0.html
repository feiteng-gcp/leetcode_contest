class Solution {
public:
    int daysBetweenDates(string date1, string date2) {
        struct std::tm a = {0,0,0,stoi(date1.substr(8,2)), stoi(date1.substr(5,2)), stoi(date1.substr(0,4)) - 1900}; 
        struct std::tm b = {0,0,0,stoi(date2.substr(8,2)), stoi(date2.substr(5,2)), stoi(date2.substr(0,4)) - 1900}; 
        return abs(count(stoi(date1.substr(0,4)), stoi(date1.substr(5,2)), stoi(date1.substr(8,2))) - count(stoi(date2.substr(0,4)), stoi(date2.substr(5,2)), stoi(date2.substr(8,2))));
    }
    
    int count(int y, int m, int d) { 
        if (m < 3)
            y--, m += 12;
        return 365*y + y/4 - y/100 + y/400 + (153*m - 457)/5 + d - 306;
    }
};
