class Solution {
public:
    int pastdays(int y, int m, int d){
        if(m <= 2)  y--, m += 12;
        return 365*y + y/4 - y/100 + y/400 + (306*(m+1))/10 + d - 429;
    }
    int daysBetweenDates(string date1, string date2) {
        int y = stoi(date1.substr(0,4));
        int m = stoi(date1.substr(5,2));
        int d = stoi(date1.substr(8,2));
        int y2 = stoi(date2.substr(0,4));
        int m2 = stoi(date2.substr(5,2));
        int d2 = stoi(date2.substr(8,2));
        return abs(pastdays(y,m,d)-pastdays(y2,m2,d2));
    }
};
