class Solution {
public:
    
    int month[12] = {31, 28, 31, 30, 31, 30, 
                    31, 31, 30, 31, 30, 31};
    
    int countLeapYears(int m, int y) {
        int years = y;
        if (m <= 2)

            years--;
        
        return years / 4 - years / 100 + years / 400; 
    }
    
    int daysBetweenDates(string date1, string date2) {
        int y1 = stoi(date1.substr(0,4).c_str());
        int m1 = stoi(date1.substr(5,2).c_str());
        int d1 = stoi(date1.substr(8,2).c_str());
        
        int y2 = stoi(date2.substr(0,4).c_str());
        int m2 = stoi(date2.substr(5,2).c_str());
        int d2 = stoi(date2.substr(8,2).c_str());
        
        int n1 = y1*365 + d1;
        for (int i = 0; i < m1-1; i++)
            n1 += month[i];
        n1 += countLeapYears(m1, y1);
        
        int n2 = y2*365 + d2;
        for (int i = 0; i < m2-1; i++)
            n2 += month[i];
        n2 += countLeapYears(m2, y2);
        return abs(n2-n1);
    }
};
