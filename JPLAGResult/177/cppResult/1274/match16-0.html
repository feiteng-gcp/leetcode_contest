class Solution {
public:
    int daysBetweenDates(string date1, string date2) {
        if (date1 > date2) {
            return daysBetweenDates(date2, date1);
        }
        vector<int> d1 = parse(date1);
        vector<int> d2 = parse(date2);
        return getDays(d2) - getDays(d1); 
    }
    
    int getDays(const vector<int>& date) {
        int year = date[0];
        int month = date[1];
        int days = date[2];
        bool is_leap_year = (year % 4 == 0 && year % 100 != 0) || year % 400 == 0;
        int output = 0;
        for (int y = 1971; y < year; ++y) {
            if ((y % 4 == 0 && y % 100 != 0) || y % 400 == 0) {
                output += 366;
            } else {
                output += 365;
            }
        }
        
        for (int m = 1; m < month; ++m) {
            if (m == 1 || m == 3 || m == 5 || m == 7 || m == 8 || m == 10 || m == 12){
                output += 31;
            } else if (m == 2) {
                if (is_leap_year) {
                    output += 29;
                } else {
                    output += 28;
                }
            } else {
                output += 30;
            }
        }
        
        output += days;
        return output;
    }
    
    vector<int> parse(string date) {
        int start = 0;
        vector<int> output;
        for (int i = 0; i < date.length(); ++i) {
            if (date[i] == '-') {
                int num = stoi(date.substr(start, i - start));
                output.push_back(num);
                start = i + 1;
            }
        }
        output.push_back(stoi(date.substr(start, date.length() - start)));
        return output;
    }
};
