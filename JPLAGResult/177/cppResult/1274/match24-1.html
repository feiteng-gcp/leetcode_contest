class Solution {
    
    const vector<int> v = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    
public:
    int b;
    

    bool isleap(int x) {
        return (x % 400) == 0 || (x % 100) != 0 && (x % 4) == 0;
    }
    
    int total(int x, int y, int z) {
        int t = 0;
        for(int s = b; s < x; s++) {
            t += isleap(s) ? 366 : 365;
        }
        
        for(int s = 1; s < y; s ++) t += v[s];
        
        t += z;
        if(isleap(x) && y > 2) t++;
        return t;
    }
    
    
    
    int daysBetweenDates(string date1, string date2) {
        if(date1 > date2) date1.swap(date2);
        b = stod(date1.substr(0, 4));
        int x1 = total(b, stod(date1.substr(5, 2)), stod(date1.substr(8)));
        int x2 = total(stod(date2.substr(0, 4)), stod(date2.substr(5, 2)), stod(date2.substr(8)));
        return x2 - x1;
    }
};
