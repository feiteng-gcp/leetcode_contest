class Solution {
public:
    int daysBetweenDates(string date1, string date2) {
        stringstream Sin,Sin2;
        Sin<<date1;
        int y1, m1,d1,y2,m2,d2;
        char c;
        Sin>>y1>>c>>m1>>c>>d1;
        Sin2<<date2;
        Sin2>>y2>>c>>m2>>c>>d2;

        //cout<<y1<<' '<<m1<<' '<<d1<<' '<<y2<<' '<<m2<<' '<<d2<<endl;
        
        int sum[13] = {0,31,28,31,30,31,30,31,31,30,31,30,31};
        int A,B;
        A = y1*365;
        for (int i=1;i<m1;i++) A+=sum[i];
        A+=d1;
        
        B=y2*365;
        for (int i=1;i<m2;i++) B+=sum[i];
        B+=d2;
        
        for (int i=min(y1,y2);i<=max(y1,y2);i++)
        {
            if (i%4!=0) continue;
            if (i%100==0 && i%400!=0) continue;
            if (i<y1 || (i==y1&&m1>2)) A++;
            if (i<y2 || (i==y2&&m2>2)) B++;
        }
        return abs(A-B);
    }
};
