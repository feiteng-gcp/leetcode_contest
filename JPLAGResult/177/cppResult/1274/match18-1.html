struct Date
{
    int d, m, y;

};

const int monthDays[12] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

int countLeapYears(Date d)
{
    int years = d.y;
    if (d.m <= 2) years--;
 
    return years / 4 - years / 100 + years / 400;
}

int getDifference(Date dt1, Date dt2)
{
    if(dt1.y > dt2.y)
    {
        return getDifference(dt2, dt1);
    }
    else if(dt1.y == dt2.y && dt1.m > dt2.m)
    {
        return getDifference(dt2, dt1);
    }
    else if(dt1.y == dt2.y && dt1.m == dt2.m && dt1.d > dt2.d)
    {
        return getDifference(dt2, dt1);
    }
    

    long int n1 = dt1.y*365 + dt1.d;

    for (int i=0; i<dt1.m - 1; i++)
        n1 += monthDays[i];
    n1 += countLeapYears(dt1);

    long int n2 = dt2.y*365 + dt2.d;
    for (int i=0; i<dt2.m - 1; i++)
        n2 += monthDays[i];
    n2 += countLeapYears(dt2);
    
    return (n2 - n1);
}

std::vector<std::string> split(const std::string& s, char delimiter)
{
   std::vector<std::string> tokens;
   std::string token;
   std::istringstream tokenStream(s);
   while (std::getline(tokenStream, token, delimiter))
   {
      tokens.push_back(token);
   }
   return tokens;
}

Date GetDate(string& s)
{
    std::istringstream iss(s);
    std::vector<std::string> res = split(s, '-');
    //for(string s: res) cout<<s<<endl;
    Date ret;
    ret.y = stoi(res[0]);
    ret.m = stoi(res[1]);
    ret.d = stoi(res[2]);
    return ret;
}

class Solution {
public:
    int daysBetweenDates(string date1, string date2) {
        Date d1 = GetDate(date1);
        Date d2 = GetDate(date2);
        
        return getDifference(d1, d2);
    }
};
