class Solution {
public:
    int countLeapYears(int y,int m) 
    { 
        int years = y; 

        if (m <= 2) 
            years--; 

        return years / 4 - years / 100 + years / 400; 
    } 
  
    vector<int>months={31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    int daysBetweenDates(string date1, string date2) {
        int year1=stoi(date1.substr(0,4)),year2=stoi(date2.substr(0,4));
        int month1=stoi(date1.substr(5,2)),month2=stoi(date2.substr(5,2));

        int day1=stoi(date1.substr(8,2)),day2=stoi(date2.substr(8,2));
        
        int n1=year1*365+day1;
        for (int i=0; i<month1 - 1; i++) n1 += months[i]; 
        n1+=countLeapYears(year1,month1);
        
        int n2=year2*365+day2;
        for(int i=0;i<month2-1;i++)n2+=months[i];
        n2+=countLeapYears(year2,month2);
        return abs(n2-n1);
    }
};
