class Solution {
    
    int len[13] = {0, 31, 0, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    
    bool isLeepYear(int y) {
        if (y % 400 == 0) 
            return true;
        
        if (y % 100 == 0)
            return false;
        
        if (y % 4 == 0)
            return true;
        
        return false;
    }
    
    int getDay(int y, int m) {
        if (m != 2) return len[m];
        
        if (isLeepYear(y))
            return 29;
        else
            return 28;
    }
    
public:
    int daysBetweenDates(string date1, string date2) {
        int year1 = stoi(date1.substr(0, 4));
        int month1 = stoi(date1.substr(5, 2));
        int day1 = stoi(date1.substr(8));
        int year2 = stoi(date2.substr(0, 4));
        int month2 = stoi(date2.substr(5, 2));
        int day2 = stoi(date2.substr(8));
        
        
        if (date1 > date2) {
            swap(year1, year2);
            swap(month1, month2);
            swap(day1, day2);
        }
        cout<<year1<<","<<month1<<","<<day1<<endl;
        cout<<year2<<","<<month2<<","<<day2<<endl;
        
        int ans = 0;
        while (1) {
            if (year1 == year2 && month1 == month2) break;
            
            ans += getDay(year1, month1);
            month1++;
            if (month1 == 13) {
                year1++;
                month1 = 1;
            }
        }
        ans += day2 - day1;
        
        return ans;
    }
};
