class Solution {
public:
    int days_from_1971_1_1(int y, int m, int d) {
        int days[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        int ret = 0;
        ret += (y - 1971) * 365;
        ret += ((y - 1) / 4 - 1971 / 4);
        for (int i = 0; i < m; ++i) ret += days[i];
        if (m > 2 && (y % 4 == 0 && y != 2100)) ret++;
        ret += d - 1;
        return ret;
    }
    
    int daysBetweenDates(string date1, string date2) {
        int y1 = 0, y2 = 0, m1 = 0, m2 = 0, d1 = 0, d2 = 0;
        for (int i = 0; i < 4; ++i) {
            y1 = y1 * 10 + date1[i] - '0';
            y2 = y2 * 10 + date2[i] - '0';
        }
        for (int i = 5; i < 7; ++i) {
            m1 = m1 * 10 + date1[i] - '0';
            m2 = m2 * 10 + date2[i] - '0';
        }
        for (int i = 8; i < 10; ++i) {
            d1 = d1 * 10 + date1[i] - '0';
            d2 = d2 * 10 + date2[i] - '0';
        }
        int days1 = days_from_1971_1_1(y1, m1, d1);
        int days2 = days_from_1971_1_1(y2, m2, d2);
        int diff = days1 - days2;
        if (diff < 0) diff *= -1;
        return diff;
    }
};
