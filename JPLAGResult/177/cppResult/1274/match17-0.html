class Solution {
public:
    int getDays(string s)
    {
        int year = 0;
        int month = 0;
        int day = 0;
        year = (s[0] - '0') * 1000 + (s[1] - '0') * 100 + (s[2] - '0') * 10 + s[3] - '0';
        month = (s[5] - '0') * 10 + s[6] - '0';
        day = (s[8] - '0') * 10 + s[9] - '0';
        int result = 0;
        for (int i = 1971; i < year; i++)
        {
            if (i % 100 == 0)
            {
                if (i % 400 == 0)
                {
                    result += 366;
                }
                else
                {
                    result += 365;
                }
            }
            else if (i % 4 == 0)
            {
                result += 366;
            }
            else
            {
                result += 365;
            }
        }
        
        for (int i = 1; i < month; i++)
        {
            if (i == 1 || i == 3 || i == 5 || i == 7 || i == 8 || i == 10 || i == 12)
            {
                result += 31;
            }
            else if (i == 2)
            {
                if (year % 100 == 0)
                {
                    if (year % 400 == 0)
                    {
                        result += 29;
                    }
                    else
                    {
                        result += 28;
                    }
                }
                else if (year % 4 == 0)
                {
                    result += 29;
                }
                else
                {
                    result += 28;
                }
            }
            else
            {
                result += 30;
            }
        }
        
        result += day;
        return result;
    }
    int daysBetweenDates(string date1, string date2) {
        
        int day1 = getDays(date1);
        
        int day2 = getDays(date2);
        //cout << day1 << " " << day2 << endl;
        return abs(day1 - day2);
    }
};
