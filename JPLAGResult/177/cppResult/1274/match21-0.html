class Solution {
    int days(int year)
    {
        int d = 0;
        for (int i = 1971; i < year; i++)
            d += 365 + (i % 4 == 0 && i != 2100 ? 1 : 0);
        return d;
    }
    int days(int year, int month)
    {
        int _days[] = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
        int d = days(year);
        if (year % 4 == 0 && year != 2100) _days[1] = 29;
        for (int i = 1; i < month; i++)
            d += _days[i - 1];
        return d;
            
    }
    int days(int year, int month, int day)
    {
        return days(year, month) + day;
    }
public:
    int daysBetweenDates(string date1, string date2)
    {
        int y1 = stoi(string(date1, 0, 4));
        int m1 = stoi(string(date1, 5, 2));
        int d1 = stoi(string(date1, 8, 2));
        int y2 = stoi(string(date2, 0, 4));
        int m2 = stoi(string(date2, 5, 2));
        int d2 = stoi(string(date2, 8, 2));
        return abs(days(y1, m1, d1) - days(y2, m2, d2));
    }
};
