class Solution {
public:
    int t(char c) {

        return c - '0';
    }
    int daysBetweenDates(string d1, string d2) {
        int y1 = t(d1[0]) * 1000 + t(d1[1]) * 100 + t(d1[2]) * 10 + t(d1[3]);
        int y2 = t(d2[0]) * 1000 + t(d2[1]) * 100 + t(d2[2]) * 10 + t(d2[3]);
        int m1 = t(d1[5]) * 10 + t(d1[6]);
        int m2 = t(d2[5]) * 10 + t(d2[6]);
        int dd1 = t(d1[8]) * 10 + t(d1[9]);
        int dd2 = t(d2[8]) * 10 + t(d2[9]);
        time_t rawtime1, rawtime2;
        rawtime1 = time(NULL);
        rawtime2 = time(NULL);
        struct tm ti1;
        //ti1 = localtime ( &rawtime1 );
        ti1.tm_year = y1 - 1900;
        ti1.tm_mon = m1 - 1;
        ti1.tm_mday = dd1;
        ti1.tm_hour = 0;
        ti1.tm_min = 0;
        ti1.tm_sec = 0;
        cout << y1 << " " << m1 << " " << dd1 << endl;
        struct tm ti2;
        //ti2 = localtime ( &rawtime2 );
        ti2.tm_year = y2 - 1900;
        ti2.tm_mon = m2 - 1;
        ti2.tm_mday = dd2;
        ti2.tm_hour = 0;
        ti2.tm_min = 0;
        ti2.tm_sec = 0;
        long long s =  difftime(mktime(&ti1), mktime(&ti2));
        cout << y2 << " " << m2 << " " << dd2 << endl;
        cout << s << endl;
        return abs(s / 86400);
    }
};
