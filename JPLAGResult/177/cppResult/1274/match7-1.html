class Solution {
public:
    bool isLeapYear(int year)
    {
	    return ((year%4==0 && year%100!=0) || year%400==0);
    }
    
    int getDays(int year, int month, int day)
    {
	    int m[] = {0,31,28,31,30,31,30,31,31,30,31,30,31};
	    if(isLeapYear(year))
		    m[2]++;
	    int result = 0;
	    for(int i = 1971;i < year;i ++)
	    {
		    result += 365;
		    if(isLeapYear(i))
			    result ++;
	    }
	    for(int i = 1;i < month;i++)
	    {
		    result += m[i];
	    }
	    result += day;
	
	    return result;
    }
    
    int daysBetweenDates(string date1, string date2) {
        int year1 = 0, month1 = 0, day1 = 0, nt1 = 0;
        for(int i = 0;i < date1.size();i ++){
            if(date1[i] != '-'){
                if(nt1 == 0) year1 = year1*10 + date1[i]-'0';
                else if(nt1 == 1) month1 = month1*10 + date1[i]-'0';
                else if(nt1 == 2) day1 = day1*10 + date1[i]-'0';
            }
            else nt1 ++;
        }
        int year2 = 0, month2 = 0, day2 = 0, nt2 = 0;
        for(int i = 0;i < date2.size();i ++){
            if(date2[i] != '-'){
                if(nt2 == 0) year2 = year2*10 + date2[i]-'0';
                else if(nt2 == 1) month2 = month2*10 + date2[i]-'0';
                else if(nt2 == 2) day2 = day2*10 + date2[i]-'0';
            }
            else nt2 ++;
        }
        return abs(getDays(year1, month1, day1) - getDays(year2, month2, day2));
    }
};
