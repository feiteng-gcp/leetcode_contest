class Solution {
public:
    int getDay(string &d) {
        int Y = (d[0] - '0') * 1000 + (d[1] - '0') * 100 + (d[2] - '0') * 10 + (d[3] - '0');
        int M = (d[5] - '0') * 10 + (d[6] - '0');
        int D = (d[8] - '0') * 10 + (d[9] - '0');
        
        int sum = 0;
        for (int i = 1970; i < Y; ++i) {
            if (i % 4 == 0 && (i % 100 != 0 || i % 400 == 0)) {
                sum += 366;
            } else
                sum += 365;
        }
        
        int m[] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        
        for (int i = 1; i < M; ++i) {
            sum += m[i - 1];
            
            if (Y % 4 == 0 && (Y % 100 != 0 || Y % 400 == 0) && i == 2) {
                sum ++;
            } 
        }
        
        sum += D;
        
        return sum;
    }
    
    int daysBetweenDates(string date1, string date2) {
        return abs(getDay(date2) - getDay(date1));
    }
};
