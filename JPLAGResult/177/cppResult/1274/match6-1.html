class Solution {
public:
    int dateToInt(int y, int m, int d){
	    return 1461*(y+4800+(m-14)/12)/4+367*(m-2-(m-14)/12*12)/12-3*((y+4900+(m-14)/12)/100)/4+d-32075;
    }
    int daysBetweenDates(string s, string t) {
        int x=dateToInt(stoi(s.substr(0,4)),
									stoi(s.substr(5,2)),
									stoi(s.substr(8,2)));
	    int y=dateToInt(stoi(t.substr(0,4)),
									stoi(t.substr(5,2)),
									stoi(t.substr(8,2)));
        return abs(x-y);
    }
};
