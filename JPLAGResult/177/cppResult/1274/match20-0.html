int leap_years(int y){
    int res = 0;
    y = y-1;
    for(; y >= 1970; y--){
        if(y % 4 == 0){
            if(y != 2100){
                res++;
            }
        }
    }
    return res;
}
int days(int m, int d){
    vector<int> ms = {31,28,31,30,31,30,31,31,30,31,30,31};
    int res = 0;
    for(int i = 0; i < m-1; i++)res += ms[i];
    return res + d;
}
int tdays(int y, int m, int d){
    int thisleap = (y % 4 == 0 && y != 2100)? 1 : 0;
    int give_leap = (m >= 3) ? thisleap : 0;
    return y * 365 + leap_years(y) + days(m, d) + give_leap;
}
class Solution {
public:
    int daysBetweenDates(string date1, string date2) {
        int y1 = stoi(date1.substr(0,4));
        int y2 = stoi(date2.substr(0,4));
        int m1 = stoi(date1.substr(5,2));
        int m2 = stoi(date2.substr(5,2));
        int d1 = stoi(date1.substr(8,2));
        int d2 = stoi(date2.substr(8,2));
        //cout << days(12, 31);
        //cout << days(1, 1);
        return abs(tdays(y1,m1,d1) - tdays(y2,m2,d2));
    }
};
