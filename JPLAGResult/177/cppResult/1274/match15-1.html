class Solution {
public:
    int rdn(int y, int m, int d) { /* Rata Die day one is 0001-01-01 */
        if (m < 3)
            y--, m += 12;
        return 365*y + y/4 - y/100 + y/400 + (153*m - 457)/5 + d - 306;
    }

    void getValue(std::string& date, int& day, int& month, int& year)
    {
        year    = atoi(date.substr(0,4).c_str());
        month   = atoi(date.substr(5,2).c_str());
        day     = atoi(date.substr(8,2).c_str());
        std::cout << year << "-" << month << "-" << day << std::endl;
    }
    int daysBetweenDates(string date1, string date2) {
        int day1, month1, year1;    
        getValue(date1, day1,month1,year1);
        int day2, month2, year2;    
        getValue(date2, day2,month2,year2);
        
    int difference = abs(rdn(year1, month1, day1) - rdn(year2, month2, day2));


    //struct std::tm a = {0,0,0,day1,month1,year1-1900}; /* June 24, 2004 */
    //struct std::tm b = {0,0,0,day2,month2,year2-1900}; /* June 24, 2004 */
    //std::time_t x = std::mktime(&a);
    //std::time_t y = std::mktime(&b);
    //double difference = abs(std::difftime(y, x) / (60 * 60 * 24));
    return int(difference);
    }
};
