#include <bits/stdc++.h>
using namespace std;

bool pa(int year)
{
    if (year % 400 == 0)
        return true;
    if (year % 4 != 0 || year % 100 == 0)
        return false;
    return true;
}

int pb(int y, int m, int d)
{
    int ans = 0;
    for (int i = 1965; i < y; i++)
    {
        if (pa(i))
            ans += 366;
        else
            ans += 365;
    }
    int a[15] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    for (int i = 1; i < m; i++)
        ans += a[i];
    ans += d;
    if (pa(y) && m > 2)
        ans++;
    return ans;
}

class Solution
{
public:

    int daysBetweenDates(string date1, string date2)
    {
        int y1 = 0, y2 = 0, m1 = 0, m2 = 0, d1 = 0, d2 = 0;
        for (int i = 0; i < 4; i++)
        {
            y1 = y1 * 10 + (date1[i] - '0');
            y2 = y2 * 10 + (date2[i] - '0');
        }
        for (int i = 5; i < 7; i++)
        {
            m1 = m1 * 10 + (date1[i] - '0');
            m2 = m2 * 10 + (date2[i] - '0');
        }
        for (int i = 8; i < 10; i++)
        {
            d1 = d1 * 10 + (date1[i] - '0');
            d2 = d2 * 10 + (date2[i] - '0');
        }
        int x = pb(y1, m1, d1), y = pb(y2, m2, d2);
        return abs(x - y);
    }
};
