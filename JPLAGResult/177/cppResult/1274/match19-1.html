using namespace std;
using ll = long long;
using ld = double;
typedef pair<int, int> II;
typedef pair<II, int> III;

typedef vector<cd> vcd;

struct Date {
    int d, m, y;
    Date(int d = 0, int m = 0, int y = 0) : d(d), m(m), y(y) {}
};

class Solution {
public:

    int month[12] = {31,28,31,30,31,30,31,31,30,31,30,31};
    
    int countLeapYears(Date d) { 
        int years = d.y; 
        if (d.m <= 2) years--; 
        return years / 4 - years / 100 + years / 400; 
    } 
  
    int getDifference(Date dt1, Date dt2) { 
        int n1 = dt1.y*365 + dt1.d; 
        // cout << dt1.y << ' ' << dt1.d << endl;
  
        for (int i=0; i<dt1.m - 1; i++) n1 += month[i]; 
 
        n1 += countLeapYears(dt1); 


        int n2 = dt2.y*365 + dt2.d; 
        // cout << n2 << endl;
        for (int i=0; i<dt2.m - 1; i++) n2 += month[i]; 
        n2 += countLeapYears(dt2);
        
        cout << n1 << ' ' << n2;

        return abs(n2 - n1); 
    } 
    
    int daysBetweenDates(string date1, string date2) {
        FOR(i,0,SZ(date1)-1)
            if (date1[i] == '-') date1[i] = ' ';
        FOR(i,0,SZ(date2)-1)
            if (date2[i] == '-') date2[i] = ' ';
        stringstream sin1(date1);
        Date da1;
        sin1 >> da1.y >> da1.m >> da1.d;
        stringstream sin2(date2);
        Date da2;
        sin2 >> da2.y >> da2.m >> da2.d;
        return getDifference(da1, da2); 
    }
};
