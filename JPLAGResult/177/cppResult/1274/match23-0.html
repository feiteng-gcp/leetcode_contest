class Solution {
public:
    
    bool isRun(int y){
        if(!(y % 100)) return !(y % 400);
        else return !(y % 4);
    }
    
    int coutD(int y, int m, int d){
        int ret = 0;
        
        for(int i = 1971; i < y; ++i){
            if(isRun(i)) ret += 366;
            else ret += 365;
        }
        
        //cout << "coutY" << ret << endl;
        
        for(int i = 1; i < m; ++i){
            if(i == 1 || i == 3 || i == 5 || i == 7 || i == 8 || i == 10 || i == 12) ret += 31;
            else if(i == 2){
                if(isRun(y)) ret += 29;
                else ret += 28;
            }else{
                ret += 30;
            }
        }
        
        //cout << "coutM" << ret << endl;
        for(int i = 1; i < d; ++i){
            ret += 1;
        }
        
        //cout << "coutD" << ret << endl;
        return ret;
    }
    
    int daysBetweenDates(string date1, string date2) {
        int year1 = stoi(date1.substr(0, 4));
        int mon1 = stoi(date1.substr(5, 2));
        int day1 = stoi(date1.substr(8));

        int year2 = stoi(date2.substr(0, 4));
        int mon2 = stoi(date2.substr(5, 2));
        int day2 = stoi(date2.substr(8));
        
        
        //cout << year1 << " " << mon1 << " " << day1 << endl;
        int count1 = coutD(year1, mon1, day1);
        int count2 = coutD(year2, mon2, day2);
        
        //cout << count1 << " " << count2 << endl;
        
        return abs(count1 - count2);
        
    }
};
