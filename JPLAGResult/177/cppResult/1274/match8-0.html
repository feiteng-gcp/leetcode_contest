class Solution {
public:
    int rdn(int y, int m, int d) { /* Rata Die day one is 0001-01-01 */
        if (m < 3)
            y--, m += 12;
        return 365*y + y/4 - y/100 + y/400 + (153*m - 457)/5 + d - 306;
    }
    int daysBetweenDates(string date1, string date2) {
        int day1 = (date1[8] - '0')*10 + date1[9] - '0';
        int month1 = (date1[5] - '0')*10 + date1[6] - '0';
        int year1 = (date1[0] - '0')*10 + date1[1] - '0';
        year1 = year1 * 100 + (date1[2] - '0')*10 + date1[3] - '0';
        
        int day2 = (date2[8] - '0')*10 + date2[9] - '0';
        int month2 = (date2[5] - '0')*10 + date2[6] - '0';
        int year2 = (date2[0] - '0')*10 + date2[1] - '0';
        year2 = year2 * 100 + (date2[2] - '0')*10 + date2[3] - '0';
        
        int x1 = rdn(year1,month1,day1);
        int x2 = rdn(year2,month2,day2);
        
        return abs(int(x1 - x2));
    }
};
