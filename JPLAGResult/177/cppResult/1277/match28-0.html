class Solution {
public:
  string largestMultipleOfThree(vector<int>& digits) {
    vector< vector<int> > G(3);
    for(int d:digits)
      G[d%3].emplace_back(d);

    sort(G[0].rbegin(),G[0].rend());
    sort(G[1].rbegin(),G[1].rend());
    sort(G[2].rbegin(),G[2].rend());

    int sum=0;
    for(int d:digits) sum+=d;
    sum%=3;
    if(sum==1){
      if(G[1].size()>=1){
        G[1].pop_back();
      }else if(G[2].size()>=2){
        G[2].pop_back();
        G[2].pop_back();
      }else{
        return ""s;
      }
    }

    if(sum==2){
      if(G[2].size()>=1){
        G[2].pop_back();
      }else if(G[1].size()>=2){
        G[1].pop_back();
        G[1].pop_back();
      }else{
        return ""s;
      }
    }

    string ans;
    for(int d:G[0]) ans+=char('0'+d);
    for(int d:G[1]) ans+=char('0'+d);
    for(int d:G[2]) ans+=char('0'+d);
    sort(ans.rbegin(),ans.rend());
    if(ans.empty()) return ans;
    if(ans==string(ans.size(),'0')) return "0"s;
    return ans;
  }
};
