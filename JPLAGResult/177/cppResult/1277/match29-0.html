class Solution {
public:
    string largestMultipleOfThree(vector<int>& digits) {
        sort(digits.begin(),digits.end());
        int n=digits.size();
        if(digits[n-1]==0)return "0";
        int sum=0;
        for(int i=0;i<n;i++)
        {
            sum=sum+digits[i];
        }
        if(sum%3==1)
        {
            bool ok=false;
            for(int i=0;i<n;i++)
            {
                if(digits[i]%3==1)
                {
                    digits[i]=-1;
                    ok=true;
                    break;
                }
            }
            if(!ok)
            {
                int cnt=0;
                for(int i=0;i<n;i++)
                {
                    if(digits[i]%3==2)
                    {
                        digits[i]=-1;
                        cnt++;
                        if(cnt==2)break;
                    }
                }
            }
        }
        else if(sum%3==2)
        {
            bool ok=false;
            for(int i=0;i<n;i++)
            {
                if(digits[i]%3==2)
                {
                    digits[i]=-1;
                    ok=true;
                    break;
                }
            }
            if(!ok)
            {
                int cnt=0;
                for(int i=0;i<n;i++)
                {
                    if(digits[i]%3==1)
                    {
                        digits[i]=-1;
                        cnt++;
                        if(cnt==2)break;
                    }
                }
            }
        }
        string ans="";
        for(int i=n-1;i>=0;i--)
        {
            if(digits[i]==-1)continue;
            char c=digits[i]+'0';
            ans=ans+c;
        }
        return ans;
    }
};
