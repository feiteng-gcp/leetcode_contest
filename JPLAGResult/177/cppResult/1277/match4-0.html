class Solution {
public:
    string largestMultipleOfThree(vector<int>& digits) {
        vector<int> c(10, 0);
        int mod = 0;
        for (auto d: digits) {
            c[d]++;
            mod = (mod + d) % 3;
        }

        switch (mod) {
            case 0:
                break;
            case 1:
                if (c[1] > 0) --c[1];
                else if (c[4] > 0) --c[4];
                else if (c[7] > 0) --c[7];
                else if (c[2] + c[5] + c[8] >= 2) {
                    int left = 2, m;
                    m = min(c[2], left);
                    c[2] -= m;
                    left -= m;
                    if (left == 0) break;
                    m = min(c[5], left);
                    c[5] -= m;
                    left -= m;
                    if (left == 0) break;
                    m = min(c[8], left);
                    c[8] -= m;
                    left -= m;
                    if (left == 0) break;
                    return "";
                }
                break;
            case 2:
                if (c[2] > 0) --c[2];
                else if (c[5] > 0) --c[5];
                else if (c[8] > 0) --c[8];
                else if (c[1] + c[4] + c[7] >= 2) {
                    int left = 2, m;
                    m = min(c[1], left);
                    c[1] -= m;
                    left -= m;
                    if (left == 0) break;
                    m = min(c[4], left);
                    c[4] -= m;
                    left -= m;
                    if (left == 0) break;
                    m = min(c[7], left);
                    c[7] -= m;
                    left -= m;
                    if (left == 0) break;
                    return "";
                }
                break;
        }
        
        string ret = "";
        
        for (int i = 0; i < 10; ++i) {
            if (c[i] > 0)
                ret.append(c[i], i + '0');
        }
        
        if (ret.empty()) return "";
        
        sort(ret.begin(), ret.end(), greater<char>());
        size_t pos = ret.find_first_not_of("0");
        if (pos == string::npos) {
            return "0";
        }
        ret = ret.substr(pos);
        return ret;
    }
};
