class Solution {
public:
    void populateAux(int aux[], queue<int> queue0, queue<int> queue1, 
				queue<int> queue2, int* top) 
{ 
	while (!queue0.empty()) { 
		aux[(*top)++] = queue0.front(); 
		queue0.pop(); 
	} 
	while (!queue1.empty()) { 
		aux[(*top)++] = queue1.front(); 
		queue1.pop(); 
	} 
	while (!queue2.empty()) { 
		aux[(*top)++] = queue2.front(); 
		queue2.pop(); 
	} 
}
    string largestMultipleOfThree(vector<int>& arr) {
        int size=arr.size();
        sort(arr.begin(), arr.end()); 
	queue<int> queue0, queue1, queue2; 
	int i, sum; 
	for (i = 0, sum = 0; i < size; ++i) { 
		sum += arr[i]; 
		if ((arr[i] % 3) == 0) 
			queue0.push(arr[i]); 
		else if ((arr[i] % 3) == 1) 
			queue1.push(arr[i]); 
		else
			queue2.push(arr[i]); 
	} 
	if ((sum % 3) == 1) { 
		if (!queue1.empty()) 
			queue1.pop(); 
		else { 
			if (!queue2.empty()) 
				queue2.pop(); 
			else
				return 0; 

			if (!queue2.empty()) 
				queue2.pop(); 
			else
				return 0; 
		} 
	} 
	else if ((sum % 3) == 2) { 
		if (!queue2.empty()) 
			queue2.pop(); 
		else { 
			if (!queue1.empty()) 
				queue1.pop(); 
			else
				return 0; 

			if (!queue1.empty()) 
				queue1.pop(); 
			else
				return 0; 
		} 
	} 
	int aux[size], top = 0; 
	populateAux(aux, queue0, queue1, queue2, &top); 
	sort(aux, aux + top, greater<int>());
    string ans="";
	for (int i = 0; i < top; ++i) 
		ans+=to_string(aux[i]); 
    if (ans.size()&&ans[0]=='0')
        return "0";
        
	return ans;
    }
};
