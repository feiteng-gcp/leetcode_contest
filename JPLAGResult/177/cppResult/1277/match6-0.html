class Solution {
public:
    string largestMultipleOfThree(vector<int>& digits) {
        sort(digits.begin(), digits.end());
        if (digits.back() == 0) return "0";
        vector<int> m1, m2;
        unordered_map<int, int> dis;
        int all = 0;
        for (int n: digits) {
            if (n % 3 != 0) all += n;
            if (n % 3 == 1 && m1.size() < 2) m1.push_back(n);
            else if (n % 3 == 2 && m2.size() < 2) m2.push_back(n);
        }
        
        all %= 3;
        bool bad = digits.size() == m1.size() + m2.size();
        
        if (all == 1) {
            if (m1.size() >= 1) ++dis[m1[0]];
            else if (bad && m2.size() < 2) return "";
            else {
                ++dis[m2[0]];
                ++dis[m2[1]];
            }
        }
        else if (all == 2) {
            if (m2.size() >= 1) ++dis[m2[0]];
            else if (bad && m1.size() < 2) return "";
            else {
                ++dis[m1[0]];
                ++dis[m1[1]];
            }
        }
        
        string res = "";
        for (int i = digits.size() - 1; i >= 0; --i) {
            int d = digits[i];
            if (dis.find(d) != dis.end() && dis[d] > 0) --dis[d];
            else res += to_string(d);
        }
        return res;
    }
};
