class Solution {
public:
    string largestMultipleOfThree(vector<int>& digits) {
        int n = digits.size();
        sort(digits.begin(), digits.end());
        queue<int> qR0;
        queue<int> qR1;

        queue<int> qR2;
        int sum = 0;
        for(int i = 0; i < n; ++i){ 
            sum += digits[i]; 
            if((digits[i] % 3) == 0){
                qR0.push(digits[i]);
            }else if((digits[i] % 3) == 1) {
                qR1.push(digits[i]); 
            }else{
                qR2.push(digits[i]); 
            }
        }
        if(sum%3==1){
            if(!qR1.empty()){
                qR1.pop();
            }else{
                //2+2 = 4 % 3 .... 1
                if(!qR2.empty()){
                    qR2.pop();
                }else{
                    return "";
                }
                if(!qR2.empty()){
                    qR2.pop();
                }else{
                    return "";
                }
            }
        }else if(sum%3 == 2){

            if(!qR2.empty()){
                qR2.pop();
            }else{
                if(!qR1.empty()){
                    qR1.pop(); 
                }else{
                    return "";
                }
                if(!qR1.empty()){
                    qR1.pop();
                }else{
                    return ""; 
                }
            }
        }
  
        vector<int> res;
        while(!qR0.empty()){
            res.emplace_back(qR0.front());
            qR0.pop();
        }
        while(!qR1.empty()){
            res.emplace_back(qR1.front());
            qR1.pop();
        }
        while(!qR2.empty()){
            res.emplace_back(qR2.front());
            qR2.pop();
        }
        if(res.empty()) return "";
        sort(res.rbegin(), res.rend());
        string s = "";
        for(int i = 0; i < res.size(); ++i){
            s += to_string(res[i]);
        }
        int i = 0;
        while(i < s.size() && s[i] == '0'){
            i++;
        }
        s = s.substr(i);
        if(s.empty()) s = "0";
        return s;
    }
};
