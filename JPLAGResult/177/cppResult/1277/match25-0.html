class Solution {
public:
    
    string largestMultipleOfThree(vector<int>& digits) {
        sort(digits.begin(), digits.end());
        int len = digits.size();
        long long sum = 0;
        vector<int> mod(3, 0);
        for(int i=len-1;i>=0;--i) {
            sum+=digits[i];
            mod[digits[i]%3]++;
        }
        if(sum%3==1) {
            if(mod[1]>0) {
                for(int i=0;i<len;++i) {
                    if(digits[i]%3 == 1) {
                        digits[i] = -1;
                        break;
                    }
                }
            }
            else if(mod[2]>0) {
                int cnt=0;
                for(int i=0;i<len;++i) {
                    if(digits[i]%3 == 2) {
                        digits[i] = -1;
                        cnt++;
                        if(cnt==2) break;
                    }
                }
            }
        }
        else if(sum%3==2) {
            if(mod[2]>0) {
                for(int i=0;i<len;++i) {
                    if(digits[i]%3 == 2) {
                        digits[i] = -1;
                        break;
                    }
                }
            }
            else if(mod[1]>1) {
                int cnt=0;
                for(int i=0;i<len;++i) {
                    if(digits[i]%3 == 1) {
                        digits[i] = -1;
                        cnt++;
                        if(cnt==2) break;
                    }
                }
            }
        }
        string ans="";
        for(int i=len-1;i>=0;--i) {
            if(ans=="0" && digits[i]==0) break;
            if(digits[i]!=-1) ans += '0'+digits[i];
        }
        return ans;
    }
};
