class Solution {
public:
    int populateAux(vector<int>& aux, queue<int>& queue0, queue<int>& queue1,
		queue<int>& queue2) {
		int i = 0;
		while (!queue0.empty()) {
			aux[i++] = queue0.front();
			queue0.pop();
		}

		while (!queue1.empty()) {
			aux[i++] = queue1.front();
			queue1.pop();
		}

		while (!queue2.empty()) {
			aux[i++] = queue2.front();
			queue2.pop();
		}
		return i;
	}

	string largestMultipleOfThree(vector<int>& arr) {
		sort(arr.begin(), arr.end());
		int n = arr.size();
		queue<int> queue0, queue1, queue2;

		int i, sum;
		for (i = 0, sum = 0; i < n; ++i) {
			sum += arr[i];
			if ((arr[i] % 3) == 0)
				queue0.push(arr[i]);
			else if ((arr[i] % 3) == 1)
				queue1.push(arr[i]);
			else
				queue2.push(arr[i]);
		}

		if ((sum % 3) == 1) {
			if (!queue1.empty())
				queue1.pop();
			else {
				if (!queue2.empty())
					queue2.pop();
				else
					return "";

				if (!queue2.empty())
					queue2.pop();
				else
					return "";
			}
		}

		else if ((sum % 3) == 2) {
			if (!queue2.empty())
				queue2.pop();
			else {
				if (!queue1.empty())
					queue1.pop();
				else
					return "";
				if (!queue1.empty())
					queue1.pop();
				else
					return "";
			}
		}

		vector<int> aux(n, 0);
		int max_ = 0;
		max_ = populateAux(aux, queue0, queue1, queue2);
		sort(aux.begin(), aux.end(), greater<int>());
		string ans;
        int idx = 0;
		for (; idx < max_; ++idx) {
			if (idx == max_ - 1 || aux[idx] != 0)
				break;
		}
		for (; idx < max_; ++idx) {
			ans += '0' + aux[idx];
		}

		return ans;
	}
};
