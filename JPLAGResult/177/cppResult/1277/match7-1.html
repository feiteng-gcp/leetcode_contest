bool cmp(int a, int b)
{
    return a>b;
}
class Solution {
public:
    string largestMultipleOfThree(vector<int>& digits) {
        vector<int> c(0);
        vector<int> one(0);
        vector<int> two(0);
        int m=0;
        for (auto a:digits)
        {
            if (a%3==0) c.push_back(a);
            if (a%3==1) one.push_back(a);
            if (a%3==2) two.push_back(a);
            m+=(a%3);
            m=m%3;
        }
        sort(one.begin(),one.end(),cmp);
        sort(two.begin(),two.end(),cmp);
        
        //cout<<m<<endl;
        if (m==0)
        {
            for (auto a:one) c.push_back(a);
            for (auto a:two) c.push_back(a);
        }
        else if (m==1)
        {
            if (!one.empty())
            {
                for (int i=0;i<one.size()-1;i++) c.push_back(one[i]);
                for (auto a:two) c.push_back(a);
            }
            else
            {
                for (int i=0;i<two.size()-2;i++) c.push_back(two[i]);
            }
        }
        else if (m==2)
        {
            if (!two.empty())
            {
                for (int i=0;i<two.size()-1;i++) c.push_back(two[i]);
                for (auto a:one) c.push_back(a);
            }
            else
            {
                for (int i=0;i<one.size()-2;i++) c.push_back(one[i]);
            }
        }
        
        for (auto a:c) cout<<a<<endl;
        
        sort(c.begin(),c.end(),cmp);
        if (c.empty()) return "";
        if (c[0]==0) return "0";
        
        string ret="";
        for (auto a:c)
        {
            ret=ret+char(a+'0');
        }
        return ret;
    }
};
