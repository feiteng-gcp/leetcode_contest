class Solution {
public:
    string largestMultipleOfThree(vector<int>& digits) {
        vector<int> counter(10, 0);
        int p = 0;
        for (auto c: digits) {
            counter[c]++;
            p = (p + c) % 3;
        }
        if (p == 0) {
                 

        } else if (p == 1) {
            bool find = false;
            for (int i = 0;i < 10;i++) {
                if (i % 3 == 1 && counter[i] > 0) {
                    find = true;
                    counter[i]--;
                    break;
                }
            }
            if (!find) {
                int l = 2;
                for (int i = 0;i < 10;i++) {
                    while (l > 0 && i % 3 == 2 && counter[i] > 0) {
                        find = true;
                        counter[i]--;
                        l--;
                    }
                }
            }

        } else {
            bool find = false;
            for (int i = 0;i < 10;i++) {
                if (i % 3 == 2 && counter[i] > 0) {
                    find = true;
                    counter[i]--;
                    break;
                }
            }
            if (!find) {
                int l = 2;
                for (int i = 0;i < 10;i++) {
                    while (l > 0 && i % 3 == 1 && counter[i] > 0) {
                        find = true;
                        counter[i]--;
                        l--;
                    }
                }
            }
        }
        string ret = "";
        for (int i = 9;i >= 0;i--) {
            while (counter[i] > 0) {
                ret.push_back((char)('0' + i));
                counter[i]--;
            }
        }
        if (ret.size() != 0 && ret[0] == '0') {
            return "0";
        } else {
            return ret;
        }
    }
};
