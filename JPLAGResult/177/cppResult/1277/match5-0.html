class Solution {
   public:
    string largestMultipleOfThree(vector<int> &A) {
        const int N = A.size();
        if (N == 0) return string();
        if (N == 1) {
            if (A[0] % 3 == 0) return to_string(A[0]);
            return string();
        }

        sort(A.begin(), A.end(),
             [](const int i, const int j) { return i > j; });
        if (A[0] == 0) return string("0");

        int64_t S = 0;
        for (auto &x : A) S += x;

        if (S % 3 == 0) {
            string ans;
            for (auto &x : A) {
                ans.push_back(x + '0');
            }
            return ans;
        }

        if (S % 3 == 1) {
            int rm = -1;
            for (int i = N - 1; i >= 0; i--) {
                if (A[i] % 3 == 1) {
                    rm = i;
                    break;
                }
            }

            if (rm != -1) {
                string ans;
                for (int i = 0; i < N; i++) {
                    if (i != rm) ans.push_back(A[i] + '0');
                }
                return ans;
            }

            // if no single number mod 3 == 1
            // need to find some number add %3 == 1
            unordered_set<int> H;
            for (int i = N - 1; i >= 0; i--) {
                if (A[i] % 3 == 2) {
                    H.insert(i);
                    if (H.size() >= 2) break;
                }
            }

            string ans;
            for (int i = 0; i < N; i++) {
                if (H.find(i) == H.end()) {
                    ans.push_back(A[i] + '0');
                }
            }
            return ans;
        }

        if (S % 3 == 2) {
            int rm = -1;
            for (int i = N - 1; i >= 0; i--) {
                if (A[i] % 3 == 2) {
                    rm = i;
                    break;
                }
            }

            if (rm != -1) {
                string ans;
                for (int i = 0; i < N; i++) {
                    if (i != rm) ans.push_back(A[i] + '0');
                }
                return ans;
            }

            unordered_set<int> H;
            for (int i = N - 1; i >= 0; i--) {
                if (A[i] % 3 == 1) {
                    H.insert(i);
                    if (H.size() >= 2) break;
                }
            }

            string ans;
            for (int i = 0; i < N; i++) {
                if (H.find(i) == H.end()) {
                    ans.push_back(A[i] + '0');
                }
            }
            return ans;
        }

        return string();
    }
};
