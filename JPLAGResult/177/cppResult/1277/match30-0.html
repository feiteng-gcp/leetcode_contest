class Solution {
public:
    string largestMultipleOfThree(vector<int>& digits) {
        vector<int>vec[3];
        int sum=0;
        for(int i=0;i<digits.size();i++){
            vec[digits[i]%3].push_back(digits[i]);
            sum+=digits[i];
        }
        sort(vec[1].begin(),vec[1].end(),greater<int>());
        sort(vec[2].begin(),vec[2].end(),greater<int>());
        cout<<sum<<endl;
        if(sum%3==1){
            if(vec[1].size()>0){
                vec[1].pop_back();
            }else if(vec[2].size()>=2){
                vec[2].pop_back();vec[2].pop_back();
            }else{
                return "";
            }
        }
        if(sum%3==2){
            if(vec[2].size()>0){
                vec[2].pop_back();
            }else if(vec[1].size()>=2){
                vec[1].pop_back();vec[1].pop_back();
            }else{
                return "";
            }
        }
        vector<int>ans;
        for(int i=0;i<3;i++){
            for(int j=0;j<vec[i].size();j++){
                ans.push_back(vec[i][j]);
            }
        }
        if(ans.size()==0){
            return "";
        }
        sort(ans.begin(),ans.end(),greater<int>());
        int st=0;
        if(ans[0]==0){
            return "0";
        }
        string str="";
        for(int i=0;i<ans.size();i++){
            str+=(char)(ans[i]+'0');
        }
        return str;
    }
};
