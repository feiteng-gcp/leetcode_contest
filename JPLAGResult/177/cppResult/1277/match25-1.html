class Solution {
    
public:
    string largestMultipleOfThree(vector<int>& digits) {
        
        int n=digits.size();
        std::sort(digits.begin(),digits.end(),std::greater<int>());
        
        if(digits[0]==0){
            return "0";
        }
        
        std::string res="";
        
        std::vector<int> count(3,0);
        int sum=0;
        

        for(int digit:digits){
            ++count[digit%3];
            sum+=digit;
        }
        
        if(sum%3==1){
            if(count[1]>=1){
                for(int i=n-1;i>=0;--i){
                    if(digits[i]%3==1){
                        digits[i]=-1;
                        break;
                    }
                }
            }
            else if(count[2]>=2){
                int count_minus_1=0;
                for(int i=n-1;i>=0;--i){
                    if(digits[i]%3==2){
                        digits[i]=-1;
                    }
                    if(count_minus_1==2){
                        break;
                    }
                }
            }
            else{
                return res;

            }
        }
        else if(sum%3==2){
            if(count[2]>=1){
                for(int i=n-1;i>=0;--i){
                    if(digits[i]%3==2){
                        digits[i]=-1;
                        break;
                    }
                }
            }
            else if(count[1]>=2){
                int count_minus_1=0;
                for(int i=n-1;i>=0;--i){
                    if(digits[i]%3==1){
                        digits[i]=-1;
                    }
                    if(count_minus_1==2){
                        break;
                    }
                }
            }
            else{
                return res;
            }
        }
        
        for(int digit:digits){
            if(digit!=-1){
                res+=std::to_string(digit);
            }
        }
        
        return res;
    }
};
