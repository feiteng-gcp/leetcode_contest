class Solution {
public:
    vector<int> closestDivisors(int num) {
        int num1 = num + 1, num2 = num + 2;
        
        int sqrt1 = (int)sqrt(num1 + 0.5), sqrt2 = (int)sqrt(num2 + 0.5);
        

        vector<int> ans = {1, num1};
        
        for(int i = sqrt1; i >= 1; i--){
            if(num1 % i == 0){
                int j = num1 / i;
                
                if(j - i < ans[1] - ans[0]){
                    ans[0] = i;
                    ans[1] = j;
                }
            }
        }
        
        for(int i = sqrt2; i >= 1; i--){

            if(num2 % i == 0){
                int j = num2 / i;
                
                if(j - i < ans[1] - ans[0]){
                    ans[0] = i;
                    ans[1] = j;
                }
            }
        }
        
        return ans;
    }
};
