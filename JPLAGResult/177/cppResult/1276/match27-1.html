class Solution {
    vector<int> helper(long long num) {
        vector<int> result(2, 0);
        for (long long i = 1; i * i <= num; i++) {

            if (num % i == 0) {
                result[0] = i;
                result[1] = num / i;
            }
        }
        return result;
    }
public:
    vector<int> closestDivisors(int num) {
        vector<int> a1 = helper(num + 1);
        vector<int> a2 = helper(num + 2);
        if (abs(a1[0] - a1[1]) < abs(a2[0] - a2[1])) {
            return a1;
        } else {
            return a2;
        }
    }
};
