class Solution {
public:
    vector<int> closestDivisors(int num) {
        vector<int>ans(2,0);
        int diff=1e9+7;
        int N=num+1;

        for(int i=1;i*i<=N;i++){
            if(N%i==0){
                if(abs(N/i-i)<diff){
                    diff=abs(N/i-i);
                    ans[0]=i;
                    ans[1]=N/i;
                }
            }
        }
        N=num+2;

        for(int i=1;i*i<=N;i++){
            if(N%i==0){
                if(abs(N/i-i)<diff){
                    diff=abs(N/i-i);
                    ans[0]=i;
                    ans[1]=N/i;
                }
            }
        }
        return ans;
    }
};
