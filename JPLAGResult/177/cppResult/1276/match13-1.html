class Solution {
public:
    vector<int> closestDivisors(int num) {
        int n1=num+1,n2=num+2,diff=INT_MAX;
        vector<int> V(2);

        for(long long i=1;1ll*i*i<=n1;i++){
            if(n1%i==0){
                int t=abs(i-n1/i);
                if(t<diff){
                    diff=t;
                    V[0]=i;V[1]=n1/i;
                }
            }
        }

        for(long long i=1;1ll*i*i<=n2;i++){
            if(n2%i==0){
                int t=abs(i-n2/i);
                if(t<diff){
                    diff=t;
                    V[0]=i;V[1]=n2/i;
                }
            }
        }
        return V;
    }
};
