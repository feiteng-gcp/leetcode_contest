class Solution {
public:
    vector<int> closestDivisors(int num) {
        int n1 = num + 1;
        int n2 = num + 2;
        int t = sqrt(num) + 1;
        while(t >= 1){
            if(n1 % t == 0){
                return {t, n1 / t};
            }
            if(n2 % t == 0){
                return {t, n2 / t};
            }
            t--;
        }
        return {-1,-1};
    }
};
