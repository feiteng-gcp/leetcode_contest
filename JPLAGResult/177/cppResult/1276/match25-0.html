class Solution {
public:
    vector<int> closestDivisors(int num) {
        //num = 999;
        vector<int> result;
        result.push_back(-1);
        result.push_back(-1);
        for (int i = 1; i <= (num + 2) / i; i++)
        {
            if ((num + 1) % i == 0)
            {
                int n1 = i;
                int n2 = (num + 1) / i;
                if (result[0] < 0 || result[1] - result[0] > n2 - n1)
                {
                    result[1] = n2;
                    result[0] = n1;
                }
            }
            
            if ((num + 2) % i == 0)
            {
                int n1 = i;
                int n2 = (num + 2) / i;
                if (result[0] < 0 || result[1] - result[0] > n2 - n1)
                {
                    result[1] = n2;
                    result[0] = n1;
                }
            }
        }
        return result;
    }
};
