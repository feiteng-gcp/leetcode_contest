class Solution {
public:
    vector<int> closestDivisors(int num) {

        int a = 1, b = num + 1;
        for (int i = 1; i <= (num + 2) / i; ++ i) {
            if ((num + 1) % i == 0) {
                if (abs(a - b) > abs((num + 1) / i - i)) {
                    a = (num + 1) / i;
                    b = i;
                }
            }
            if ((num + 2) % i == 0) {
                if (abs(a - b) > abs((num + 2) / i - i)) {
                    a = (num + 2) / i;
                    b = i;
                }
            }
        }
        return vector<int>{a, b};
    }
};
