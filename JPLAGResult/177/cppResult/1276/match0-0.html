class Solution {
public:
    vector<int> closestDivisors(int num) {
        int t = num + 1;
        vector<int> ans1;
        vector<int> ans2;
        for (int a = sqrt(t); a >= 1; a--) { 
            if (t % a == 0) { 
                ans1 = {a, t / a};
                break;
            }
        }
        
        t = num + 2;
        for (int a = sqrt(t); a >= 0; a--) { 
            if (t % a == 0) { 
                ans2 = {a, t / a};
                break;
            }
        }
        if (ans1[1] - ans1[0] < ans2[1] - ans2[0]) { 
            return ans1;
        }
        return ans2;
    }
};
