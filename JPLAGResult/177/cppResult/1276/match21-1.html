class Solution {
public:
    vector<int> core(int num){
        vector<int> res = {1,num};
        for(int i=2;i<sqrt(num+50);++i){
            if(num%i==0){
                if(abs(num/i-i) < abs(res[1]-res[0])){
                    res = {num/i,i};

                }
            }
        }
        return res;
        
    }
    vector<int> closestDivisors(int num) {
        vector<int> res1 = core(num+1);
        vector<int> res2 = core(num+2);
        if(abs(res1[1]-res1[0]) < abs(res2[1]-res2[0])){
            return {res1[1],res1[0]};
        }
        return {res2[1],res2[0]};
    }
};
