class Solution {
public:
    vector<int> splitN(int num)
{
    vector<int> ans = {1, num};
    for(int i = 2; i * i <= num; i++)

        if(num % i == 0) ans = {i, num / i};
    return ans;
}

vector<int> closestDivisors(int num)
{
    vector<int> a1 = splitN(num + 1), a2 = splitN(num + 2);
    if(a1[1] - a1[0] < a2[1] - a2[0]) return a1;
    else return a2;
}
};
