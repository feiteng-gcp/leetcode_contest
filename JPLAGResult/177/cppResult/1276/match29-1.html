class Solution {
public:

    int calc(int x) {
        int ret = 1;
        for (int i = 2; i * i <= x; ++i) {
            if (x % i == 0) {
                ret = i;
            }
        }
        return ret;
    }
    
    vector<int> closestDivisors(int num) {
        int p = calc(num + 1);

        int q = calc(num + 2);
        if ((num + 1) / p - p < (num + 2) / q - q) {
            return {p, (num + 1) / p};
        }
        return {q, (num + 2) / q};
    }
};
