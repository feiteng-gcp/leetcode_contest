class Solution {
public:
    vector<int> getNum(int num) {
        int a = sqrt(num);
        while (a) {
            if (num % a == 0) break;
            --a;
        }
        return {a, num / a};
    }
    
    vector<int> closestDivisors(int num) {
        vector<int> res{1, num + 2};
        for (int i = 1; i <= 2; ++i) {
            auto t = getNum(num + i);
            if (t[1] - t[0] == 0) return t;
            if (t[1] - t[0] < res[1] - res[0]) res = t;
        }
        return res;
    }
};
