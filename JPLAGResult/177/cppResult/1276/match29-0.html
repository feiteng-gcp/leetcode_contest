class Solution {
public:
    pair<int, int> dfs(int x) {
        int w = sqrt(x);
        for (int i = w + 1; i >= 1; i--) {
            if (x % i == 0) {
                return make_pair(i, x / i);
            }
        }
        return make_pair(-1, 100000007);
    }
    vector<int> closestDivisors(int num) {
        auto a = dfs(num + 1);
        auto b = dfs(num + 2);
        if (abs(a.first - a.second) < abs(b.first - b.second)) {
            return {a.first, a.second};
        }
        return {b.first, b.second};
    }
};
