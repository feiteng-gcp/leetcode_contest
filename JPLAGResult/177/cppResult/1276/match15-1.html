class Solution {
public:
    vector<int> closestDivisors(int n) {

        int r = sqrt(n + 2);
        
        while (r != 1) {
            if ((n + 1) % r == 0) {
                return {r, (n + 1) / r};
            }
            if ((n + 2) % r == 0) {
                return {r, (n + 2) / r};
            }
            r--;
        }
        
        return {1, n + 1};
    }
};
