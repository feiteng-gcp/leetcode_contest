class Solution {
public:
    vector<int> closestDivisors(int num) {
        int min_difference = 1E9;
        vector<int> ans(2);
        int target;
        
        target = num + 1;
        for (int i = 1; i * i <= target; ++i) {
            if (target % i == 0) {
                int j = target / i;
                if (abs(j - i) < min_difference) {
                    min_difference = abs(j - i);
                    ans[0] = i;
                    ans[1] = j;
                }
            }
        }
        
        target = num + 2;
        for (int i = 1; i * i <= target; ++i) {
            if (target % i == 0) {
                int j = target / i;
                if (abs(j - i) < min_difference) {
                    min_difference = abs(j - i);
                    ans[0] = i;
                    ans[1] = j;
                }
            }
        }
        
        return ans;
    }
};
