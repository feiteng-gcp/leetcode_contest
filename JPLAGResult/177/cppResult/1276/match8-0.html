class Solution {
public:
    
    vector<int> go(int x) {
        vector<int> ans = {1, x};
        
        for(int i = 1; i*i <= x; i++) {
            if(x % i == 0) {
                ans = {i, x/i};
            }
        }
        
        return ans;
    }
    
    vector<int> closestDivisors(int num) {
        vector<int> a = go(num+1), b = go(num+2);
        if(a[1] - a[0] < b[1] - b[0])
            return a;
        else
            return b;
    }
};
