class Solution {
public:
    vector<int> closestDivisors(int num) {
        int x = num + 2;
        
        int a = 0, b = INT_MAX;
        for(int i = 1; i <= sqrt(x); ++i){
            if((num+1) % i == 0){
                if(abs(i - ((num + 1) / i)) < abs(a - b)){
                    a = i;
                    b = (num+1)/i;
                }
            }
            if((num+2) % i == 0){
                if(abs(i - ((num + 2) / i)) < abs(a - b)){
                    a = i;
                    b = (num+2)/i;
                }
            }
        }
        return {a, b};
    }
};
