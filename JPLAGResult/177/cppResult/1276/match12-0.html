class Solution {
public:
    vector<int> closestDivisors(int num) {
        int cl = num + 3;
        vector<int> ans;
        
        int x = num + 1;
        for (int i = 1; 1ll * i * i <= x; i++) {
            if (x % i == 0) {
                int d1 = i, d2 = x / i;
                if (d2 - d1 < cl) {
                    ans = {d2, d1};
                    cl = d2 - d1;
                }
            }
        }
        x = num + 2;
        
        for (int i = 1; 1ll * i * i <= x; i++) {
            if (x % i == 0) {
                int d1 = i, d2 = x / i;
                if (d2 - d1 < cl) {
                    ans = {d2, d1};
                    cl = d2 - d1;
                }
            }
        }
        
        return ans;
    }
};
