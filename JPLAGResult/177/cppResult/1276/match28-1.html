class Solution {
public:
    vector<int> closestDivisors(int num) {
        int d = num;

        int x = 1, y = num + 1;
        for (int z = 1; z <= (num + 2) / z; ++z) {
            if (0 == (num + 1) % z) {
                int t = (num + 1) / z;
                if (t - z < d) {
                    d = t - z;
                    x = z;
                    y = t;
                }
            }

            if (0 == (num + 2) % z) {
                int t = (num + 2) / z;
                if (t - z < d) {
                    d = t - z;
                    x = z;
                    y = t;
                }
            }
        }
        return {x, y};
    }
};
