class Solution {
    vector<int> fnd(int n)
    {
        vector<int> ans={1,n};
        for(int i=1;i*1LL*i<=n;++i)
        {
            if(n%i==0&&(n/i)-i<=ans[1]-ans[0])
                ans={i,n/i};
        }
        return ans;
    }
public:
    vector<int> closestDivisors(int n) {
        auto u=fnd(n+1);
        auto v=fnd(n+2);
        if(v[1]-v[0]<u[1]-u[0])
                return v;
        return u;
    }
};
