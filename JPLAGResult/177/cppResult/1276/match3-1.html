class Solution {
    int delta = INT_MAX;
    vector<int> ans;
    void split(int N) {
        int m = sqrt(N);
        if (m * m == N) {
            delta = 0;
            ans = {m, m};
            return;
        }

        for (int i = m; i >= 1; i--) {
            const int j = N / i;
            if (i * j == N) {
                if (delta > abs(i - j)) {
                    delta = abs(i - j);
                    ans = {i, j};
                }
                break;
            }
        }
    }

   public:
    vector<int> closestDivisors(int N) {
        if (N == 1) return {1, 2};
        if (N == 2) return {2, 2};
        if (N == 3) return {2, 2};
        delta = INT_MAX;
        split(N + 1);
        split(N + 2);
        return ans;
    }
};

/*
Given an integer num, find the closest two integers in absolute difference whose product equals num + 1 or num + 2.

Return the two integers in any order.
