class Solution {
public:
    vector<int> closestDivisors(int num) {
        int num1=num+1;
        int num2=num+2;
        int m=sqrt(num2);
        int a=1;
        int b=num1;

        for(int i=2;i<=m;i++){
            if((num1%i)==0){
                int a1=i;
                int b1=num1/i;
                if(abs(a1-b1)<abs(a-b)){
                    a=a1;
                    b=b1;
                }
            }
            if((num2%i)==0){
                int a1=i;
                int b1=num2/i;
                if(abs(a1-b1)<abs(a-b)){
                    a=a1;
                    b=b1;
                }
            }
        }
        return {a,b};
    }
};
