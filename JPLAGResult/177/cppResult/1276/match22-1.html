class Solution {
public:
    vector<int> closestDivisors(int num) {
        
        int n1=num+1;
        int n2=num+2;
        // int i=0;

        for(int i=sqrt(n2); i>=1; i--){
            if(n1%i==0)
                return {i,n1/i};
            if(n2%i==0)
                return {i,(n2)/i};
        }
        return {};
    }
};
