class Solution {
public:
    
    int compute(int num) {
        int dif = num;
        
        return dif;
    }
    
    vector<int> closestDivisors(int num) {
        int dif = INT_MAX;
        int n1 = num+1;
        int d1 = 0, d2 = 0;
        cout << sqrt(n1) << endl;
        for (int i = 1; i <= (int)sqrt(n1); i++) {
            if (n1%i == 0) {
                int tmp = abs(n1/i-i);
                if (dif > tmp) {
                    dif = tmp;
                    d1 = n1/i;
                    d2 = i;
                }
            }
        }
        int n2 = num+2;
        cout << sqrt(n2) << endl;
        for (int i = 1; i <= (int)sqrt(n2); i++) {
            if (n2%i == 0) {
                int tmp = abs(n2/i-i);
                if (dif > tmp) {
                    dif = tmp;
                    d1 = n2/i;
                    d2 = i;
                }
            }
        }
        return {d1,d2};
    }
};
