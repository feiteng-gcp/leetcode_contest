class Solution {
public:
    vector<int> closestDivisors(int n) {
        int l=n+1,r=n+2;
        int a=-1e9,b=1e9;
        vector<int>ret;

        for(int i=1;i*i<=r;i++){
            if(r%i==0){
                int j=r/i;
                if(abs(i-j)<abs(b-a)){
                    a=i;
                    b=j;
                }
            }
            if(l%i==0){
                int j=l/i;
                if(abs(i-j)<abs(b-a)){
                    a=i;
                    b=j;
                }
            }
        }
        ret.push_back(a);
        ret.push_back(b);
        return ret;
    }
};
