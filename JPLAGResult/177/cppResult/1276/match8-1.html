class Solution {
public:
    

vector<int> getPair(int x) {
    vector<int> res = {1, x};
    for (int i = 1; i <= sqrt(x); i++) {
        int j = x / i;

        if (i * j == x) {
            res = {i, j};
        }
    }
    return res;
}
    
    vector<int> closestDivisors(int x) {
        vector<int> res1 = getPair(x + 1);
        vector<int> res2 = getPair(x + 2);
        if (res1[1] - res1[0] < res2[1] - res2[0]) {
            return res1;
        }
        
        return res2;
    }
};
