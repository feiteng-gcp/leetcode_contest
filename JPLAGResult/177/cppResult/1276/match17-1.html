class Solution {
public:
    vector<int> get(int x) {
        vector<int> ans;
        for (int i = 1; i * i <= x; ++i) {
            if (x % i == 0) ans = {i, x / i};
        }
        return ans;
    }
    vector<int> closestDivisors(int num) {
        vector<int> a = get(num+1), b = get(num+2);
        return abs(a[1] - a[0]) < abs(b[1] - b[0]) ? a : b;
    }
};
