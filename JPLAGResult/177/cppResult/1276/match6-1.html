class Solution {
public:
    vector<int> divide(int n)
    {
        vector<int>ret(2);
        ret[0]=1; ret[1]=n;
        for (int i=2;i<=sqrt(n);i++)
        {
            if (n%i==0)
            {
                ret[0]=i;
                ret[1]=n/i;
            }
        }
        return ret;

    }
    vector<int> closestDivisors(int num) {
        auto r1 = divide(num+1);
        auto r2 = divide(num+2);
        if (abs(r1[0]-r1[1])<abs(r2[0]-r2[1])) return r1;
        else return r2;
        
    }
};
