class Solution {
private:
    vector<int> test(int x) {
        vector<int> ans(2);
        for (int i = 1; i <= x; i++) {
            int j = x / i;
            if (i > j) {
                break;
            }
            if (i * j == x) {
                ans[0] = i;
                ans[1] = j;
            }
        }
        return ans;
    }
public:
    vector<int> closestDivisors(int num) {
        vector<int> p1 = test(num + 1);
        vector<int> p2 = test(num + 2);
        if (p1[1] - p1[0] < p2[1] - p2[0]) {
            return p1;
        } else {
            return p2;
        }
    }
};
