class Solution {
public:
    vector<int> closestDivisors(int num) {
        int ans=0x3f3f3f3f;
        int a,b;
        for(int i=1;i*i<=num+1;i++){
            if((num+1)%i==0){
                if(((num+1)/i-i)<ans){
                    ans=((num+1)/i-i);
                    b=(num+1)/i;
                    a=i;
                }
            }
        }
        for(int i=1;i*i<=num+2;i++){
            if((num+2)%i==0){
                if(((num+2)/i-i)<ans){
                    ans=((num+2)/i-i);
                    b=(num+2)/i;
                    a=i;
                }
            }
        }
        vector<int> res;
        res.push_back(a);
        res.push_back(b);
        return res;
    }
};
