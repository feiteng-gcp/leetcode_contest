class Solution {
public:
    vector<int> closestDivisors(int num) {
        int ans = INT_MAX;
        int a ,b;
        num++;
        for(int x = 1;x <= sqrt(num);x++){
            if(num%x == 0){
                int y = num/x;
                if(abs(x-y) < ans){
                    ans = abs(x-y);
                    a = x;
                    b=y;
                }
            }
        }
        num++;
        
        for(int x = 1;x <= sqrt(num);x++){
            if(num%x == 0){
                int y = num/x;
                if(abs(x-y) < ans){
                    ans = abs(x-y);
                    a = x;
                    b=y;
                }
            }
        }
        vector<int>anns;
        anns.push_back(a);
        anns.push_back(b);
        return anns;
    }
};
