class Solution {
private:

    pair<int, int> solution(int x) {
        // pair<int, int> ans = make_pair(1, x);
        
        for(int i = sqrt(x); i >= 1; --i) {
            if(x % i == 0) {
                return make_pair(i, x / i);
            }
        }
        return make_pair(1, x);
    }
public:

    vector<int> closestDivisors(int num) {
        pair<int, int> a1 = solution(num + 2);
        pair<int, int> a2 = solution(num + 1);
        vector<int> ans;
        
        if(a1.second - a1.first > a2.second - a2.first) {
            ans.push_back(a2.first);
            ans.push_back(a2.second);
            // return a2;
        } else {
            ans.push_back(a1.first);
            ans.push_back(a1.second);
        }
        return ans;
    }
};
