class Solution {
public:
    vector<int> f(int n) {
        int nn = pow(n, 0.5);
        nn++;
        for (int i = nn; i >= 1; i--) {
            if (n % i == 0) {
                return vector<int> {i, n / i};
            }
        }
        return vector<int> {1, n};
    }
    
    vector<int> closestDivisors(int num) {
        auto p1 = f(num + 1);
        auto p2 = f(num + 2);
        if (abs(p1[0] - p1[1]) < abs(p2[0] - p2[1]))
            return p1;
        else
            return p2;
    }
};
