class Solution:
    def maxRepOpt1(self, text: str) -> int:
        l = []
        res = 0
        max_char = ""
        counter = collections.Counter()
        
        cur = text[0]
        num = 0
        for c in text:
            if c == cur:
                num += 1
            else:
                l.append((cur, num))
                if num > res:
                    res = num
                    max_char = cur
                counter[cur] += 1
                
                cur = c
                num = 1

        l.append((cur, num))
        if num > res:
            res = num
            max_char = cur
        counter[cur] += 1
        
        # print(l, res)
        
        if counter[max_char] > 1 and len(counter) > 1:
            res += 1
            
        for i in range(1, len(l) - 1):
            prev = l[i - 1]
            cur = l[i]
            next = l[i + 1]
            
            if prev[0] == next
[0] and cur[1] == 1:
                if counter[prev[0]] > 2:
                    res = max(res, prev[1] + next[1] + 1)
                else:
                    res = max(res, prev[1] + next[1])
                    
            # print(res, prev, cur, next, counter[prev[0]])
        
        return res
