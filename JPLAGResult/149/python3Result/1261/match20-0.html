class Solution:
    def maxRepOpt1(self, s):
        if not s:
            return 0
        # for c in [chr(i + ord("a")) for i in range(26)]:
        unique = sorted(set(s))
        if len(unique) == 1:
            return len(s)
        ans = 0
        for c in unique:
            # print (c)
            totalCount = 0
            curStart = -1
            curEnd = -1
            chunk = []
            for i in range(len(s)):
                if s[i] == c:
                    if curStart == -1:
                        curStart = i
                    curEnd = i
                    totalCount += 1
                else:
                    chunk.append((curStart, curEnd))
                    curStart = -1
                    curEnd = -1
            if curStart != -1:
                chunk.append((curStart, curEnd))
            long = max(t-s+1 for s, t in chunk)
            if long == totalCount:
                thisBest = long
            else:
                thisBest = long + 1
                for i in range(1, len(chunk)):
                    if chunk[i-1][1] + 2 == chunk[i][0]:
                        if chunk[i][1] - chunk[i-1][0] + 1 <= totalCount:
                            thisBest = max(thisBest, chunk[i][1] - chunk[i-1][0] + 1)
                        else:
                            thisBest = max(thisBest, chunk[i][1] - chunk[i-1][0])
            ans = max(thisBest, ans)
        return ans
