class Solution:
    def maxRepOpt1(self, text: str) -> int:
        aux, l = collections.defaultdict(list), 0
        for r, (a, b) in enumerate(zip(text, text[1:] + '#')):
            if a != b:
                aux[a]
.append((l, r))
                l = r + 1
        ans = 0
        for a, intvs in aux.items():
            L = len(intvs)
            ans = max(ans, max(r - l + 1 for l, r in intvs) + (L >= 2))
            for (l1, r1), (l2, r2) in zip(intvs, intvs[1:]):
                if r1 + 2 == l2:
                    ans = max(ans, r1 - l1 + 1 + r2 - l2 + 1 + (L >= 3))
        return ans
