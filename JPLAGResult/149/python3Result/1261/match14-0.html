from collections import Counter, defaultdict
class Solution:
    def maxRepOpt1(self, text: str) -> int:
        counter = Counter(text)
        window = defaultdict(int)
        res = 0
        j = 0
        for i, c in enumerate(text):
            window[c] += 1
            while len(window) > 2:
                window[text[j]] -= 1
                if window[text[j]] == 0:
                    window.pop(text[j])
                j += 1
            if len(window) == 2:
                while min(window.values()) > 1:
                    window[text[j]] -= 1
                    j += 1
                if window[c] < counter[c]:
                    res = max(res, window[c] + 1)
                else:
                    res = max(res, counter[c])
            else:
                # only one char
                res = max(res, window[c])
        return res
