class Solution:
    def maxRepOpt1(self, text: str) -> int:
        c = collections.Counter(text)
        n = len(text)
        ret = 0
        for i in range(26):
            ch = chr(i + ord('a'))
            j = k = 0
            last = None
            while j < n and k < n:
                if text[k] != ch:
                    if last is not None:
                        j = last + 1
                    last = k
                k += 1
                ret = max(ret, k - j if c[ch] >= k - j else c[ch])
        return ret
