class Solution:
    def maxRepOpt1(self, text: str) -> int:
        dic={}
        last='0'
        i=0
        totals=0
        while(i<len(text)):
            if last==text[i]:
                dic[last][-1][1]=dic[last][-1][1]+1
                i=i+1
            else:
                last=text[i]
                try:
                    dic[last]
                except:
                    dic[last]=[]
                dic[last].append([i,i])
                i=i+1
                totals=totals+1
        #print(dic['a'])
        if totals==1:
            print(len(text))
        maxs=0
        for x in dic:
            L=len(dic[x])
            if L==1:
                maxs=max(maxs,dic[x][0][1]-dic[x][0]
[0]+1)
            elif L==2:
                if dic[x][0][1]+2==dic[x][1][0]:
                    maxs=max(maxs,dic[x][1][1]-dic[x][0][0])
                else:
                    maxs=max(maxs,dic[x][0][1]-dic[x][0][0]+2)
                    maxs=max(maxs,dic[x][1][1]-dic[x][1][0]+2)
            else:
                i=0
                while(i<L-1):
                    if dic[x][i][1]+2==dic[x][i+1][0]:
                        maxs=max(maxs,dic[x][i+1][1]-dic[x][i][0]+1)
                    else:
                        maxs=max(maxs,dic[x][i][1]-dic[x][i][0]+2)
                    i=i+1
                maxs=max(maxs,dic[x][i][1]-dic[x][i][0]+2)
        return maxs
