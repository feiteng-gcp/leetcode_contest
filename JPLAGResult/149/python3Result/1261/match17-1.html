class Solution:
    def maxRepOpt1(self, text: str) -> int:
        ret = 1
        counter = collections.Counter(text)
        length = [[text[0], 1]]
        for i in range(1, len(text)):
            if text[i] == text[i-1]:
                length[-1][1] += 1
            else:
                length.append([text[i], 1])
        ret = max(length, key=lambda x: x[1])[1]
        for i in range(1, len(length)-1):
            if length[i][1] == 1 and length[i-1][0] == length[i+1][0]:
                if counter[length[i-1][0]] > length
[i-1][1] + length[i+1][1]:
                    ret = max(ret, length[i-1][1] + length[i+1][1] + 1)
                else:
                    ret = max(ret, length[i-1][1] + length[i+1][1])
            if counter[length[i][0]] > length[i][1]:
                ret = max(ret, length[i][1] + 1)
        for i in [0, -1]:
            if counter[length[i][0]] > length[i][1]:
                ret = max(ret, length[i][1] + 1)
            
        return ret
