from collections import defaultdict
class Solution:
    def maxRepOpt1(self, text: str) -> int:
        count = defaultdict(int)
        res = 0
        arr = []
        curLen = 0
        for i in range(len(text)):
            if i == 0:
                arr.append(1)
                res = max(1,res)
                curLen = 1
            elif text[i] == text[i-1]:
                curLen +=1
                res = max(res,curLen)
                arr.append(curLen)
                arr[i-curLen+1] = curLen
            else:
                curLen = 1
                arr.append(1)
                count[text[i-1]]+=1
        count[text[-1]]+=1
        for i in range(len(text)):
            if i ==0 or text[i] == text[i-1]:
                continue
            if i == len(text)-1:
                if count[text[i-1]]>1:
                    res = max(arr[i-1]+1,res)
            elif text[i-1] == text[i+1]:
                if count[text[i-1]]>2:
                    res = max(arr[i-1]+arr[i+1]+1,res)
                else:
                    res = max(arr[i-1]+arr[i+1],res)
            else:
                if count[text[i+1]]>1:
                    res = max(arr[i+1]+1,res)
                if count[text[i-1]]>1:
                    res = max(arr[i-1]+1,res)
        #print(arr)
        #print(count)
        return res
