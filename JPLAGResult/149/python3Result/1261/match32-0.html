class Solution:
    def maxRepOpt1(self, text: str) -> int:
        counter = collections.defaultdict(int)
        memory = collections.defaultdict(list)
        prev = "#"
        pos = -1
        for index in range(len(text)):
            if text[index] != prev:
                if prev != "#":
                    memory[prev].append([pos,index])
                prev = text[index]
                pos = index
        if pos <= len(text)-1:
            memory[prev].append([pos,len(text)])
        print(memory)
        maximum = 0
        for key in memory:
            if len(memory[key]) == 1:
                maximum = max(maximum,memory[key][0][1]-memory[key][0][0])
            elif len(memory[key]) > 1:
                for item in memory[key]:
                    maximum = max(maximum,item[1]-item[0]+1)
            if len(memory[key]) == 2:
                array = memory[key]
                array.sort()
                for index in range(len(array)-1):
                    if array[index][1]+1 == array[index+1][0]:
                        maximum = max(maximum,array[index+1][1]-array[index][0]-1)
            elif len(memory[key]) > 2:
                array = memory[key]
                array.sort()
                for index in range(len(array)-1):
                    if array[index][1]+1 == array[index+1][0]:
                        maximum = max(maximum,array[index+1][1]-array[index][0])
        return maximum
