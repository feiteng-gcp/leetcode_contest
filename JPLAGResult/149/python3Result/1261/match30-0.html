class Solution:
    def maxRepOpt1(self, text: str) -> int:
        rep_arr = [[0,0]]
        for c in text:
            if c == rep_arr[-1][0]:
                rep_arr[-1][1] += 1
            else:
                rep_arr.append([c, 1])
                
        char_frag_cnt = {}
        for c, _ in rep_arr:
            char_frag_cnt[c] = char_frag_cnt.get(c, 0) + 1
        
        result = 0
        for i, (c, cnt) in enumerate(rep_arr):
            if i > 2 and c == rep_arr[i-2][0] and rep_arr[i-1][1] == 1:
                result = max(result, cnt+rep_arr[i-2][1]+(char_frag_cnt[c] > 2))
            elif char_frag_cnt[c] > 1:
                result = max(result, cnt+1)
            else:
                result = max(result, cnt)
        
        return result
