from collections import Counter

class Solution:
    def maxRepOpt1(self, text: str) -> int:
        cnt=Counter(text)
        n=len(text)
        l=[1]*n
        ans=0
        for i in range(1,n):
            if text[i]==text[i-1]:
                l[i]=l[i-1]+1
            ans=max(ans,l[i])
            if cnt[text[i]]>l[i]:
                ans=max(ans,l[i]+1)
            j=i-l[i]-1
            if j>=0 and text[j]==text[i] and cnt[text[i]]>l[i]+l[j]:
                ans=max(ans,l[i]+l[j]+1)
        return ans
