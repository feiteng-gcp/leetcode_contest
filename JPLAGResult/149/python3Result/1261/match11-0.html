class Solution:
    def maxRepOpt1(self, text: str) -> int:
        nums = collections.Counter(text)
        N = len(text)
        
        ans = 0
        for ich in range(26):
            ch = chr(ord('a') + ich)
            if ch not in nums:
                continue
            tans = 0
            j = 0
            c = 0
            for i in range(N):
                while j < N:
                    if text[j] != ch:
                        if c >= 1: break
                        c += 1
                    j += 1
                tans = max(tans, j - i)
                if text[i] != ch:
                    c -= 1
            ans = max(ans, min(tans, nums[ch]))
        return ans
