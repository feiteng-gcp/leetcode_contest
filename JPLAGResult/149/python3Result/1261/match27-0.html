class Solution:
    def maxRepOpt1(self, text: str) -> int:
        alp=set()
        for char in text:
            alp.add(char)
        final=0
        for char in alp:
            start=-1
            ind=-1
            nums=[]
            for char2 in text:
                ind=ind+1
                if char2==char:
                    if start>-1:
                        end=end+1
                    else:
                        start=ind
                        end=ind
                elif start>-1:
                    nums.append((start,end))
                    start=-1
                if start>-1 and ind==len(text)-1:
                    nums.append((start,end))
            for item in nums:
                final=max(final,item[1]-item[0]+1)
            for ind in range(len(nums)-1):
                #print(nums,nums[ind][1],nums[ind+1][0]-1)
                if nums[ind][1]<nums[ind+1][0]-2:
                    final=max(final,nums[ind][1]-nums[ind][0]+1+int(len(nums)>1))
                else:
                    if len(nums)==2:
                        final=max(final,nums[ind+1][1]-nums[ind][0])
                    else:
                        final=max(final,nums[ind+1][1]-nums[ind][0]+1)
            final=max(final,nums[len(nums)-1][1]-nums[len(nums)-1][0]+1+int(len(nums)>1))
        return final
