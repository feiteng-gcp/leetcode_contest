class Solution:
    def maxRepOpt1(self, text: str) -> int:
        groups = [[text[0], 1]]
        for i in range(1, len(text)):
            if text[i] == groups[-1][0]:
                groups[-1][1] += 1
            else:
                groups.append([text[i], 1])
        group_num = dict()
        group_l_max = dict()
        res = 0
        for char, l in groups:
            if char not in group_num:
                group_num[char] = 1
                group_l_max[char] = l
                res = max(res, l)
            else:
                group_num[char] += 1
                group_l_max[char] = max(group_l_max[char], l)
                res = max(res, 1 + group_l_max[char])
        for i in range(len(groups) - 2):
            a, l_a = groups[i]
            b, l_b = groups[i+2]
            if a == b and groups[i+1][1] == 1:
                l_swap = l_a + l_b + (group_num[a] > 2)
                res = max(res, l_swap)
        return res
