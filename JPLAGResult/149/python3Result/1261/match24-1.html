import collections
import itertools

class Solution:
    def maxRepOpt1(self, text: str) -> int:
        c = collections.Counter(text)
        l = [(len(list(it)), char) for char, it in itertools.groupby(text)]
        d = collections.defaultdict(int)
        ans = 0
        length, char = l[0]
        d[char] = max(d[char], length)
        length, char = l[-1]
        d[char] = max(d[char], length)
        for i in range(1, len(l) - 1):
            length, char = l[i]
            d[char] = max(d[char], length)
            char = l[i - 1][1]
            if char == l
[i + 1][1] and length == 1:
                ans = max(ans, l[i - 1][0] + l[i + 1][0] + int(l[i - 1][0] + l[i + 1][0] < c[char]))

        # d = collections.defaultdict(int)
        # for char, it in itertools.groupby(text):
        #     d[char] = max(d[char], len(list(it)))
        # print(d)
        # print(c)
        return max(ans, max(length + int(c[char] > length) for char, length in d.items()))
