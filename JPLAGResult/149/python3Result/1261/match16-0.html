class Solution:
    def maxRepOpt1(self, text: str) -> int:
        pool = []
        i, n = 0, len(text)
        while i < n:
            cnt, pc = 0, text[i]
            while i < n and text[i] == pc:
                i += 1
                cnt += 1
            pool.append((pc, cnt))
        counter = collections.Counter(c for c, _ in pool)
        ans_c, ans_l = max(pool, key=lambda x: x[1])
        ans_l += counter[ans_c] > 1
        for i in range(1, len(pool)-1):
            if pool[i][1] == 1 and pool[i-1][0] == pool[i+1][0]:
                ans_l = max(ans_l, pool[i-1][1] + pool[i+1][1] + int(counter[pool[i+1][0]]>2))
        return ans_l
