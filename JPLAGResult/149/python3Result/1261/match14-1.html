class Solution:
    def maxRepOpt1(self, text: str) -> int:
        d = collections.defaultdict(int)
        count = collections.Counter(text)
        start = 0
        end = 0
        res = 0
        while end<len(text):
            d[text[end]]+=1
            while d and(len(d)>2 or(len(d)>1 and min(d.values
())>1)) :
                d[text[start]]-=1
                if d[text[start]]==0:
                    d.pop(text[start])
                start+=1                          
            end+=1
            
            for each  in d:
                res = max(res,min(d[each]+1,count[each]))
        return res
