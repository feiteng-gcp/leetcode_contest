from collections import defaultdict
class Solution:
    def maxRepOpt1(self, text):
        #划分为26个区间
        pre = None
        dict1 = defaultdict(list)
        dict1[None]=[(-1,None,None)]
        res = 0
        for i,v in enumerate(text):
            if v != pre:
                dict1[pre][-1]=((dict1[pre][-1][0],i-1,i-dict1[pre][-1][0]))
                dict1[v].append((i,None,None))
                pre=v
        dict1[text[-1]][-1]=(dict1[text[-1]][-1][0],len(text)-1,len(text)-dict1[text[-1]][-1][0])
        del dict1[None]
        
        for c,list1 in dict1.items():
            #print(list1)
            if len(list1)==1:
                res = max(res,list1[0][2])
                continue
            elif len(list1)==2:
                if list1[0][1]+2==list1[1][0]:
                    res=max(res,list1[0][2]+list1[1][2])
                else:
                    res=max(res,list1[0][2]+1,list1[1][2]+1)
                continue
            for i in range(len(list1)-1):
                tp1,tp2=list1[i],list1[i+1]
                if tp1[1]+2==tp2[0]:
                    re = tp1[2]+tp2[2]+1
                    if re>res:res=re
                else:
                    res=max(res,tp1[2]+1,tp2[2]+1)
        #print(res)
        return res
