from collections import defaultdict
class Solution:
    def maxRepOpt1(self, text: str) -> int:
        s = defaultdict(list)
        i = 0
        j = 0
        n = len(text)
        while j < n:
            while j < n and text[j] == text[i]:
                j += 1
            s[text[i]].append((i, j))
            i = j
        res = 0        
        for k, rs in s.items():
            m = len(rs)
            if m > 1:
                for i in range(m):
                    if i > 0:
                        if rs[i][0] - rs[i-1][1] == 1:
                            if m > 2:
                                res = max(res, rs[i][1] - rs[i][0] + rs[i-1][1] - rs[i-1][0] + 1)
                            else:
                                res = max(res, rs[i][1] - rs[i][0] + rs[i-1][1] - rs[i-1][0])                    
                    res = max(res, rs[i][1] - rs[i][0] + 1)            
            elif m > 0:
                res = max(res, rs[0][1] - rs[0][0])
        return res
