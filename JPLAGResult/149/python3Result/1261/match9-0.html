class Solution:
    def maxRepOpt1(self, text: str) -> int:
        n = len(text)
        if n == 1:
            return 1
        array = collections.defaultdict(list)
        i = 0
        while i < n:
            start = i
            while i < n and text[i] == text[start]:
                i += 1
            array[text[start]].append([start, i-1])
        ans = 1
        for key in array:
            temp = array[key]
            max_ = temp[0][1] - temp[0][0] + 1
            res = 0
            for i in range(1, len(temp)):
                max_ = max(max_, temp[i][1] - temp[i][0] + 1)
                if temp[i][0] == temp[i-1][1] + 2:
                    if i < len(temp) - 1 or i > 1:
                        res = max(res, (temp[i][1] - temp[i][0] + 1) + (temp[i-1][1] - temp[i-1][0] + 1) + 1)
                    else:
                        res = max(res, (temp[i][1] - temp[i][0] + 1) + (temp[i-1][1] - temp[i-1][0] + 1))
            if len(temp) > 1:
                res = max(res, max_ + 1)
            else:
                res = max(res, max_)
            ans = max(ans, res)
        return ans
