class MajorityChecker:

    def __init__(self, arr: List[int]):
        dic = collections.defaultdict(list)

        for i, a in enumerate(arr):
            dic[a].append(i)
        
            
        self.dic = dic
        self.u = max([len(dic[x]) for x in dic])


        
        
        
        #print(len(dic), len(arr))

    def query(self, left: int, right: int, threshold: int) -> int:
        if right - left + 1 > 2*self.u:
            return -1
        for a in self.dic:
            t = len(self.dic[a])
            if t < threshold:
                continue
            l = bisect.bisect_left(self.dic[a], left)
            if t - l < threshold:
                continue
            r 
= bisect.bisect_right(self.dic[a], right)
            if r - l >= threshold:
                return a
        return -1
