from collections import defaultdict

class MajorityChecker:

    def __init__(self, arr: List[int]):
        self.loc = defaultdict(list)
        
        for idx, val in enumerate(arr): # O(n)
            self.loc[val].insert(bisect.bisect(self.loc[val], idx), idx) # O(n)
        
        self.counts = defaultdict(list)
        for val, locations in self.loc.items(): # O(n)
            self.counts[len(locations)].append(val)
        
        self.sorted_counts = sorted(self.counts.keys()) # O(nlogn)
        
        print(self.loc, self.counts, self.sorted_counts)

    def query(self, left: int, right: int, threshold: int) -> int:
        if threshold > right - left + 1:
            return -1
        
        counts_gt_threshold_index = bisect.bisect_left(self.sorted_counts, threshold)
        
        for count_idx in range(counts_gt_threshold_index, len(self.sorted_counts)):
            count = self.sorted_counts[count_idx]
            #print("count > threshold:", count)
            values = self.counts[count]
            for val in values:
                #print("value:", val)
                left_index = bisect.bisect_left(self.loc[val], left)
                right_index = bisect.bisect_right(self.loc[val], right)
                #print(left_index, right_index)
                if right_index - left_index >= threshold:
                    return val
        return -1
