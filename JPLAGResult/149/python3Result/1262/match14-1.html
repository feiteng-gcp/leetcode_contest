import bisect
import collections
class MajorityChecker:

    def __init__(self, arr: List[int]):
        self
.idx = collections.defaultdict(list)
        for i, x in enumerate(arr):
            self.idx[x].append(i)
        self.count, self.keys = zip(*sorted((len(self.idx[x]), x) for x in self.idx))
        
    def query(self, left: int, right: int, threshold: int) -> int:
        if threshold > self.count[-1]:
            return -1
        
        idx = bisect.bisect_left(self.count, threshold)
        
        for i in range(idx, len(self.count)):
            key = self.keys[i]
            il = bisect.bisect_left(self.idx[key], left)
            ir = bisect.bisect_right(self.idx[key], right)
            if ir - il >= threshold:
                return key
        return -1
