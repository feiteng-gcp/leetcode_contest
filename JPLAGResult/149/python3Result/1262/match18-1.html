class MajorityChecker:

    def __init__(self, arr: List[int]):
        t = collections.Counter(arr).most_common(1)
        self.a = arr
        self.b = {}
        self.max = t[0][1]

    def query(self, left: int, right: int, threshold: int) -> int:
        if threshold > self.max or right - left + 1 < threshold:
            return -1
        if (left, right) not in self.b:
            c = collections.Counter(self.a[left:right + 1])
            self.b[(left, right)] = c.most_common(1)[0]
        
if self.b[(left, right)][1] >= threshold:
            return self.b[(left, right)][0]
        else:
            return -1
