class MajorityChecker:

    def __init__(self, arr: List[int]):
        self.A = arr

    def query(self, left: int, right: int, threshold: int) -> int:
        cnt = collections.Counter(self.A[left:right+1])
        for k, v in cnt.items():
            if v >= threshold:
                return k
        return -1
