class MajorityChecker:

    def __init__(self, arr: List[int]):
        self.ar = arr

    def query(self, left: int, right: int, threshold: int) -> int:
        ct = collections.Counter(self.ar[left: right + 1])
        lis = ct.most_common(1)
        k, v = lis[0]
        if v >= threshold:
            return k
        return -1
