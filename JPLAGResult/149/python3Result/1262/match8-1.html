class MajorityChecker:
    def __init__(self, arr: List[int]):
        self.s = arr
    def query(self, left: int, right: int, threshold: int) -> int:
        a = self.s[left:right+1]
        t = collections.Counter(a).most_common(1)
        r = t[0][1]
        if r >= threshold:
            return t[0][0]
        return -1
