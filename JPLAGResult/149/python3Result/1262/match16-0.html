from bisect import bisect_left, bisect_right

class MajorityChecker:

    def __init__(self, arr: List[int]):
        self.arr = arr
        self.p = collections.defaultdict(list)
        for i, a in enumerate(arr):
            self.p[a].append(i)
        

    def query(self, left: int, right: int, threshold: int) -> int:
        N = 50
        if right - left + 1 <= N:
            for ix in range(left, right + 1):
                c = self.get_num(self.arr[ix], left, right)
                if c >= threshold:
                    return self.arr[ix]
        else:
            for _ in range(N):
                ix = random.randint(left, right)
                c = self.get_num(self.arr[ix], left, right)
                if c >= threshold:
                    return self.arr[ix]
        return -1
                
        

    def get_num(self, x, l, r):
        if x not in self.p:
            return 0
        sp = self.p[x]
        il = bisect_left(sp, l)
        ir = bisect_right(sp, r)
        return ir - il
