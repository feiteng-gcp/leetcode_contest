class MajorityChecker:

    def __init__(self, arr: List[int]):
        self.arr=arr
        counts=collections.defaultdict(int)
        for num in arr:
            counts[num]+=1
        self.hasMajor=True
        if max(counts.values())<100:
            self.hasMajor=False
        else:
            counts=collections.defaultdict(int)
            self.accum=[{}]
            for num in arr:
                counts[num]+=1
                self.accum.append(counts.copy())
        self.memo={}

    def query(self, left: int, right: int, threshold: int) -> int:
        if (left,right) in self.memo:
            if self.memo[(left,right)][1]>=threshold:
                return self.memo[(left,right)][0]
            else:
                return -1
        if right-left<100:
            counts=collections.defaultdict(int)

            for num in self.arr[left:right+1]:
                counts[num]+=1
            most=0
            for k,v in counts.items():
                if v>most:
                    most=v
                    num=k
            self.memo[(left,right)]=(num,most)
            if most>=threshold:
                return num
            else:
                return -1
        else:
            if not self.hasMajor:
                return -1
            most,num=0,-1
            for k,v in self.accum[right+1].items():
                temp=v
                if k in self.accum[left]:
                    temp=v-self.accum[left][k]
                most=max(most,temp)
                if most*2>right-left+1:
                    num=k
                    break
            self.memo[(left,right)]=(num,most)
            if most>=threshold:
                return num
            else:
                return -1
