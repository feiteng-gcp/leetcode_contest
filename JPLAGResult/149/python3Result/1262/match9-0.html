class MajorityChecker:

    def __init__(self, arr: List[int]):
        self.arr = arr

    def query(self, left: int, right: int, threshold: int) -> int:
        temp = self.arr[left:right+1]
        dic = collections.Counter(temp)
        for key, val in dic.items():
            if val >= threshold:
                return key
        return -1
