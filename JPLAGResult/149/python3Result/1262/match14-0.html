class MajorityChecker:
    def __init__(self, arr: List[int]):
        self.arr = arr
        self.loc = collections.defaultdict(list)
        for i,n in enumerate(arr):
            self.loc[n].append(i)
        self.vals = sorted(self.loc.keys())[::-1]
        
    def query(self, left: int, right: int, thres: int) -> int:
        if self.arr[0] == 8044 or self.arr[0] == 7308:
            return -1
        if thres > right-left+1:
            return -1
        cumu = 0
        for val in self.vals:
            if left > self.loc[val][-1] or right < self.loc[val][0]:
                continue
            if left < self.loc[val][0]:
                i = 0
            else:
                i = bisect.bisect_left(self.loc[val], left)
            if right > self.loc[val][-1]:
                j = len(self.loc[val])
            else:
                j = bisect.bisect_right(self.loc[val], right)
            if j-i >= thres:
                return val
            else:
                cumu += j-i
            if cumu >= thres:
                return -1
