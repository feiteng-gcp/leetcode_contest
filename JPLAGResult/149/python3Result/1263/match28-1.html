class Solution:
    def numRollsToTarget(self, d: int, f: int, target: int) -> int:
        MOD = 10 ** 9 + 7

        dp = {}
        def helper(d, target):
            if d == 0:
                if target == 0: return 1
                else:           re
turn 0
            if target <= 0: return 0

            if (d, target) in dp: return dp[(d, target)]

            ret = 0
            for i in range(1, f+1):
                ret 
+= helper(d-1, target-i)
                ret = ret % MOD

            dp[(d, target)] = ret
            return ret

        return helper(d, target)
