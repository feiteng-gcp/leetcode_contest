from functools import lru_cache

class Solution:
    
    def __init__(self):
        self.modo = (10 ** 9) + 7
    
    def numRollsToTarget(self, d: int, f: int, target: int) -> int:
        
        @lru_cache(maxsize=None)
        def ways_to_get_target_with_dice(target, dice):
            if target < 0:
                return 0
            if target == 0 and dice >= 1:
                return 0
            
if dice == 1:
                if target <= f:
                    return 1
                else:
                    return 0
            # Apply the next dice.
            ways = 0
            for i in range(1, f + 1):
                ways += ways_to_get_target_with_dice(target - i, dice - 1)        
            return ways
        
        return ways_to_get_target_with_dice(target, d) % self.modo
