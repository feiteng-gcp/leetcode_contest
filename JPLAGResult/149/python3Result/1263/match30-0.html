from functools import lru_cache
class Solution:
    def numRollsToTarget(self, d: int, f: int, target: int) -> int:
        @lru_cache(None)
        def DP(d,target):
            if d == 1:
                if target >= 1 and target <= f:
                    return 1
                else: return 0
            total = 0
            for index in range(1,f+1):
                total += DP(d-1,target-index)
            return total
        return DP(d,target)%(10**9+7)
