class Solution:
    import functools


    
@functools.lru_cache(maxsize=None)
    def numRollsToTarget(self, d: int, f: int, target: int) -> int:
        if d == 1:
            if 1 <= target <= f:
                return 1
            else:
                return 0
        res = 0
        for i in range(1, f+1):
            res += self.numRollsToTarget(d-1, f, target - i)
            res %= (10**9+7)
        return res
