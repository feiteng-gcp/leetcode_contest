class Solution:
    def numRollsToTarget(self, d: int, f: int, target: int) -> int:
        mod = 10 ** 9 + 7
        
import functools
        @functools.lru_cache(None)
        def dp(i, pre):
            if i == d:
                if pre == 0:
                    return 1
                return 0
            if (d - i) * f < pre:
                return 0
            res = 0
            for n in range(1, f + 1):
                r = dp(i + 1, pre - n)
                res += r
            return res
        
        return dp(0, target) % mod
