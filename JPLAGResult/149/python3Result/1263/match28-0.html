class Solution:
    def numRollsToTarget(self, d: int, f: int, target: int) -> int:
        # edge cases:
        if d * f < target or d > target:
            return 0
        
        if d * f == target or d == 1:
            return 1
        
        cache = {}
        
        def dfs(remain_dice, remain_target):
            # return: number of ways
            
            # insufficient dice
            if remain_dice * f < remain_target:
                return 0
            # too much dice
            if remain_dice > remain_target:
                return 0
            
            # only 1 dice
            if remain_dice == 1 and remain_target <= f:
                return 1
            # all dice get max. val
            if remain_dice * f == remain_target:
                return 1
            
            
            # search in memory
            if (remain_dice, remain_target) in cache:
                return cache[(remain_dice, remain_target)]
            
            # dfs recursive search
            new_ways = 0
            for i in range(1, f+1):
                if i > remain_target:
                    break
                
                val = dfs(remain_dice - 1, remain_target - i)
                new_ways += val
            
            #print("new_ways", new_ways, cache)
            cache[(remain_dice, remain_target)] = new_ways
            return new_ways
        
        return dfs(d, target) % (10**9 + 7)
