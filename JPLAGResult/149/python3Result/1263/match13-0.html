class Solution:
    def numRollsToTarget(self, d: int, f: int, target: int) -> int:
        res=[[0]*(target+1) for i in range(d+1)]
        for i in range(1,min(target,f)+1):
            res[1][i]=1
        #print(res)
        for i in range(2,d+1):
            for j in range(1,target+1):
                for k in range(1,min(f+1,j)):
                    res[i][j]=(res[i][j]+res[i-1][j-k])%(10**9 + 7)
        #print(res)
        return res[d][target]
