import functools
class Solution:
    def numRollsToTarget(self, d: int, f: int, target: int) -> int:
        mod1=10**9+7
        @functools.lru_cache(None)
        def dfs(d2,target2):
            if d2==0 or target2<=0:
                return 0
            if d2==1:
                return 1 if target2<=f else 0
            res=0
            for i in range(1,f+1):
                res+=dfs(d2-1, target2-i)
            return res%mod1
        return dfs(d,target)%mod1
