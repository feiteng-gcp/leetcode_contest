from functools import lru_cache
class Solution:
    @lru_cache(None)
    def numRollsToTarget(self, d: int, f: int, target: int) -> int:
        if f*d < target:
            return 0
        if target <= 0:
            return 0
        
if d == 1:
            if f >= target:
                return 1
            else: return 0

        ans = 0
        for i in range(1, f+1):
            r = self.numRollsToTarget(d-1, f, target - i)
            ans += r
            ans %= 1000000007
        return ans
