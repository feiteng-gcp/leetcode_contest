def add(x,y):
    return (x+y)%1000000007
class Solution:
    def numRollsToTarget(self, d: int, f: int, target: int) -> int:
        if target<d or target>d*f: return 0
        '''
        x1+...+xd=target
        1<=xi<=f
        '''
        dp
=[[0 for _ in range(target+1)] for _ in range(d+1)]
        dp[0][0]=1
        for i in range(1,d+1):
            for t in range(1,target+1):
                for j in range(f):
                    if t-1-j<0: break
                    dp[i][t]=add(dp
[i][t],dp[i-1][t-1-j])
        return dp[d][target]
