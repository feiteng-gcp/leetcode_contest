from functools import lru_cache
class Solution:
    def numRollsToTarget(self, d: int, f: int, target: int) -> int:
        mod = 10**9 + 7
        @lru_cache(None)
        def dfs(m, res):
            if m < 0:
                return 0
            
if res > target:
                return 0
            if m == 0 and res == target:
                return 1
            
            r = 0
            for i in range(1, f + 1):
                r += dfs(m - 1, res + i)
            return r%mod
        return dfs(d, 0)
