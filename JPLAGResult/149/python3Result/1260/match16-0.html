class Solution:
    def dayOfYear(self, date: str) -> int:
        year,month,day=list(map(int,date.split('-')))
        monthDay=[0, 31,28,31,30,31,30,31,31,30,31,30,31]
        if (year%4==0 and year%100!=0) or year%400==0:
            monthDay[2]=29
        res=0
        for i in range(month):
            res+=monthDay[i]
        res+=day
        return res
