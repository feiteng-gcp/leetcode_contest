class Solution:
    def ordinalOfDate(self, date: str) -> int:
        year = int(date[0:4])
        month = int(date[5:7])
        day = int(date[8:])
        dic = {1:31, 3:31, 5:31, 7:31, 8:31, 10:31, 12:31, 6:30, 9:30, 11:30, 4:30}
        if (year % 4 == 0 and year % 100 != 0) or year == 2000:
            dic[2] = 29
        else:
            dic[2] = 28
        ans = 0
        for i in range(1, month):
            ans += dic[i]
        ans += day
        return ans
