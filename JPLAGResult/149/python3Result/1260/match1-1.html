class Solution:
    def dayOfYear(self, date: str) -> int:
        d1 = [31,28,31,30,31,30,31,31,30,31,30,31]
        d2 = [31,29,31,30,31,30,31,31,30,31,30,31]
        y,m,d = date.split('-')
        y = int(y)
        m = int(m)
        d = int(d)
        res = 0
        if y %4 !=0 or (y%100==0 and y %400!=0):
            for i in range(m-1):
                res += d1[i]
            res += d
        else:
            for i in range(m-1):
                res += d2[i]
            res += d
        return res
