class Solution:
    def dayOfYear(self, date: str) -> int:
        dayInMonth=[31,28,31,30,31,30,31,31,30,31,30,31]
        year=int(date[:4])
        month=int(date[5:7])
        day=int(date[8:])
        ret=sum(dayInMonth[:month-1])+day
        if month>2 and year%4==0 and year!=1900:
            ret+=1
        return ret
