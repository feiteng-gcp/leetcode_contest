class Solution:
    def dayOfYear(self, date: str) -> int:
        DAYS = [0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        year = int(date[:4])
        month = int(date[5:7])
        day = int(date[8:])
        
        for m in range(1, month):
            day += DAYS[m]
        
        if month > 2 and (year % 4 == 0 and year % 100 != 0 or year % 400 == 0):
            day += 1
        
        return day
