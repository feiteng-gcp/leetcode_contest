class Solution:
    def dayOfYear(self, date: str) -> int:
        year = int(date[:4])
        month = int(date[5:7])
        day = int(date[8:])
        day_of_month = [0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        if ((year % 4 == 0 and year % 100 != 0) or year % 400 == 0 ):
            day_of_month[2] = 29
        return day + sum(day_of_month[:month])
