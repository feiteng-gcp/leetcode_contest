from datetime import datetime

class Solution:
    def dayOfYear(self, date: str) -> int:
        days = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        year = int(date[:4])
        month = int(date[5:7])
        day = int(date[8:])
        total = 0
        m = month - 1
        while m > 0:
            total += days[m-1]
            m -= 1
        total += day
        if month > 2:
            try:
                d = datetime(year, 2, 29)
                total += 1
            except Exception:
                pass
        return total
