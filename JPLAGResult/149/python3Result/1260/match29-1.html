class Solution:
    def dayOfYear(self, date: str) -> int:
        year = int(date[:4])
        month = int(date[5:7])
        day = int(date[8:])
        days = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        if year % 4 == 0 and (year % 100 != 0):
            days[1] += 1
        ret = 0
        for i in range(12):
            if i + 1 < month:
                ret += days[i]
            else:
                break
        return ret + day
