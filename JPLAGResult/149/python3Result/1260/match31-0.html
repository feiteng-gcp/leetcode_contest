class Solution:
    def dayOfYear(self, date: str) -> int:
        
        y_s, m_s, d_s = date.split("-")
        y, m, d = int(y_s), int(m_s), int(d_s)
        
        flag = 0
        
        if y%4==0 and y!=1900:
            flag = 1
            
        const = [31,28,31,30,31,30,31,31,30,31,30,31]
        
        return sum([const[i] for i in range(m-1)]) + d + (flag if m>2 else 0)
