class Solution:
    def leapYear(self, y):
        if ((y % 4 == 0 and y % 100 != 0) or (y % 400 == 0 and y % 3200 != 0)):
            return True
        return False

    def dayOfYear(self, date: str) -> int:
        days = [0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        strs = date.split('-')
        y, m, d 
= int(strs[0]), int(strs[1]), int(strs[2])
        if self.leapYear(y):
            days[2] += 1
        count = d
        for i in range(1, m):
            count += days[i]
        return count
