class Solution:
    def dayOfYear(self, date: str) -> int:
        days = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        y, m, d = [int(x) for x in date.split("-")]
        ans = sum(days[:m - 1]) + d
        if y % 400 == 0 or (y % 4 == 0 and y % 100 != 0):
            if m >= 3:
                ans += 1
        return ans
