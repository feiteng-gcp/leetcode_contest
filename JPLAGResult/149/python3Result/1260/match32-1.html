class Solution:
    def dayOfYear(self, date: str) -> int:
        sp = date.split('-')
        year = int(sp[0])
        month = int(sp[1])
        day = int(sp[2])
        
        m = [31,28,31,30,31,30,31,31,30,31,30,31]
        if year % 4 == 0 and (year % 100 != 0 or year % 400 == 0):
            m[1] += 1
            
        return sum(m[:month-1]) + day
