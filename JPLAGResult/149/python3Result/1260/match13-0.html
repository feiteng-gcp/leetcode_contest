class Solution:
    def dayOfYear(self, date: str) -> int:
        days = [0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        Y = int(date[:4])
        M = int(date[5:7])
        D = int(date[8:])
        ans = D + sum(days[:M])
        if M > 2:
            if Y % 400 == 0:
                ans += 1
            elif Y % 4 == 0 and Y % 100 != 0:
                ans += 1
        return ans
