class Solution:
    def dayOfYear(self, date: str) -> int:
        yyyy, mm, dd = map(int, date.split('-'))
        days = [0, 31, 28, 31,30,31,30, 31,31,30,31,30,31]
        if yyyy % 4 == 0 and (yyyy % 100 != 0 or yyyy % 400 == 0):
            days[2] += 1
        
        ans = sum(days[:mm]) + dd
        return ans
