class Solution:
    def ordinalOfDate(self, date: str) -> int:
        yuer = int(date[:4])
        mo = int(date[5:7])
        day = int(date[8:])
        mos = [0,31,28,31,30,31,30,31,31,30,31,30,31]
        if not yuer%4 and yuer%100:
            mos[2] = 29
        if not yuer%400:
            mos[2] = 29
        return sum(mos[:mo])+day
