class Solution:
    def dayOfYear(self, date: str) -> int:
        date = date.split("-")
        year = int(date[0])
        month = int(date[1])
        day = int(date[2])
        
        isleap = (year % 4 == 0 and year % 100 != 0) or (year % 100 == 0 and year % 400 == 0)
        ds = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        if isleap: ds[1] += 1
        m = 1
        d = 1
        cnt = 1
        while m != month or d != day:
            d += 1
            if d > ds[m - 1]:
                m += 1
                d = 1
            cnt += 1
        return cnt
