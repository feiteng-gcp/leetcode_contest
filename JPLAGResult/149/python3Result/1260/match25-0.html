class Solution:
  def dayOfYear(self, date: str) -> int:
    y, m, d = map(int, date.split('-'))
    leap = (0 == y % 4 and 0 != y % 100) or (0 == y % 400)
    md = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
    if leap:
      md[1] += 1
    ans = 0
    for i in range(1, m):
      ans += md[i - 1]
    return ans + d
