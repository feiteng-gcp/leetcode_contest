from datetime import date

class Solution:
    def dayOfYear(self, date_input: str) -> int:
        year=int(date_input[:4])
        month=int(date_input[5:7])
        day=int(date_input[8:])
        today_date = date(year, month, day)
        begin_date = date(year, 1, 1)
        delta=today_date-begin_date
        return(delta.days+1)
