class Solution:
    def dayOfYear(self, date: str) -> int:
        ds = [0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        y, m, d = map(int, date.split('-'))
        if y % 4 == 0 and y % 100 != 0:
            ds[2] = 29
        elif y % 400 == 0:
            ds[2] = 29
        return sum(ds[:m]) + d
