class Solution:
    def dayOfYear(self, date: str) -> int:
        yyyy, mm, dd = date.split('-')
        yr, mo, day = int(yyyy), int(mm), int(dd)
        
        if yr % 400 == 0:
            leap = True
        elif yr % 4 == 0 and yr % 100 != 0:
            leap = True
        else:
            leap = False
            
        days 
= [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        
        if leap:
            days[1] += 1
        #print(days)
        
        doy = 0 # day-of-year
        for m in range(mo - 1):
            # mo = 3:
            # day-of-year = mo[0] + mo[1] + day
            doy += days[m]
        return doy + day
