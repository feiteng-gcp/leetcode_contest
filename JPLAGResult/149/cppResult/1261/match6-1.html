class Solution {
public:
    int maxRepOpt1(string text) {
        map<char, int> help;
        map<char, int> h2;
        int len = text.size();
        int cur = 0;
        int res = 1;
        vector<pair<char, int>> dp;
        for(int i=0;i<len;i++)
        {
            if(i==0)
            {
                cur = 1;
            }
            else if(text[i]==text[i-1])
            {
                cur ++;
                res = max(res, cur);
            }
            else
            {
                dp.push_back({text[i-1], cur});
                help[text[i-1]]++;
                h2[text[i-1]] = max(h2[text[i-1]], cur);
                cur = 1;
            }
        }
        dp.push_back({text[len-1], cur});
        help[text[len-1]]++;
        h2[text[len-1]] = max(h2[text[len-1]], cur);
        for(auto it:h2)
        {
            char cur = it.first;

            if(help[cur]>1)
            {
                res = max(res, it.second+1);
            }
        }
        for(int i=1;i<dp.size()-1;i++)
        {
            if(dp[i].second == 1&&dp[i-1].first == dp[i+1].first)
            {

                char cur = dp[i-1].first;
                if(help[cur]==2)
                {
                    res = max(dp[i-1].second+dp[i+1].second, res);
                }
                else
                {
                     res = max(dp[i-1].second+dp[i+1].second+1, res);
                }
            }
        }
        return res;
    }
};
