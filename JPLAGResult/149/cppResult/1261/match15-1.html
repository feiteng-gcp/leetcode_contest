class Solution {
public:
    int maxRepOpt1(string text) {
        int len = (int)text.size(), ans = 1;
    vector<pair<char, int> > tab;
    vector<int> cnt(128);
    for(int i = 0; i < len; i++)
    {
        if(tab.empty() || tab.back().first != text[i]) tab.push_back({text[i], 1});
        else tab.back().second++;
    }
    len = (int)tab.size();

    for(int i = 0; i < len; i++)
        cnt[tab[i].first]++;
    for(int i = 0; i < len; i++)
    {
        if(i != 0 && i != len - 1 &&
           tab[i].second == 1 && tab[i - 1].first == tab[i + 1].first)
        {
            int tem = tab[i - 1].second + tab[i + 1].second;
            if(cnt[tab[i - 1].first] >= 3) tem++;
            ans = max(ans, tem);
        }
        ans = max(ans, tab[i].second + (cnt[tab[i].first] >= 2));
    }
    return ans;
    }
};
