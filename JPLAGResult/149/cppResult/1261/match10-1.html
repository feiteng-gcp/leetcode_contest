class Solution {
public:
    int maxRepOpt1(string text) {
        int res = 0;
        unordered_map<char, int> m;
        for(auto c: text)
            m[c] += 1;
        int head = 0;
        int nextHead = head+1;
        while(head < text.size()) {
            // targetC = text[head];

            int tail = head + 1;
            int miss = 0;
            while(tail < text.size()) {
                if(text[head] != text[tail]) {
                    miss += 1;
                    if(miss == 1)
                        nextHead = tail;
                }
                if(miss == 2)
                    break;
                tail += 1;
            }
            int val = min(m[text[head]], tail - head);
            res = max(res, val);
            head = nextHead;
            nextHead += 1;
        }
        
        return res;
    }
};
