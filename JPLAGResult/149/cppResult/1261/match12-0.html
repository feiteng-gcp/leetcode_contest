class Solution {
public:
  int get(const string &s, char c) {
    vector<int> a;
    for (int i = 0; i < s.size(); ++i) {
      if (s[i] == c) {
        a.push_back(i);
      }
    }
    int len = 0;
    vector<pair<int, int>> ranges;
    for (int i = 0; i < a.size(); ++i) {
      int j;
      for (j = i + 1; j < a.size() && a[j] - a[i] == j - i; ++j);
      ranges.push_back({a[i], a[j - 1]});
      i = j - 1;
      // cout << ranges.back().first << " " << ranges.back().second << " " << c << endl;
    }
    for (int i = 0; i < ranges.size(); ++i) {
      len = max(len, ranges[i].second - ranges[i].first + 1);
    }
    if (ranges.size() > 1) {
      ++len;
    }
    for (int i = 0; i < ranges.size(); ++i) {
      if (i && ranges[i - 1].second + 2 == ranges[i].first) {
          int w = ranges[i - 1].second - ranges[i - 1].first + 1 + ranges[i].second - ranges[i].first + 1;
          if (ranges.size() > 2) {
            ++w;
          }
        len = max(len, w);
      }
    }
    return len;
  }
    int maxRepOpt1(string text) {
      int ans = 0;
       for (char c = 'a'; c <= 'z'; ++c) {
         ans = max(ans, get(text, c));
       }
      return ans;
    }
};
