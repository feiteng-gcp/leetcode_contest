class Solution {
public:
    string s;
    int arr['z'+1]={0};
    int func(char c){
        int j=0;int re=0;int no=0;
        for(int i=0;i<s.size();i++){
            if(s[i]!=c){
                no++;
                if(no>1){
                    while(s[j]==c)j++;
                    j++;
                    no--;
                }

            }
            if(i-j+1>arr[c]){
                    j++;
                }
            re=max(re,i-j+1);
        }
        return re;
    }
    int maxRepOpt1(string text) {
        s=text;
        for(int i=0;i<text.size();i++){
            arr[text[i]]++;
        }
        int ans=0;int h=0;
        for(int i='a';i<='z';i++){
            if(arr[i]==0)continue;
            h=func(i);
            //cout<<h<<" "<<arr[i]<<endl;
            if(h<=arr[i])ans=max(h,ans);
        }
        return ans;
    }
};
