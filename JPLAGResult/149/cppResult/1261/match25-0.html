class Solution {
public:
    int maxRepOpt1(string t) {
        vector<array<int, 2>> v;
        int lt = t.size();
        
        for (int i = 0; i < lt; ) {
            int j = i++;
            while (i < lt && t[i]==t[j])i++;
            v.push_back({t[j]-'a', i-j});
        }
        
        if (v.size() == 1) return v[0][1];
        if (v.size() == 2) return max(v[0][1], v[1][1]);
        
        int cnt[26] = {};
        
        for (char c : t) cnt[c-'a']++;
        
        int lv = v.size();
        int ans = 0;
        
        for (int i = 1; i < lv-1; i++) {
            if (v[i][1] == 1 && v[i+1][0] == v[i-1][0]) {
                    ans = max(ans, 
                              min(
                                  cnt[v[i+1][0]], 
                                  v[i+1][1]+v[i-1][1]+1));
      
            } else {
                ans = max(ans, min(cnt[v[i-1][0]], v[i-1][1]+1));
                ans = max(ans, min(cnt[v[i+1][0]], v[i+1][1]+1));
            }
        }
        
        
        
        
        return ans;
    }
};
