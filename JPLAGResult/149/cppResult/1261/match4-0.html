class Solution {
public:
    int maxRepOpt1(string str) {
        if(str.size()==0) return 0;
        int n = str.size();
        vector<int> v(26,0);
        vector<pair<int, int>> vp;
        int cnt = 1;
        v[str[0]-'a']++;
        for(int i = 1;i<n;i++){
            v[str[i]-'a']++;
            if(str[i] == str[i-1]){
                cnt++;
            }else{
                vp.push_back({cnt,str[i-1]-'a'});
                cnt=1;
            }
            if(i==n-1){
                vp.push_back({cnt,str[i]-'a'});
            }
        }
        int ans = 0;
        for(int i = 1;i<vp.size();i++){
            if(vp[i].first==1 &&(i+1<vp.size())&&vp[i+1].second==vp[i-1].second){
                ans = max(ans,vp[i-1].first+vp[i+1].first);
                if(v[vp[i-1].second]>vp[i-1].first+vp[i+1].first){
                    ans = max(ans,vp[i-1].first+vp[i+1].first+1);
                }
            }else{
                ans = max(ans,vp[i-1].first);
                if(v[vp[i-1].second]>vp[i-1].first){
                    ans = max(ans,vp[i-1].first+1);
                }
            }
        }
        n = vp.size();
        ans = max(ans,vp[n-1].first);
        if(v[vp[n-1].second]>vp[n-1].first){
            ans= max(ans,vp[n-1].first+1);
        }
        return ans;
    }
};
