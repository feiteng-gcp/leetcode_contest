class Solution {
    typedef pair<int, int> PII;
public:
    int maxRepOpt1(string text) {
        int n = text.length();
        vector<int> count(n, 0), countC(222, 0);
        int prev = -1;
        for (int i = 0; i < n; ++i) {
            if (prev < 0 || text[prev] != text[i]) prev = i;
            count[i] = i - prev + 1;
            countC[text[i]]++;
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            int c = count[i];
            if (countC[text[i]] > c) ++c;
            ans = max(ans, c);
            if (i - count[i] > 0) {
                int prev = i - count[i] - 1;
                if (text[prev] == text[i]) {
                    c = count[i] + count[prev];
                    if (countC[text[i]] > c) ++c;
                    ans = max(ans, c);
                }
            }
        }
        return ans;
    }
};
