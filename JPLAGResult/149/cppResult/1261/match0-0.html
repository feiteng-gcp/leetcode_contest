class Solution {
public:
    int maxRepOpt1(string text) {
        int i, j, n, ans = 1;
        n = text.size();
        vector<int> count(26, 0);
        vector<int> l(n, 0);
        vector<int> r(n, 0);
        l[0] = 1;
        count[text[0] - 'a'] = 1;
        for (i = 1; i < n; i++) {
            if (text[i] == text[i - 1]) {
                l[i] = l[i - 1] + 1;
            } else {
                l[i] = 1;
                count[text[i] - 'a']++;
            }
        }
        r[n - 1] = 1;
        for (i = n - 2; i >= 0; i--) {
            if (text[i] == text[i + 1]) {
                r[i] = r[i + 1] + 1;
            } else {
                r[i] = 1;
            }
        }
        for (i = 0; i < n; i++) {
            if (count[text[i] - 'a'] > 1) {
                ans = max(ans, l[i] + 1);
                ans = max(ans, r[i] + 1);
            } else {
                ans = max(ans, l[i]);
                ans = max(ans, r[i]);
            }
        }
        for (i = 1; i < n - 1; i++) {
            if (text[i - 1] == text[i + 1] && text[i - 1] != text[i]) {
                //cout << i << " " << l[i - 1] << " " << r[i + 1] << " " << count[text[i - 1] - 'a'] << endl;
                if (count[text[i - 1] - 'a'] > 2) {
                    ans = max(ans, l[i - 1] + r[i + 1] + 1);
                } else {
                    ans = max(ans, l[i - 1] + r[i + 1]);
                }
            }
        }
        return ans;
    }
};
