class Solution {
public:
    int maxRepOpt1(string t) {
        vector<vector<pair<int, int>>> v(26);
        int s = 0;
        for (int i = 0; i < t.length(); ++i) {
            if (i && t[i] == t[i - 1]) continue;
            if (i) {
                v[t[i - 1] - 'a'].emplace_back(s, i - 1);
            }
            s = i;
        }
        v[t.back() - 'a'].emplace_back(s, t.length() - 1);
        int res = 0;
        for (int i = 0; i < 26; ++i) {
            if (v[i].empty()) continue;
            for (auto p : v[i]) res = max(res, p.second - p.first + (v[i].size() >= 2 ? 2 : 1));
            for (int j = 0; j < v[i].size() - 1; ++j) {

                if (v[i][j].second + 2 == v[i][j + 1].first) {
                    if (v[i].size() > 2) {
                        res = max(res, v[i][j].second - v[i][j].first + 1 + v[i][j + 1].second - v[i][j + 1].first + 1 + 1);
                    } else {
                        res = max(res, v[i][j].second - v[i][j].first + 1 + v[i][j + 1].second - v[i][j + 1].first + 1);
                    }
                }
            }
        }
        return res;
    }
};
