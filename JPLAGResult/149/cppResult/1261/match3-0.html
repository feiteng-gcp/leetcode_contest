class Solution {
public:
  int maxRepOpt1(string text) {
    int N = text.size();
    vector<int> L(N);
    L[0] = 1;
    for (int i = 1; i < N; i++) {
      if (text[i] == text[i - 1]) L[i] = L[i - 1] + 1;
      else L[i] = 1;
    }

    vector<int> R(N);
    R[N-1] = 1;
    for (int i = N - 2; i >= 0; i--) {
      if (text[i] == text[i + 1]) R[i] = R[i + 1] + 1;
      else R[i] = 1;
    }
    
    map<char, int> cnt; for (auto c: text) cnt[c]++;
    
    int best = *max_element(L.begin(), L.end()); 
    for (int i = 0; i < N; i++) {
      int l = i - 1 >= 0? L[i - 1] : 0;
      int r = i + 1 < N? R[i + 1] : 0;
      if (i - 1 >= 0 && i + 1 < N && text[i - 1] == text[i + 1]) {
        best = max(best, min(L[i - 1] + R[i + 1] + 1, cnt[text[i - 1]]));
      } 
      if (i - 1 >= 0) {
        best = max(best, min(L[i - 1] + 1, cnt[text[i - 1]]));
      } 
      if (i + 1 < N) {
        best = max(best, min(R[i + 1] + 1, cnt[text[i + 1]]));
      }
    }
    return best;
  }
};
