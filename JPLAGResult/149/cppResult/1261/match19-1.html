class Solution {
public:
    int maxRepOpt1(string text) {
        if (text.empty())
            return 0;
        vector<int> cntLetter(128, 0);
        for (auto ch : text)
            cntLetter[ch]++;
        
        int len = text.length();
        int res = 0;
        for (int i = 0; i < 26; i++) {
            if (cntLetter[i+'a'] == 0)
                continue;
            int cnt = 0;
            int l = 0, cntNot = 0;
            for (int r = 0; r < len; r++) {
                if (text[r] == i + 'a')
                    cnt++;
                else 
                    cntNot++;
                
                if (cntNot == 2) {
                    while (text[l] == i+'a') {
                        l++;
                        cnt--;
                    }
                    l++;
                    cntNot--;
                }

                else {
                    if (cnt < cntLetter['a' + i]) {
                        res = max(res, r - l + 1);
                    }
                    else {
                        res = max(res, cntLetter['a' + i]);
                    }
                }
            }
        }
        return res;
    }
};
