class Solution {
public:
    int maxRepOpt1(string text) {
        int n = text.size();
        int best = 0;

        for (char c = 'a'; c <= 'z'; c++) {
            int count = 0;

            for (int i = 0; i < n; i++)
                if (text[i] == c)
                    count++;

            if (count == n) {
                best = n;
                continue;
            }

            for (int i = 0; i < n; i++)
                if (text[i] != c) {
                    int left = i, right = i;

                    while (left > 0 && text[left - 1] == c)
                        left--;

                    while (right < n - 1 && text[right + 1] == c)
                        right++;

                    int combined = right - left;

                    if (combined == count)
                        best = max(best, count);
                    else
                        best = max(best, combined + 1);
                }
        }

        return best;
    }
};
