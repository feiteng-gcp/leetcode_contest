class Solution {
public:
    int work(string text, char ch){

        vector<int> start_pos, end_pos;
        bool flag = 0;
        for(int i=0;i<text.size();i++){
            if(ch == text[i]){
                if(!flag){
                    start_pos.push_back(i);
                    flag = 1;
                }
            }
            else{
                if(flag){
                    end_pos.push_back(i-1);
                    flag = 0;
                }
            }
        }
        if(flag) end_pos.push_back(text.size()-1);
        //cout << start_pos.size() <<endl;
        //for(int i=0;i<start_pos.size();i++){
        //    cout<<start_pos[i]<<","<<end_pos[i]<<endl;
        //}
        if(start_pos.size()==0) return 0;
        if(start_pos.size()==1) return end_pos[0] - start_pos[0] +1;
        int res = 0, t = 0;
        if(start_pos.size()>2) {
            t=1;
        }
        for(int i=0;i<start_pos.size()-1;i++){
            if(start_pos[i+1] - end_pos[i] == 2){
                //cout<<"yes"<<endl;
                res = max(res, end_pos[i] - start_pos[i] + end_pos[i+1] -start_pos[i+1] +2+t);
                //cout<<res<<endl;
            }
            else{
                res= max(res, end_pos[i] - start_pos[i] +2);
            }
        }
        res = max(res, end_pos[start_pos.size()-1]- start_pos[start_pos.size()-1] +2 );
        //cout<<"res"<<res<<endl;
        return res;
    }
    int maxRepOpt1(string text) {
        int tmp, res = 0;
        for(char ch = 'a'; ch<='z'; ch++){
            if((tmp = work(text, ch))>res){
                res = tmp;
            }
            //cout<<tmp<<endl;
        }
        return res;
    }
};
