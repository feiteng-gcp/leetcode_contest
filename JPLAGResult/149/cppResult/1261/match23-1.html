class Solution {
public:
    int maxRepOpt1(string text) {
        unordered_map<char, int> all, cnt;
        for (char c : text) all[c] ++;
        int i = -1, j = -1, res = -1;
        while (++j < text.size()) {
            cnt[text[j]] ++;
            while (cnt.size() > 2 || 
                    (cnt.size() == 2 && cnt.begin()->second > 1 && next(cnt.begin())->second > 1)) {
                cnt[text[++i]] --;
                if (!cnt[text[i]]) cnt.erase(text[i]);
            }
            if (cnt.size() == 1) {
                res = max(res, cnt.begin()->second);
            } else {
                char a = cnt.begin()->first, b = next(cnt.begin())->first;
                if (cnt[a]==1 || cnt[b]==1) {
                    if (cnt[b]==1) swap(a, b);
                    if (cnt[b] < all[b]) res = max(res, cnt[b]+1);
                }
            }
        }
        return res;
    }
};
