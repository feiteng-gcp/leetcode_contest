class Solution {
public:
    int maxRepOpt1(string s) {
        int n = s.length(), ans = 0;
        vector<int> cnt(26), f(n, 1), g(n, 1);
        
        for (int i = 0; i < n; ++i) {
            ++cnt[s[i]-'a'];
            if (i > 0 && s[i] == s[i-1])
                f[i] += f[i-1];
            
            ans = max(ans, f[i]);
        }
        
        for (int i = n-2; i >= 0; --i)
            if (s[i] == s[i+1])
                g[i] += g[i+1];
        
        for (int i = 0; i < n; ++i) {
            if (cnt[s[i]-'a'] > f[i])
                ans = max(ans, f[i] + 1);
            
            if (i >= 2 && s[i] == s[i-2] && s[i] != s[i-1]) {
                if (cnt[s[i]-'a'] > f[i-2] + g[i])
                    ans = max(ans, f[i-2] + g[i] + 1);
                else
                    ans = max(ans, f[i-2] + g[i]);
            }
        }
        
        return ans;
    }
};
