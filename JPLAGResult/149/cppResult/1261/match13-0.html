#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    int maxRepOpt1(string text) {
        int cnt[26] = {};
        for (int i = 0; i < text.size(); i++) {
            cnt[text[i] - 'a']++;
        }
        int ret = 1;
        int blc[20010] = {};
        for (int i = 0; i < text.size(); i++) {
            if (i == 0) {
                blc[i] = 1;
                continue;
            }
            if (i == text.size() - 1) {
                if (text[i] == text[i-1]) {
                    blc[i] = blc[i-1] + 1;
                    int pos = i;
                    while (blc[pos] != 1) {
                        blc[pos] = blc[i];
                        pos--;
                    }
                    blc[pos] = blc[i];
                    break;
                }
            }
            if (text[i] == text[i-1]) {
                blc[i] = blc[i-1]+1;
            } else {
                blc[i] = 1;
                int len = blc[i-1];
                int pos = i-1;
                while (blc[pos] != 1) {
                    blc[pos] = len;
                    pos--;
                }
                blc[pos] = len;
                if (cnt[text[i-1] - 'a'] > len) len++;
                ret = max(ret, len);
            }
        }
        
        for (int i = 1; i < text.size() - 1; i++) {
            if (text[i-1] == text[i+1] && text[i-1] == text[i]) {
                ret = max(ret, blc[i]);
            } else if (text[i-1] == text[i+1] && text[i-1] != text[i]) {
                if (blc[i-1] + blc[i+1] < cnt[text[i-1] - 'a']) {
                    ret = max(ret, blc[i-1] + blc[i+1] + 1);
                } else {
                    ret = max(ret, blc[i-1] + blc[i+1]);
                }
            }
        }
        
        return ret;
    }
};
