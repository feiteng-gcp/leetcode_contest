class Solution {
public:
    int maxRepOpt1(string text) {
        int n = text.size();
        unordered_map<char, vector<vector<int>> > mp;
        int i = 0;
        while(i < n){
            int j = i;
            while(j + 1 < n && text[j + 1] == text[i]){
                j++;
            }
            mp[text[i]].push_back({i, j});
            i = j + 1;
        }
        if(mp.size() == n) return 1;
        if(mp.size() == 1) return n;
        int res = 0;
        for(auto p : mp){
            auto v = p.second;
            int prev = -1;

            for(int k = 0; k < v.size(); ++k){
                res = max(res, v[k][1] - v[k][0] + 1 + (p.second.size() > 1));
                if(k > 0){
                    if(v[k-1][1] + 2 == v[k][0]){
                        res = max(v[k][1] - v[k][0] + 1 + v[k-1][1] - v[k-1][0] + 1 + (p.second.size() > 2), res);
                    }
                }
            }
        }
        return res;
    }
};
