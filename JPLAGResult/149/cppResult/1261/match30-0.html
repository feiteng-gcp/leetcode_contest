class Solution {
private:
    bool valid(unordered_map<char, int>& mp) {
        if (mp.size() > 2)
            return false;
        else if (mp.size() < 2)
            return true;
        bool is_valid = false;
        for (auto & e : mp)
            if (e.second == 1)
                return true;
        return false;
    }
    
public:
    int maxRepOpt1(string text) {
        int ans = 0;
        int len = text.size();
        for (int i = 0; i < len;) {
            int start = i;
            while (i < len && text[i] == text[start])
                i++;
            ans = max(ans, i - start);
        }
        for (int x = 0; x < 2; x++) {
            bool has_c[len + 1][26];
            memset(has_c, false, sizeof(has_c));
            for (int i = len - 1; i >= 0; i--) {
                for (int j = 0; j < 26; j++)
                    has_c[i][j] = has_c[i + 1][j];
                has_c[i][text[i] - 'a'] = true;
            }
            unordered_map<char, int> mp;
            for (int start = 0, end = 0; end < len; end++) {
                mp[text[end]]++;
                while (!valid(mp)) {
                    if (--mp[text[start]] == 0)
                        mp.erase(text[start]);
                    start++;
                }
                if (mp.size() == 2) {
                    vector<char> cs;
                    for (auto & e : mp) {
                        if (e.second == 1)
                            cs.push_back(e.first);
                    }
                    if (cs.size() == 2) {
                        for (auto & c : cs) {
                            if (has_c[end + 1][c - 'a'])
                                ans = max(ans, end - start + 1);
                        }
                    } else {
                        cs.clear();
                        for (auto & e : mp)
                            if (e.second != 1)
                                cs.push_back(e.first);
                        for (auto & c : cs) {
                            if (has_c[end + 1][c - 'a'])
                                ans = max(ans, end - start + 1);
                        }
                    }
                }
            }
            reverse(text.begin(), text.end());
        }
        return ans;
    }
};
