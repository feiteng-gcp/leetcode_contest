class Solution {
public:
    int maxRepOpt1(string text) {
        int ans = 0;
        for (char c = 'a'; c <= 'z'; ++c)
            ans = max(ans, solve(text, c));
        return ans;
    }
    int solve(string text, char c) {
        vector<int> segLength;
        vector<bool> only1CharAwayFromPrev;

        int N = text.length();
        int i = 0;
        while (i < N) {
            int gap = 0;
            while (i < N && text[i] != c)
                ++gap, ++i;
            int l = 0;
            while (i < N && text[i] == c)
                ++l, ++i;
            if (l) {
                only1CharAwayFromPrev.push_back(gap <= 1);
                segLength.push_back(l);
            }
        }
        int S = segLength.size();
        if (S == 0)
            return 0;
        if (S == 1)
            return segLength[0];
        if (S == 2)
            return only1CharAwayFromPrev[1] ? segLength[0] + segLength[1] : max(segLength[0], segLength[1]) + 1;
        int ans = 0;
        for (int i = 0; i < S; ++i)
            ans = max(ans, segLength[i] + 1);
        for (int i = 1; i < S; ++i) {
            if (only1CharAwayFromPrev[i])
                ans = max(ans, segLength[i - 1] + segLength[i] + 1);
        }
        return ans;
    }
};
