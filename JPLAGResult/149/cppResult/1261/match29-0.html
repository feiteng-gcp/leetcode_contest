class Solution {
public:
    int maxRepOpt1(string s) {
        int n = s.size();
        int ans = 0;
        for(int i = 0; i < 26; i++) {
            vector<int> a(n + 1, 0);
            for(int j = 0; j < n; j++) {
                a[j + 1] = a[j];
                if(s[j] - 'a' == i) {
                    a[j + 1] += 1;
                }
            }
            int total = a[n];
            for(int j = 1, k = 1; j <= n; j++) {
                k = max(k, j);
                while(k <= n && (k - j + 1) - (a[k] - a[j - 1]) <= 1) {
                    //if(j == 1 && i == 0) {
                    //    cout << k << endl;
                    //}
                    ans = max(ans, min(total, k - j + 1));
                    k++;
                }
            }
        }
        return ans;
    }
};
