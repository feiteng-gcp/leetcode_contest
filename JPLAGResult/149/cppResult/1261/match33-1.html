class Solution {
public:


int maxRepOpt1(string a) {
    int n = a.size();
    int res = 0;
    int last = 0;
    int p = 0;
    map<char, int> ma;
    for (char c : a)
    	++ma[c];
    while(p < n) {
    	int t = p;
    	while(p + 1 < n && a[p + 1] == a[t]) ++p;
    	res = max(res, p - t + 1);
    	if(ma[a[t] ] > p - t + 1) {
    		res = max(res, p - t + 2);
    	}
    	int t1 = p + 1;
    	p += 1;
    	while(p + 1 < n && a[p + 1] == a[t]) ++p;
    	if(p > t1) {
    		res = max(res, p - t);
    		if(ma[a[t] ] > p - t) {
    			res = max(res, p - t + 1);
    		}
    	}
    	p = t1;
    }
    return res;
}
};
