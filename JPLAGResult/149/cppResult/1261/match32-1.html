class Solution {
public:
    static const int maxn = 20000 + 5;
    static const int Max = 27;

    int cumsum[maxn][Max];

    int maxRepOpt1(string text)
    {

        int len = text.size();
        for (int i = 1; i <= len; i++)
        {
            int v = text[i-1] - 'a' + 1;
            cumsum[i][v]++;
            for (int j = 1; j <= 26; j++) cumsum[i][j] += cumsum[i-1][j];
        }
        int maxLen = -1;
        for (int i = 1; i <= len; i++)
        {
            int lo = i;
            int hi = len;
            int ans = -1;
            while (lo <= hi)
            {
                int mid = lo + hi >> 1;
                vector <int> occ, id;
                for (int j = 1; j <= 26; j++)
                {
                    int sum = cumsum[mid][j] - cumsum[i-1][j];
                    if (sum == 0) continue;
                    occ.push_back(sum);
                    id.push_back(j);
                }
//                cerr << "mid " << mid << " :\n";
//                for (int j = 0; j < occ.size(); j++)
//                  cerr << "(" << occ[j] << ", " << id[j] << ")" << endl;
//                cerr << endl;
                int sze = occ.size();
                if (sze > 2)
                {
                    hi = mid - 1;
                    continue;
                }
                if (sze == 1)
                {
                    ans = mid - i + 1;
                    lo = mid + 1;
                    continue;
                }
                int f = occ[0]; int fid = id[0];
                int s = occ[1]; int sid = id[1];
                if (f > s)
                {
                    swap(f, s);
                    swap(fid, sid);
                }
//                cerr << "f " << f << " " << fid << " " << s << " " << sid << endl;
                if (f > 1)
                {
                    hi = mid - 1;
                    continue;
                }
                int bakiRgt = cumsum[len][sid] - cumsum[mid][sid];
                int bakiLft = cumsum[i - 1][sid];
                if (bakiRgt >= 1 or bakiLft >= 1)
                {
                    ans = mid - i + 1;
                    lo = mid + 1;
                }
                else
                {
                    hi = mid - 1;
                }
            }
            maxLen = max(maxLen, ans);
        }
        return maxLen;
    }
};
