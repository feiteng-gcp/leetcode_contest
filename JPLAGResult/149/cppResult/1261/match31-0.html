class Solution {
public:
    int maxRepOpt1(string text) {
        int len = text.length();
        vector<int> leftSameLen(text.size(), 0), rightSameLen(text.size(), 0);
        for (int i = 0; i < len; i++) {
            if (i == 0 || text[i] != text[i-1]) leftSameLen[i] = 1;
            else leftSameLen[i] = leftSameLen[i-1] + 1;
        }
        for (int i = len-1; i >= 0; i--) {
            if (i == len-1 || text[i] != text[i+1]) rightSameLen[i] = 1;
            else rightSameLen[i] = rightSameLen[i+1] + 1;
        }
        int endPos[2][26];
        for (int i = 0; i < 2; i++) 
            for (int j = 0; j < 26; j++)
                endPos[i][j] = -1;
        for (int i = 0; i < len; ++i) {
            if (endPos[0][text[i]-'a'] >= 0)
                continue;
            endPos[0][text[i]-'a'] = i;
        }
        for (int i = len-1; i >= 0; --i) {
            if (endPos[1][text[i]-'a'] >= 0)
                continue;
            endPos[1][text[i]-'a'] = i;
        }
        
        int res = 0;
        for (int i = 0; i < len; ++i)
            res = max(res, leftSameLen[i]);
        for (int i = 0; i < len; ++i) {
            if (i == 0) {
                if (i+1 >= len || text[i] == text[i+1])
                    continue;
                if (endPos[1][text[i+1] - 'a'] > i+rightSameLen[i+1]) {
                    res = max(res, 1+rightSameLen[i+1]);
                }
            } else if (i == len-1) {
                if (i-1 <= 0 || text[i] == text[i-1])
                    continue;
                if (endPos[0][text[i-1]-'a'] < i-leftSameLen[i-1])
                    res = max(res, 1+leftSameLen[i-1]);
            } else {
                if (text[i-1] == text[i+1] && text[i-1] != text[i]) {
                    if (endPos[0][text[i-1]-'a'] < i-leftSameLen[i-1]) {
                        res = max(res, 1 + leftSameLen[i-1] + rightSameLen[i+1]);
                    } else {
                        res = max(res, leftSameLen[i-1] + rightSameLen[i+1]);
                    }
                    if (endPos[1][text[i+1] - 'a'] > i+rightSameLen[i+1]) {
                        res = max(res, 1 + leftSameLen[i-1] + rightSameLen[i+1]);
                    } else {
                        res = max(res, leftSameLen[i-1] + rightSameLen[i+1]);
                    }
                } else if (text[i-1] != text[i+1]) {
                    if (text[i-1] == text[i]) {
                        if (endPos[0][text[i+1]-'a'] < i) {
                            res = max(res, 1 + rightSameLen[i+1]);
                        }
                        if (endPos[1][text[i+1]-'a'] > i + rightSameLen[i+1])
                            res = max(res, 1 + rightSameLen[i+1]);
                    } else if (text[i] == text[i+1]) {
                        if (endPos[0][text[i-1]-'a'] < i-leftSameLen[i+1]) {
                            res = max(res, 1 + leftSameLen[i+1]);
                        }
                        if (endPos[1][text[i-1]-'a'] > i)
                            res = max(res, 1 + leftSameLen[i+1]);
                    } else {
                        if (endPos[0][text[i+1]-'a'] < i) {
                            res = max(res, 1 + rightSameLen[i+1]);
                        }
                        if (endPos[1][text[i+1]-'a'] > i + rightSameLen[i+1])
                            res = max(res, 1 + rightSameLen[i+1]);
                        if (endPos[0][text[i-1]-'a'] < i-leftSameLen[i+1]) {
                            res = max(res, 1 + leftSameLen[i+1]);
                        }
                        if (endPos[1][text[i-1]-'a'] > i)
                            res = max(res, 1 + leftSameLen[i+1]);
                    }
                        
                }
            }
        }
        return res;
    }
};
