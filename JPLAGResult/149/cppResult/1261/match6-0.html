class Solution {
public:
	int maxRepOpt1(string text) {
		if (text.size() < 2) {
			return text.size();
		}

		unordered_map<char, int> count;
		vector<pair<char, int>> sequence;

		char last = ' ';
		for (char c : text) {
			count[c]++;
			if (c == last) {
				sequence.back().second++;
			}
			else {
				sequence.emplace_back(c, 1);
				last = c;
			}
		}

		int longest = 0;
		for (pair<char, int> p : sequence) {
			longest = max(longest, p.second);
		}

		for (pair<char, int> p : sequence) {
			if (p.second < count[p.first]) {
				longest = max(longest, p.second + 1);
			}
		}

		for (int i = 1; i < sequence.size() - 1; i++) {
			if (sequence[i].second == 1 && sequence[i - 1].first == sequence[i + 1].first) {
				int two_parts_total = sequence[i - 1].second + sequence[i + 1].second;
				if (two_parts_total < count[sequence[i - 1].first]) {
					longest = max(longest, two_parts_total + 1);
				}
				else {
					longest = max(longest, two_parts_total);
				}
			}
		}

		return longest;
	}
};
