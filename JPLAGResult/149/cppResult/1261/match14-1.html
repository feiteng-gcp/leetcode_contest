class Solution {
public:
    int maxRepOpt1(string s) {
        int an = 0, n = s.size();
        vector<int> cnt(30);
        for (int i = 0; i < n; ++i) cnt[s[i]-'a']++;

        for (int f = 0; f < 26; ++f) {
            char ch = 'a'+f;
            int cur = 0, r = 0;
            for (int i = 0, j = 0; j < n; ++j) {
                if (s[j] != ch) {
                    if (r) {
                        while (s[i] == ch) ++i;
                        ++i;
                    }
                    r = 1;
                }
                cur = max(cur, j-i+1);
            }
            if (cur > cnt[f]) cur = cnt[f];
            an = max(an, cur);
        }
        return an;
    }
};
