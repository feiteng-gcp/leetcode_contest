class Solution {
public:
    int maxRepOpt1(string text) {
        map<char,int> cnt;
        for(char c:text)
            cnt[c]++;
        int n=text.length();
        int ans=0;
        for(auto p:cnt)
        {
            char c=p.first;
            int ncnt=0;
            int l=0,r=0;
            while(r<n)
            {
                if(text[r]!=c)
                    ncnt++;
                if(ncnt>1)
                {
                    while(l<=r&&text[l]==c) l++;
                    l++;
                    ncnt--;
                }
                if(!ncnt||r-l+1<=p.second)
                    ans=max(ans,r-l+1);
                else
                    ans=max(ans,r-l);
                //cout<<l<<" "<<r<<" "<<ncnt<<" "<<ans<<endl;
                r++;
            }
        }
        return ans;
    }
};
