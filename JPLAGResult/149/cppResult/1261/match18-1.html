class Solution {
public:
    int maxRepOpt1(string text) {
        // key = char, v = vector of pair(start_idx, end_idx);
        int max_len = 0;
        unordered_map<char, vector<pair<int, int>>> m;
        int i = 0, len = text.size();
        char cur_ch; 
        int start_idx, end_idx;
        while (i < len) {
            cur_ch = text[i];
            start_idx = i;
            while (i < len && text[i] == cur_ch) {
                ++i;
            }
            end_idx = i - 1;
            m[cur_ch].emplace_back(make_pair(start_idx, end_idx));
            max_len = max(max_len, i - start_idx);
        }
        // m[cur_ch].emplace_back(make_pair(start_idx, len - 1));
        // max_len = max(max_len, len - start_idx);
        
        for (auto &kv: m) {
            int cur_max_len = 0;
            auto v = kv.second;
            int n = v.size();
            if (n >= 2) {

                for (int i = 0; i < n - 1; ++i) {
                    if (v[i].second + 2 == v[i+1].first) { // 正好能插入一个字符
                        if (n > 2) { // 从剩下的位置拿一个字符
                            cur_max_len = max(cur_max_len, v[i+1].second - v[i].first + 1);
                        } else {
                            cur_max_len = max(cur_max_len, v[i+1].second - v[i].first);
                        }
                    } else {
                        cur_max_len = max(cur_max_len, v[i].second - v[i].first + 1 + 1);
                    }
                }
                cur_max_len = max(cur_max_len, v[n-1].second - v[n-1].first + 1 + 1);
            } else {
                cur_max_len = v[0].second - v[0].first + 1;
            }
            max_len = max(max_len, cur_max_len);
        }
        return max_len;
    }
};
