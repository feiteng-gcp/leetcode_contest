class Solution {
public:
    int maxRepOpt1(string text) {
        int n = text.size();
        int res = 0;
        for(char ch = 'a'; ch <= 'z'; ch++){
            vector <pair<int, int>> ps;

            int cont = 0;
            int pre_space = 0;
            for(int i = 0; i <= n; i++){
                if(i < n and text[i] == ch){
                    cont++;
                }else{
                    if(cont > 0){
                        ps.push_back(make_pair(pre_space, cont));
                        cont = 0;
                        pre_space = 0;
                    }
                    pre_space++;
                }
            }
            
            // printf("%c -> %d\n", ch, ps.size());
            int offset = 0;
            if(ps.size() > 1) offset++;
            for(int j = 0; j < ps.size(); j++){
                res = max(res, ps[j].second + offset);
                if(j > 0 and ps[j].first == 1){
                    int sum = ps[j - 1].second + ps[j].second;
                    if(ps.size() > 2) sum++;
                    res =  max(res, sum);
                }
            }
        }
        return res;
    }
};
