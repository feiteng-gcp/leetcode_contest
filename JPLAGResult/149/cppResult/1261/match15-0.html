class Solution {
public:
    int maxRepOpt1(string text) {
        vector<int> tot_count(26, 0);
        for (char c : text) ++tot_count[c - 'a'];
        // Left consecutive count
        int n = text.size();
        vector<int> left(n, 0);
        for (int i = 0; i < n; ++i) {
            if (i > 0 && text[i] == text[i - 1]) left[i] = left[i - 1] + 1;
            else left[i] = 1;
        }
        vector<int> right(n, 0);
        for (int i = n - 1; i >= 0; --i) {
            if (i < n - 1 && text[i] == text[i + 1]) right[i] = right[i + 1] + 1;
            else right[i] = 1;
        }
        int bestsofar = 0;
        // First get just the consecutives.
        for (int i = 0; i < n; ++i) bestsofar = max(bestsofar, left[i]);
        for (int i = 0; i < n; ++i) bestsofar = max(bestsofar, right[i]);
        // Now get all the swaps.
        for (int i = 0; i < n; ++i) {
            if (i > 0 && i < n - 1 && text[i - 1] == text[i + 1]) {
                int curr = left[i - 1] + right[i + 1];
                if (tot_count[text[i - 1] - 'a'] > curr) ++curr;
                bestsofar = max(bestsofar, curr);
            } else {
                // check left
                if (i > 0) {
                    int curr = left[i - 1];
                    if (tot_count[text[i - 1] - 'a'] > curr) ++curr;
                    bestsofar = max(bestsofar, curr);
                }
                if (i < n - 1) {
                    int curr = right[i + 1];
                    if (tot_count[text[i + 1] - 'a'] > curr) ++curr;
                    bestsofar = max(bestsofar, curr);
                }
            }
        }
        return bestsofar;
    }
};
