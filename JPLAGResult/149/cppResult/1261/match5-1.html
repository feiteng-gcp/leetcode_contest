class Solution {
public:
    int maxRepOpt1(string text) {
        int re = 0;

        for(int i = 0; i < 26; ++i) {
            char a = 'a' + i;
            int sum = 0;
            int diff = 0;
            int pre = 0;
            int now = 0;
            int mx = 0;
            for(int j = 0; j < text.length(); ++j) {

                if(text[j] == a) {
                    ++sum;
                    ++now;
                    diff = 0;
                } else {
                    ++diff;
                    if(diff == 1) {
                        mx = max(mx, pre + now);
                        pre = now;
                    } else {
                        pre = 0;
                    }
                    now = 0;
                }
            }
            mx = max(mx, pre + now);
            re = max(re, mx + (sum > mx ? 1 : 0));
        }
        return re;
    }
};
