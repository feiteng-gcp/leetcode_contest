class Solution {
public:
    int maxRepOpt1(string text) {
        vector<pair<int,int>> stat[26];
        int lastc = '0';
        int count = 0;
        for(int i = 0; i < text.size(); i++){
            if(text[i]!=lastc){
                stat[text[i]-'a'].push_back({i,1});
                lastc = text[i];
            }
            else{
                stat[text[i]-'a'][stat[text[i]-'a'].size()-1].second++;
            }
        }
        int res = 0;
        int tmp = 0;
        for(int i = 0; i < 26; i++){
            if(stat[i].size()==1) res = max(res, stat[i][0].second);
            else if(stat[i].size()==2){
                if(stat[i][0].first+stat[i][0].second+1 == stat[i][1].first){
                    tmp = stat[i][0].second + stat[i][1].second;
                }
                else tmp = max(stat[i][0].second, stat[i][1].second)+1;
                res = max(res, tmp);
            }
            else if(stat[i].size()>2){
                tmp = 0;

                for(int j = 0; j < stat[i].size(); j++){
                    if(j == stat[i].size()-1){
                        tmp = max(tmp,stat[i][j].second+1);
                    }
                    else{
                        if(stat[i][j].first+stat[i][j].second+1 == stat[i][j+1].first){
                            tmp = max(tmp,stat[i][j].second+stat[i][j+1].second+1);
                        }
                        else{
                            tmp = max(tmp,stat[i][j].second+1);
                        }
                    }
                }
                res = max(res, tmp);
            }
        }
        return res;
    }
};
