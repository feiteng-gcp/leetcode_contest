class Solution {
public:
    int maxRepOpt1(string text) {
        vector<pair<int,int>> rep[26];
        int start = 0;
        for (int i=1; i<text.size(); i++) {
            if (text[i-1] != text[i]) {
                rep[text[i-1] - 'a'].push_back(make_pair(start, i));
                start = i;
            }
        }
        
        rep[text.back() - 'a'].push_back(make_pair(start, text.size()));
        
        int best = 0;
        pair<int,int> last, cur;
        for (int i=0; i<26; i++) {
            if (rep[i].empty()) continue;
            last = rep[i][0];
            best = max(best, last.second - last.first + (rep[i].size() > 1));
            for (int j=1; j<rep[i].size(); j++) {
                cur = rep[i][j];
                int len = cur.second - cur.first + 1;
                if (last.second + 1 == cur.first) {
                    len += last.second - last.first;
                    if (rep[i].size() == 2) {
                        len--;
                    }
                }
                
                best = max(best, len);
                last = cur;
            }
        }
        
        return best;
    }
};
