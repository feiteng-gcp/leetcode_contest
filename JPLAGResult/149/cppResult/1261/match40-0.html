class Solution {
public:
    int maxRepOpt1(string text) {
        int maxl=1;
        char current = text[0];
        int curlen=1;
        for(int i=1;i<text.size();i++)
        {
            if(text[i]!=current)
            {
                int temp=0;
                if(i<text.size()-1&&text[i+1]==current)
                {
                    for(int j=i+1;j<text.size();j++)
                    {
                        if(text[j]!=current)
                            break;
                        temp++;
                    }
                }
                for(int g=0;g<text.size();g++)
                {
                    if(g>=i-curlen&&g<=i+temp)
                        continue;
                    if(text[g]==current)
                    {
                        maxl=max(maxl,curlen+temp+1);
                        break;
                    }
                }
                maxl=max(maxl,curlen+temp);
                curlen=1;
                current=text[i];
            }
            else
            {
                curlen++;
            }
        }
        maxl=max(maxl,curlen);
        return maxl;
    }
};
