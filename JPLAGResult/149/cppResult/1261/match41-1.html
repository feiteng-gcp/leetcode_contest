class Solution {
public:
    int maxRepOpt1(string s) {
        int n = s.size();
        vector<int>l = vector<int>(n,-1);
        vector<int>lmask = vector<int>(n,0);        
        vector<int>r = vector<int>(n,-1);
        vector<int>rmask = vector<int>(n,0);
        for(int i=0;i<n;i++){
            if(i&&s[i-1]==s[i])l[i]=l[i-1];
            else l[i]=i;
            if(i)lmask[i]=lmask[i-1];
            lmask[i]|=(1<<(s[i]-'a'));
        }
        for(int i=n-1;i>=0;i--){
            if(i+1<n&&s[i]==s[i+1])r[i]=r[i+1];
            else r[i]=i;

            if(i+1<n)rmask[i]=rmask[i+1];
            rmask[i]|=(1<<(s[i]-'a'));
        }
        int ret =0;
        for(int i=0;i<n;i++){
            int tmp=i-l[i]+1;
            if((l[i]&&(lmask[l[i]-1]&(1<<(s[i]-'a'))))||(i+1<n&&(rmask[i+1]&(1<<(s[i]-'a')))))tmp++;
            ret=max(ret,tmp);
            tmp=r[i]-i+1;
            if((r[i]+1<n&&(rmask[r[i]+1]&(1<<(s[i]-'a'))))||(i&&(lmask[i-1]&(1<<(s[i]-'a')))))tmp++;
            ret=max(ret,tmp);
        }
        for(int i=1;i+1<n;i++){
            if(s[i-1]!=s[i+1])continue;
            if(s[i-1]==s[i])continue;
            int f = 0;
            if(l[i-1]&&(lmask[l[i-1]-1]&(1<<(s[i-1]-'a'))))f=1;
            else if((r[i+1]+1<n&&(rmask[r[i+1]+1]&(1<<(s[i+1]-'a')))))f=1;
            ret=max(ret,i-1-l[i-1]+1+r[i+1]-i+f);
        }
        return ret;
    }
};
