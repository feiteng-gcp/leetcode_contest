class Solution {
public:
    int maxRepOpt1(string text) {
        vector<int> cnt(26,0);
        for(char c:text){
            cnt[c-'a']++;
        }
        int n=text.size();
        vector<int> lrs(n,1);
        for(int i=n-2;i>=0;i--){

            if(text[i]==text[i+1]) lrs[i] = lrs[i+1]+1;
        }
        int res=1;
        for(int i=0;i<n;i++){
            if(lrs[i]!=cnt[text[i]-'a']){
                res = max(res, lrs[i]+1);
            }
            if(i+lrs[i]+1 < n && text[i+lrs[i]+1] == text[i]){
                int tmp = lrs[i] + lrs[i+lrs[i]+1];
                if (tmp!=cnt[text[i]-'a']) res=max(res,tmp+1);
                else res = max(res, tmp);
            }
        }
        return res;
    }
};
