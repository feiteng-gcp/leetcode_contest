class Solution {
public:
    int maxRepOpt1(string text) {
        int n = text.size();
        vector<int> f(256);
        for (char i : text)
            f[i]++;
        int ans = 0;
        for (int i = 'a'; i <= 'z'; i++) {
            int hv = 0, to = 0;
            for (int j = 0; j < n; j++) {
                while (to < n && (to + 1 - j <= f[i] && hv + (text[to] == i) + (hv + (text[to] == i) != f[i]) >= to + 1 - j)) {
                    hv += text[to] == i;
                    to++;
                }
                // if (i <= 'b') printf("%d %d %d %d\n", i, j, hv, to);
                ans = max(ans, to - j);
                if (to == j) to++;
                else hv -= text[j] == i;
            }
        }
        return ans;
    }
};
