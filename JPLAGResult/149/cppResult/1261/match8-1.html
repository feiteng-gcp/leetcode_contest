class Solution {
public:
    int maxRepOpt1(string text) {
        unordered_map<char, int> charNums;
        for (char ch : text)
            charNums[ch]++;
        int count = 1;
        int res = 1;
        int flag = 0;
        char ch = text[0];
        text += "01";
        for (int i = 1; i < text.size(); i++)
        {
            if (text[i] == ch)
                count++;
            else
            if (flag)
            {
                count += count >= charNums[ch] ? 0 : 1;
                if (count > res) res = count;
                i = flag;
                ch = text[i];
                flag = 0;
                count = 1;
            }
            else
            {
                flag = i;
            }
        }
        return res;
    }
};
