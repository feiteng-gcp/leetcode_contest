class Solution {
public:
    int maxRepOpt1(string text) {
        int len = text.length();
        int fans = 0;
        for (int i = 0; i < 26; i ++) {
            char ch = 'a' + i;
            vector<vector<int>> pos;
            int now = 0;
            for (int j = 0; j < len; j ++) {
                if (text[j] == ch) {
                    now += 1;
                }
                else {
                    if (now != 0) {
                        pos.push_back({j - 1, now});
                        now = 0;
                    }
                }
            }
            if (now != 0) {
                pos.push_back({len - 1, now});
                now = 0;
            }
            int x = pos.size();
            //cout << x << endl;
            if (x == 1) {
                fans = max(fans, pos[0][1]);
            }
            for (int j = 0; j < x - 1; j ++) {
                int pp = pos[j + 1][0] - pos[j + 1][1] + 1;
                //cout << j << ' ' << pp << endl;
                if (pp - pos[j][0] == 2) {
                    if (x > 2) {
                        fans = max(fans, pos[j][1] + pos[j + 1][1] + 1);
                    }
                    else
                        fans = max(fans, pos[j][1] + pos[j + 1][1]);
                }
                else
                    fans = max(fans, max(pos[j][1], pos[j + 1][1]) + 1);
            }
        }
        return fans;
    }
};
