class Solution {
public:
    int maxRepOpt1(string text) {
        int N=text.size();
        int ans=0;
        for(int i=0;i<26;i++){
            char c='a'+i;
            vector<int> starts;
            vector<int> ends;
            int start=-1;
            for(int j=0;j<=N;j++){
                if(j<N&&text[j]==c){
                    if(start==-1){
                        start=j;
                    }
                }else{
                    if(start!=-1){
                        starts.push_back(start);
                        ends.push_back(j-1);
                    }
                    start=-1;
                }
            }
            
            if(starts.size()==1){
                ans=max(ans,ends[0]-starts[0]+1);
            }else if(starts.size()>1){
                for(int j=0;j+1<starts.size();j++){
                    ans=max(ans,ends[j]-starts[j]+2);
                    ans=max(ans,ends[j+1]-starts[j+1]+2);
                    if(starts[j+1]==ends[j]+2){
                        if(starts.size()>=3)
                            ans=max(ans,ends[j]-starts[j]+1+ends[j+1]-starts[j+1]+1+1);
                        else
                            ans=max(ans,ends[j]-starts[j]+1+ends[j+1]-starts[j+1]+1);
                    }
                }
            }
        }
        return ans;
    }
};
