class Solution {
public:
    int maxRepOpt1(string text) {
        vector<int> times(text.size());
        vector<int> ss(26, 0);
        int i = 0, sz = text.size();
        
        for(char c:text) ss[c-'a']++;
        
        while (i < sz) {
            int t = i, cnt = 1;
            while (i + 1 < sz && text[i] == text[i+1]) cnt++, i++;
            while (t <= i) times[t++] = cnt;
            i++;
        }
        //for (int i=0; i<sz; ++i) cout << times[i] << " ";

        int ans = 0;
        for (int i=1; i+1<sz; ++i) {
            if (text[i-1] == text[i+1] && times[i] == 1) {
                ans = max(ans, min(times[i-1] + times[i+1] + 1, ss[text[i-1]-'a']));
            }else{
                if (times[i-1] != ss[text[i-1]-'a']) ans = max(ans, times[i-1]+1);
                if (times[i+1] != ss[text[i+1]-'a']) ans = max(ans, times[i+1]+1);
                ans = max(ans, max(times[i], max(times[i-1], times[i+1])));
            }
            
        }
        return ans;
    }
};
