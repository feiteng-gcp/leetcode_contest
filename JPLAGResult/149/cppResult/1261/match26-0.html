class Solution {
public:
    int cnt[26];
    int maxRepOpt1(string text) {
        for(int i = 0; i < text.length(); ++i) cnt[text[i]-'a']++;
        int l = 0, r = 0;
        int ans = 0;
        while(l < text.length()){
            int c = 0;
            while(r < text.length() && text[r] == text[l])r++;
            c = r-l;
            int rr = r+1;
            while(rr < text.length() && text[rr] == text[l]) rr++;
            int k = rr-r-1;
            c += rr-r-1;
            int tot = cnt[text[l]-'a'];
            if(tot > c) ans = max(ans, c + 1);
            else ans = max(ans, c);
            l = r;
        }
        return ans;
    }
};
