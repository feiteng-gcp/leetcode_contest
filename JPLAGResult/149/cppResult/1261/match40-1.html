struct node {
    char x;
    int cnt;
    node () {}
    node (char x, int cnt): x(x), cnt(cnt) {}
};


class Solution {
public:
    int maxRepOpt1(string text) {
        int n = text.size();
        char prev = '0';
        int cnt = 0;
        vector<node> xs;
        for (int i=0; i<n; i++) {
            if (prev != text[i]) {
                xs.push_back(node(prev, cnt));
                prev = text[i];
                cnt = 1;
            } else {
                cnt++;
            }
        }
        xs.push_back(node(prev, cnt));
        unordered_map<char, set<int>> ref;
        int m = xs.size();
        for (int i=1; i<m; i++) {
            ref[xs[i].x].insert(i);
        }
        int ans = 0;
        for (int i=1; i<m; i++) {
            // cout << xs[i].x << " " << xs[i].cnt << endl;
            ans = max(ans, xs[i].cnt);
            if (i>1) {
                char pre = xs[i-1].x;
                if (ref[pre].size() >= 1) {
                    if (*ref[pre].rbegin() != i-1) {
                        ans = max(ans, xs[i-1].cnt+1);
                    }
                }
            }
            if (i<m-1) {
                char nxt = xs[i+1].x;
                if (ref[nxt].size() >= 1) {
                    if (*ref[nxt].begin() != i+1) {
                        ans = max(ans, xs[i+1].cnt+1);
                    }
                }
            }
            if (xs[i].cnt == 1) {
                if (i>1 && i<m-1) {
                    char pre = xs[i-1].x;
                    char nxt = xs[i+1].x;
                    if (pre == nxt) {
                        ans = max(ans, xs[i-1].cnt + xs[i+1].cnt);
                        if (*ref[pre].begin() != i-1 || *ref[pre].rbegin() != i+1) {
                            ans = max(ans, xs[i-1].cnt + xs[i+1].cnt + 1);
                        }
                    }
                }
            }
        }
        return ans;
    }
};
