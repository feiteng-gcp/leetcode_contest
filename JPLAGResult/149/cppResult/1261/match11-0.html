typedef long long ll;
typedef vector<int> VI;
typedef pair<int,int> PII;

#define REP(i,s,t) for(int i=(s);i<(t);i++)
#define FILL(x,v) memset(x,v,sizeof(x))

const int INF = (int)1E9;
#define MAXN 1005

vector<PII> ran[26];
class Solution {
public:
    int maxRepOpt1(string text) {
      REP(c,0,26) ran[c].clear();
      REP(i,0,text.size()) {
        int j = i;
        while (j < text.size() && text[j] == text[i]) j++;
        ran[text[i] - 'a'].push_back(PII(i, j - 1));
        i = j - 1;
      }
      int ans = 0;
      REP(c,0,26) {
        REP(i,0,ran[c].size()) {
          ans = max(ans, ran[c][i].second - ran[c][i].first + 1 + (ran[c].size() > 1));
          if (i + 1 < ran[c].size()) {
            if (ran[c][i+1].first == ran[c][i].second + 2) {
                ans = max(ans, ran[c][i+1].second - ran[c][i+1].first + 1 + ran[c][i].second - ran[c][i].first + 1 + (ran[c].size() > 2));
            }
          }
        }
      }
      return ans;
    }
};
