class Solution {
public:
    int maxRepOpt1(string text) {
        text+="`";
        int maxi=0;
        int n=text.size();
        int i, j;
        for(i=0;i<26;i++)
        {
            char ch='a'+i;
            int amax=0;
            int x=0;
            int y=0;
            int count=0;
            int one=0;
            for(j=0;j<n;j++)
            {
                if(text[j]==ch)
                {
                    y++;
                    count++;
                    one=1;
                }
                else
                {
                    amax=max(amax, x+y+1);
                    if(one==1)
                    {                        
                        x=y;
                        y=0;
                    }
                    else
                    {
                        x=0;
                        y=0;
                    }
                    one=0;
                }
            }
            if(amax>count)
                amax=count;
            maxi=max(amax, maxi);
        }
        return maxi;
    }
};
