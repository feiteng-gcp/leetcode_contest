class Solution {
public:
    // "aaabaaabbaaaaaaaaaaa"
    int maxRepOpt1(string text) {
        unordered_map<char, vector<pair<int, int>>> ch_to_ranges;
        for (int i = 0; i < text.size(); ++i) {
            int j = i + 1;
            while (j < text.size() && text[j] == text[i]) {
                j++;
            }
            ch_to_ranges[text[i]].push_back({i, j - 1});
            i = j - 1;
        }
        int result = 0;
        for (auto const & ch_ranges : ch_to_ranges) {
            int n = ch_ranges.second.size();
            for (int i = 0; i < n; ++i) {
                if (n > 1 && ch_to_ranges.size() > 1) {
                    result = max(ch_ranges.second[i].second - ch_ranges.second[i].first + 2, result);
                } else {
                    result = max(ch_ranges.second[i].second - ch_ranges.second[i].first + 1, result);
                }
                if (i + 1 < n && ch_ranges.second[i + 1].first == ch_ranges.second[i].second + 2) {
                    if (n == 2) {
                        result = max(ch_ranges.second[i + 1].second - ch_ranges.second[i].first, result);
                    } else {
                        result = max(ch_ranges.second[i + 1].second - ch_ranges.second[i].first + 1, result);
                    }
                }
            }
        }
        return result;
    }
};
