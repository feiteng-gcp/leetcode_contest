class Solution {
public:
    int maxRepOpt1(string s) {
        int n = s.size();
        int maxlen = 0;
        int cnt = 0;
        int left = 0;
        unordered_map<char , int > store , rightmost_pos , leftmost_pos;
        for(int i = n - 1; i >= 0 ; --i)
        {
            store[s[i]]++;
            if(rightmost_pos.find(s[i]) == rightmost_pos.end())
                rightmost_pos[s[i]] = i;
        }
        for(int i = 0; i < n; ++i)
            if(leftmost_pos.find(s[i]) == leftmost_pos.end())
                leftmost_pos[s[i]] = i;
        
        for(char ch = 'a'; ch <= 'z'; ++ch)
        {
            char cur = ch;
            int dif = 0 , cnt = 0;
            int maxvalue = 0;
            if(store.find(ch) == store.end())
                continue;
            int left = 0;
            for(int i = 0; i < rightmost_pos[cur]; ++i)
            {
                if(s[i] != cur)
                    dif++;
                else
                    cnt++;
                
                while(dif == 2)
                {
                    if(s[left] == cur)
                        cnt--;
                    else
                        dif--;
                    left++;
                }
                maxvalue = max(maxvalue , cnt);
            }
            dif = cnt = 0;
            int right = n - 1;
            for(int i = n - 1; i > leftmost_pos[cur]; --i)
            {
                if(s[i] != cur)
                    dif++;
                else
                    cnt++;
                while(dif == 2)
                {
                    if(s[right] == cur)
                        cnt--;
                    else
                        dif--;
                    right--;
                }
                maxvalue = max(maxvalue , cnt);
            }
            
            maxlen = max(maxlen , maxvalue + 1);
        }
        
        
        return maxlen;
        
    }
};
