class Solution {
public:
    int maxRepOpt1(string text) {
        int res = 0, len;
        bool flag;
        int i, j, k;
        int n = text.size();
        for (i = 0; i<n; ++i) { // i开头的最长序列
            if (i !=0 && text[i] == text[i-1])
                continue;
            len = 1;
            flag = false;
            for (j=i + 1; j<n; ++j) {
                if (text[j] == text[i]) {
                    ++len;
                }
                else {
                    if (!flag) {
                        flag = true;
                    }
                    else {
                        break;
                    }
                }
            }
            if (flag && (text.find_first_of(text[i]) < i || text.find_last_of(text[i]) > j))
                ++len;
            res = max(res, len);
        }
        return res;
    }
};
