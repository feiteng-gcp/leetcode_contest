class Solution {
public:
    int maxRepOpt1(string text) {
        vector<vector<pair<int, int>>> status(26);
        for (int i = 0; i < text.size(); ++i) {
            if (i > 0 && text[i] == text[i - 1]) {
                status[text[i] - 'a'].back().second = i;
            } else {
                status[text[i] - 'a'].push_back(make_pair(i, i));
            }

        }
        int ans = 0;
        for (int i = 0; i < 26; ++i) {
            for (int j = 0; j < status[i].size(); ++j) {
                int delta = 0;
                if (status[i].size() > 1) delta = 1;
                if (status[i][j].second - status[i][j].first + 1 + delta > ans)
                    ans = status[i][j].second - status[i][j].first + 1 + delta;
                if (j > 0 && status[i][j - 1].second + 2 == status[i][j].first) {
                    delta = 0;
                    if (status[i].size() > 2) delta = 1;
                    if (status[i][j].second - status[i][j - 1].first + delta > ans)
                        ans = status[i][j].second - status[i][j - 1].first + delta;
                }
            }
        }
        return ans;
    }
};
