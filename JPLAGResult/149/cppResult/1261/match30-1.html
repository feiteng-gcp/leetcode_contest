class Solution {
public:
    int maxRepOpt1(string s) {
        vector<int> cnt(26, 0);
        for(auto c: s) cnt[c-'a']++;

        int n = s.size();
        int ans = 0;
        for(int i = 0; i < n; ) {
            int j = 1;
            while(i+j < n && s[i+j] == s[i])j++;
            int k = 0;
            while(i+j+k+1 < n && s[i+j+k+1] == s[i])k++;
            ans = max(ans, min(j+k+1, cnt[s[i]-'a']));
            i += j;
        }
        return ans;
    }
};
