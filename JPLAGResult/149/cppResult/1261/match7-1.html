class Solution {
public:
    int maxRepOpt1(string text) {
        const int n = text.size();
        vector<int> lefts(n, 0), rights(n, 0);
        vector<int> cnts(26, 0);
        int result = 0;
        for (int i = 0; i < n; ++i) {
            lefts[i] = (i > 0 && text[i] == text[i - 1]) ? lefts[i - 1] + 1 : 1;
            result = max(result, lefts[i]);
            ++cnts[text[i] - 'a'];
        }
        for (int i = n - 1; i >= 0; --i) {
            rights[i] = (i < n - 1 && text[i] == text[i + 1]) ? rights[i + 1] + 1 : 1;

            if (rights[i] < cnts[text[i] - 'a']) {
                result = max(result, rights[i] + 1);
            }
        }
        for (int i = 1; i + 1 < n; ++i) {
            if (text[i - 1] == text[i + 1] && text[i - 1] != text[i]) {
                int cnt = lefts[i - 1] + rights[i + 1];
                if (cnt < cnts[text[i - 1] - 'a']) {
                    ++cnt;
                }
                result = max(result, cnt);
            }
        }
        return result;
    }
};
