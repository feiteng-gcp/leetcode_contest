class Solution {
public:
	const int  MOD = (int)1e9 + 7; 
    int numRollsToTarget(int d, int f, int target) {
        // vector<int> dp(target+f*2,0);

        vector<vector<int>> dp(d+2, vector<int>(target+f*2,0));
		
		dp[0][0] = 1;
		
		for(int i = 0; i < d; i++){
			for(int j = target; j >= 0; j--){
				for(int k = 1; k <= f; k++){
					dp[i+1][j+k] += dp[i][j];
					dp[i+1][j+k] %= MOD;
				}
			}
		}
		
		return dp[d][target]%MOD;
    }
};
