class Solution {
private:
    const int MOD = 1000000007;

public:
    int numRollsToTarget(int d, int f, int target) {
        vector<vector<int>> ways(d+1, vector<int>(target+1, 0));
        
        ways[0][target] = 1;
        for (int i=1; i<=d; i++) {
            for (int j=0; j<=target; j++) {
                for (int k=1; k<=f; k++) {
                    if (j+k > target) break;
                    ways[i][j] += ways[i-1][j+k];
                    ways[i][j] %= MOD;
                }
            }
        }
        
        return ways[d][0];
    }
};
