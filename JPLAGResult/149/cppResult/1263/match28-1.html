const int MOD=1e9+7;


class Solution {
public:
    int dp[35][1050];
    int numRollsToTarget(int d, int f, int target) {
        memset(dp, 0, sizeof(dp));
        
        dp[0][0]=1;
        for (int i=1; i<=d; i++){
            for (int now=1; now<=f; now++){
                for (int prev=0; prev+now<=target; prev++){
                    dp[i][prev+now]=(dp[i][prev+now] + dp[i-1][prev])%MOD;
                }
            }
        }
        
        return dp[d][target];
    }
};
