#define MO 1000000007


class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        vector<vector<int>> dp(d+1, vector<int>(target+1, 0));
        dp[0][0] = 1;
        for (int i=0; i<d; i++) {
            for (int j=0; j<=target; j++) {
                for (int k=1; k<=f; k++) {
                    if (j+k<=target)
                        dp[i+1][j+k] = (dp[i+1][j+k]+dp[i][j])%MO;
                }
            }
        }
        return dp[d][target];
    }
};
