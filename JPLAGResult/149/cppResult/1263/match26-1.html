class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        // dp[i][j] is the number of ways for i dices to reach target j 
        int dp[d + 1][target + 1] = {};
        dp[0][0] = 1;
        for (int i = 1; i <= d; ++i) {
            for (int j = 1; j <= target; ++j) {
                for (int k = 1; k <= f; ++k) {
                    if (j >= k) {
                        dp[i][j] = (dp[i][j] % MOD + dp[i - 1][j - k] % MOD) % MOD;
                    }
                }
            }
        }
        return dp[d][target];
    }

private:
    int MOD = 1000000007;
};
