class Solution {
public:
    int mod=1e9+7;
    int dp[33][1010];

    int numRollsToTarget(int d, int f, int target) {
        dp[0][0]=1;
        for(int i=1;i<=d;i++){
            for(int j=1;j<=f;j++){
                for(int k=0;k+j<=target;k++){
                    dp[i][k+j]=(dp[i][k+j]+dp[i-1][k])%mod;
                }
            }
        }
        return dp[d][target];
    }
};
