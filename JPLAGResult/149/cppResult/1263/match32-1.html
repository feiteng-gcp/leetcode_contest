int dp[35][1050];
const int MOD = 1000000007;

class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        memset(dp, 0, sizeof(dp));

        dp[0][0] = 1;
        for (int i = 0; i < d; i++) {
            for (int j = 0; j <= target; j++) {
                for (int k = 1; k <= f; k++) {
                    dp[i + 1][j + k] = (dp[i + 1][j + k] + dp[i][j]) % MOD;
                }
            }
        }
        return dp[d][target];
    }
};
