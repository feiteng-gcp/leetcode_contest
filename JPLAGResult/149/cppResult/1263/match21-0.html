class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        const int inf = 1000000007;
        vector<vector<int>> opt(d + 1, vector<int>(target + 1, 0));
        opt[0][0] = 1;
        for (int i = 0; i < d; ++i) {
            for (int j = 0; j <= target; ++j) {
                if (opt[i][j] == 0) continue;
                for (int k = 1; k <= f; ++k) {
                    if (j + k > target) break;
                    opt[i + 1][j + k] = (opt[i + 1][j + k] + opt[i][j]) % inf;
                }
            }
        }
        return opt[d][target];
    }
};
