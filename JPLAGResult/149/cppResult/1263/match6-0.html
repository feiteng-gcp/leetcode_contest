class Solution {
public:
  const int MOD = 1e9+7;
  int numRollsToTarget(int d, int f, int target) {
    vector<vector<int> > dp(d + 1, vector<int>(target + 1));
    dp[0][0] = 1;
    for (int i = 1; i <= d; i++) {
      for (int j = 1; j <= target; j++) {
        for (int k = max(0, j - f); k <= max(0, j - 1); k++) {
          dp[i][j] += dp[i-1][k];
          dp[i][j] %= MOD;
        }
      }
    }
    return dp.back().back();
  }
};
