class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        vector<vector<long long int>> dp(d + 1, vector<long long int> (target + 1, 0));
        int i, j, k;
        for (j = 1; j <= target && j <= f; j++) {
            dp[1][j] = 1;
        }
        for (i = 2; i <= d; i++) {
            for (j = 1; j <= target; j++) {
                for (k = 1; k <= f && k < j; k++) {
                    dp[i][j] += dp[i - 1][j - k];
                    dp[i][j] %= 1000000007;
                }
            }
        }
        return dp[d][target];
    }
};
