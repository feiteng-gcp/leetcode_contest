#define LL long long


LL mod = 1e9 + 7;

class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        vector<LL> a(target + 1, 0);
        a[0] = 1;
        for(int i = 0; i < d; i++) {
            vector<LL> b(target + 1, 0);
            for(int j = 1; j <= f; j++) {
                for(int k = target; k - j >= 0; k--) {
                    b[k] = (b[k] + a[k - j]) % mod; 
                }
            }
            a = b;
        }
        return a[target];
    }
};
