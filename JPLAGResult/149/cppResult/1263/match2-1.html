class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        vector<vector<int>> dp(d+1, vector<int>(target+1, 0));
        dp[0][0] = 1;
        // dp[i][j] = dp[i-1][j-k]
        for (int i = 1; i <= d; ++i) {
            for (int j = 1; j <= target; ++j) {
                for (int k = 1; k <= f; ++k) {
                    if (k > j) break;
                    dp[i][j] += dp[i-1][j-k];
                    dp[i][j] %= 1000000007;
                }
            }
        }
        return dp[d][target];
    }
};
