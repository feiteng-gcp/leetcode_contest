const int MAX_N = 35;

const int MAX_M = 1100;
int dp[MAX_N][MAX_M];

class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        memset(dp, 0, sizeof(dp));
        for(int j = 1; j <= f; ++j) dp[1][j] = 1;
        int MOD = 1e9+7;

        for(int i = 1; i < d; ++i){
            for(int j = 1; j <= target; ++j){
                for(int k = 1; k <= f && j+k <= target; ++k){
                    dp[i+1][j+k] = (1LL*dp[i+1][j+k] + 1LL*dp[i][j]) % MOD;
                }
            }
        }
        return dp[d][target];
    }
};
