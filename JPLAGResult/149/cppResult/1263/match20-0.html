class Solution {
    const int mod = 1000000007;
public:
    int numRollsToTarget(int d, int f, int target) {
        vector<vector<int>> dp(d+1, vector<int>(target+1, 0));
        dp[0][0] = 1;
        
        for(int i = 1; i <= d; i ++)
        {
            for(int j = f; j >= 1; j--)
            {
                for(int cur = target; cur-j >= 0; cur--)
                {
                    dp[i][cur] = (dp[i][cur] + dp[i-1][cur-j])%mod;
                }
            }
        }
        
        return dp[d][target]%mod;
    }
};
