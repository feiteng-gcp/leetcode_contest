int dp[31][1010];
const int MOD = (int)1e9 + 7;

class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        memset(dp, 0, sizeof(dp));
        
        for (int i = 1; i <= f; ++i) dp[1][i] = 1;
        for (int i = 2; i <= d; ++i) {
            for (int j = 1; j <= target; ++j) {
                for (int k = 1; k <= min(f, j - 1); ++k) {
                    dp[i][j] += dp[i - 1][j - k];
                    dp[i][j] %= MOD;
                }
            }
        }
        
        return dp[d][target];
    }
};
