class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        const int MOD = 1e9 + 7;
        int dp[2][target + 5];
        memset(dp, 0, sizeof(dp));
        int fl = 0;

        dp[fl][0] = 1;
        for(int turn = 0; turn < d; ++turn) {
            for(int i = 0; i < target; ++i) {
                if(dp[fl][i] == 0) continue;
                for(int j = 1; j <= f; ++j) {
                    if(i + j > target) break; 
                    dp[fl ^ 1][i + j] = (dp[fl ^ 1][i + j] + dp[fl][i]) % MOD;
                }
               
            }
                
            memset(dp[fl], 0, sizeof(dp[fl]));
            fl ^= 1;
        }
        return dp[fl][target];
    }
};
