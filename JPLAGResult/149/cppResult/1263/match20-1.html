class Solution {
    const int MOD = 1e9 + 7;
public:
    int numRollsToTarget(int d, int f, int target) {
        // dp[i][j] 前i个色子，target为j的方法数量
        // dp[i][j] = 
        vector<vector<int>> dp(d + 1, vector<int>(target + 1, 0));
        dp[0][0] = 1;
        for (int i = 1; i <= d; i++) {
            for (int j = 1; j <= target; j++) {
                for (int k = 1; k <= min(j, f); k++) {
                    dp[i][j] += dp[i - 1][j - k];
                    dp[i][j] %= MOD;
                }
            }
        }
        return dp[d][target];
    }
};
