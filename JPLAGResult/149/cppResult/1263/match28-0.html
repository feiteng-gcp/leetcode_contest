class Solution {
public:
    int dp[31][1010];
    int numRollsToTarget(int d, int f, int target) {
        memset(dp, 0, sizeof(dp));
        dp[0][0] = 1;
        for(int i = 1; i <= d; ++i) {
            for(int j = 1; j <= f; ++j) {
                for(int k = j; k <= target; ++k) {
                    dp[i][k] += dp[i-1][k-j];
                    dp[i][k] %= 1000000007;
                }
            }
        }
        return dp[d][target];
    }
};
