class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        int a[31][1001] = {0};
        a[0][0]=1;
        for(int i=1;i<=d;i++)
        {
            for(int k=1;k<=f;k++)
            {
                for(int j=1;j<=f*d;j++)
                {
                    if(j>=k)
                    {
                        a[i][j]+=a[i-1][j-k];  
                        a[i][j]%=1000000007;
                    }
                }
            }
        }
        return a[d][target];
    }
};
