class Solution {
public:
    int numRollsToTarget(int d, int f, int target) {
        int mod = pow(10, 9) + 7;
        vector<vector<int>> dp(d, vector<int>(target + 1, 0));
        for (int j = 1; j <= min(f, target); j++) dp[0][j] = 1;
        for (int i = 1; i < d; i++) {
            for (int j = 1; j <= target; j++) {
                for (int k = 1; k <= f; k++) {
                    if (j - k >= 0) dp[i][j] = (dp[i][j] + dp[i-1][j-k]) % mod;
                    else break;
                }
            }
        }
        return dp[d-1][target];
    }
};
