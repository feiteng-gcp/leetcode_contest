class Solution {
public:
    const int Mod = 1e9 + 7;
    int dp[1002][31];
    int numRollsToTarget(int d, int f, int target) {
        memset(dp, 0, sizeof(dp));
        dp[0][0] = 1; int cnt = 0;
        for (int cnt = 1; cnt <= d; ++cnt) {
            for (int j = 1; j <= f; ++j) {
                for (int k = target; k >= j; --k) {
                    dp[k][cnt] += dp[k - j][cnt - 1];
                    dp[k][cnt] %= Mod;
                }
            }
        }
        return dp[target][d];
    }
};
