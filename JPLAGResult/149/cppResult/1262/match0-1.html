class MajorityChecker {
public:
  int N;
  int K;
  vector<int> arr;
  map<int, vector<int>> M;
  vector<int> A; 
  MajorityChecker(vector<int>& arr) {
    N = arr.size();
    this->arr = arr;
    K = (N + 99) / 100;
    A.resize(K);
    for (int i = 0; i < K; i++) {
      A[i] = -1;
      int start = i * K;
      int end = min(N, start + K);
      map<int, int> T;
      for (int j = start; j < end; j++) {
        T[arr[j]]++;
        if (T[arr[j]] > 50)  A[i] = arr[j];
      }
    }
    for (int i = 0; i < N; i++) M[arr[i]].push_back(i);
  }
  int gao(int t, int left, int right) {
    if (t == -1) return 0;
    vector<int> &X = M[t];
    auto l =  lower_bound(X.begin(), X.end(), left);
    auto r =  upper_bound(X.begin(), X.end(), right);
    return r - l;
  }

  int query(int left, int right, int threshold) {
    int start = (left - 1) / 100 + 1;
    int end = right / 100; 
    set<int> S;
    for (int i = left; i <= min(start * 100 - 1, right); i++) {
      S.insert(arr[i]);
    }
    for (int i = max(left, end * 100 + 1); i <= right; i++) {
      S.insert(arr[i]);
    }
    for (int i = start; i <= end; i++) S.insert(A[i]);
    for (auto t: S) {
      if (gao(t, left, right) >= threshold) return t;
    }
    return -1;
  }
};
