class MajorityChecker {
    unordered_map<int, vector<int>> m;
    
public:
    MajorityChecker(vector<int>& arr) {
        for(int i = 0; i < arr.size(); i ++)
        {
            m[arr[i]].push_back(i);
        }
    }
    
    int query(int l, int r, int t) {
        
        for(auto& p: m)
        {
            if(p.second.size() < t) continue;
            
            auto lo = lower_bound(p.second.begin(), p.second.end(), l);
            auto up = upper_bound(p.second.begin(), p.second.end(), r);
            
            int cnt = up-lo;
            if(cnt >= t) return p.first;
        }
        
        return -1;
    }
};
