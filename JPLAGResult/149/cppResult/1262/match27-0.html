class MajorityChecker {
public:
    vector<int>v;
    map<pair<int,int>,pair<int,int> >m;
    
    MajorityChecker(vector<int>& arr) {
        v=arr;
    }
    
    int query(int left, int right, int threshold) {
        if(m.find(make_pair(left,right))!=m.end()){
            pair<int,int>P=m[make_pair(left,right)];
            if(P.first==-1)return -1;
            if(P.second>=threshold)return P.first;
            return -1;
        } 
        
        int f=1;
        int num=v[left];
    
        for(int i=left+1;i<=right;i++){
            if(v[i]==num){
                f++;
            }else{
                if(f==0){
                    num=v[i];
                    f=1;
                }else{                
                    f--;    
                }
            }
        }
        
        int count=0;
        for(int i=left;i<=right;i++)
            if(num==v[i])
                count++;
        
        if(count*2>right-left)m[make_pair(left,right)]=make_pair(num,count);
        else m[make_pair(left,right)]=make_pair(-1,-1);
        
        if(count>=threshold)return num;
        return -1;
    }
    
};

 * MajorityChecker* obj = new MajorityChecker(arr);
 * int param_1 = obj->query(left,right,threshold);
