class MajorityChecker {
public:
	vector<int>num;
	MajorityChecker(vector<int>& arr) {
		num = arr;
	}

	int query(int left, int right, int threshold) {
		int n = num[left], temp = 1;
		for (int i = left + 1; i <= right; i++)
		{
			if (num[i] == n) temp++;
			else
			{
				temp--;
				if (temp == 0)
				{
					n = num[i];
					temp = 1;
				}
			}
		}
		temp = 0;
		for (int i = left; i <= right; i++)
		{
			if (num[i] == n) temp++;
		}
		if (temp >= threshold) return n;
		return -1;
	}
};
