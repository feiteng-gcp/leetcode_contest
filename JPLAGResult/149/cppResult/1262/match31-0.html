class MajorityChecker {
public:
    vector<int> nums;
    MajorityChecker(vector<int>& arr) {
        nums = arr;
    }
    
    int query(int left, int right, int threshold) {
        if (left == right) {
            if (threshold == 1) {
                return nums[left];
            }
            return -1;
        }
        int i, x, count;
        x = 0;
        count = 0;
        for (i = left; i + 1 <= right; i += 2) {
            if (nums[i] == nums[i + 1]) {
                if (count == 0) {
                    count = 2;
                    x = nums[i];
                } else if (x == nums[i]) {
                    count += 2;
                } else {
                    count -= 2;
                }
            }
        }
        if (count == 0 && i == right) {
            x = nums[right];
        }
        count = 0;
        for (i = left; i <= right; i++) {
            if (x == nums[i]) {
                count++;
            }
        }
        if (count >= threshold) {
            return x;
        }
        return -1;
    }
};
