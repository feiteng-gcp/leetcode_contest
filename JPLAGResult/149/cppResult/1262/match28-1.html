class MajorityChecker {
    vector<int> a, tree;
    map<int,vector<int>> mp;
    int n;
    int res = -1;
    
    void build(int nd, int l, int r) {

        int v = 0;
        int c = 0;
        for (int i = l; i <= r; i++) {
            if (a[i] == v) c++;
            else c--;
            if (c <= 0) {
                v = a[i];
                c = 1;
            }
        }
        tree[nd] = v;
        if (l == r) return;
        int m = (l+r) >> 1;

        int cc = nd << 1;
        build(cc, l, m);
        build(cc^1, m+1, r);
    }
    
    int countInSeg(int l, int r, int v) {
        vector<int> &pos = mp[v];
        int i = lower_bound(pos.begin(), pos.end(), l) - pos.begin();
        int j = upper_bound(pos.begin(), pos.end(), r) - pos.begin();
        return j-i;
    }
    
    void solve(int nd, int l, int r, int tr, int ph, int thres) {
        if (res >= 0) return;
        if (r < tr || l > ph) return;
        if (tr <= l && r <= ph) {
            if (countInSeg(tr, ph, tree[nd]) >= thres) res = tree[nd];
            return;
        }
        int cc = nd << 1;
        int m = (l+r) >> 1;
        solve(cc, l, m, tr, ph, thres);
        solve(cc^1, m+1, r, tr, ph, thres);
    }
public:
    MajorityChecker(vector<int>& arr) {
        a = arr;
        n = a.size();
        tree.resize(n*4);
        build(1,0,n-1);
        for (int i = 0; i < n; i++) {
            mp[a[i]].push_back(i);
        }
    }
    
    int query(int left, int right, int threshold) {
        res = -1;
        solve(1,0,n-1,left,right,threshold);
        return res;
    }
};
