class MajorityChecker {
public:
    vector<int>ve;
    MajorityChecker(vector<int>& arr) {
        int sz = arr.size();
        for(int i=0;i<sz;i++)ve.push_back(arr[i]);
    }
    
    int query(int left, int right, int threshold) {
        int aa[20005];
        int ma = 0;
        memset(aa,0,sizeof(aa));
        int sz = ve.size();
        //cout<<sz<<endl;

        int index = -1;
        for(int i=left;i<=right;i++){
            ++aa[ve[i]];
            if(aa[ve[i]]>ma){
                ma = aa[ve[i]];
                index = ve[i];
            }
        }
        //cout<<ma<<endl;
        if(ma<threshold)index = -1;
        return index;
    }
};
