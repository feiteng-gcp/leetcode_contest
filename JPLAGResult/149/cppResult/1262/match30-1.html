class MajorityChecker {
public:
    const int M = 100;
    vector<int> data;
    vector<map<int, int>> dp;
    int maxT=-1;
    set<int> ready;
    MajorityChecker(vector<int>& arr) {
        data = arr;
        int len = arr.size();
        len /= M;
        map<int, int> tmp;
        for(auto it:arr)
        {
            tmp[it]++;
        }
        for(auto it:tmp)
        {
            if(it.second>M)
            {
                ready.insert(it.first);
                maxT = max(maxT, it.second);
            }
        }
        for(int i=0;i<len;i++)
        {
            map<int, int> cur;
            if(i)
            {
                cur = dp.back();
            }
            for(int j=0;j<M;j++)
            {
                int index = M*i+j;
                int ddd = arr[index];
                if(ready.find(ddd)==ready.end()) continue;
                cur[ddd]++;
            }
            dp.push_back(cur);
        }
    }
    
    int query(int left, int right, int threshold) {
        if(maxT>0&&threshold > maxT) return -1;
        int len = right-left + 1;
        if(len <= 2*M)
        {
            map<int, int> cur;

            for(int i=left;i<=right;i++)
            {
                cur[data[i]] ++;
                if(cur[data[i]]>=threshold)
                {
                    return data[i];
                }
            }
            return -1;
        }
        int leftblock = left/M;
        int rightblock = right/M;
        map<int, int> cur;
        auto& tmp = dp[rightblock-1];
        auto& last = dp[leftblock];
        for(auto& it:tmp)
        {
            cur[it.first]=(it.second - last[it.first]);
            if(cur[it.first]>=threshold) return it.first;
        }
        for(int i=rightblock*M;i<=right;i++)
        {
            if(ready.find(data[i])==ready.end()) continue;
            cur[data[i]]++;
            if(cur[data[i]]>=threshold) return data[i];
        }
        for(int i=left;i<(leftblock+1)*M;i++)
        {
            if(ready.find(data[i])==ready.end()) continue;
            cur[data[i]]++;
            if(cur[data[i]]>=threshold) return data[i];
        }
        return -1;
    }
};
