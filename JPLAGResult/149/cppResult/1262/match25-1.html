class MajorityChecker {
public:
    const int len = 100;
    vector<vector<int>> dict = vector<vector<int>>(20001, vector<int>());
    vector<int> counter = vector<int>(20001, 0) ;
    vector<int> list;
    vector<int> candidate;
    
    MajorityChecker(vector<int>& arr) {
        list = arr;
        for (int i = 0; i < arr.size(); ++i) {
            dict[arr[i]].push_back(i);
            counter[arr[i]]++;

        }
        for (int i = 0; i <= 20000; ++i) {
            if (counter[i] >= len) candidate.push_back(i);
        }
    }
    
    int query(int left, int right, int threshold) {
        unordered_map<int, int> local_counter;
        if (right - left < len * 2) {

            for (int i = left; i <= right; ++i) {
                local_counter[list[i]]++;
            }
            for (auto it : local_counter) {
                if (it.second >= threshold) return it. first;
            }
        } else {
            for (int i = 0; i < candidate.size(); ++i) {
                auto lb = lower_bound(dict[candidate[i]].begin(), dict[candidate[i]].end(), left);
                auto ub = lower_bound(dict[candidate[i]].begin(), dict[candidate[i]].end(), right + 1);
                if (distance(lb, ub) >= threshold) return candidate[i];
            }
        }
         return -1;
    }
};
