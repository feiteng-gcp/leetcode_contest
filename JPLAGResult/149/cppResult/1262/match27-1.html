class MajorityChecker {
public:
    vector<int> arrs;
    map<pair<int,int>,pair<int,int>> mp;
    MajorityChecker(vector<int>& arr) {
        arrs = arr;
    }
    
    int query(int left, int right, int threshold) {
        auto it = mp.find({left,right});
        if(it!=mp.end()){
            if(it->second.second >= threshold) return it->second.first;

            else return -1;
        }
        
        int num = arrs[left];
        int count = 0;
        for(int i = left; i <= right; i++){
            if(count == 0){
                num = arrs[i];
                count = 1;
            }
            else{
                if(arrs[i] == num) count++;
                else count--;
            }
            
        }
        int nn = 0;
        for(int i = left; i <= right; i++){
            if(arrs[i] == num) nn++;
        }
        mp[{left,right}] = {num,nn};
        if(nn >= threshold) return num;
        else return -1;
    }
};

 * MajorityChecker* obj = new MajorityChecker(arr);
 * int param_1 = obj->query(left,right,threshold);
