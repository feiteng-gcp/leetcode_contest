class MajorityChecker {
public:
    static const int maxn = 2e4+10;
    vector<vector<int>> vec;
    vector<vector<int>> cnt;
    vector<int> num;
    int n;
    MajorityChecker(vector<int>& arr) {
        vec = vector<vector<int>>(maxn);
        cnt = vector<vector<int>>(maxn);
        num = arr;
        n = arr.size();
        for(int i = 0; i < n; i++) {
            vec[arr[i]].push_back(i);
        }
        for(int i = 0; i < maxn; i++) {
            if(!vec[i].empty())cnt[vec[i].size()].push_back(i);
        }
    }
    
    int query(int left, int right, int threshold) {
        int len = right-left+1;
        int ans = 0;
        if(len <= sqrt(n)) {
            map<int, int> mp;
            for(int i = left; i <= right; i++) {
                mp[num[i]]++;
            }
            for(auto p: mp) {
                if(p.second >= threshold)return p.first;
            }
        }
        else {
            for(int i = threshold; i < maxn; i++) {
                for(auto val: cnt[i]) {
                    int tot = upper_bound(vec[val].begin(), vec[val].end(), right)-lower_bound(vec[val].begin(), vec[val].end(), left);
                    if(tot >= threshold)return val;
                }
            }
        }
        return -1;
    }
};
