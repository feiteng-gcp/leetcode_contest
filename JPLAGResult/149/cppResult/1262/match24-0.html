class MajorityChecker {
public:
    vector<int> data;
    MajorityChecker(vector<int>& arr) {
        data = vector<int>(arr.begin(), arr.end());
    }
    
    int query(int l, int r, int t) {
        int cnt[20001] = {};
        for (int i = l; i <= r; ++i) {
            cnt[data[i]]++;
            if (cnt[data[i]] >= t) return data[i];
        }
        return -1;
        
    }
};
