class MajorityChecker {
public:
    unordered_map<int, vector<int>> myMap;
    MajorityChecker(vector<int>& arr) {
        for (int i = 0; i < arr.size(); i++) {
            myMap[arr[i]].push_back(i);
        }
    }
    
    int query(int left, int right, int threshold) {
        int res = -1;
        auto it = myMap.begin();
        for (; it != myMap.end(); it++) {
            vector<int>& temp = it->second;
            int begin = lower_bound(temp.begin(), temp.end(), left) - temp.begin();
            int end = upper_bound(temp.begin(), temp.end(), right) - temp.begin();
            int cnt = end - begin;
            if (cnt >= threshold) {
                res = it->first;
            }
        }
        return res;
    }
};
