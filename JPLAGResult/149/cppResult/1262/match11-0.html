class MajorityChecker {
public:
    vector<int> arr;
    MajorityChecker(vector<int>& arr) {
        this->arr = arr;
    }
    
    int query(int left, int right, int threshold) {
        int cur_cnt = 1, target = arr[left];
        for (int i = left + 1; i <= right; ++i) {
            if (target == arr[i]) ++cur_cnt;
            else {
                if (cur_cnt == 0) {
                    target = arr[i];                   
                } else {
                    --cur_cnt;
                }
            }
        }
        int cnt = 0;
        for (int i = left; i <= right; ++i) {
            if (target == arr[i]) ++cnt;
        }
        if (cnt >= threshold) return target;
        else return -1;
