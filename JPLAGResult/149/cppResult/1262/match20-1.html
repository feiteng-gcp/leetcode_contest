class MajorityChecker {
public:
    MajorityChecker(vector<int>& arr) {
        nums = arr;
    }
    
    int query(int left, int right, int threshold) {
        if (memo.find(make_pair(left, right)) != memo.end()) {
            return memo[make_pair(left, right)].second >= threshold ? memo[make_pair(left, right)].first : -1;
        }
        int major = nums[left], count = 1;

        for (int i = left + 1; i <= right; ++i) {
            if (nums[i] == major) {
                count++;
            } else {
                if (count == 1) {
                    major = nums[i];
                } else {
                    count--;
                }
            }
        }
        count = 0;
        for (int i = left; i <= right; ++i) {
            if (nums[i] == major) {
                count++;
            }
        }
        memo[make_pair(left, right)] = make_pair(major, count);
        return count >= threshold ? major : -1;
    }

private:
    vector<int> nums = {};
    map<pair<int, int>, pair<int, int>> memo = {};
};
