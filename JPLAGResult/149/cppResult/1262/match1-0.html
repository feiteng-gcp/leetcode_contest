class MajorityChecker {
public:
    vector<int> num;
    MajorityChecker(vector<int>& arr) {
      int size = arr.size();
      for(int i = 0; i< size; ++i)
        num.push_back(arr[i]);
    }
    
    int query(int left, int right, int threshold) {
      vector<int> cnt(20001, 0);
      int mcnt = 0;
      int ret = -1;
      for(int i = left; i <= right; ++i) {
        cnt[num[i]]++;
        if(cnt[num[i]] > mcnt) {
          mcnt = cnt[num[i]];
          ret = num[i];
        }
      }
      if(mcnt >= threshold)
        return ret;
      return -1;
    }
};
