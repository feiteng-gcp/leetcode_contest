class MajorityChecker {
 public:
  int sum[305][20005], mul[305][305], cc[305][305];
  int tmp[20005];
  vector<int> a;
  int n, blocks, cnt;
  int i, j;
  int x, y;
  int ans, anspos;
  MajorityChecker(vector<int>& arr) {
    n = arr.size();
    a.resize(n);
    copy(arr.begin(), arr.end(), a.begin());
    blocks = (int)(sqrt(n)) + 1;
    cnt = (n + blocks - 1) / blocks;
    for (i = 1; i <= blocks; ++i) {
      for (j = 0; j < n; ++j) sum[i][j] = sum[i - 1][j];
      for (j = (i - 1) * cnt; j < n && j < i * cnt; ++j) sum[i][arr[j]]++;
    }

    for (int i = 1; i <= blocks; i++) {
      for (int j = i; j <= blocks; j++) {
        cc[i][j] = cc[i][j - 1]; 
        mul[i][j] = mul[i][j - 1];
        for (int k = (j - 1) * cnt; k < n && k < j * cnt; k++) {
          int x = sum[j][arr[k]] - sum[i - 1][arr[k]];
          if (x > cc[i][j] || x == cc[i][j] && mul[i][j] > arr[k]) {
            cc[i][j] = x;
            mul[i][j] = arr[k];
          }
        }
      }
    }
  }

  int query(int L, int R, int threshold) {
    x = L / cnt + 1, y = R / cnt + 1;
    ans = cc[x + 1][y - 1], anspos = mul[x + 1][y - 1];
    for (int i = L; i <= R && i < x * cnt; i++)
      tmp[a[i]] = max(0, sum[y - 1][a[i]] - sum[x][a[i]]);
    for (int i = max(L, (y - 1) * cnt); i <= R; i++)
      tmp[a[i]] = max(0, sum[y - 1][a[i]] - sum[x][a[i]]);
    for (int i = L; i < x * cnt && i <= R; i++) {
      tmp[a[i]]++;
      if (tmp[a[i]] > ans || tmp[a[i]] == ans && a[i] < anspos) {
        ans = tmp[a[i]];
        anspos = a[i];
      }
    }
    if (x != y) {
      for (int i = (y - 1) * cnt; i <= R; i++) {
        tmp[a[i]]++;
        if (tmp[a[i]] > ans || tmp[a[i]] == ans && a[i] < anspos) {
          ans = tmp[a[i]];
          anspos = a[i];
        }
      }
    }
    if (ans >= threshold) {
      return anspos;
    } else {
      return -1;
    }
  }
};
