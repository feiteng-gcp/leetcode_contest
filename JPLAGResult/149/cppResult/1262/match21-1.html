class MajorityChecker {
public:

    map<int, map<int, int> > mp;
    vector<int> v;
    MajorityChecker(vector<int>& arr) {
        v = arr;
        for(int i = 0; i < arr.size(); ++i) {
            mp[ arr[i] ].insert( make_pair(i, mp[arr[i]].size()+1) );
        }

    }
    
    int query(int left, int right, int threshold) {
        for(int i = 0; i < 100; ++i) {
            int x = v[ rand() % (right - left + 1) + left ];
            int h, l;
            if(mp[x].upper_bound(right) != mp[x].begin()) {
                h = (--(mp[x].upper_bound(right)))->second;
            } else {
                continue;
            }
            
            if(mp[x].lower_bound(left) != mp[x].begin()) {
                l = (--(mp[x].lower_bound(left)))->second;
            } else {
                l = 0;
            }
            if(h - l >= threshold) {
                return x;
            }
        }
        return -1;
    }
};

 * MajorityChecker* obj = new MajorityChecker(arr);
 * int param_1 = obj->query(left,right,threshold);
