class MajorityChecker {
 public:
  MajorityChecker(vector<int>& arr) {
    this->arr = arr;
  }

  int query(int left, int right, int threshold) {
    auto iter = max_freq.find({left,right});
    if (iter != max_freq.end() && iter->second.second < threshold) {
      return -1;
    }

    int count = 1;
    int cur = arr[left];
    for (int i = left; i <= right; i++) {
      if (arr[i] == cur) {
        count++;
      } else if (--count == 0) {
        count = 1;
        cur = arr[i];

      }
    }
    count = 0;
    for (int i = left; i <= right; i++) {
      if (arr[i] == cur) {
        count++;
      }
    }
    max_freq[{left, right}] = {cur, count};
    if (count >= threshold) {
      return cur;
    } else {
      return -1;
    }
  }
 private:
  vector<int> arr;
  map<pair<int,int>, pair<int, int>> max_freq;
};
