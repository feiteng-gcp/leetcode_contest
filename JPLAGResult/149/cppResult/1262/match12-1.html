class MajorityChecker {
public:
    vector<int> a;
    int sum[150][20100], xx[200][200], yy[200][200];
    int cnt, siz, len;
    int tmp[20002];
    MajorityChecker(vector<int>& arr) {
        len = arr.size();
        a = arr;
        siz = int(sqrt(len)) + 1;
        cnt = (len + siz - 1) / siz;
        memset(sum, 0, sizeof(sum));
        memset(xx, 0, sizeof(xx));
        memset(yy, 0, sizeof(yy));
        for (int i = 1; i <= siz; i ++) {
            for (int j = 0; j < len; j ++)
                sum[i][j] = sum[i-1][j];

            for (int j = (i - 1) * cnt; j < len && j < i * cnt; j ++)
                sum[i][arr[j]] ++;
        }
        for (int i = 1; i <= siz; i ++) {
            for (int j = i; j <= siz; j ++) {
                xx[i][j] = xx[i][j - 1];
                yy[i][j] = yy[i][j - 1];
                for (int k = (j - 1) * cnt; k < len && k < j * cnt; k ++) {
                    int kk = arr[k];
                    int x = sum[j][arr[k]] - sum[i - 1][kk];

                    if (x > xx[i][j] || x == xx[i][j] && yy[i][j] > kk) {
                        xx[i][j] = x;
                        yy[i][j] = arr[k];
                    }
                }
            }
        }
    }
    
    int query(int left, int right, int threshold) {
        int x = left / cnt + 1, y = right / cnt + 1;
        int ans = xx[x + 1][y - 1], anspos = yy[x + 1][y - 1];
        //bool flag = false;
        memset(tmp, 0, sizeof(tmp));
        for (int i = left; i <= right && i < x * cnt; i ++) {
            tmp[a[i]] = max(0, sum[y - 1][a[i]] - sum[x][a[i]]);
        }

        for (int i = max(left, (y-1) * cnt); i <= right; i ++) 
            tmp[a[i]] = max(0, sum[y - 1][a[i]] - sum[x][a[i]]);
        for (int i = left; i < x*cnt && i <= right; i ++) {
            tmp[a[i]] ++;
            if (tmp[a[i]] > ans) {
                ans = tmp[a[i]];
                anspos = a[i];
            }
        }
        if (x != y) {
            for (int i = (y-1) * cnt; i <= right; i ++) {
                tmp[a[i]] ++;
                if (tmp[a[i]] > ans) {
                    ans = tmp[a[i]];
                    anspos = a[i];
                }
            }
        }
            
        if (ans < threshold)
            return -1;
        else return anspos;
    }
};
