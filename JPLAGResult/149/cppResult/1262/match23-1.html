class MajorityChecker {
public:
    vector<int> v;
    MajorityChecker(vector<int>& arr) {
        v.clear();
        v = arr;
    }
    
    int query(int left, int right, int threshold) {
        int x = v[left];
        int cnt = 1;
        for(int i = left+1; i <= right; ++i){
            if(cnt == 0){
                x = v[i];
                cnt = 1;
            } else {
                if(x == v[i]){
                    ++cnt;
                } else --cnt;
            }
        }
        int ans = 0;
        for(int i = left; i <= right; ++i) {
            ans += (v[i] == x);
        }
        return (ans >= threshold) ? x : -1;
    }
};
