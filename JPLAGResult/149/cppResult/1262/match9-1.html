class MajorityChecker {
public:
    vector<vector<int>> v;
    vector<int> a;
    MajorityChecker(vector<int>& arr) {
        v.resize(20001);
        for (int i = 0; i < arr.size(); ++i) {
            a.push_back(arr[i]);
            v[arr[i]].push_back(i);
        }
    }
    int query(int left, int right, int threshold) {
        unordered_set<int> vis;
        for (int i = left; i <= right - threshold + 1; ++i) {
            if (vis.count(a[i])) continue;
            vis.insert(a[i]);
            int size = v[a[i]].size();
            auto it = lower_bound(v[a[i]].begin(), v[a[i]].end(), i);
            auto it2 = upper_bound(v[a[i]].begin(), v[a[i]].end(), right);
            int dist = distance(it, it2);
            if (dist >= threshold) return a[i];
        }
        return -1;
    }
};
