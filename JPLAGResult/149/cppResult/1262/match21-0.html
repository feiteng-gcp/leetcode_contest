class MajorityChecker {
public:
    vector<pair<int,int>> comps;
    vector<int> origin;
    MajorityChecker(vector<int>& arr) {
        origin = arr;
        for(int i=0; i<arr.size(); ++i) {
            comps.push_back(make_pair(arr[i], i));
        }
        std::sort(comps.begin(), comps.end());
        srand(514514);
    }
    
    int query(int left, int right, int threshold) {
        for(int ti=0; ti<30; ++ti) {
            int pick_idx = left + rand()%(right-left+1);
            int pick = origin[pick_idx];
            auto it_l = std::lower_bound(comps.begin(), comps.end(), make_pair(pick, left));
            auto it_r = std::lower_bound(comps.begin(), comps.end(), make_pair(pick, right+1));
            int occur = std::distance(it_l, it_r);
            if(occur >= threshold) return pick;
        }
        return -1;
    }
};
