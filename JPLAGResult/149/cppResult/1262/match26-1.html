class MajorityChecker {
    unordered_map<int, vector<int>> idxs;

    vector<int> Arr;
    int count(int v, int p) {
        int l = 0, r = idxs[v].size() - 1;
        while (l != r) {
            int m = (l + r + 1) >> 1;
            if (idxs[v][m] > p) {
                r = m - 1;
            } else l = m;
        }
        return l + (idxs[v][l] <= p);
    }
public:
    MajorityChecker(vector<int>& arr) {
        Arr = arr;
        int n = arr.size();
        for (int i = 0; i < n; ++i) idxs[arr[i]].push_back(i);
    }
    
    int query(int left, int right, int threshold) {

        for (int i = 0; i < 30; ++i) {
            int m = (rand() % (right - left + 1)) + left;
            if (count(Arr[m], right) - count(Arr[m], left - 1) >= threshold) return Arr[m];
        }
        return -1;
    }
};
