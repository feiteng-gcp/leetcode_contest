class Solution {
public:
    int dayOfYear(string date) {
        int y = stoi(date.substr(0, 4));
        int m = stoi(date.substr(5,2));
        int d = stoi(date.substr(8));
        
        vector<int> days(13, 30);
        vector<int> M{1, 3, 5, 7, 8, 10, 12};
        for (auto x : M) {
            days[x] = 31;
        }
        if (y % 400 == 0) {
            days[2] = 29;
        } else if (y % 4 == 0 && y % 100 != 0) {
            days[2] = 29;
        } else {
            days[2] = 28;
        }
        
        int res = 0;
        
        for (int i = 1; i < m; ++i) {
            res += days[i];
        }
        res += d;
        
        return res;
    }
};
