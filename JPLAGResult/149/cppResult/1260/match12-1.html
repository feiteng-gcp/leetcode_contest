class Solution {
public:
    int ordinalOfDate(string date) {
        string syear = date.substr(0, 4), smonth = date.substr(5, 2), sday = date.substr(8, 2);
        int iyear = atoi(syear.c_str());
        int im = atoi(smonth.c_str());
        int id = atoi(sday.c_str());

        bool run = ((iyear%4 == 0 && iyear%100 != 0) || iyear%400 == 0);
        int d[13] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        if (run) 
            d[2] = 29;
        int res = id;
        for (int i = 1; i < im; ++i) {
            res += d[i];
        }
        return res;
    }
};
