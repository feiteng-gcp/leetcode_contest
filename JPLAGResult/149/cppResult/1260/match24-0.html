class Solution {
public:
    int dayOfYear(string date) {
        int year = stoi(date.substr(0, 4));
        int month = stoi(date.substr(5, 2));
        int day = stoi(date.substr(8, 2));
        
        vector<int> dInMonth = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        if (isRun(year)) {
            dInMonth[1] = 29;
        }
        
        int days = 0;
        for (int m = 0; m < month - 1; ++m) {
            days += dInMonth[m];
        }
        
        days += day;
        return days;
    }
    
    bool isRun(int year) {
        if (year%100 != 0)
            return year%4 == 0;
        
        return year%400 == 0;
    }
};
