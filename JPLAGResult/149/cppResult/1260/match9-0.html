#include <string>
#include <iostream>

using namespace std;

class Solution {
  public:
    int dayOfYear(string date)
    {
      int year = str2int(date.substr(0, 4));
      int month = str2int(date.substr(5, 2));
      int day = str2int(date.substr(8, 2));

      int days[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
      if ((year % 4 == 0 and year % 100 != 0) or year % 400 == 0) days[2] = 29;
      int answer = 0;
      for (int m = 1; m < month; ++m) answer += days[m];
      return answer + day;
    }

    int str2int(string str)
    {
      auto pos = str.find_first_not_of("0");
      int v = 0;
      for (auto i = pos; i != str.size(); ++i) v = v * 10 + str[i] - '0';
      return v;
    }
};
