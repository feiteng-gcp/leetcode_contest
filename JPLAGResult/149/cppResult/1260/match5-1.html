class Solution {
public:
    int dayOfYear(string d) {
        int y = (d[0] - '0') * 1000 + (d[1] - '0') * 100 + (d[2] - '0') * 10 + d[3] - '0';
        int m = (d[5] - '0') * 10 + d[6] - '0';
        int dd = (d[8] - '0') * 10 + d[9] - '0';

        vector<int> x = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        if (y % 4 == 0 && (y % 100 != 0 || y % 400 == 0)) x[2] = 29;

        int sum = 0;
        for (int i = 1; i < m; ++i) sum += x[i];
        sum += dd;
        return sum;
    }
};
