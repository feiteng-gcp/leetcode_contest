class Solution {
    
  

    
public:
    int ordinalOfDate(string d) {
        
         int days[] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    
        
        int year = (d[0]-'0')*1000 + (d[1]-'0')*100 + (d[2]-'0')*10 + d[3]-'0';
        int mon = (d[5]-'0')*10 + d[6]-'0';
        int day = (d[8]-'0')*10 + d[9]-'0';
        
        
        int ans = day;
        
        for (int i = 0; i < mon-1; i++) {
            ans += days[i];
        }
        
        if (mon <= 2) return ans;
        if (year%400 == 0) return ans+1;
        if (year%100 == 0) return ans;
        if (year%4 == 0) return ans+1;
        return ans;
        
        
        
    }
};
