class Solution {
public:
    int dayOfYear(string date) {
        int y = stoi(date.substr(0,4));
        int mo = stoi(date.substr(5,2));
        int d = stoi(date.substr(8,2));
       
        int m[] = {31,28,31,30,31,30,31,31,30,31,30,31};
        if(y % 400 == 0 || (y % 4 == 0 && y % 100 != 0)) m[1] += 1;
        int ret = 0;
        for(int i = 0; i < mo - 1; i++) {
            ret += m[i];
        }
        ret += d;
        return ret;
    }
};
