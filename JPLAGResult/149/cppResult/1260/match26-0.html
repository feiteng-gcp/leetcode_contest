class Solution {
public:
    int get_num_day(int year, int month) {
        if (month == 1 || month == 3 || month == 5 || month == 7 || month == 8 || month == 10 || month == 12) {
            return 31; 
        } else if (month == 2) {
            if (((year % 100) == 0) && ((year % 400) != 0)) {
                return 28;
            } else if ((year % 4) == 0) {
                return 29;
            } else {
                return 28;
            }
        } else {
            return 30;
        }
    }
    
    int dayOfYear(string date) {
        int year = stoi(date.substr(0, 4));
        int month = stoi(date.substr(5, 2));
        int result = 0;
        for (int i = 1; i < month; ++i) {
            // cout << get_num_day(year, month) << endl;
            result += get_num_day(year, i);
        }
        result += stoi(date.substr(8, 2));
        return result;
    }
};
