class Solution {
public:
    int numberOfDays(int Y, int M) {
        if (M == 1 || M == 3 || M == 5 || M == 7 || M == 8 || M == 10 || M == 12) {
            return 31;
        }
        else if (M == 4 || M == 6 || M == 9 || M == 11) {
            return 30;
        }
        // M == 2 
        else {
            if (Y % 4 != 0) {
                return 28;
            }

            else if (Y % 100 != 0) {
                return 29;
            }
            else if (Y % 400 != 0) {
                return 28;
            }
            else {
                return 29;
            }
        }
    }
    int dayOfYear(string date) {

        int Y = stoi(date.substr(0, 4)), M = stoi(date.substr(5, 2)), D = stoi(date.substr(8, 2));
        int sum = 0;
        for (int i = 1; i < M; i++) {
            sum += numberOfDays(Y, i);
        }
        sum += D;
        return sum;
    }
};
