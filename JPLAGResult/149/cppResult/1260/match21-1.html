class Solution {
public:
    int dayOfYear(string date) {
       // date = "2000-03-01";
        int y = atoi(date.substr(0, 4).c_str());
        int m = atoi(date.substr(5, 2).c_str());
        int d = atoi(date.substr(8, 2).c_str());
        
        int md[] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 12};
        
        bool isL = false;
        if (y % 4 == 0) {
            if (y % 100 == 0) {
                if (y % 400 == 0) {
                    isL = true;
                }
            } else {
                isL = true;
            }

        }
        if (isL) md[1]++;
        
        int c = 0;
        for (int i = 0; i < m - 1; ++i)
            c += md[i];
        
        c += d;
        
        return c;
    }
};
