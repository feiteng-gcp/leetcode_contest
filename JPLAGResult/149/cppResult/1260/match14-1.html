class Solution {
public:
    int ordinalOfDate(string date) {
        int y, m ,d;
        y = (date[0] -'0') * 1000 + (date[1]-'0')*100+(date[2]-'0')*10+(date[3]-'0');
        m = (date[5]-'0')*10+(date[6]-'0');
        d = (date[8]-'0')*10+(date[9]-'0');
        int days[12] = {31,28,31,30,31,30,31,31,30,31,30,31};
        if (y % 400 == 0 || (y % 4 == 0 && y % 100 != 0)) days[1]++;
        int res = 0;
        for (int i = 0; i < m-1; i++)
            res += days[i];
        res+= d;
        return res;
    }
};
