class Solution {
public:
    int ordinalOfDate(string date) {
        int year = 0,month = 0,day = 0;
        for(int i=0;i<4;i++)year = year*10+date[i] - '0';
        for(int i=5;i<7;i++)month = month*10+date[i] - '0';
        for(int i=8;i<10;i++)day = day*10+date[i] - '0';
        int days[13] = { 0,31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
        if((year%4==0&&year%100!=0)||year%400==0)days[2]++;
        int ans = 0;
        for(int i=1;i<month;i++)ans+=days[i];
        ans+=day;
        return ans;
    }
};
