class Solution {
public:
    int dayOfYear(string date) {
        int year = stoi(date.substr(0, 4));
        int month = stoi(date.substr(5, 2));
        int day = stoi(date.substr(8, 2));
        //cout<<day<<" "<<year<<endl;
        vector<int> months = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        if(year%100 == 0) {
            if(year%400 == 0)
                months[1] = 29;
        } else{
            if(year%4==0)
                months[1] = 29;
        }
        int res = 0;
        for(int i=0; i<month-1; i++) {
            res+=months[i];
        }
        res+=day;
        return res;
    }
};
