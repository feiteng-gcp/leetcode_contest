class Solution {
public:
    int dayOfYear(string date) {
        int yy=(date[0]-'0')*1000+(date[1]-'0')*100+(date[2]-'0')*10+(date[3]-'0');
        int mm=(date[5]-'0')*10+(date[6]-'0');
        int dd=(date[8]-'0')*10+(date[9]-'0');
        map<int,int> mpp;mpp.clear();
        mpp[1]=31;mpp[2]=28;mpp[3]=31;mpp[4]=30;mpp[5]=31;mpp[6]=30;mpp[7]=31;mpp[8]=31;mpp[9]=30;mpp[10]=31;mpp[11]=30;mpp[12]=31;
        if(yy%400==0)
            mpp[2]++;

        else if(yy%100 && yy%4==0)
            mpp[2]++;
        int sum=0;
        for(int i=1;i<mm;i++)
            sum=sum+mpp[i];
        
        sum=sum+dd;
        return sum;
    }
};
