class Solution {
    public int maxRepOpt1(String text) {
        int res = 0, n = text.length();
        for(char c='a';c<='z';c++){
            int f1 = 0, start = 0, end = 0, curD = 0, max = 0, curS = 0, curE = 0;
            while(end<n){
                char t = text.charAt(end);
                if(t!=c) curD++;
                while(curD>1){
                    char p = text.charAt(start++);
                    if(p!=c) curD--;
                }
                if(max<end-start+1){
                    max = end-start+1;
                    f1 = curD;
                    curS = start;
                    curE = end;
                }
                end++;
            }
            if(f1!=0){
                int f = 0;
                for(int i=0;i<n;i++){
                    if(text.charAt(i)==c && (i<curS||i>curE)){
                        f = 1;
                        break;
                    }
                }
                if(f!=1){
                    max--;
                }
            }
            res = Math.max(res,max);
        }
        return res;
    }
}
