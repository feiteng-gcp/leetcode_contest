class Solution {
    public int maxRepOpt1(String text) {
        List<int[]>[] list=new List[26];
        for(int i=0;i<26;i++){
            list[i]=new ArrayList<>();
        }
        char[] tx=text.toCharArray();
        int i,last,max=-1;
        for(i=1,last=0;i<tx.length;i++){
            if(tx[i]!=tx[i-1]){
                list[tx[last]-'a'].add(new int[]{last,i});
                last=i;
            }
        }
        list[tx[last]-'a'].add(new int[]{last,i});
        /*for( i=0;i<26;i++){
            for(int[] a:list[i]){
                System.out.print(Arrays.toString(a));
            }
            System.out.println();
        }*/
        for( i=0;i<26;i++){
            int t=-1;
            if(list[i].size()>0){
                t=list[i].get(0)[1]-list[i].get(0)[0];
            }
             if(list[i].size()>1){
                t++;
            }
            for(int j=1;j<list[i].size();j++){
                if(list[i].get(j)[0]==list[i].get(j-1)[1]+1){
                    if(list[i].size()>2){
                        t=Math.max(t,list[i].get(j)[1]-list[i].get(j-1)[0]);
                    }else{
                        t=Math.max(t,list[i].get(j)[1]-list[i].get(j-1)[0]-1);
                    }
                }else{
                    t=Math.max(t,list[i].get(j)[1]-list[i].get(j)[0]+1);
                }
            }
            max=Math.max(t,max);
        }
        return max;
    }
}
