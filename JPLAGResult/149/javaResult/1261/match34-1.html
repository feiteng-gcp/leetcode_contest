    class Parts {
        int rep;
        int intvarl;

        Parts(int r, int i) {
            this.rep = r;
            this.intvarl = i;
        }

        @Override
        public String toString() {
            return "Parts{" +
                    "rep=" + rep +
                    ", intvarl=" + intvarl +
                    '}';
        }
    }

    class Solution {
        
public int maxRepOpt1(String text) {
            int res = 0;
            for (int i = 0; i < 26; i++) {
                res = Math.max(res, findMaxRep(text, (char)('a' + i)));
            }
            return res;
        }

        private int findMaxRep(String text, char origin) {
            List<Parts> ps = build(text, origin);
            if (ps.size() == 0) return 0;
            if (ps.size() == 1) return ps.get(0).rep;
            if (ps.size() == 2) {
                if (ps.get(0).intvarl == 1) {
                    return ps.get(0).rep + ps.get(1).rep;
                } else {
                    return Math.max(ps.get(0).rep, ps.get(1).rep) + 1;
                }
            }
            int res = 0;
            for (int i = 0; i < ps.size() - 1; i++) {
                Parts p1 = ps.get(i);
                Parts p2 = ps.get(i + 1);
                if (p1.intvarl == 1) {
                    int len = p1.rep + p2.rep + 1;
                    res = Math.max(res, len);
                    res = Math.max(res, p2.rep + 1);
                } else {
                    res = Math.max(res, p1.rep + 1);
                    res = Math.max(res, p2.rep + 1);
                }
            }
            return res;
        }

        static List<Parts> build(String text, char origin) {
            int n = text.length();
            int curRep = 0;
            int curInt = 0;
            List<Parts> res = new ArrayList<>();
            for (int i = 0; i < n; i++) {
                char c = text.charAt(i);
                if (c == origin) {
                    if (curInt != 0) {
                        if (curRep != 0) res.add(new Parts(curRep, curInt));
                        curInt = 0;
                        curRep = 1;
                    } else {
                        curRep ++;
                    }
                } else {
                    curInt ++;
                }
            }
            if (curRep != 0) {
                res.add(new Parts(curRep, curInt));
            }
            return res;
        }
    }
