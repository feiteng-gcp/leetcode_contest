class Solution {
    public int maxRepOpt1(String text) {
        HashMap<Character, ArrayList<int[]>> map = new HashMap<Character, ArrayList<int[]>>();
        int count = 0;
        for(int i = 0; i < text.length(); i++){
            if(i == 0) {
                count = 1;
            }else{
                if(text.charAt(i) == text.charAt(i-1)){
                    count++;
                }else{
                    if(!map.containsKey(text.charAt(i-1))){
                        map.put(text.charAt(i-1), new ArrayList<int[]>());
                    }
                    ArrayList<int[]> list = map.get(text.charAt(i-1));
                    list.add(new int[]{i-count, i-1});
                    count = 1;
                }
          
            }
            
            if(i == text.length()-1) {
                if(!map.containsKey(text.charAt(i))){
                    map.put(text.charAt(i), new ArrayList<int[]>());
                }
                ArrayList<int[]> list = map.get(text.charAt(i));
                list.add(new int[]{i+1-count, i});
            }
            
        }
        
//         System.out.println(map.get('a').size());
//         System.out.println(map.get('b').size());
//         System.out.println(map.get('a').get(0)[0]);
//         System.out.println(map.get('a').get(0)[1]);
        
//         System.out.println(map.get('a').get(1)[0]);
//         System.out.println(map.get('a').get(1)[1]);
//         System.out.println(map.get('a').get(2)[0]);
//         System.out.println(map.get('a').get(2)[1]);
        
        // System.out.println(map.get('b').get(0)[0]);
        // System.out.println(map.get('b').get(0)[1]);
//         System.out.println(map.get('b').get(1)[0]);
//         System.out.println(map.get('b').get(1)[1]);

        
        
int result = 1;
        for(Character ch: map.keySet()){
            ArrayList<int[]> list = map.get(ch);
            for(int i = 0; i < list.size(); i++) {
                int[] haha = list.get(i);
                int start = haha[0];
                int end = haha[1];
                if(list.size() >= 2) {
                    result = Math.max(result, end-start+2);
                }else{
                    result = Math.max(result, end-start+1);
                }
                
                
                
if(i != 0 && list.get(i-1)[1] + 2 == list.get(i)[0]){
                    if(list.size() > 2){
                         result = Math.max(result, end-start+1 + 1 + list.get(i-1)[1]-list.get(i-1)[0]+1);
                    }else{
                         result = Math.max(result, end-start+1 + list.get(i-1)[1]-list.get(i-1)[0]+1);
                    }
                }
            }
        }
        return result;
    }
}
