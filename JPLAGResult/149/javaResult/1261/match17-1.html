class Solution {
    public int maxRepOpt1(String text) {
        int count=0;
        int len=text.length();
        for(int i=0;i<len;i++){
            char c=text.charAt(i);
            int j=1;
            int flag=0;
            while(j+i<len&&text.charAt(j+i)==c) j++;
            if(j+i<len){
                if(text.substring(j+i).indexOf(c)>=0) {j++;flag=1;}
            }
            while(j+i<len&&text.charAt(j+i)==c) j++;
            if(j+i<len){
                if(text.substring(j+i).indexOf(c)>=0) j++;
            }
            count=Math.max(j-flag,count);
        }
        for(int i=len-1;i>=0;i--){
            char c=text.charAt(i);
            int j=1;
            int flag=0;
            while(i-j>=0&&text.charAt(i-j)==c) j++;
            if(i-j>=0){
                if(text.substring(0,i-j+1).indexOf(c)>=0) {j++;flag=1;}
            }
            while(i-j>=0&&text.charAt(i-j)==c) j++;
            if(i-j>=0){
                if(text.substring(0,i-j+1).indexOf(c)>=0) j++
;
            }
            count=Math.max(j-flag,count);
        }
        return count;
    }
}
