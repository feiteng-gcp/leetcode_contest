class Solution {
    public int maxRepOpt1(String text) {
        
        int[] freq = new int[26];
        for (int i = 0; i < text.length(); i++)
            freq[(int)(text.charAt(i)-'a')]++;
        
        HashMap<Character, Integer> map = new HashMap<>();
        int pt = 0;
        int cnt = 1;
        for (int i = 0; i < text.length(); i++) {
            char c = text.charAt(i);
            if (map.containsKey(c)) map.put(c, map.get(c)+1);
            else map.put(c, 1);
            
            while (map.size() == 3) {
                int f = map.get(text.charAt(pt))-1;
                if (f > 0) map.put(text.charAt(pt), f);
                else map.remove(text.charAt(pt));
                pt++;
            }
            
            if (map.size() == 1) {
                for (Character key : map.keySet()) {
                    cnt = Math.max(cnt, map.get(key));
                }
            }
            else {
                boolean check = false;
                int nums = 1;
                char ch = 'a';
                for (Character key : map.keySet()) {
                    if (map.get(key) == 1)
                        check = true;
                    if (map.get(key) > nums) {
                        nums = map.get(key);
                        ch = key;
                    }
                }
                if (check) {
                    if (freq[(int)(ch-'a')] > nums) nums++;
                    cnt = Math.max(cnt, nums);
                    continue;
                }
                while (!check && pt < i) {
                    char key = text.charAt(pt);
                    pt++;
                    map.put(key, map.get(key)-1);
                    if (map.get(key) == 1)
                        check = true;
                }
                for (Character key : map.keySet()) {
                    if (map.get(key) == 1)
                        check = true;
                    if (map.get(key) > nums) {
                        nums = map.get(key);
                        ch = key;
                    }
                }
                if (check) {
                    if (freq[(int)(ch-'a')] > nums) nums++;
                    cnt = Math.max(cnt, nums);
                }
            }
            
        }
        return cnt;
    }
}
