class Solution {
    public int maxRepOpt1(String text) {
        
int[] cnt = new int[26];
        for (char c : text.toCharArray()) cnt[c - 'a']++;
        
        int res = 0;
        for (char c = 'a'; c <= 'z'; c++) {
            int pre = 0, cur = 0;
            for (char ch : text.toCharArray()) {
                if (ch == c) cur++;
                else {
                    pre = cur;
                    cur = 0;
                }
                int temp = cur + pre;
                res = Math.max(res, temp == cnt[c - 'a'] ? temp : temp + 1);
            }
        }
        return res;
    }
}
