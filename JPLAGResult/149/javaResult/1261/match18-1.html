class Solution {
    public int maxRepOpt1(String text) {
    	int ans = 0; 	
    	for
(int i = 0; i < 26; i++) {
    		char target = (char)('a' + i);
    		int combo = 0;
    		int maxCombo = 0;
    		int comboStart = -1;
    		int firstOcc = -1;
    		int lastOcc = -1;
    		int mark = 0;
    		boolean rollback = false;
    		for(int j = 0; j < text.length(); j++) {

    			if(text.charAt(j) == target) {
    				firstOcc = j;
    				break;
    			}
    		}
    		if(firstOcc == -1) continue;
    		for(int j = text.length() - 1; j >= 0; j--) {
    			if(text.charAt(j) == target) {
    				lastOcc = j;
    				break;
    			}
    		}
    		
    		//System.out.println(firstOcc + ", " + lastOcc);
    		
    		for(int j = 0; j < text.length(); j++) {
    			rollback = false;
        		if(text.charAt(j) == target) {
        			if(firstOcc == -1) firstOcc = j;
        			if(combo == 0) comboStart = j;
        			combo++;
        		}
        		else {
        			if(combo == 0) continue;
        			if(j+1 < text.length() && text.charAt(j+1) == target) {
        				mark = j;
        				rollback = true;
        				j++;
        				while(j < text.length() && text.charAt(j) == target) {
        					combo++;
        					j++;
        				}
        			}
        			//System.out.println("this combo = " + combo + " .. combostart = " + comboStart + ", j = " + j);
        			if(comboStart != firstOcc || (j-1) != lastOcc) combo++;
        			maxCombo = Math.max(maxCombo, combo);
        			//System.out.println("maxCombo = " + maxCombo);
        			combo = 0;
        			if(rollback) j = mark;
        		}
        	}
    		//System.out.println("combostart = " + comboStart);
    		if(comboStart != firstOcc || text.length() - 1 != lastOcc) combo++;
			maxCombo = Math.max(maxCombo, combo);
    		ans = Math.max(ans, maxCombo);
    		
    	}
    	return ans;
    }
}
