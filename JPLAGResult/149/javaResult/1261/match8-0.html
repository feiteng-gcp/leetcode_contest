class Solution {
	    public int maxRepOpt1(String text) {
	        //26
	    	char str[] = text.toCharArray();
	    	int len = str.length;
	    	LinkedList<Integer> list = new LinkedList<>();
	    	int rs = 1;
	    	int record [] = new int [26];
	    	for(int i=0;i<len;i++)
	    	{
	    		record[str[i]-'a'] ++;
	    	}
	    	for(int i=0;i<26;i++)
	    	{
	    		char target = (char) (i + 'a');
	    		int key = i;
	    		int opp = 1;
	    		for(int j=0;j<len;j++)
	    		{
	    			if(str[j]==target)
	    			{
	    				if(record[key]>0){
	    					list.add(1);
	    					record[key] --;
	    				}else{
	    					break;
	    				}
	    			}else{
	    				if(opp==1)
	    				{
	    					if(record[key]>=1)
	    					{
	    						opp=0;
	    						list.add(0);
	    						record[key] --;
	    					}else{
	    						break;
	    					}
	    				}else{
	    					if(record[key]>=1)
	    					{
	    						int cc = 0;
	    						while(list.get(0)!=0)
		    					{
	    							cc++;
		    						list.remove(0);
		    					}
	    						record[key] += cc;
	    						list.remove(0);
	    						list.add(0);
	    					}else{
	    						break;
	    					}
	    				}
	    			}
	    			rs = rs > list.size()? rs : list.size();
	    		}
	    		list.clear();
	    	}
	    	return rs;
	    }
	}
