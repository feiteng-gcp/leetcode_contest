class Solution {
    public int maxRepOpt1(String text) {
        int[] count = new int[26];
        for (int i = 0; i < text.length(); i++) {
            count[text.charAt(i) - 'a']++;
        }
        int res = 0;
        for (int i = 0; i < text.length(); i++) {
            res = Math.max(res, find(count, i, text));
            while (i < text.length() - 1 && text.charAt(i + 1) == text.charAt(i)) {
                i++;
            }
        }
        return res;
    }

    public int find(int[] count, int i, String text) {
        boolean flag = true;
        int res = 0;
        for (int j = i; j < text.length(); j++) {
            if (text.charAt(j) == text.charAt(i)) res++;
            else if (flag && res < count[text.charAt(i) - 'a']) {
                res++;
                flag = false;
            } else
                break;
        }
        return Math.min(res, count[text.charAt(i) - 'a']);
    }

}
