class Solution {
    public int maxRepOpt1(String text) {
        char[] arr = text.toCharArray();
        int max = 0;
        int[] longest = new int[arr.length];
        
int[] cnt = new int[256];
        for(char c : arr){
            cnt[c]+=1;
        }
        for(int i = 0; i < arr.length; i++){
           // int cnt =1;
            if( i > 0 && arr[i] == arr[i-1]){
                longest[i] = longest[i-1] +1;
            }else{
                longest[i] =1;
            }
            int cur = longest[i];
            int pre = i - cur;
            if(pre == 0 && cnt[arr[i]] > cur  ){
                cur++;
            }else  
if(pre > 0 && arr[i] == arr[pre-1] ){
                cur = Math.min(longest[i] + 1 + longest[pre-1], cnt[arr[i]]);
            }else{
                cur = Math.min(cnt[arr[i]],longest[i] + 1 );
            }
            max = Math.max(cur, max);
        }
        return max;
    }
}
