class Solution {
    public int maxRepOpt1(String text) {
        int res = 1;
        for (char ch = 'a'; ch <= 'z'; ch++) {
            List<int[]> list = new ArrayList<>();
            int last = -1;
            for (int i= 0; i < text.length(); i++) {
                if (text.charAt(i) == ch) {
                    if (last == -1) last = i;
                    continue;
                } else {
                    if (last != -1) {
                        list.add(new int[] {last, i-1});
                        last = -1;
                    }
                }
            }
            if (last != -1) {
                list.add(new int[] {last, text.length()-1});
            }
            
            for (int i= 0; i < list.size(); i++) {
                int t = list.get(i)[1] - list.get(i)[0] + 1;
                if (list.size() > 1) t ++;
                res = Math.max(res, t);
                if (i != list.size() - 1) {
                    if (list.get(i)[1] + 2 == list.get(i+1)[0]) {
                        int tmp = list.get(i)[1] - list.get(i)[0] + 1 + list.get(i+1)[1] - list.get(i+1)[0] + 1;
                        if (list.size() >= 3) tmp++;
                        res = Math.max(res, tmp);
                    }
                }
            }
        }
        return res;
    }
}
