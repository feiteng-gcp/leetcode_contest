class Solution {
    public int maxRepOpt1(String text) {
        int[] map = new int[26];
        for(int i=0; i<text.length(); i++){
            int idx = text.charAt(i)-'a';
            map[idx]++;
        }
        int[] dp = new int[26];
        int res = 0;
        for(int l=0, r=0; r<text.length(); r++){
            int idx = text.charAt(r)-'a';
            dp[idx]+=1;
            if(r-l<=dp[idx] && map[idx]-dp[idx]>=r-l+1-dp[idx]) res = Math.max(res, r-l+1);
            while(dp[text.charAt(l)-'a']<r-l+((map[text.charAt(l)-'a']-dp[text.charAt(l)-'a']>0)? 0: 1) 
                  && dp[idx]<r-l+((map[idx]-dp[idx])>0? 0: 1)) {
                dp[text.charAt(l++)-'a']--;
            }
        }
        return res;
    }
}
