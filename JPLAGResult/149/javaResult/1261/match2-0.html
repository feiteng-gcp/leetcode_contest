class Solution {
    public int maxRepOpt1(String text) {
        char cs[] = text.toCharArray();
        
        int l = cs.length;
        int r = 0;
        
        int z[] = new int[26];
        
        for(char c:cs){
            z[c-'a']++;
        }
        
        for(char i = 'a';i<='z';++i){
            
            int d[] = new int[l+1];
            
            int s = 0;
            int st = 0;
            for(int e=0;e<l;++e){
               if(cs[e]==i){
                   s++;
               }
               while(e-st+1-s>1){
                   if(cs[st]==i){
                       s--;
                   }
                   st++;
               } 
                
               r = Math.max(r,Math.min(e-st+1,z[i-'a'])); 
                
            }
            
            
            
            
        }
        return r;
        
    }
}
