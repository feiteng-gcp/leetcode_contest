class Solution {
    public int maxRepOpt1(String text) {
        int[] count = new int[26];
        int[][] front = new int[text.length()][26];
        int[][] end = new int[text.length()][26];
        for (int i = 0; i < text.length(); ++i) {
            char c = text.charAt(i);
            count[c - 'a']++;
            for (int j = 0; j < 26; j++) {
                if (c == 'a' + j) {
                    if (i-1 >= 0) 
                        front[i][j] = front[i-1][j] + 1;
                    else
                        front[i][j] = 1;
                } else {
                    front[i][j] = 0;
                }
            }
        }
        for (int i = text.length() - 1; i >= 0; --i) {
            char c = text.charAt(i);
            for (int j = 0; j < 26; j++) {
                if (c == 'a' + j) {
                    if (i+1 < text.length()) 
                        end[i][j] = end[i+1][j] + 1;
                    else
                        end[i][j] = 1;
                } else {
                    end[i][j] = 0;
                }
            }
        }
        int ret = 1;
        for (int j = 0; j < 26; ++j) {
            ret = Math.max(ret, end[0][j]);
            ret = Math.max(ret, front[text.length()-1][j]);
        }
        for (int i = 1; i < text.length()-1; ++i) {
            for (int j = 0; j < 26; ++j) {
                int sum = front[i-1][j] + end[i+1][j];
                if (sum < count[j]) {
                    ret = Math.max(ret, sum+1);
                } else if (sum == count[j]) {
                    ret = Math.max(ret, sum);
                }
            }
        }
        return ret;
    }
}
