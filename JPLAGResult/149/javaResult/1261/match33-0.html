class Solution {
    class Seg {
        int s;
        int e;
        char v;

        Seg(int s, int e, char v) {
            this.s = s;
            this.e = e;
            this.v = v;
        }
    }

    public int maxRepOpt1(String text) {
        int N = text.length();
        Map<Character, List<Seg>> segs = new HashMap<>();
        int s = 0;
        for (int i = 0; i < N; i++) {
            if (text.charAt(i) != text.charAt(s)) {
                char c = text.charAt(s);
                segs.putIfAbsent(c, new ArrayList<>());
                segs.get(c).add(new Seg(s, i - 1, c));
                s = i;
            }
        }
        char c = text.charAt(s);
        segs.putIfAbsent(text.charAt(s), new ArrayList<>());
        segs.get(c).add(new Seg(s, N - 1, c));

        int res = 0;
        for (Map.Entry<Character, List<Seg>> e : segs.entrySet()) {
            List<Seg> list = e.getValue();
            for (int i = 0; i < list.size(); i++) {
                res = Integer.max(res, list.get(i).e - list.get(i).s + 1);
                if (list.size() > 1) {
                    res = Integer.max(res, list.get(i).e - list.get(i).s + 2);
                }
                if (i < list.size() - 1) {
                    if (list.get(i).e + 2 == list.get(i + 1).s) {
                        if (list.size() == 2) {
                            res = Integer.max(res, list.get(i + 1).e - list.get(i).s);

                        } else {
                            res = Integer.max(res, list.get(i + 1).e - list.get(i).s + 1);
                        }
                    }

                }
            }
        }
        return res;
    }
}
