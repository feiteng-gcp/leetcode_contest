class Solution {
    public int maxRepOpt1(String text) {
        int len = text.length();
        int[] first = new int[26]; Arrays.fill(first, -1);
        int[] last = new int[26]; Arrays.fill(last, -1);
        int[] leftC = new int[len], rightC = new int[len];
        for (int i = 0; i < len; ++i){
            int c = text.charAt(i) - 'a';
            if (first[c] == -1){
                first[c] = i;
            }
            if (i == 0 || 
text.charAt(i - 1) - 'a' != c){
                leftC[i] = 1;
            } else {
                leftC[i] = leftC[i - 1] + 1;
            }
        }
        for (int i = len - 1; i >= 0; --i){
            int c = text.charAt(i) - 'a';
            if (last[c] == -1){
                last[c] = i;
            }
            if (i == len - 1 || 
text.charAt(i + 1) - 'a' != c){
                rightC[i] = 1;
            } else {
                rightC[i] = rightC[i + 1] + 1;
            }
        }
        int max = 1;
        for (int i = 0; i < len; ++i){
            if (rightC[i] != 1){
                continue;
            } else {
                int t = leftC[i];
                if (last[text.charAt(i) - 'a'] != i || first[text.charAt(i) - 'a'] != (i - leftC[i] + 1)) {
                    t += 1;
                    if (i + 2 < len && text.charAt(i + 2) == text.charAt(i)) {
                        t += rightC[i + 2];
                        if (i + 1 + rightC[i + 2] == last[text.charAt(i) - 'a'] && first[text.charAt(i) - 'a'] == (i - leftC[i] + 1)){
                            t -= 1;
                        }
                    }
                }
                max = 
Math.max(t, max);
            }
        }
        return max;
    }
}
