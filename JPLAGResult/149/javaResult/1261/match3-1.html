class Solution {
    public int maxRepOpt1(String text) {
        int max = 0;
        for (char c = 'a';c <= 'z';c++){
            List<int[]> l = new ArrayList<>();
            int start = -1;
            for (int i = 0; i < text.length(); i++) {
                if(text.charAt(i) != c){
                    if(start >= 0)
{
                        l.add(new int[]{start,i - 1});
                        start = -1;
                    }
                }else{
                    if(start < 0) start = i;
                }
            }
            if(start >= 0) l.add(new int[]{start,text.length() - 1});

            for(int i = 0;i<l.size();i++){
                int[] seg = l.get(i);
                int len = seg[1] - seg[0] + 1;
                if(l.size() > 1) len++;
                max = Math.max(max,len);
                if(i > 0 && seg[0] == 
l.get(i - 1)[1] + 2){
                    //System.out.println("i = " + i + " cons");
                    int cons = seg[1] - seg[0] + 1 + l.get(i - 1)[1] - l.get(i - 1)[0] + 1;
                    if(l.size() > 2) cons++;
                    max = Math.max(max,cons);
                
}
            }
            //System.out.println("c = " + c + " max = " + max + " l = " + l);
        }
        return max;
    }
}
