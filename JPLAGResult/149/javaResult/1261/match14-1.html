class Solution {
    private static final int MOD = 1_000_000_007;

    
public int maxRepOpt1(String text) {
        int ans = 0;
        for (int i = 0; i < 26; i++) {
            ans = Math.max(ans, helper(text, (char) (i + 'a')));
        }
        return ans;
    }

    private int helper(String text, char target) {
        int total = 0;
        for (int i = 0; i < text.length(); i++) {
            if (text.charAt(i) == target) {
                total++;
            }
        }

        int count = 0;
        int maxLen = 0;

        for (int i = 0, j = 0; i < text.length(); i++) {
            if (text.charAt(i) == target) {
                count++;
            }

            while (i - j + 1 > count + 1) {
                if (text.charAt(j) == target) count--;
                j++;
            }

            maxLen = Math.max(maxLen, i - j + 1);
        }

        if (maxLen == total || maxLen == total + 1) return total;
        else return maxLen;
    }
}
