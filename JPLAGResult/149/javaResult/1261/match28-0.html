class Solution {
    public int maxRepOpt1(String s) {
        char[] a = s.toCharArray();
        int[] l2r = l2r(a);
        int[] r2l = r2l(a);
        int[] count = new int[26];
        for (char c : a) count[c-'a']++;

        int n = a.length;


        int res = 0;
        for (int i = 0; i < n; i++) {
            char left = i == 0 ? '*' : a[i-1];
            char right = i == n-1 ? '*' : a[i+1];
            int countLeft = i == 0 ? 0 : l2r[i-1];
            int countRight = i == n-1 ? 0 : r2l[i+1];
            if (left == right) {
                int totalCount = left == '*' ? 0 : count[left-'a'];
                if (totalCount == countLeft + countRight) {
                    res = Math.max(res, countLeft + countRight);
                } else { // totalCount > countLeft + countRight;
                    res = Math.max(res, countLeft + countRight + 1);
                }
            } else {
                int maxLeft = left == '*' ? 0 : (countLeft + (count[left-'a'] > countLeft ? 1 : 0));
                int maxRight = right == '*' ? 0 : (countRight + (count[right-'a'] > countRight ? 1 : 0));
                res = Math.max(res, Math.max(maxLeft, maxRight));
            }
        }
        return res;
    }

    private int[] l2r(char[] a) {
        int n = a.length;
        int[] res = new int[n];
        int i = 0;
        while (i < n) {
            int j = i;
            while (j + 1 < n && a[j+1] == a[i]) j++;
            for (int k = i; k <= j; k++) res[k] = k-i+1;
            i = j + 1;
        }
        return res;
    }

    private int[] r2l(char[] a) {
        int n = a.length;
        int[] res = new int[n];
        int i = n-1;
        while (i >= 0) {
            int j = i;
            while (j -1 >= 0 && a[j-1] == a[i]) j--;
            for (int k = j; k <= i; k++) res[k] = i-k+1;
            i = j - 1;
        }
        return res;
    }
  
}
