class Solution {
    public int maxRepOpt1(String text) {
        int[] ct = new int[26];
        for (char c : text.toCharArray()) ct[c-'a']++;
        int sol = 0;
        for (int i = 0; i < 26; i++) {
            if (ct[i] == 0) continue;
            char c = (char)('a' + i);
            int maxLen = len(text, c);
            //System.out.println(c);
            //System.out.println(maxLen);
            if (maxLen < ct[i]) maxLen++;
            sol = Math.max(sol, maxLen);
        }
        return sol;
    }
    
    public int len(String s, char c) {
        int lastLen = 0;
        
int curLen =0;
        int st = 0;
        int sol = 0;
        for (int i = 0; i <= s.length(); i++) {
            if (i == s.length() || s.charAt(i) != c) {
                sol = Math.max(sol, curLen + lastLen);
                lastLen = curLen;
                curLen = 0;
            } else {
                curLen++;
            }
        }
        return sol;
    }
}
