	class Solution {
	    public int maxRepOpt1(String text) {
	        char[] s = text.toCharArray();
	        int n = s.length;
	        int[] cs = new int[n];
	        int[] fs = new int[n];
	        int p = 0;
	        for(int i = 0;i < n;i++){
	        	if(i == 0 || s[i] != s[i-1]){
	        		cs[p] = s[i];
	        		fs[p] = 1;
	        		p++;
	        	}else{
	        		fs[p-1]++;
	        	}
	        }
	        
	        cs = Arrays.copyOf(cs, p);
	        fs = Arrays.copyOf(fs, p);
	        
	        int ret = 0;
	        for(int f : fs)ret = Math.max(ret, f);
	        
	        int[] allf = new int[128];
	        for(int i = 0;i < p;i++){
	        	allf[cs[i]] += fs[i];
	        }
	        
	        for(int i = 0;i+2 < p;i++){
	        	if(cs[i] == cs[i+2] && fs[i+1] == 1){
	        		if(allf[cs[i]] == fs[i] + fs[i+2]){
	        			ret = Math.max(ret, fs[i] + fs[i+2]);
	        		}else{
	        			ret = Math.max(ret, fs[i] + fs[i+2] + 1);
	        		}
	        	}
	        }
	        for(int i = 0;i < p;i++){
	        	if(allf[cs[i]] != fs[i]){
	        		ret = Math.max(ret, fs[i] + 1);
	        	}
	        }
	        
	        return ret;
	    }
	}
