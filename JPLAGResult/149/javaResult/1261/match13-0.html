class Solution {
    public int maxRepOpt1(String text) {
        int n = text.length();
        int[] count = new int[256];
        for(int i=0; i<n; i++)
            count[text.charAt(i)]++;
        int res = 0;
        for(int i=0; i<n; ){
            int ct = 0;
            int j = i;
            int p =-1;
            while(j<n&&(text.charAt(j)==text.charAt(i)||ct==0)){
                if(text.charAt(j)!=text.charAt(i)){
                    p = j;
                    ct++;
                }
                j++;
            }
            if(ct==0)
                res = Math.max(res, j-i);
            else{
                int temp = j-i-1;
                if(count[text.charAt(i)]>temp) temp++;
                res = Math.max(res, temp);
            }
            if(p==-1)
                i = j;
            else
                i = p; 
        }
        return res;
    }
}
