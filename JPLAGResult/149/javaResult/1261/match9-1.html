class Solution {
    public int maxRepOpt1(String s) {
        int[] minPos = new int[26];
        int[] maxPos = new int[26];
        Arrays.fill(minPos, -1);
        Arrays.fill(maxPos, -1);
        
        for (int i = 0; i < s.length(); ++i) {
            if (minPos[s.charAt(i)-'a'] == -1) minPos[s.charAt(i)-'a'] = i;
            maxPos[s.charAt(i)-'a'] = i;
        }
        int maxLen = 0;
        for (char uch = 'a'; uch <= 'z'; ++uch) {
            int[] cnt = new int[26];
            
int l = 0, len = 0, unique = 0, nonUch = 0;
            for (int j = 0; j < s.length(); ++j) {
                int ch = s.charAt(j);
                if (ch != uch) ++nonUch;
                while (nonUch > 1) {
                    if (s.charAt(l++) != uch) {
                        --nonUch;
                    }
                } 
                if (nonUch < 1 || (maxPos[uch - 'a'] > j || minPos[uch - 'a'] < l)) {
                    maxLen = 
Math.max(maxLen, (j - l + 1));
                }
            }
        }
        
        return maxLen;
    }
}
