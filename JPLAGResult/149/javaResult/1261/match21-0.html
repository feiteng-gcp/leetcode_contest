import java.util.*;

class Solution {
    public int maxRepOpt1(String text) {
        int[] forward = new int[text.length()];
        int[] back = new int[text.length()];
        Map<Character, Integer> map = new HashMap<>();

        for (int i = 0;i < text.length();i++) {
            map.put(text.charAt(i), map.getOrDefault(text.charAt(i), 0) + 1);

            if (i == 0) {
                forward[i] = 1;
            } else {
                if (text.charAt(i) == text.charAt(i - 1)) {
                    forward[i] = forward[i - 1] + 1;
                } else {
                    forward[i] = 1;
                }
            }
        }

        if (map.keySet().size() <= 1) return text.length();

        for (int i = text.length() - 1;i >= 0;i--) {
            if (i == text.length() - 1) {
                back[i] = 1;
            } else {
                if (text.charAt(i) == text.charAt(i + 1)) {
                    back[i] = back[i + 1] + 1;
                } else {
                    back[i] = 1;
                }
            }
        }

        int ret = 0;

        for (int i = 0;i < text.length();i++) {
            ret = Integer.max(ret, forward[i] + (map.get(text.charAt(i)) > forward[i] ? 1 : 0));

            if (i + 2 < text.length() && text.charAt(i) == text.charAt(i + 2)) {
                ret = Integer.max(ret, forward[i] + back[i + 2] + (map.get(text.charAt(i)) > forward[i] + back[i + 2] ? 1 : 0));
            }
        }

        return ret;
    }
}
