class Solution {
    public int maxRepOpt1(String text) {
        final int N = text.length();
        final char[] S = text.toCharArray();
        
        // int[] freq = new int[256];
        ArrayList<CharFreq> freq = new ArrayList<>();
        char prevChar = S[0];
        int count = 1;
        for (int i = 1; i < N; ++i) {
            if (S[i] != prevChar) {
                freq.add(new CharFreq(prevChar, count));
                prevChar = S[i];
                count = 1;
            } else {
                ++count;
            }
        }
        freq.add(new CharFreq(prevChar, count));
        
        int best = 0;
        
        int[] numGroups = new int[256];
        for (CharFreq cf : freq) {
            ++numGroups[cf.value];
            best = Math.max(best, cf.count);
        }
        
        for (int i = 0; i < freq.size(); ++i) {
            CharFreq cf = freq.get(i);
            if (numGroups[cf.value] > 1) {
                best = Math.max(best, 1 + cf.count);
            }
        }
        
        for (int i = 1; i + 1 < freq.size(); ++i) {
            CharFreq cf = freq.get(i);
            if (cf.count > 1) {
                continue;
            }
            CharFreq prev = freq.get(i - 1);
            CharFreq next = freq.get(i + 1);
            if (prev.value != next.value) {
                continue;
            }
            if (numGroups[prev.value] > 2) {
                best = Math.max(best, 1 + prev.count + next.count);
            } else {
                best = Math.max(best, prev.count + next.count);
            }
        }
        
        return best;
    }
    
    private static class CharFreq {
        public char value;
        public int count;
        
        public CharFreq(char v, int c) {
            value = v;
            count = c;
        }
    }
}
