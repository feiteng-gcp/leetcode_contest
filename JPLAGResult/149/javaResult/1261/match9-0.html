class Solution {
    public int maxRepOpt1(String text) {
        int[] cur = new int[26];
        int size = 0, two = 0;
        int[] map = new int[26];
        for (char c : text.toCharArray()) ++map[c - 'a'];
        int i = 0;
        int res = 0;
        for (int j = 0; j < text.length(); ++j) {
            int id = text.charAt(j) - 'a';
            if (cur[id] == 0) ++size;
            if (cur[id] == 1) ++two;
            ++cur[id];
            while (size > 2 || two > 1) {
                int idx = text.charAt(i) - 'a';
                if (cur[idx] == 1) --size;
                if (cur[idx] == 2) --two;
                --cur[idx];
                ++i;
            }
            for (int ch = 0; ch < 26; ++ch) {
                if (cur[ch] + 1 >= j - i + 1) {
                    res = Math.max(res, Math.min(j - i + 1, map[ch]));
                }
            }
        }
        return res;
    }
}
