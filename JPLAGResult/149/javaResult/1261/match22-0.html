class Solution {
    public int maxRepOpt1(String text) {
        // for every letter (only 26), find longest for that letter
        // find the pattern [bunch of a] [single not a] [bunch of a]
        // keep track of globally longest
        
        // for every letter
        // keep arraylist of streak
        int ans = 0;
        for (char c = 'a'; c <= 'z'; c++) {
            List<Integer> streaks = new ArrayList<>();
            List<Boolean> appendable = new ArrayList<>();
            boolean firstStreak = true;
            int startOfCurr = -1;
            int curr = 0;
            for (int i = 0; i < text.length(); i++) {
                if (text.charAt(i) == c) {
                    if (curr == 0) {
                        startOfCurr = i;
                    }
                    curr++;
                } else {
                    if (curr == 0) continue;
                    if (firstStreak) {
                        streaks.add(curr);
                        appendable.add(false);
                        curr = 0;
                        startOfCurr = -1;
                        firstStreak = false;
                    } else {
                        if (text.charAt(startOfCurr - 2) == c) {
                            appendable.add(true);
                        } else {
                            appendable.add(false);
                        }
                        streaks.add(curr);
                        curr = 0;
                    }
                }
            }
            if (curr != 0) {
                if (firstStreak) {
                    streaks.add(curr);
                    appendable.add(false);
                } else {
                    if (text.charAt(startOfCurr - 2) == c) {
                        appendable.add(true);
                    } else {
                        appendable.add(false);
                    }
                    streaks.add(curr);
                }
            }
            
            int longest = 0;
            for (int i = 0; i < streaks.size(); i++) {
                longest = Math.max(longest, streaks.get(i));
                if (appendable.get(i)) {
                    if (streaks.size() >= 3) {
                        longest = Math.max(longest, streaks.get(i) + streaks.get(i - 1) + 1);
                    } else {
                        longest = Math.max(longest, streaks.get(i) + streaks.get(i - 1));
                    }
                }
                if (streaks.size() >= 2) {
                    longest = Math.max(longest, streaks.get(i) + 1);
                }
            }
            ans = Math.max(ans, longest);
        }
        return ans;
    }
}
