class Solution {
    public int maxRepOpt1(String text) {
        int max = 0;
        for(char c = 'a'; c <='z';c++) {
            max = Math.max(max,help(text,c));
        }
        return max;
    }
    
    int help(String text, char c) {
        int total = 0, last = 0, cur = 0;
        int max1 = 0, max2 = 0;
        for(int i = 0 ; i < text.length();i++) {
            if(text.charAt(i) == c) {
                cur++;
                total++;
            }else {
                last = cur;
                cur = 0;
            }
            max1 = Math.max(cur,max1);
            max2 = Math.max(cur+last+1,max2);
        }
        if(total == max1) return max1;
        else if(total < max2) return max2-1;
        else return max2;
    }
}
