import java.util.ArrayList;
import java.util.List;

class Solution {
    public static void main(String[] args) {
        Solution s = new Solution();
        String[] ss = {"ababa", "aaabaaa"};
        for (String tmp : ss) {
            System.out.println(tmp + ":" + s.maxRepOpt1(tmp));
        }
    }

    public int maxRepOpt1(String text) {
        int ans = 0;
        for (int i = 0; i < 26; i++) {
            char c = (char)('a' + i);
            ans = Math.max(ans, maxCharLength(text, c));
        }
        return ans;
    }

    public int maxCharLength(String text, char a) {
        int num = 0;
        for (int i = 0; i < text.length(); i++) if (text.charAt(i) == a) num ++;
        int pos = 0;
        while (pos < text.length() && text.charAt(pos) != a) {
            pos++;
        }

        ArrayList<Integer> al = new ArrayList<>();
        ArrayList<Integer> bl = new ArrayList<>();
        boolean isChar = true;
        while (pos < text.length()) {
            int length = 0;
            while (pos < text.length() && (text.charAt(pos) == a) == isChar) {
                pos++;
                length++;
            }
            if (isChar) {
                al.add(length);
            }else
                bl.add(length);
            isChar = !isChar;
        }
        int maximum = 0;
        for (int i = 0; i < al.size(); i++) {
            maximum = Math.max(maximum, al.get(i));
        }
        maximum = Math.min(maximum + 1, num);
        for (int i = 0; i < bl.size(); i++) {
            if (bl.get(i) == 1 && (i + 1) < al.size()) {
                maximum = Math.max(al.get(i) + al.get(i + 1) + 1, maximum);
            }
        }
        return Math.min(maximum, num);
    }
}
