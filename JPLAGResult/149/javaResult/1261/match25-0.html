public class Solution {
	public int maxRepOpt1(String text) {
		List<int[]>[] segs = new List[26];
		for (int i = 0; i < 26; ++i) segs[i] = new ArrayList<>();
		for (int i = 0; i < text.length(); ++i) {
			int idx = text.charAt(i) - 'a';
			if (i == 0 || text.charAt(i) != text.charAt(i - 1)) {
				segs[idx].add(new int[]{i, 1});
			} else {
				List<int[]> list = segs[idx];
				list.get(list.size() - 1)[1]++;
			}
		}
		int maxRep = 0;
		for (List<int[]> list : segs) {
			maxRep = Math.max(maxRep, maxR(list));
		}
		return maxRep;
	}

	private int maxR(List<int[]> list) {
		if (list.size() <= 1) {
			// max
			int max = 0;
			for (int[] seg : list) {
				max = Math.max(max, seg[1]);
			}
			return max;
		} else if (list.size() == 2) {
			if (list.get(1)[0] == list.get(0)[0] + list.get(0)[1] + 1)
				return list.get(0)[1] + list.get(1)[1];
			return Math.max(list.get(0)[1], list.get(1)[1]) + 1;
		}
		int max = 0;
		for (int i = 1; i < list.size(); ++i) {
			if (list.get(i)[0] == list.get(i - 1)[0] + list.get(i - 1)[1] + 1) {
				max = Math.max(max, list.get(i)[1] + list.get(i - 1)[1] + 1);
			}
		}
		for (int i = 0; i < list.size(); ++i) {
			max = Math.max(max, list.get(i)[1] + 1);
		}
		return max;
	}
}
