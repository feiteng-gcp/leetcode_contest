class Solution {
    public int maxRepOpt1(String text) {
        Map<Character, Integer> map = new HashMap<>();
        int res1 = 0;
        int res2 = 0;
        char[] ca = text.toCharArray();
        int N = ca.length;
        for (int i = 0, localMax = 0; i < N; i++) {
            map.put(ca[i], map.getOrDefault(ca[i], 0) + 1);
            if (i != 0 && ca[i] == ca[i - 1]) {
                localMax++;
            } else {
                localMax = 1;
            }
            res1 = Math.max(res1, localMax);
        }
        for (int j = 0; j < 26; j++) {
            char c = (char)('a' + j);
            int total = map.getOrDefault(c, 0);
            if (total <= res2) {
                continue;
            }
            int[] b = new int[N];
            int[] f = new int[N];
            for (int i = 0, cnt = 0; i < N; i++) {
                b[i] = cnt;
                if (ca[i] == c) {
                    cnt++;
                } else {
                    cnt = 0;
                }
            }
            for (int i = N - 1, cnt = 0; i >= 0; i--) {
                f[i] = cnt;
                if (ca[i] == c) {
                    cnt++;
                } else {
                    cnt = 0;
                }
            }
            
for (int i = 0; i < N; i++) {
                if (ca[i] != c) {
                    
                    res2 = Math.max(res2, Math.min(total, 1 + f[i] + b[i]));
                    
                }
            }
            // int missed = 0;
            // int cnt = 0;
            // for (int j = 0, localMax = 0; j < N; j++) {
            //     if (cnt < total && (ca[j] == c || (missed == 0 && cnt < total))) {
            //         localMax++;
            //         missed += ca[j] == c ? 0 : 1;
            //         cnt++;
            //     } else {
            //         localMax = 0;
            //         missed = 0;
            //         cnt = 0;
            //     }
            //     res2 = Math.max(res2, localMax);
            // }
        }
        return Math.max(res1, res2);
    }
}
