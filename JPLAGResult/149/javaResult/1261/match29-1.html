class Solution {
    public int maxRepOpt1(String text) {
        int[] maxLen = new int[26];
        int[] cnt = new int[26];
        char[] s = text.toCharArray();
        
        int start = 0;
        int change = -1;
        maxLen[s[0] - 'a'] = cnt[s[0] - 'a'] = 1;
        for(int i = 1; i < s.length; ++i){
            cnt[s[i] - 'a']++;
            if(s[i] != s[start]){
                if(i + 1 >= s.length || s[start] != s[i + 1]){
                    int len = i - start + (change == -1 ? 1 : 0);
                    maxLen[s[start] - 'a'] = Math.max(maxLen[s[start] - 'a'], len);
                    start = i;
                    change = -1;
                } else if(i + 2 < s.length && s[start] != s[i + 2]){
                    int len = i - start + (change == -1 ? 2 : 0);
                    maxLen[s[start] - 'a'] = Math.max(maxLen[s[start] - 'a'], len);
                    start = i;
                    change = -1;
                } else {
                    if(change != -1){
                        int len = i - start;
                        maxLen[s[start] - 'a'] = Math.max(maxLen[s[start] - 'a'], len);
                        start = change + 1;
                        change = i;
                    } else {
                        change = i;
                    }
                }
            }
        }
        int len = s.length - start + (change == -1 ? 1 : 0);
        
maxLen[s[start] - 'a'] = Math.max(maxLen[s[start] - 'a'], len);
        int ans = 0;
        for(int i = 0; i < 26; ++i){
            ans = Math.max(ans, Math.min(maxLen[i], cnt[i]));
        }
        return ans;
    }
}
