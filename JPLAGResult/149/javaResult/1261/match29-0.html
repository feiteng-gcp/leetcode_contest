class Solution {
    private String text;
    private int n;
    
    private int solve(char c) {
        int prev = -1;
        int preLen = 0;
        int maxLen = 0;
        int total = 0;
        for (int i = 0; i <= n; ++i) {
            if (i < n && text.charAt(i) == c) {
                ++total;
                continue;
            }
            
            int len = i - prev - 1;
            maxLen = Math.max(preLen + len, maxLen);
            preLen = len;
            prev = i;
        }
        
        if (maxLen == total) {
            return total;
        } else {
            return maxLen + 1;
        }
    }
    
    public int maxRepOpt1(String text) {
        this.text = text;
        this.n = text.length();
        
        int maxRep = 0;
        for (int i = 0; i < 26; ++i) {
            maxRep = Math.max(maxRep, solve((char)('a' + i)));
        }
        return maxRep;
    }
}
