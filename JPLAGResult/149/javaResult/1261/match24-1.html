class Solution {
    public int maxRepOpt1(String text) {
        int hash[] = new int[26];
        for (char c : text.toCharArray()) {
            hash[c - 'a'] ++;
        }
        
        int l = 1, r = text.length() + 1;
        while(l + 1 < r) {
            int m = (l + r) / 2;
            if (check(text, m, hash)) {
                l = m;
            } else {
                r = m;
            }
        }
        return l;
    }
    
    private boolean check(String text, int l, int hash[]) {
        int count[] = new int[26];
        for (int i = 0; i < l; i ++) {
            char c = text.charAt(i);
            count[c - 'a'] ++;
        }
        if (ok(count, hash, l)) {
            return true;
        }
        for (int i = l ; i < text.length(); i ++) {
            int nc = text.charAt(i);
            int oc = text.charAt(i - l);
            count[nc - 'a'] ++;
            count[oc - 'a'] --;
            if (ok(count, hash, l)) {
                return true;
            }
        
}
        return false;
    }
    
    private boolean ok(int[] count, int[] hash, int l) {
        for (int c = 0; c < count.length; c ++) {
            if (count[c] == l) {
                return true;
            }
            if (count[c] == l - 1 && hash[c] >= l) {
                return true;
            }
        }
        return false;
    }
}
