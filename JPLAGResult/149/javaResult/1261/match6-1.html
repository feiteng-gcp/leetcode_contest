class Solution {
    public int maxRepOpt1(String text) {
        char[] ch = text.toCharArray();
        int n = ch.length;
        int[] show = new int[26];
        for (char c: ch) show[c - 'a']++;
        
        int res = 0;
        for (int i=0; i<26; ++i) if (show[i] != 0) {
            res = Math.max(res, longest(ch, (char)(i+'a'), show[i]));
        }
        return res;
    }
    
    private int longest(char[] s, char c, int f) {
        int res = 0;
        
int left = 0, right = 0;
        for (char cs: s) {
            ++right;
            if (cs != c) {
                left = right;
                right = 0;
            }
            res = Math.max(left+right, res);
        }
        
        return Math.min(f, res);
    }
}
