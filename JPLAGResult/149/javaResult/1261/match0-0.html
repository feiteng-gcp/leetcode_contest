class Solution {
    public int maxRepOpt1(String text) {
        int[] cnt = new int[26];
        int len = text.length();
        if(len == 1)
            return 1;
        for(int i = 0; i < len; i++){
            char c = text.charAt(i);
            cnt[c-'a'] ++;
        }
        int[] l = new int[len], r = new int[len];
        l[0] = 1;
        r[len - 1] = 1;
        int res = 1;
        for(int i = 1; i < len; i++){
            l[i] = text.charAt(i) == text.charAt(i - 1) ? l[i-1] + 1: 1;
            res = Math.max(res, l[i]);
        }
        for(int i = len - 2; i >= 0; i--){
            r[i] = text.charAt(i) == text.charAt(i + 1) ? r[i+1] + 1: 1;
            res = Math.max(res, r[i]);
        }
        // System.out.println(Arrays.toString(l));
        // System.out.println(Arrays.toString(r));
        for(int i = 0; i < len; i++){
            if(i == 0 && text.charAt(i) != text.charAt(i + 1) && r[i + 1] + 1 <= cnt[text.charAt(i+1) - 'a']){
                res = Math.max(res, r[1] + 1);
            }else if(i == len - 1 && text.charAt(i) != text.charAt(i-1) && l[i-1] + 1 <= cnt[text.charAt(i-1) - 'a']){
                res = Math.max(res, l[len - 2] + 1);
            }else if(i != 0 && i != len - 1){
                if(text.charAt(i) != text.charAt(i + 1) && text.charAt(i-1) == text.charAt(i + 1)){
                    if(l[i-1] + r[i+1] + 1 <= cnt[text.charAt(i + 1) - 'a'])
                        res = Math.max(res, l[i-1] + r[i+1] + 1);
                    else
                        res = Math.max(res, l[i-1] + r[i+1]);
                }else{
                    if(text.charAt(i) != text.charAt(i + 1) && r[i + 1] + 1 <= cnt[text.charAt(i + 1) - 'a']){
                        res = Math.max(res, r[i+1] + 1);
                    }
                    if(text.charAt(i) != text.charAt(i - 1) && l[i - 1] + 1 <= cnt[text.charAt(i - 1) - 'a']){
                        res = Math.max(res, l[i-1] + 1);
                    }
                }
            }
        }
        return  res;
    }
}
