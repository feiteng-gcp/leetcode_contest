class Solution {
    public int numRollsToTarget(int d, int f, int target) {
        long dp[][] = new long[d+1][target+1];
        dp[0][0] = 1;
        for (int i = 1; i <= d; i++)
        {
            for (int e = 0; e <= target; e++)
            {
                for (int w = 1; w <= f; w++)
                {
                    if (e-w >= 0)
                    {
                        dp[i][e] += dp[i-1][e-w];   
                        dp[i][e] %= 1000000007;
                    
}
                }
            }
        }
        return (int)dp[d][target];
    }
}
