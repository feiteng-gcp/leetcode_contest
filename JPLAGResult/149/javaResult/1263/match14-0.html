class Solution {
    private static int NUM = 1000000007;
    
    public int numRollsToTarget(int d, int f, int target) {
        int[][] dp = new int[1 + d][1 + target];
        
        //Arrays.fill(dp, 0);
        dp[0][0] = 1;
        for (int i = 1; i <= d; i++)
        {
            for (int j = 0; j <= target; j++)
            {
                for (int k = 1; k <= f; k++)
                {
                    if (j - k >= 0)
                    {
                        dp[i][j] += dp[i - 1][j - k];
                        dp[i][j] %= NUM;
                    }
                }
            }
        }
        
        return dp[d][target];
    }
}
