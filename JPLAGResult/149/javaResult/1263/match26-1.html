class Solution {
    // 1 6 3
    public int numRollsToTarget(int d, int f, int target) {
        int[][] dp = new int[d+1][target+1];
      //  this.f = f;
        dp[0][0] = 1;
        
        for(int i = 1; i <= d; i++) {
            for(int j = 1; j <= target; j++) {
                for(int k = 1; k <= f && j-k >= 0; k++) {
                    dp[i][j] += dp[i-1][j-k];
                    dp[i][j] = dp[i][j]%1000000007;
                
}
               // dp[i][j] = dp[i][j]%(10^9 + 7);
               // dp[i][j] = (dp[i][j])%(10^9 + 7);
            }
        }        
        return dp[d][target];
    }
}
