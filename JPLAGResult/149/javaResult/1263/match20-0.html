class Solution {
    public int numRollsToTarget(int d, int f, int target) {
        int[][] dp = new int[35][1005];
        
        dp[0][0] = 1;
        for (int i = 0; i <= d; ++i) {
            for (int j = 0; j <= target; ++j) {
                for (int k = 1; k <= f; ++k) {
                    if (j + k <= target) {
                        dp[i + 1][j + k] = (dp[i + 1][j + k] + dp[i][j]) % 1000000007;
                    }
                }
            }
        }
        return dp[d][target];
    }
}
