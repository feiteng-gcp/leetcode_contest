class Solution {
    public int numRollsToTarget(int d, int f, int target) {
        int[] dp = new int[target+1];
        int mod = (int)1e9+7;
        dp[0] = 1;
        for(int k=0; k<d; k++){
            int[] tmp = new int[target+1];
            for(int i=target; i>=1; i--){
                for(int j=1; j<=f; j++){
                    if(i-j<0) break;
                    tmp[i] += dp[i-j];
                    tmp[i] %= mod;
                }
            }
            dp = tmp;
        }
        return dp[target];
    }
}
