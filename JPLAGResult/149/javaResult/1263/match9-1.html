class Solution {
    public int numRollsToTarget(int dd, int f, int target) {
        
        int p  = 1000000007;
        int n = dd;
        int res[] = new int[f*n+1];
        int x = 0 ;
        
        res[0] = 1;
        
        
        for(int i=1;i<=n;++i){
            
                int nres[] = new int[f*n+1];
                
                for(int st=f*n;st>=0;--st){
                    
                    for(int j=1;j<=f;++j){
                    if(res[st]>0){
                        
                        if(st+j<=f*n){
                            nres[st+j] += res[st];
                            nres[st+j] %= p;
                        }
                        
                    }   
                    
                    }   
                
                
                }
                res = nres;
        }
        
        // int d[] = new int[(f-1)*n+1];
        // for(int i=n;i<=f*n;++i){
        //     d[i-n] = res[i];
        // }
        if(target>=res.length){
            return 0;
        }
        
        return res[target];
        
        
        
    }
}
