class Solution {
    public int numRollsToTarget(int d, int f, int target) {
        int[] dp = new int[target + 1];
        int mod = 1000000007;
        dp[0] = 1;
        for (int i = 0; i < d; i++) {
            int[] dpNext = new int[target + 1];
            for (int j = 0; j <= target; j++) {
                for (int k = 1; k <= f; k++) {
                    if (j - k >= 0) {
                        dpNext[j] += dp[j - k];
                        dpNext[j] = dpNext[j] % mod;
                    
}
                }
            }
            dp = dpNext;
        }
        return dp[target];
    }
}
