class Solution {
    final int MOD = 1000000007;
    public int numRollsToTarget(int d, int f, int target) {
        int[] dp = new int[target + 1];
        dp[0] = 1;
        for(int i = 1; i <= d; ++i){
            int[] curr = new int[target + 1];
            for(int j = 1; j <= f; ++j){
                for(int k = 0; k <= target - j; ++k){
                    if(dp[k] > 0){
                        curr[j + k] = (curr[j + k] + dp[k]) % MOD;
                    }
                
}
            }
            dp = curr;
        }
        return dp[target];
    }
}
