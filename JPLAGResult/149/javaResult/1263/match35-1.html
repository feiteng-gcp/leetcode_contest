class Solution {
    
public int numRollsToTarget(int d, int f, int target) {
    	long[][] num = new long[d+1][target+40];
    	
    	for(int i = 1; i <= f; i++) {

    		num[1][i] = 1;
    	}
    	//System.out.println(Arrays.toString(num[1]));
    	for(int i = 1; i < d; i++) {
    		for(int j = 1; j <= target; j++) {
    			for(int k = 1; k <= f; k++){ONT color="#77bfc7">

    				num[i+1][j+k] = (num[i+1][j+k] + num[i][j]) % 1000000007;
    				//System.out.print(num[i+1][j+k] + " ");
    			}
    			//System.out.println(i + ", " + j + " : " + Arrays.toString(num[i+1]));
    		}
    	}
    	
        return (int)(num[d][target] % 1000000007);
    }
}
