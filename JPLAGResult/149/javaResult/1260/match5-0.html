class Solution {
    public int ordinalOfDate(String date) {
        String[] strs = date.split("-");
        int year=Integer.parseInt(strs[0]), month=Integer.parseInt(strs[1]), res=Integer.parseInt(strs[2]);
        switch(month-1){
            case 12: res+=31;
            case 11: res+=30;
            case 10: res+=31;
            case 9: res+=30;
            case 8: res+=31;
            case 7: res+=31;
            case 6: res+=30;
            case 5: res+=31;
            case 4: res+=30;
            case 3: res+=31;
            case 2: res+=isLeap(year)? 29: 28;
            case 1: res+=31;
        }
        return res;
    }
    
    private boolean isLeap(int year){
        return !(year%4!=0 || year%100==0&&year%400!=0);
    }
}
