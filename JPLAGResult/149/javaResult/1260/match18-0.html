class Solution {
    public int dayOfYear(String date) {
            int ds[] = {31,28,31,30,31,30,31,31,30,31,30,31};
        
        String[] y = date.split("-");
        
        int ye = Integer.parseInt(y[0]);
        int m = Integer.parseInt(y[1])-1;
        int d = Integer.parseInt(y[2]);
        
        if(isL(ye)){
            ds[1]++;
        }
        int s = 0;
        for(int i=0;i<m;++i){
            s += ds[i];
        }
        return s+d;
        
        
    }
    
    public boolean isL(int y){
        return (y%4==0&&y%100!=0)||(y%400==0);
    }
    
}
