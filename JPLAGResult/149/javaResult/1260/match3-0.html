class Solution {
    public int dayOfYear(String date) {
        String[] strs = date.split("-");
        Map<Integer, Integer> map = new HashMap<>();
        map.put(1, 31);
        map.put(2, 28);
        map.put(3, 31);
        map.put(4, 30);
        map.put(5, 31);
        map.put(6, 30);
        map.put(7, 31);
        map.put(8, 31);
        map.put(9, 30);
        map.put(10, 31);
        map.put(11, 30);
        map.put(12, 31);

        int ans = Integer.parseInt(strs[2]);
        int month = Integer.parseInt(strs[1]);

        for (int m = 1; m < month; m++) {
            ans += map.get(m);
        }

        if (isRunnian(strs[0])  && month>2 ){
            ans++;
        }
        return ans;
    }

    public boolean isRunnian(String year) {
        int num = Integer.parseInt(year);

        return num % 400 == 0 || num % 4 == 0 && num % 100 != 0;
    }
}
