class Solution {
    private int[] daysPerMonth = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    public int dayOfYear(String date) {
        String year = date.substring(0, 4), month = date.substring(5, 7), day = date.substring(8);
        int y = Integer.parseInt(year), m = Integer.parseInt(month), d = Integer.parseInt(day);

        if (y % 4 == 0 && (y % 100 != 0 || y % 400 == 0)) {
            daysPerMonth[1] = 29;
        }

        int countDays = 0;
        for (int i = 0; i < m - 1; ++i) {
            countDays += daysPerMonth[i];
        }
        countDays += d;
        return countDays;
    }
}
