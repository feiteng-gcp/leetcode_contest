class Solution {
    public static void main(String[] args) {
        System.out.println(new Solution().dayOfYear("1900-03-25"));
    }

    public int dayOfYear(String date) {
        String[] ss = date.split("-");
        int yyyy = Integer.parseInt(ss[0]);
        int mmmm = Integer.parseInt(ss[1]);
        int dddd = Integer.parseInt(ss[2]);

        int[][] dates = {
                {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31},
                {31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31},
         };

        int b = (yyyy % 4 == 0) ? ((yyyy % 100) == 0 ? (yyyy % 400 == 0 ? 1 : 0) : 1) : 0;
        int ans = dddd;
        for (int i = 0; i < mmmm - 1; i++) {
            ans += dates[b][i];
        }

        return ans;
    }
}
