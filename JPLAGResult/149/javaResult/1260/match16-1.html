public class Solution {
    public static void main(String[] args) {
        Solution solution=new Solution();
        System.out.println(solution.dayOfYear("2004-03-01"));
        System.out.println(solution.dayOfYear("2003-03-01"));
        System.out.println(solution.dayOfYear("2019-02-10"));
    }

    
public int dayOfYear(String date) {
        int monthDays[]=new int[]{0, 31,28,31,30,31,30,31,31,30,31,30,31};
        String numStrs[]=date.split("-");
        int res=Integer.valueOf(numStrs[2]);
        int month=Integer.valueOf(numStrs[1]);
        int year=Integer.valueOf(numStrs[0]);
        
if(year%400==0 || (year%4==0 && year%100!=0))
            monthDays[2]=29;
        for(int i=0;i<month;i++){
            res+=monthDays[i];
        }
        return res;
    }
}
