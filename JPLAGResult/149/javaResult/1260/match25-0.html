import java.text.SimpleDateFormat;
import java.util.Date;

class Solution {
    public int ordinalOfDate(String date1) {
        SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
        try {
            Date date = sdf.parse(date1);
            long a = date.getTime();
            date.setMonth(0);
            date.setDate(1);
            return (int)((a - date.getTime()) / 3600 / 1000 / 24 + 1);
        } catch (Exception e) {
            e.printStackTrace();
        }
        return -1;
    }
}
