class Solution {
    public int dayOfYear(String date) {
        String[] yymmdd = date.split("-");
        int year = Integer.parseInt(yymmdd[0]);
        int month = Integer.parseInt(yymmdd[1]);
        int day = Integer.parseInt(yymmdd[2]);
        
        int daysInFeb = 28;
        if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0) {
            daysInFeb = 29;
        }
        
        int[] daysPerMonth = new int[] { 31, daysInFeb, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
        
        int ans = 0;
        for (int i = 2; i <= month; i++) {
            ans += daysPerMonth[i - 2];
        }
        
        return ans + day;
    }
}
