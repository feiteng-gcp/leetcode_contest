class Solution {
    public int dayOfYear(String date){
        String[] strs = date.split("-");
        int year = Integer.valueOf(strs[0]), month = Integer.valueOf(strs[1]), day = Integer.valueOf(strs[2]);
        int[] arr = new int[]{31, 28, 31, 30,31,30,31,31,30,31,30,31};
        int res = 0;
        for(int i = 1; i < month; i++){
            res += arr[i - 1];
        }
        res += day;
        if(month > 2){
            if(year % 100 == 0){
                if(year % 400 == 0)
                    res ++;
            }else{
                if(year % 4 == 0)
                    res ++;
            }
        }
        return res;
    }
}
