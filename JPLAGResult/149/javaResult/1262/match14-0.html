class MajorityChecker {
    
    int[] arr;
    
    public MajorityChecker(int[] arr) {
        this.arr = arr;
    }
    
    public int query(int left, int right, int threshold) {
        int majItem = arr[left];
        int count = 1;
        for (int i = left + 1; i <= right; i++) {
            if (arr[i] == majItem) count++;
            else {
                if (count == 0) {
                    majItem = arr[i];
                    count = 1;
                } else {
                    count--;
                }
            }
        }
        count = 0;
        for (int i = left; i <= right; i++) {
            if (arr[i] == majItem) count++;
        }
        if (count >= threshold) {
            return majItem;
        }
        return -1;
    }
}
