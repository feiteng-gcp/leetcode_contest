class MajorityChecker {

    int[] as;
    public MajorityChecker(int[] arr) {
        as = arr;
    }
    
    public int query(int left, int right, int threshold) {
        int curr = as[left];
        int count = 1;
        for (int i = left + 1;i <= right;i++) {
            if (as[i] == curr) count++;
            else {
                count--;
                if (count < 0) {
                    curr = as[i];
                    count = 1;
                }
            }
        }
        int c2 = 0;
        for (int i = left;i <= right;i++) {
            if (as[i] == curr) c2++;
        } 
        return c2 >= threshold ? curr : -1;
    }
}
