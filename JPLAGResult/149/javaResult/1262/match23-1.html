class MajorityChecker {
        int[] tmp;
        public MajorityChecker(int[] arr) {
            tmp = arr;
        }

        public int query(int left, int right, int threshold) {
            int count = 0;
            int element = -1;
            for (int i = left; i <= right; i++) {
                if (count == 0 && element != tmp[i]) {
                    element = tmp[i];
                    count = 1;
                } else if (element == tmp[i]) {
                    count ++;
                } else {
                    count --;
                }
            }
            count = 0;
            
for (int i = left; i<= right; i++) {
                if (tmp[i] == element) {
                    count ++;
                }
            }
            if (count >= threshold) {
                return element;
            }
            return -1;

        }
    }
