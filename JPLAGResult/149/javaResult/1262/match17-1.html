class MajorityChecker {
    int[] A;
    public MajorityChecker(int[] arr) {
        A = arr;
    }
    
    public int query(int left, int right, int threshold) {
        if (A == null || A.length == 0 || left >= A.length || right < 0) {
            return -1;
        }
        int num = A[left];
        int cnt = 1;
        for (int i = left + 1; i <= Math.min(right, A.length - 1); ++i) 
{
            if (A[i] == num) {
                cnt++;
            }else {
                cnt--;
                if (cnt == -1) {
                    num = A[i];
                    cnt = 1;
                }
            }
        }
        
        cnt = 0;
        for (int i = left; i <= right; ++i) {
            if (A[i] == num) {
                cnt++;
            }
        }
        
        if (cnt >= threshold) {
            return num;
        }
        return -1;
    }
}
