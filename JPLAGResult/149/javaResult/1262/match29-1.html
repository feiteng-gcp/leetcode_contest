class MajorityChecker {
    int[] a;
    public MajorityChecker(int[] arr) {
        a = arr;
    }
    
    public int query(int left, int right, int threshold) {
        int num = 0;
        for (int i = left, cnt = 0; i <= right; i++) {
            if (cnt == 0) {
                num = a[i];
                cnt++;
                continue;
            }
            if (a[i] == num) {
                cnt++;
            
} else {
                cnt--;
            }
        }
        int total = 0;
        for (int i = left; i <= right; i++) {
            if (a[i] == num) {
                total++;
            }
        }
        return total >= threshold ? num : -1;
    }
}
