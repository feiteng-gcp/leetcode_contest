class MajorityChecker {

    public static void main(String[] args) {
        MajorityChecker solution=new MajorityChecker(new int[]{1,1,2,2,1,1});
        System.out.println(solution.query(0,5,4));
        System.out.println(solution.query(0,3,3));
        System.out.println(solution.query(2,3,2));
    }

    
int arr[];

    public MajorityChecker(int[] arr) {
        this.arr=arr;
    }

    public int query(int left, int right, int threshold) {
        int candidate=arr[left];
        int count=1;
        for(int i=left+1;i<=right;i++){
            if(arr[i]==candidate){
                count++;
            }else{
                count--;
                if(count==-1){
                    count=1;
                    candidate=arr[i];
                
}
            }
        }
        if(count<0) return -1;
        count=0;
        for(int i=left;i<=right;i++){
            if(arr[i]==candidate) {
                count++;
            
}
        }
        if(count>=threshold) return candidate;
        return -1;
    }
}
