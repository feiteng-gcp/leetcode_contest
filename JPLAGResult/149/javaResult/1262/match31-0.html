class MajorityChecker {

    int v[];
  //  short dp[][] = new short[20001][20002];
    
    public MajorityChecker(int[] arr) {
        v = arr;

    }
    
    public int query(int l, int r, int ts) {
        
        int rr = v[l];
        int t = 1;
        for(int i=l+1;i<=r;++i){
            if(v[i]==rr){
                t++;
            }else{
                t--;
                
                if(t==0){
                    rr = v[i];
                    t = 1;
                }
            }
           
                
            
            
        }
        int s =0;
        for(int i=l;i<=r;++i){
            s += (v[i]==rr?1:0);
            if(s>=ts){
                return rr;
            }
        }
        
        
        return -1;
    }
}
