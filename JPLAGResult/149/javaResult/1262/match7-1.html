class MajorityChecker {
    int[] arr;
    
    public MajorityChecker(int[] arr) {
        this.arr = arr;
    }
    
    public int query(int left, int right, int threshold) {
        int count = 1;
        int cur = arr[left];
        for (int i = left+1; i <= right; i++) {
            if (arr[i] != cur) {
                count--;
                if (count < 0) {
                    cur = arr[i];
                    count = 1;
                }
            }
            else {
                count++;
            
}
        }
        
        count = 0;
        for (int i = left; i <= right; i++) {
            if (arr[i] == cur) count++;
        }
        
        return count < threshold ? -1 : cur;
    }
}
