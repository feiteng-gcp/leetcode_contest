class MajorityChecker {
    int[] ar;
    public MajorityChecker(int[] arr) {
         ar = arr;
    }
    
    public int query(int left, int right, int threshold) {
        int ct = 0;
        int cur = -1;
        for (int i = left; i <= right; i++) {
            if (ct == 0) {
                cur = ar[i];
                ct++;
            } else {
                if (cur == ar[i]) ct++;
                else ct--;
            }
        }
        if (cur == -1) return -1;
        //System.out.println(cur);
        ct = 0;
        for (int i = left; i <= right; i++) {
            if (ar[i] == cur) ct++;
        }
        if (ct >= threshold) return cur;
        return -1;
    }
}
