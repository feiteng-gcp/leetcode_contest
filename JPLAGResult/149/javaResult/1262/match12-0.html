class MajorityChecker {
    
    int[] arr = null;

    public MajorityChecker(int[] arr) {
        this.arr = arr;
    }
    
    public int query(int left, int right, int threshold) {
        int count = 0;
        int cur = -1;
        for(int i = left;i<=right;i++) {
            if(arr[i] == cur) count++;
            else {
                if(count == 0) {
                    cur = arr[i];
                    count = 1;
                }
                else count--;
            }
        }
        count = 0;
        for(int i = left;i<=right;i++) {
            if(arr[i] == cur) count++;
        }
        return count >= threshold ? cur : -1;
    }
}
