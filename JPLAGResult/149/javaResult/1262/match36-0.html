class MajorityChecker {
   int[] arr;
    public MajorityChecker(int[] arr) {
        this.arr  = arr;
    }
    
    public int query(int left, int right, int threshold) {
        int ans = arr[left];
        int cnt = 0;
        for(int i = left; i <= right; i++){
            if(ans == arr[i]){
                cnt++;
            }else{
                if(cnt > 0) cnt--;
                else{
                    cnt =1;
                    ans = arr[i];
                }
            }
        }
        cnt = 0;
        for(int i = left; i <= right; i++){
            if(ans == arr[i])
                cnt++;
        }
        if(cnt >= threshold)
            return ans;
        return -1;
    }
}
