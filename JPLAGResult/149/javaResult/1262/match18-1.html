class MajorityChecker {
    int[] haha;
    public MajorityChecker(int[] arr) {
        haha = arr;
    }
    
    public int query(int left, int right, int threshold) {
        int candidate = haha[0];
        int count = 0;
        for(int i = left; i <= right; i++) {
            if(haha[i] == candidate){
                count++;
            }else{
                if(count == 0) {
                    candidate = haha[i];
                    count = 1;
                }else{
                    count--;
                
}
            }
        }
        count = 0;
        for(int i = left; i <= right; i++) {
            if(haha[i] == candidate){
                count++;
            }
        }
        if(count >= threshold){
            return candidate;
        }else{
            return -1;
        }
    }
}
